13#[derive(Clone, Debug)]
14enum PrivateKey {
15    Bls(PrivKey),
16    Schnorr(StateSignKey),
17}
18
19impl FromStr for PrivateKey {
20    type Err = anyhow::Error;
21
22    fn from_str(s: &str) -> Result<Self, Self::Err> {
23        let tb64 = TaggedBase64::parse(s)?;
24        if let Ok(key) = tb64.clone().try_into() {
25            Ok(Self::Bls(key))
26        } else if let Ok(key) = tb64.try_into() {
27            Ok(Self::Schnorr(key))
28        } else {
29            bail!("unrecognized key type")
30        }
31    }
32}
33
34/// Get the public key corresponding to a private key.
35#[derive(Clone, Debug, Parser)]
36pub struct Options {
37    /// The private key to get the public key for.
38    key: PrivateKey,
39
40    // Whether or not to derive the libp2p peer ID from the private key.
41    #[clap(long, short)]
42    libp2p: bool,
43}
44
45pub fn run(opt: Options) {
46    match (opt.libp2p, opt.key) {
47        // Non-libp2p
48        (false, PrivateKey::Bls(key)) => println!("{}", PubKey::from_private(&key)),
49        (false, PrivateKey::Schnorr(key)) => {
50            println!("{}", StateKeyPair::from_sign_key(key).ver_key())
51        },
52
53        // Libp2p
54        (true, PrivateKey::Bls(key)) => {
55            println!(
56                "{}",
57                derive_libp2p_peer_id::<BLSPubKey>(&key).expect("Failed to derive libp2p peer ID")
58            );
59        },
60        (true, _) => {
61            eprintln!("Key type unsupported for libp2p peer ID derivation");
62        },
63    }
64}
