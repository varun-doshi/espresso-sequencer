7/// Options for resetting persistent storage.
8///
9/// This will remove all the persistent storage of a sequencer node or marketplace solver, effectively resetting it to
10/// its genesis state. Do not run this program while the sequencer or solver is running.
11#[derive(Clone, Debug, Subcommand)]
12pub enum Commands {
13    /// Contains subcommands for resetting sequencer storage.
14    #[command(subcommand)]
15    Sequencer(SequencerStorage),
16    /// resetting marketplace solver storage.
17    Solver(marketplace_solver::DatabaseOptions),
18}
19
20#[derive(Clone, Debug, Subcommand)]
21pub enum SequencerStorage {
22    /// Reset file system storage.
23    Fs(persistence::fs::Options),
24    /// Reset SQL storage.
25    Sql(Box<persistence::sql::Options>),
26}
27
28pub async fn run(opt: Commands) -> anyhow::Result<()> {
29    match opt {
30        Commands::Sequencer(query_resetter) => match query_resetter {
31            SequencerStorage::Fs(opt) => {
32                tracing::warn!("resetting sequencer file system storage {opt:?}");
33                reset_storage(opt).await
34            },
35            SequencerStorage::Sql(opt) => {
36                tracing::warn!("resetting sequencer SQL storage {opt:?}");
37                reset_storage(*opt).await
38            },
39        },
40
41        Commands::Solver(opt) => {
42            tracing::warn!("resetting solver SQL storage {opt:?}");
43            let opts = opt.reset();
44            opts.connect().await?;
45
46            Ok(())
47        },
48    }
49}
50
51async fn reset_storage<O: DataSourceOptions>(opt: O) -> anyhow::Result<()> {
52    // Reset query service storage.
53    O::DataSource::create(opt.clone(), Default::default(), true).await?;
54    // Reset consensus storage.
55    opt.reset().await?;
56
57    Ok(())
58}
