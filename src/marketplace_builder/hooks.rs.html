32/// Fetch registered namespaces from the solver and construct the list of namespaces to skip.
33///
34/// # Returns
35/// - `Some` namespaces if the fetching succeeds, even if the list is empty.
36/// - `None` if the fetching fails.
37pub async fn fetch_namespaces_to_skip(solver_base_url: Url) -> Option<HashSet<NamespaceId>> {
38    let solver_client = connect_to_solver(solver_base_url);
39    match solver_client
40        .get::<Vec<RollupRegistration>>("rollup_registrations")
41        .send()
42        .await
43    {
44        Ok(registrations) => {
45            let mut namespaces_to_skip = HashSet::new();
46            for registration in registrations {
47                if registration.body.reserve_url.is_some() || !registration.body.active {
48                    namespaces_to_skip.insert(registration.body.namespace_id);
49                }
50            }
51            Some(namespaces_to_skip)
52        },
53        Err(e) => {
54            error!("Failed to get the registered rollups: {:?}.", e);
55            None
56        },
57    }
58}

60/// Reserve builder hooks for espresso sequencer.
61///
62/// Provides bidding and transaction filtering on top of base builder functionality.
63pub(crate) struct EspressoReserveHooks {
64    /// IDs of namespaces to filter and bid for
65    pub(crate) namespaces: HashSet<NamespaceId>,
66    /// Base API to contact the solver
67    pub(crate) solver_base_url: Url,
68    /// Builder API base to include in the bid
69    pub(crate) builder_api_base_url: Url,
70    /// Keys for bidding
71    pub(crate) bid_key_pair: EthKeyPair,
72    /// Bid amount
73    pub(crate) bid_amount: FeeAmount,
74}
75
76#[async_trait]
77impl BuilderHooks<SeqTypes> for EspressoReserveHooks {
78    #[inline(always)]
79    async fn process_transactions(
80        &self,
81        mut transactions: Vec<<SeqTypes as NodeType>::Transaction>,
82    ) -> Vec<<SeqTypes as NodeType>::Transaction> {
83        transactions.retain(|txn| self.namespaces.contains(&txn.namespace()));
84        transactions
85    }
86
87    #[inline(always)]
88    async fn handle_hotshot_event(&self, event: &Event<SeqTypes>) {
89        let EventType::ViewFinished { view_number } = event.event else {
90            return;
91        };
92
93        let fee_account = self.bid_key_pair.fee_account();
94        let bid_amount = self.bid_amount;
95        let namespaces = self.namespaces.iter().cloned().collect();
96        let builder_api_base_url = self.builder_api_base_url.clone();
97        let bid_key_pair = self.bid_key_pair.clone();
98        let solver_base_url = self.solver_base_url.clone();
99
100        spawn(async move {
101            let bid_tx = match BidTxBody::new(
102                fee_account,
103                bid_amount,
104                view_number + 3, // We submit a bid 3 views in advance.
105                namespaces,
106                builder_api_base_url,
107                Default::default(),
108            )
109            .signed(&bid_key_pair)
110            {
111                Ok(bid) => bid,
112                Err(e) => {
113                    error!("Failed to sign the bid txn: {:?}.", e);
114                    return;
115                },
116            };
117
118            let solver_client = connect_to_solver(solver_base_url);
119            if let Err(e) = solver_client
120                .post::<()>("submit_bid")
121                .body_json(&bid_tx)
122                .unwrap()
123                .send()
124                .await
125            {
126                error!("Failed to submit the bid: {:?}.", e);
127                return;
128            }
129
130            info!("Submitted bid for view {}", *view_number);
131        });
132    }
133}

135/// Fallback builder hooks for espresso sequencer.
136///
137/// Provides transaction filtering on top of base builder functionality for unregistered rollups.
138pub(crate) struct EspressoFallbackHooks {
139    /// Base URL to contact the solver.
140    pub(crate) solver_base_url: Url,
141    pub(crate) namespaces_to_skip: Arc<RwLock<Option<HashSet<NamespaceId>>>>,
142}
143
144#[async_trait]
145impl BuilderHooks<SeqTypes> for EspressoFallbackHooks {
146    #[inline(always)]
147    async fn process_transactions(
148        &self,
149        mut transactions: Vec<<SeqTypes as NodeType>::Transaction>,
150    ) -> Vec<<SeqTypes as NodeType>::Transaction> {
151        let namespaces_to_skip = self.namespaces_to_skip.read().await;
152
153        match namespaces_to_skip.as_ref() {
154            Some(namespaces_to_skip) => {
155                transactions.retain(|txn| !namespaces_to_skip.contains(&txn.namespace()));
156                transactions
157            },
158            // Solver connection has failed and we don't have up-to-date information on this
159            None => {
160                error!("Not accepting transactions due to outdated information");
161                Vec::new()
162            },
163        }
164    }
165
166    #[inline(always)]
167    async fn handle_hotshot_event(&self, event: &Event<SeqTypes>) {
168        let EventType::ViewFinished { view_number } = event.event else {
169            return;
170        };
171
172        // Re-query the solver every 20 views
173        if view_number.rem_euclid(20) != 0 {
174            return;
175        }
176
177        let solver_base_url = self.solver_base_url.clone();
178        let namespaces_to_skip_lock = Arc::clone(&self.namespaces_to_skip);
179
180        spawn(async move {
181            let namespaces_to_skip = fetch_namespaces_to_skip(solver_base_url).await;
182            *namespaces_to_skip_lock.write().await = namespaces_to_skip;
183        });
184    }
185}
