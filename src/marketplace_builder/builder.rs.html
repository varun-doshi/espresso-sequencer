104    async fn start_service(
105        global_state: Arc<GlobalState<SeqTypes, DynamicHooks>>,
106        events_api_url: Url,
107        builder_api_url: Url,
108    ) -> anyhow::Result<()> {
109        // create the proxy global state it will server the builder apis
110        let app = Arc::clone(&global_state)
111            .into_app()
112            .context("Failed to construct builder API app")?;
113
114        spawn(async move {
115            tracing::info!("Starting builder API app at {builder_api_url}");
116            let res = app
117                .serve(builder_api_url, MarketplaceVersion::instance())
118                .boxed() // https://github.com/rust-lang/rust/issues/102211
119                .await;
120            tracing::error!(?res, "Builder API app exited");
121        });
122
123        // spawn the builder service
124        tracing::info!("Running builder against hotshot events API at {events_api_url}",);
125
126        let stream =
127            EventServiceStream::<SeqTypes, SequencerApiVersion>::connect(events_api_url).await?;
128
129        spawn(async move {
130            let res = global_state.start_event_loop(stream).await;
131            tracing::error!(?res, "Builder service exited");
132            if res.is_err() {
133                panic!("Builder should restart.");
134            }
135        });
136
137        Ok(())
138    }

140    #[allow(clippy::too_many_arguments)]
141    pub async fn init(
142        is_reserve: bool,
143        builder_key_pair: EthKeyPair,
144        bootstrapped_view: ViewNumber,
145        tx_channel_capacity: NonZeroUsize,
146        event_channel_capacity: NonZeroUsize,
147        instance_state: NodeState,
148        events_api_url: Url,
149        builder_api_url: Url,
150        api_timeout: Duration,
151        maximize_txns_count_timeout_duration: Duration,
152        base_fee: FeeAmount,
153        bid_config: Option<BidConfig>,
154        solver_base_url: Url,
155    ) -> anyhow::Result<Self> {
156        tracing::info!(
157            address = %builder_key_pair.fee_account(),
158            ?bootstrapped_view,
159            %tx_channel_capacity,
160            %event_channel_capacity,
161            ?api_timeout,
162            ?instance_state.chain_config.max_block_size,
163            ?maximize_txns_count_timeout_duration,
164            "initializing builder",
165        );
166
167        let hooks: DynamicHooks = if is_reserve {
168            let bid_config = bid_config.expect("Missing bid config for the reserve builder.");
169            Box::new(hooks::EspressoReserveHooks {
170                namespaces: bid_config.namespaces.into_iter().collect(),
171                solver_base_url,
172                builder_api_base_url: builder_api_url.clone(),
173                bid_key_pair: builder_key_pair.clone(),
174                bid_amount: bid_config.amount,
175            })
176        } else {
177            // Fetch the namespaces upon initialization. It will be fetched every 20 views when
178            // handling events.
179            let namespaces_to_skip = fetch_namespaces_to_skip(solver_base_url.clone()).await;
180            Box::new(hooks::EspressoFallbackHooks {
181                solver_base_url,
182                namespaces_to_skip: RwLock::new(namespaces_to_skip).into(),
183            })
184        };
185
186        // create the global state
187        let global_state = GlobalState::new(
188            marketplace_builder_core::service::BuilderConfig {
189                builder_keys: (builder_key_pair.fee_account(), builder_key_pair),
190                api_timeout,
191                tx_capture_timeout: maximize_txns_count_timeout_duration,
192                txn_garbage_collect_duration: Duration::from_secs(60),
193                txn_channel_capacity: tx_channel_capacity.get(),
194                tx_status_cache_capacity: 81920,
195                base_fee: base_fee.as_u64().expect("Base fee too high"),
196            },
197            hooks,
198        );
199
200        Self::start_service(
201            Arc::clone(&global_state),
202            events_api_url.clone(),
203            builder_api_url.clone(),
204        )
205        .await?;
206
207        tracing::info!("Builder init finished");
208
209        Ok(Self {
210            global_state,
211            hotshot_events_api_url: events_api_url,
212            hotshot_builder_apis_url: builder_api_url,
213        })
214    }

422        if let EventType::QuorumProposal { proposal, .. } = event.event {
423            let view_number = *proposal.data.view_number();
424            let commitment = Leaf2::from_quorum_proposal(&proposal.data).payload_commitment();
425            return Some((view_number, commitment));
426        }
427        None

430    /// Wait for a quorum proposal event and get its view number and commitment.
431    async fn wait_for_proposal_view_number_and_commitment(
432        events: &mut (impl Stream<Item = Event> + Unpin),
433    ) -> (u64, VidCommitment) {
434        let start = Instant::now();
435        loop {
436            if start.elapsed() > Duration::from_secs(5) {
437                panic!("Didn't get a quorum proposal in 5 seconds");
438            }
439            let event = events.next().await.unwrap();
440            if let Some((view_number, commitment)) =
441                proposal_view_number_and_commitment(event).await
442            {
443                return (view_number, commitment);
444            }
445        }
446    }

449    async fn wait_for_transaction(
450        events: &mut (impl Stream<Item = Event> + Unpin),
451        transaction: Transaction,
452    ) {
453        let start = Instant::now();
454        loop {
455            if start.elapsed() > Duration::from_secs(5) {
456                panic!("Didn't get the transaction in 5 seconds");
457            }
458            let event = events.next().await.unwrap();
459            if let EventType::Transactions { transactions: txns } = event.event {
460                if txns == vec![transaction.clone()] {
461                    return;
462                }
463            }
464        }
465    }

486    /// Submit transactions via the private mempool and fetch the bundle.
487    async fn submit_and_get_bundle_with_private_mempool(
488        builder_client: Client<ServerError, MarketplaceVersion>,
489        transactions: Vec<Transaction>,
490        urls: Urls,
491    ) -> (Bundle<SeqTypes>, u64) {
492        // Subscribe to events.
493        let events_service_client = Client::<
494            hotshot_events_service::events::Error,
495            SequencerApiVersion,
496        >::new(urls.event.clone());
497        events_service_client.connect(None).await;
498        let mut events = events_service_client
499            .socket("hotshot-events/events")
500            .subscribe::<Event>()
501            .await
502            .unwrap();
503
504        // Submit transactions via the private mempool.
505        let txn_submission_client: Client<ServerError, SequencerApiVersion> =
506            Client::new(urls.builder.clone());
507        txn_submission_client.connect(None).await;
508        txn_submission_client
509            .post::<Vec<Commitment<Transaction>>>("txn_submit/batch")
510            .body_json(&transactions)
511            .unwrap()
512            .send()
513            .await
514            .unwrap();
515
516        // Get the parent view number and commitment.
517        let parent_view_number;
518        let parent_commitment;
519        let start = Instant::now();
520        loop {
521            if start.elapsed() > Duration::from_secs(5) {
522                panic!("Didn't get a quorum proposal in 5 seconds");
523            }
524            let event = events.next().await.unwrap().unwrap();
525            if let Some((view_number, commitment)) =
526                proposal_view_number_and_commitment(event).await
527            {
528                parent_view_number = view_number;
529                parent_commitment = commitment;
530                break;
531            }
532        }
533
534        // Fetch the bundle.
535        (
536            get_bundle(builder_client, parent_view_number, parent_commitment).await,
537            parent_view_number,
538        )
539    }

676                    .submit(Default::default())
677                    .query_fs(
678                        Default::default(),
679                        persistence::fs::Options::new(tmpdir.path().to_owned()),
680                    )
681                    .hotshot_events(HotshotEvents {
682                        events_service_port: ports.event,
683                    }),
684            )
685            .network_config(network_config)
686            .build();
687        let network = TestNetwork::new(config, MockSequencerVersions::new()).await;
688
689        // Register a rollup using the mock solver.
690        // Use `_mock_solver` here to avoid it being dropped.
691        let (_mock_solver, solver_base_url) = init_mock_solver_and_register_rollup().await;
692
693        let keypair = FeeAccount::test_key_pair();
694        let address = keypair.address();
695        let base_fee = FeeAmount::from(10);
696
697        // Start and connect to a fallback builder.
698        let init = BuilderConfig::init(
699            false,
700            FeeAccount::test_key_pair(),
701            ViewNumber::genesis(),
702            NonZeroUsize::new(1024).unwrap(),
703            NonZeroUsize::new(1024).unwrap(),
704            NodeState::default(),
705            urls.event.clone(),
706            urls.builder.clone(),
707            Duration::from_secs(2),
708            Duration::from_secs(2),
709            base_fee,
710            None,
711            solver_base_url,
712        );
713        let _ = init.await.unwrap();
714        let builder_client = connect_to_builder(urls.clone()).await;
715
716        // Construct transactions.
717        let registered_transaction =
718            Transaction::new(REGISTERED_NAMESPACE.into(), vec![1, 1, 1, 1]);
719        let unregistered_transaction =
720            Transaction::new(UNREGISTERED_NAMESPACE.into(), vec![1, 1, 1, 2]);
721
722        let (bundle, parent_view_number) = match mempool {
723            Mempool::Public => {
724                let server = &network.server;
725                let mut events = server.event_stream().await;
726
727                // Get the parent information before submitting transactions.
728                let (parent_view_number, parent_commitment) =
729                    wait_for_proposal_view_number_and_commitment(&mut events).await;
730
731                // Submit transactions and wait until they are received.
732                server
733                    .submit_transaction(registered_transaction.clone())
734                    .await
735                    .unwrap();
736                wait_for_transaction(&mut events, registered_transaction).await;
737                server
738                    .submit_transaction(unregistered_transaction.clone())
739                    .await
740                    .unwrap();
741                wait_for_transaction(&mut events, unregistered_transaction.clone()).await;
742
743                // Get the bundle.
744                (
745                    get_bundle(builder_client, parent_view_number, parent_commitment).await,
746                    parent_view_number,
747                )
748            },
749            Mempool::Private => {
750                submit_and_get_bundle_with_private_mempool(
751                    builder_client,
752                    vec![registered_transaction, unregistered_transaction.clone()],
753                    urls,
754                )
755                .await
756            },
757        };
758
759        assert_eq!(bundle.transactions, vec![unregistered_transaction.clone()]);
760
761        let txn_commit = <[u8; 32]>::from(unregistered_transaction.clone().commit()).to_vec();
762        let signature = bundle.signature;
763        assert!(signature.verify(txn_commit, address).is_ok());
764
765        let fee = base_fee * unregistered_transaction.minimum_block_size();
766
767        let fee_signature = <<SeqTypes  as NodeType>::BuilderSignatureKey as BuilderSignatureKey>::sign_sequencing_fee_marketplace(
768                    &keypair,
769                    fee.as_u64().unwrap(),
770                    parent_view_number + 1,
771                )
772                .unwrap();
773
774        let sequencing_fee = BuilderFee {
775            fee_amount: fee.as_u64().unwrap(),
776            fee_account: FeeAccount::from(address),
777            fee_signature,
778        };
779
780        assert_eq!(bundle.sequencing_fee, sequencing_fee);
781    }
