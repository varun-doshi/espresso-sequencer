1use std::{path::PathBuf, time::Duration};
2
3use anyhow::Result;
4use clap::Parser;
5use espresso_types::{parse_duration, SeqTypes};
6use ethers::types::Address;
7use sequencer_utils::{
8    logging,
9    stake_table::{update_stake_table, PermissionedStakeTableUpdate},
10};
11use url::Url;
12
13#[derive(Debug, Clone, Parser)]
14struct Options {
15    /// RPC URL for the L1 provider.
16    #[clap(
17        short,
18        long,
19        env = "ESPRESSO_SEQUENCER_L1_PROVIDER",
20        default_value = "http://localhost:8545"
21    )]
22    rpc_url: Url,
23
24    /// Request rate when polling L1.
25    #[clap(
26        long,
27        env = "ESPRESSO_SEQUENCER_L1_POLLING_INTERVAL",
28        default_value = "7s",
29        value_parser = parse_duration,
30    )]
31    pub l1_polling_interval: Duration,
32
33    /// Mnemonic for an L1 wallet.
34    ///
35    /// This wallet is used to deploy the contracts, so the account indicated by ACCOUNT_INDEX must
36    /// be funded with with ETH.
37    #[clap(
38        long,
39        name = "MNEMONIC",
40        env = "ESPRESSO_SEQUENCER_ETH_MNEMONIC",
41        default_value = "test test test test test test test test test test test junk"
42    )]
43    mnemonic: String,
44
45    /// Account index in the L1 wallet generated by MNEMONIC to use when deploying the contracts.
46    #[clap(
47        long,
48        name = "ACCOUNT_INDEX",
49        env = "ESPRESSO_DEPLOYER_ACCOUNT_INDEX",
50        default_value = "0"
51    )]
52    account_index: u32,
53
54    /// Permissioned stake table contract address.
55    #[clap(long, env = "ESPRESSO_SEQUENCER_PERMISSIONED_STAKE_TABLE_ADDRESS")]
56    contract_address: Address,
57
58    /// Path to the toml file containing the update information.
59    ///
60    /// Schema of toml file:
61    /// ```toml
62    /// stakers_to_remove = [
63    ///   {
64    ///     stake_table_key = "BLS_VER_KEY~...",
65    ///   },
66    /// ]
67    ///
68    /// new_stakers = [
69    ///   {
70    ///     stake_table_key = "BLS_VER_KEY~...",
71    ///     state_ver_key = "SCHNORR_VER_KEY~...",
72    ///     da = true,
73    ///     stake = 1, # this value is ignored, but needs to be set
74    ///   },
75    /// ]
76    /// ```
77    #[clap(
78        long,
79        env = "ESPRESSO_SEQUENCER_PERMISSIONED_STAKE_TABLE_UPDATE_TOML_PATH",
80        verbatim_doc_comment
81    )]
82    update_toml_path: PathBuf,
83    #[clap(flatten)]
84    logging: logging::Config,
85}
86
87#[tokio::main]
88async fn main() -> Result<()> {
89    let opts = Options::parse();
90    opts.logging.init();
91    let update = PermissionedStakeTableUpdate::<SeqTypes>::from_toml_file(&opts.update_toml_path)?;
92
93    update_stake_table(
94        opts.rpc_url,
95        opts.l1_polling_interval,
96        opts.mnemonic,
97        opts.account_index,
98        opts.contract_address,
99        update,
100    )
101    .await?;
102
103    Ok(())
104}
