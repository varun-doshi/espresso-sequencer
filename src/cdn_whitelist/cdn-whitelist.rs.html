1//! The whitelist is an adaptor that is able to update the allowed public keys for
2//! all brokers. Right now, we do this by asking the orchestrator for the list of
3//! allowed public keys. In the future, we will pull the stake table from the L1.
4
5use std::{str::FromStr, sync::Arc};
6
7use anyhow::{Context, Result};
8use cdn_broker::reexports::discovery::{DiscoveryClient, Embedded, Redis};
9use clap::Parser;
10use espresso_types::SeqTypes;
11use hotshot_orchestrator::client::OrchestratorClient;
12use hotshot_types::{network::NetworkConfig, traits::signature_key::SignatureKey};
13use surf_disco::Url;
14
15#[derive(Parser, Debug)]
16#[command(author, version, about, long_about = None)]
17/// Whitelist is a service that updates the allowed public keys for the CDN.
18struct Args {
19    /// The discovery client endpoint (including scheme) to connect to.
20    /// With the local discovery feature, this is a file path.
21    /// With the remote (redis) discovery feature, this is a redis URL (e.g. `redis://127.0.0.1:6789`).
22    #[arg(short, long, env = "ESPRESSO_CDN_WHITELIST_DISCOVERY_ENDPOINT")]
23    discovery_endpoint: String,
24
25    /// The URL the orchestrator is running on. This should be something like `http://localhost:5555`
26    #[arg(short, long, env = "ESPRESSO_SEQUENCER_ORCHESTRATOR_URL")]
27    orchestrator_url: String,
28
29    /// Whether or not to use the local discovery client
30    #[arg(short, long)]
31    local_discovery: bool,
32}
33
34#[tokio::main]
35async fn main() -> Result<()> {
36    // Parse the command line arguments
37    let args = Args::parse();
38
39    // Initialize tracing
40    tracing_subscriber::fmt::init();
41
42    // Create a new `OrchestratorClient` from the supplied URL
43    let orchestrator_client = OrchestratorClient::new(
44        Url::from_str(&args.orchestrator_url).with_context(|| "Invalid URL")?,
45    );
46
47    tracing::info!(
48        "Waiting for config from orchestrator on {}",
49        args.orchestrator_url
50    );
51
52    // Attempt to get the config from the orchestrator.
53    // Loops internally until the config is received.
54    let config: NetworkConfig<SeqTypes> = orchestrator_client.get_config_after_collection().await;
55
56    tracing::info!("Received config from orchestrator");
57
58    // Extrapolate the state_ver_keys from the config and convert them to a compatible format
59    let whitelist = config
60        .config
61        .known_nodes_with_stake
62        .iter()
63        .map(|k| Arc::from(k.stake_table_entry.stake_key.to_bytes()))
64        .collect();
65
66    if args.local_discovery {
67        <Embedded as DiscoveryClient>::new(args.discovery_endpoint, None)
68            .await?
69            .set_whitelist(whitelist)
70            .await?;
71    } else {
72        <Redis as DiscoveryClient>::new(args.discovery_endpoint, None)
73            .await?
74            .set_whitelist(whitelist)
75            .await?;
76    }
77
78    tracing::info!("Posted config to discovery endpoint");
79
80    Ok(())
81}
