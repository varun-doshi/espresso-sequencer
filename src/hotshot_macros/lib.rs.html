53impl CrossTestDataBuilder {

57            && self.types.is_some()
58            && self.versions.is_some()
59            && self.test_name.is_some()
60            && self.metadata.is_some()
61            && self.test_name.is_some()
62            && self.ignore.is_some()
63    }

91/// trait make a string lower and snake case
92trait ToLowerSnakeStr {
93    /// make a lower and snake case string
94    fn to_lower_snake_str(&self) -> String;
95}
96
97impl ToLowerSnakeStr for ExprPath {
98    fn to_lower_snake_str(&self) -> String {
99        self.path
100            .segments
101            .iter()
102            .fold(String::new(), |mut acc, s| {
103                acc.push_str(&s.ident.to_string().to_lowercase());
104                acc.push('_');
105                acc
106            })
107            .to_lowercase()
108    }
109}
110
111impl ToLowerSnakeStr for syn::GenericArgument {
112    /// allow panic because this is a compiler error
113    #[allow(clippy::panic)]
114    fn to_lower_snake_str(&self) -> String {
115        match self {
116            syn::GenericArgument::Lifetime(l) => l.ident.to_string().to_lowercase(),
117            syn::GenericArgument::Type(t) => match t {
118                syn::Type::Path(p) => p.to_lower_snake_str(),
119                _ => {
120                    panic!("Unexpected type for GenericArgument::Type: {t:?}");
121                },
122            },
123            syn::GenericArgument::Const(c) => match c {
124                syn::Expr::Lit(l) => match &l.lit {
125                    syn::Lit::Str(v) => format!("{}_", v.value().to_lowercase()),
126                    syn::Lit::Int(v) => format!("{}_", v.base10_digits()),
127                    _ => {
128                        panic!("Unexpected type for GenericArgument::Const::Lit: {l:?}");
129                    },
130                },
131                _ => {
132                    panic!("Unexpected type for GenericArgument::Const: {c:?}");
133                },
134            },
135            _ => {
136                panic!("Unexpected type for GenericArgument: {self:?}");
137            },
138        }
139    }
140}
141
142impl ToLowerSnakeStr for TypePath {
143    fn to_lower_snake_str(&self) -> String {
144        self.path
145            .segments
146            .iter()
147            .fold(String::new(), |mut acc, s| {
148                acc.push_str(&s.ident.to_string().to_lowercase());
149                if let PathArguments::AngleBracketed(a) = &s.arguments {
150                    acc.push('_');
151                    for arg in &a.args {
152                        acc.push_str(&arg.to_lower_snake_str());
153                    }
154                }
155
156                acc.push('_');
157                acc
158            })
159            .to_lowercase()
160    }
161}
162
163impl ToLowerSnakeStr for ExprTuple {
164    /// allow panic because this is a compiler error
165    #[allow(clippy::panic)]
166    fn to_lower_snake_str(&self) -> String {
167        self.elems
168            .iter()
169            .map(|x| {
170                let Expr::Path(expr_path) = x else {
171                    panic!("Expected path expr, got {x:?}");
172                };
173                expr_path
174            })
175            .fold(String::new(), |mut acc, s| {
176                acc.push_str(&s.to_lower_snake_str());
177                acc
178            })
179    }
180}

182impl TestData {
183    /// generate the code for a single test
184    fn generate_test(&self) -> TokenStream2 {
185        let TestData {
186            ty,
187            imply,
188            version,
189            test_name,
190            metadata,
191            ignore,
192            builder_impl,
193        } = self;
194
195        let slow_attribute = if ignore.value() {
196            // quote! { #[cfg(feature = "slow-tests")] }
197            quote! { #[ignore] }
198        } else {
199            quote! {}
200        };
201        quote! {
202            #[cfg(test)]
203            #slow_attribute
204            #[tokio::test(flavor = "multi_thread")]
205            #[tracing::instrument]
206            async fn #test_name() {
207                hotshot::helpers::initialize_logging();
208
209                hotshot_testing::test_builder::TestDescription::<#ty, #imply, #version>::gen_launcher((#metadata)).launch().run_test::<#builder_impl>().await;
210            }
211        }
212    }
213}

215/// macro specific custom keywords
216mod keywords {
217    syn::custom_keyword!(Metadata);
218    syn::custom_keyword!(Ignore);
219    syn::custom_keyword!(TestName);
220    syn::custom_keyword!(Types);
221    syn::custom_keyword!(Impls);
222    syn::custom_keyword!(BuilderImpls);
223    syn::custom_keyword!(Versions);

226impl Parse for TypePathBracketedArray {
227    /// allow panic because this is a compiler error
228    #[allow(clippy::panic)]
229    fn parse(input: ParseStream<'_>) -> Result<Self> {
230        let content;
231        syn::bracketed!(content in input);
232        let mut elems = Punctuated::new();
233
234        while !content.is_empty() {
235            let first: TypePath = content.parse()?;
236            elems.push_value(first);
237            if content.is_empty() {
238                break;
239            }
240            let punct = content.parse()?;
241            elems.push_punct(punct);
242        }
243
244        Ok(Self { elems })
245    }
246}

248impl Parse for CrossTestData {
249    /// allow panic because this is a compiler error
250    #[allow(clippy::panic)]
251    fn parse(input: ParseStream<'_>) -> Result<Self> {
252        let mut description = CrossTestDataBuilder::create_empty();
253
254        while !description.is_ready() {
255            if input.peek(keywords::Types) {
256                let _ = input.parse::<keywords::Types>()?;
257                input.parse::<Token![:]>()?;
258                let types = input.parse::<TypePathBracketedArray>()?; //ExprArray>()?;
259                description.types(types);
260            } else if input.peek(keywords::Impls) {
261                let _ = input.parse::<keywords::Impls>()?;
262                input.parse::<Token![:]>()?;
263                let impls = input.parse::<ExprArray>()?;
264                description.impls(impls);
265            } else if input.peek(keywords::BuilderImpls) {
266                let _ = input.parse::<keywords::BuilderImpls>()?;
267                input.parse::<Token![:]>()?;
268                let impls = input.parse::<ExprArray>()?;
269                description.builder_impls(impls);
270            } else if input.peek(keywords::Versions) {
271                let _ = input.parse::<keywords::Versions>()?;
272                input.parse::<Token![:]>()?;
273                let versions = input.parse::<ExprArray>()?;
274                description.versions(versions);
275            } else if input.peek(keywords::TestName) {
276                let _ = input.parse::<keywords::TestName>()?;
277                input.parse::<Token![:]>()?;
278                let test_name = input.parse::<Ident>()?;
279                description.test_name(test_name);
280            } else if input.peek(keywords::Metadata) {
281                let _ = input.parse::<keywords::Metadata>()?;
282                input.parse::<Token![:]>()?;
283                let metadata = input.parse::<Expr>()?;
284                description.metadata(metadata);
285            } else if input.peek(keywords::Ignore) {
286                let _ = input.parse::<keywords::Ignore>()?;
287                input.parse::<Token![:]>()?;
288                let ignore = input.parse::<LitBool>()?;
289                description.ignore(ignore);
290            } else {
291                panic!(
292                    "Unexpected token. Expected one of: Metadata, Ignore, Impls, BuilderImpls, Versions, Types, Testname"
293                );
294            }
295            if input.peek(Token![,]) {
296                input.parse::<Token![,]>()?;
297            }
298        }
299        description
300            .build()
301            .map_err(|e| syn::Error::new(proc_macro2::Span::call_site(), format!("{e}")))
302    }
303}

305/// Helper function to do the actual code gen
306/// allow panic because this is a compiler error
307#[allow(clippy::panic)]
308fn cross_tests_internal(test_spec: CrossTestData) -> TokenStream {
309    let impls = test_spec.impls.elems.iter().map(|t| {
310        let Expr::Path(p) = t else {
311            panic!("Expected Path for Impl! Got {t:?}");
312        };
313        p
314    });
315
316    let types = test_spec.types.elems.iter();
317
318    let versions = test_spec.versions.elems.iter().map(|t| {
319        let Expr::Path(p) = t else {
320            panic!("Expected Path for Version! Got {t:?}");
321        };
322        p
323    });
324
325    let builder_impls = test_spec.builder_impls.elems.iter().map(|t| {
326        let Expr::Path(p) = t else {
327            panic!("Expected Path for BuilderImpl! Got {t:?}");
328        };
329        p
330    });
331
332    let mut result = quote! {};
333    for ty in types.clone() {
334        let mut type_mod = quote! {};
335        for imp in impls.clone() {
336            for builder_impl in builder_impls.clone() {
337                for version in versions.clone() {
338                    let test_data = TestDataBuilder::create_empty()
339                        .test_name(test_spec.test_name.clone())
340                        .metadata(test_spec.metadata.clone())
341                        .ignore(test_spec.ignore.clone())
342                        .version(version.clone())
343                        .imply(imp.clone())
344                        .builder_impl(builder_impl.clone())
345                        .ty(ty.clone())
346                        .build()
347                        .unwrap();
348                    let test = test_data.generate_test();
349
350                    let impl_str = format_ident!("{}", imp.to_lower_snake_str());
351                    let impl_result = quote! {
352                        pub mod #impl_str {
353                            use super::*;
354                            #test
355                        }
356                    };
357                    type_mod.extend(impl_result);
358                }
359            }
360        }
361        let ty_str = format_ident!("{}", ty.to_lower_snake_str());
362        let typ_result = quote! {
363            pub mod #ty_str {
364                use super::*;
365                #type_mod
366            }
367        };
368        result.extend(typ_result);
369    }
370    let name = test_spec.test_name;
371    quote! {
372        pub mod #name {
373            use super::*;
374            #result
375        }
376    }
377    .into()
378}

380/// Generate a cartesian product of tests across all types
381/// Arguments:
382/// - `Impls: []` - a list of types that implement nodetype
383/// - `Metadata`: `TestMetadata::default()` - test metadata
384/// - `Types: []` - a list types that implement `NodeImplementation` over the types in `Impls`
385/// - `TestName: example_test` - the name of the test
386/// - `Ignore`: whether or not this set of tests are ignored
387///   Example usage: see tests in this module
388#[proc_macro]
389pub fn cross_tests(input: TokenStream) -> TokenStream {
390    let test_spec = parse_macro_input!(input as CrossTestData);
391    cross_tests_internal(test_spec)
392}

394/// Macro to test multiple `TaskState` scripts at once.
395///
396/// Usage:
397///
398///   `test_scripts[inputs, script1, script2, ...]`
399///
400/// The generated test will:
401///   - take the first entry of `inputs`, which should be a `Vec<HotShotEvent<TYPES>>`,
402///   - feed this to each task in order, validating any output received before moving on to the next one,
403///   - repeat the previous steps, with the aggregated outputs received from all tasks used as the new input set,
404///   - repeat until no more output has been generated by any task, and finally
405///   - proceed to the next entry of inputs.
406///
407/// # Panics
408///
409/// The macro panics if the input stream cannot be parsed.
410/// The test will panic if the any of the scripts has a different number of stages from the input.
411#[allow(clippy::too_many_lines)]
412#[proc_macro]
413pub fn test_scripts(input: proc_macro::TokenStream) -> TokenStream {
414    // Parse the input as an iter of Expr
415    let inputs: Vec<_> = syn::parse::Parser::parse2(
416        syn::punctuated::Punctuated::<syn::Expr, syn::Token![,]>::parse_terminated,
417        input.into(),
418    )
419    .unwrap()
420    .into_iter()
421    .collect();
422
423    let test_inputs = &inputs[0];
424
425    let test_inputs_name = quote::quote!(#test_inputs).to_string();
426
427    let scripts = &inputs[1..];
428
429    let output_index_names: Vec<_> = scripts
430        .iter()
431        .map(|i| format_ident!("{}_output_index", quote::quote!(#i).to_string()))
432        .collect();
433
434    let task_expectations: Vec<_> = scripts
435        .iter()
436        .map(|i| format_ident!("{}_expectations", quote::quote!(#i).to_string()))
437        .collect();
438
439    let script_names: Vec<_> = scripts
440        .iter()
441        .map(|i| quote::quote!(#i).to_string())
442        .collect();
443
444    let expanded = quote! { {
445
446    use hotshot_testing::script::{
447        panic_extra_output_in_script, panic_missing_output_in_script, validate_output_or_panic_in_script,
448        validate_task_state_or_panic_in_script,
449    };
450
451    use hotshot_testing::{predicates::{Predicate, PredicateResult}};
452    use async_broadcast::broadcast;
453    use hotshot_task_impls::events::HotShotEvent;
454    use tokio::time::timeout;
455    use hotshot_task::task::{Task, TaskState};
456    use hotshot_types::traits::node_implementation::NodeType;
457    use std::sync::Arc;
458
459    async {
460
461    let (to_task, mut from_test) = broadcast(1024);
462    let (to_test, mut from_task) = broadcast(1024);
463
464    let mut loop_receiver = from_task.clone();
465
466    #(let mut #task_expectations = #scripts.expectations;)*
467
468    #(assert!(#task_expectations.len() == #test_inputs.len(), "Number of stages in {} does not match the number of stages in {}", #script_names, #test_inputs_name);)*
469
470    for (stage_number, input_group) in #test_inputs.into_iter().enumerate() {
471
472    #(let mut #output_index_names = 0;)*
473
474        for input in &input_group {
475            #(
476                tracing::debug!("Test sent: {:?}", input);
477
478                to_task
479                    .broadcast(input.clone().into())
480                    .await
481                    .expect("Failed to broadcast input message");
482
483
484                let _ = #scripts.state
485                    .handle_event(input.clone().into(), &to_test, &from_test)
486                    .await
487                    .inspect_err(|e| tracing::info!("{e}"));
488
489                while from_test.try_recv().is_ok() {}
490
491                let mut result = PredicateResult::Incomplete;
492
493                while let Ok(Ok(received_output)) = timeout(#scripts.timeout, from_task.recv_direct()).await {
494                    tracing::debug!("Test received: {:?}", received_output);
495
496                    let output_asserts = &mut #task_expectations[stage_number].output_asserts;
497
498                    if #output_index_names >= output_asserts.len() {
499                            panic_extra_output_in_script(stage_number, #script_names.to_string(), &received_output);
500                    };
501
502                    let assert = &mut output_asserts[#output_index_names];
503
504                    result = validate_output_or_panic_in_script(
505                        stage_number,
506                        #script_names.to_string(),
507                        &received_output,
508                        &**assert
509                    )
510                    .await;
511
512                    if result == PredicateResult::Pass {
513                        #output_index_names += 1;
514                    }
515                }
516            )*
517        }
518
519        while let Ok(input) = loop_receiver.try_recv() {
520            #(
521                tracing::debug!("Test sent: {:?}", input);
522
523                to_task
524                    .broadcast(input.clone().into())
525                    .await
526                    .expect("Failed to broadcast input message");
527
528                let _ = #scripts.state
529                    .handle_event(input.clone().into(), &to_test, &from_test)
530                    .await
531                    .inspect_err(|e| tracing::info!("{e}"));
532
533                while from_test.try_recv().is_ok() {}
534
535                let mut result = PredicateResult::Incomplete;
536                while let Ok(Ok(received_output)) = timeout(#scripts.timeout, from_task.recv_direct()).await {
537                    tracing::debug!("Test received: {:?}", received_output);
538
539                    let output_asserts = &mut #task_expectations[stage_number].output_asserts;
540
541                    if #output_index_names >= output_asserts.len() {
542                            panic_extra_output_in_script(stage_number, #script_names.to_string(), &received_output);
543                    };
544
545                    let mut assert = &mut output_asserts[#output_index_names];
546
547                    result = validate_output_or_panic_in_script(
548                        stage_number,
549                        #script_names.to_string(),
550                        &received_output,
551                        &**assert
552                    )
553                    .await;
554
555                    if result == PredicateResult::Pass {
556                        #output_index_names += 1;
557                    }
558                }
559            )*
560        }
561
562        #(
563            let output_asserts = &#task_expectations[stage_number].output_asserts;
564
565            if #output_index_names < output_asserts.len() {
566                panic_missing_output_in_script(stage_number, #script_names.to_string(), &output_asserts[#output_index_names]);
567            }
568
569            let task_state_asserts = &mut #task_expectations[stage_number].task_state_asserts;
570
571            for assert in task_state_asserts {
572                validate_task_state_or_panic_in_script(stage_number, #script_names.to_string(), &#scripts.state, &**assert).await;
573            }
574        )*
575    } }
576
577    }
578
579    };
580
581    expanded.into()
582}

584/// Macro to run the test suite with `TaskState` scripts at once with the ability to.
585/// Randomize the input values using a consistent seed value.
586///
587/// **Note** When using `random!` you should use `all_predicates` in the output to ensure
588/// that the test does not fail due to events happening out of order!
589///
590/// Usage:
591///
592///   `run_test[inputs, script1, script2, ...]`
593///
594/// The generated test will:
595///   - take the first entry of `inputs`, which should be a `Vec<HotShotEvent<TYPES>>`,
596///   - if the input is random, it'll generate a `seed` and shuffle the inputs.
597///   - if the input is serial, it will execute in order.
598///   - print the seed being used.
599///   - feed this to each task in order, validating any output received before moving on to the next one,
600///   - repeat the previous steps, with the aggregated outputs received from all tasks used as the new input set,
601///   - repeat until no more output has been generated by any task, and finally
602///   - proceed to the next entry of inputs.
603///   - print the seed being used (again) to make finding it a bit easier.
604///
605/// # Panics
606///
607/// The macro panics if the input stream cannot be parsed.
608/// The test will panic if the any of the scripts has a different number of stages from the input.
609#[proc_macro]
610pub fn run_test(input: TokenStream) -> TokenStream {
611    // Parse the input as an iter of Expr
612    let inputs: Vec<_> = syn::parse::Parser::parse2(
613        syn::punctuated::Punctuated::<syn::Expr, syn::Token![,]>::parse_terminated,
614        input.into(),
615    )
616    .unwrap()
617    .into_iter()
618    .collect();
619
620    // Separate the first input (which should be the InputOrder enum)
621    let test_inputs = &inputs[0];
622    let scripts = &inputs[1..];
623
624    // Generate code for shuffling and flattening inputs
625    let expanded = quote! {
626        {
627            use rand::{
628                SeedableRng, rngs::StdRng,
629                seq::SliceRandom
630            };
631            use hotshot_task_impls::events::HotShotEvent;
632            use hotshot_task::task::TaskState;
633            use hotshot_types::traits::node_implementation::NodeType;
634            use hotshot_testing::script::InputOrder;
635
636            async {
637                let seed: u64 = rand::random();
638                tracing::info!("Running test with seed {seed}");
639                let mut rng = StdRng::seed_from_u64(seed);
640                let mut shuffled_inputs = Vec::new();
641
642                for (stage_number, input_order) in #test_inputs.into_iter().enumerate() {
643                    match input_order {
644                        InputOrder::Random(mut events) => {
645                            events.shuffle(&mut rng);
646                            shuffled_inputs.push(events);
647                        },
648                        InputOrder::Serial(events) => {
649                            shuffled_inputs.push(events);
650                        }
651                    }
652                }
653
654                test_scripts![shuffled_inputs, #(#scripts),*].await;
655                tracing::info!("Suite used seed {seed}");
656            }
657        }
658    };
659
660    expanded.into()
661}
