25/// A type alias to help readability
26pub type Network<TYPES, I> = Arc<<I as NodeImplementation<TYPES>>::Network>;
27
28/// Wrapper for a function that takes a `node_id` and returns an instance of `T`.
29pub type Generator<T> = Rc<dyn Fn(u64) -> T>;
30
31/// generators for resources used by each node
32pub struct ResourceGenerators<TYPES: NodeType, I: TestableNodeImplementation<TYPES>> {
33    /// generate channels
34    pub channel_generator: AsyncGenerator<Network<TYPES, I>>,
35    /// generate new storage for each node
36    pub storage: Generator<TestStorage<TYPES>>,
37    /// configuration used to generate each hotshot node
38    pub hotshot_config: Generator<HotShotConfig<TYPES>>,
39    /// config that contains the signature keys
40    pub validator_config: Generator<ValidatorConfig<TYPES>>,
41    /// generate a new marketplace config for each node
42    pub marketplace_config: Generator<MarketplaceConfig<TYPES, I>>,
43}
44
45/// test launcher
46pub struct TestLauncher<TYPES: NodeType, I: TestableNodeImplementation<TYPES>, V: Versions> {
47    /// generator for resources
48    pub resource_generators: ResourceGenerators<TYPES, I>,
49    /// metadata used for tasks
50    pub metadata: TestDescription<TYPES, I, V>,
51    /// any additional test tasks to run
52    pub additional_test_tasks: Vec<Box<dyn TestTaskStateSeed<TYPES, I, V>>>,
53}
54
55impl<TYPES: NodeType, I: TestableNodeImplementation<TYPES>, V: Versions> TestLauncher<TYPES, I, V> {
56    /// launch the test
57    #[must_use]
58    pub fn launch<N: ConnectedNetwork<TYPES::SignatureKey>>(self) -> TestRunner<TYPES, I, V, N> {
59        TestRunner::<TYPES, I, V, N> {
60            launcher: self,
61            nodes: Vec::new(),
62            solver_server: None,
63            late_start: HashMap::new(),
64            next_node_id: 0,
65            _pd: PhantomData,
66        }
67    }
68    /// Modifies the config used when generating nodes with `f`
69    #[must_use]
70    pub fn map_hotshot_config(mut self, f: impl Fn(&mut HotShotConfig<TYPES>) + 'static) -> Self {
71        let mut test_config = self.metadata.test_config.clone();
72        f(&mut test_config);
73
74        let hotshot_config_generator = self.resource_generators.hotshot_config.clone();
75        let hotshot_config: Generator<_> = Rc::new(move |node_id| {
76            let mut result = (hotshot_config_generator)(node_id);
77            f(&mut result);
78
79            result
80        });
81
82        self.metadata.test_config = test_config;
83        self.resource_generators.hotshot_config = hotshot_config;
84
85        self
86    }
87}
