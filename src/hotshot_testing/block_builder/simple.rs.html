56impl SimpleBuilderImplementation {
57    pub async fn create<TYPES: NodeType>(
58        num_nodes: usize,
59        changes: HashMap<u64, BuilderChange>,
60        change_sender: Sender<BuilderChange>,
61    ) -> (SimpleBuilderSource<TYPES>, SimpleBuilderTask<TYPES>) {
62        let (pub_key, priv_key) =
63            TYPES::BuilderSignatureKey::generated_from_seed_indexed([1; 32], 0);
64
65        let transactions = Arc::new(RwLock::new(HashMap::new()));
66        let blocks = Arc::new(RwLock::new(HashMap::new()));
67        let should_fail_claims = Arc::new(AtomicBool::new(false));
68
69        let source = SimpleBuilderSource {
70            pub_key,
71            priv_key,
72            transactions: transactions.clone(),
73            blocks: blocks.clone(),
74            num_nodes: Arc::new(RwLock::new(num_nodes)),
75            should_fail_claims: Arc::clone(&should_fail_claims),
76        };
77
78        let task = SimpleBuilderTask {
79            transactions,
80            blocks,
81            decided_transactions: LruCache::new(NonZeroUsize::new(u16::MAX.into()).expect("> 0")),
82            should_fail_claims,
83            change_sender,
84            changes,
85        };
86
87        (source, task)
88    }
89}

135#[async_trait]
136impl<TYPES: NodeType> v0_99::data_source::BuilderDataSource<TYPES> for SimpleBuilderSource<TYPES>
137where
138    <TYPES as NodeType>::InstanceState: Default,
139{
140    /// To get the list of available blocks
141    async fn bundle(
142        &self,
143        _parent_view: u64,
144        _parent_hash: &VidCommitment,
145        view_number: u64,
146    ) -> Result<Bundle<TYPES>, BuildError> {
147        let transactions = self
148            .transactions
149            .read(|txns| {
150                Box::pin(async {
151                    txns.values()
152                        .filter(|txn| {
153                            // We want transactions that are either unclaimed, or claimed long ago
154                            // and thus probably not included, or they would've been decided on
155                            // already and removed from the queue
156                            txn.claimed
157                                .map(|claim_time| claim_time.elapsed() > Duration::from_secs(30))
158                                .unwrap_or(true)
159                        })
160                        .cloned()
161                        .map(|txn| txn.transaction)
162                        .collect::<Vec<TYPES::Transaction>>()
163                })
164            })
165            .await;
166
167        let fee_amount = 1;
168        let sequencing_fee: BuilderFee<TYPES> = BuilderFee {
169            fee_amount,
170            fee_account: self.pub_key.clone(),
171            fee_signature: TYPES::BuilderSignatureKey::sign_sequencing_fee_marketplace(
172                &self.priv_key.clone(),
173                fee_amount,
174                view_number,
175            )
176            .expect("Failed to sign fee!"),
177        };
178
179        let signature =
180            TYPES::BuilderSignatureKey::sign_bundle::<TYPES>(&self.priv_key, &transactions)
181                .unwrap();
182
183        {
184            // claim transactions
185            let mut transactions_lock = self.transactions.write().await;
186            let transaction_hashes = transactions.iter().map(|txn| txn.commit());
187            let time = Instant::now();
188
189            for hash in transaction_hashes {
190                if let Some(txn) = transactions_lock.get_mut(&hash) {
191                    txn.claimed = Some(time);
192                }
193            }
194        }
195
196        Ok(Bundle {
197            transactions,
198            signature,
199            sequencing_fee,
200        })
201    }
202
203    /// To get the builder's address
204    async fn builder_address(&self) -> Result<TYPES::BuilderSignatureKey, BuildError> {
205        Ok(self.pub_key.clone())
206    }
207}

209#[async_trait]
210impl<TYPES: NodeType> v0_1::data_source::BuilderDataSource<TYPES> for SimpleBuilderSource<TYPES>
211where
212    <TYPES as NodeType>::InstanceState: Default,
213{
214    async fn available_blocks(
215        &self,
216        _for_parent: &VidCommitment,
217        _view_number: u64,
218        _sender: TYPES::SignatureKey,
219        _signature: &<TYPES::SignatureKey as SignatureKey>::PureAssembledSignatureType,
220    ) -> Result<Vec<AvailableBlockInfo<TYPES>>, BuildError> {
221        let transactions = self
222            .transactions
223            .read(|txns| {
224                Box::pin(async {
225                    txns.values()
226                        .filter(|txn| {
227                            // We want transactions that are either unclaimed, or claimed long ago
228                            // and thus probably not included, or they would've been decided on
229                            // already and removed from the queue
230                            txn.claimed
231                                .map(|claim_time| claim_time.elapsed() > Duration::from_secs(30))
232                                .unwrap_or(true)
233                        })
234                        .cloned()
235                        .map(|txn| txn.transaction)
236                        .collect::<Vec<TYPES::Transaction>>()
237                })
238            })
239            .await;
240
241        if transactions.is_empty() {
242            // We don't want to return an empty block if we have no transactions, as we would end up
243            // driving consensus to produce empty blocks extremely quickly when mempool is empty.
244            // Instead, we return no blocks, so that view leader will keep asking for blocks until
245            // either we have something non-trivial to propose, or leader runs out of time to propose,
246            // in which case view leader will finally propose an empty block themselves.
247            return Ok(vec![]);
248        }
249
250        let block_entry =
251            build_block::<TYPES>(transactions, self.pub_key.clone(), self.priv_key.clone()).await;
252
253        let metadata = block_entry.metadata.clone();
254
255        self.blocks
256            .write()
257            .await
258            .insert(block_entry.metadata.block_hash.clone(), block_entry);
259
260        Ok(vec![metadata])
261    }
262
263    async fn claim_block(
264        &self,
265        block_hash: &BuilderCommitment,
266        _view_number: u64,
267        _sender: TYPES::SignatureKey,
268        _signature: &<TYPES::SignatureKey as SignatureKey>::PureAssembledSignatureType,
269    ) -> Result<AvailableBlockData<TYPES>, BuildError> {
270        if self.should_fail_claims.load(Ordering::Relaxed) {
271            return Err(BuildError::Missing);
272        }
273
274        let payload = {
275            let mut blocks = self.blocks.write().await;
276            let entry = blocks.get_mut(block_hash).ok_or(BuildError::NotFound)?;
277            entry.payload.take().ok_or(BuildError::Missing)?
278        };
279
280        let now = Instant::now();
281
282        let claimed_transactions = payload
283            .block_payload
284            .transaction_commitments(&payload.metadata);
285
286        let mut transactions = self.transactions.write().await;
287        for txn_hash in claimed_transactions {
288            if let Some(txn) = transactions.get_mut(&txn_hash) {
289                txn.claimed = Some(now);
290            }
291        }
292
293        Ok(payload)
294    }
295
296    async fn claim_block_with_num_nodes(
297        &self,
298        block_hash: &BuilderCommitment,
299        view_number: u64,
300        sender: TYPES::SignatureKey,
301        signature: &<TYPES::SignatureKey as SignatureKey>::PureAssembledSignatureType,
302        num_nodes: usize,
303    ) -> Result<AvailableBlockData<TYPES>, BuildError> {
304        *self.num_nodes.write().await = num_nodes;
305        self.claim_block(block_hash, view_number, sender, signature)
306            .await
307    }
308
309    async fn claim_block_header_input(
310        &self,
311        block_hash: &BuilderCommitment,
312        _view_number: u64,
313        _sender: TYPES::SignatureKey,
314        _signature: &<TYPES::SignatureKey as SignatureKey>::PureAssembledSignatureType,
315    ) -> Result<AvailableBlockHeaderInputV1<TYPES>, BuildError> {
316        if self.should_fail_claims.load(Ordering::Relaxed) {
317            return Err(BuildError::Missing);
318        }
319
320        let mut blocks = self.blocks.write().await;
321        let entry = blocks.get_mut(block_hash).ok_or(BuildError::NotFound)?;
322        entry.header_input.take().ok_or(BuildError::Missing)
323    }
324
325    async fn builder_address(&self) -> Result<TYPES::BuilderSignatureKey, BuildError> {
326        Ok(self.pub_key.clone())
327    }
328}

330impl<TYPES: NodeType> SimpleBuilderSource<TYPES> {
331    pub async fn run(self, url: Url)
332    where
333        <TYPES as NodeType>::InstanceState: Default,
334    {
335        let builder_api_0_1 = hotshot_builder_api::v0_1::builder::define_api::<
336            SimpleBuilderSource<TYPES>,
337            TYPES,
338        >(&Options::default())
339        .expect("Failed to construct the builder API");
340
341        let builder_api_0_3 = hotshot_builder_api::v0_99::builder::define_api::<
342            SimpleBuilderSource<TYPES>,
343            TYPES,
344        >(&Options::default())
345        .expect("Failed to construct the builder API");
346
347        let mut app: App<SimpleBuilderSource<TYPES>, Error> = App::with_state(self);
348        app.register_module::<Error, _>(LEGACY_BUILDER_MODULE, builder_api_0_1)
349            .expect("Failed to register builder API 0.1")
350            .register_module::<Error, _>(MARKETPLACE_BUILDER_MODULE, builder_api_0_3)
351            .expect("Failed to register builder API 0.3");
352
353        spawn(app.serve(url, hotshot_builder_api::v0_1::Version::instance()));
354    }
355}

357#[derive(Debug, Clone)]
358struct SubmittedTransaction<TYPES: NodeType> {
359    claimed: Option<Instant>,
360    transaction: TYPES::Transaction,
361}

363#[derive(Clone)]
364pub struct SimpleBuilderTask<TYPES: NodeType> {
365    #[allow(clippy::type_complexity)]
366    transactions: Arc<RwLock<HashMap<Commitment<TYPES::Transaction>, SubmittedTransaction<TYPES>>>>,
367    blocks: Arc<RwLock<HashMap<BuilderCommitment, BlockEntry<TYPES>>>>,
368    decided_transactions: LruCache<Commitment<TYPES::Transaction>, ()>,
369    should_fail_claims: Arc<AtomicBool>,
370    changes: HashMap<u64, BuilderChange>,
371    change_sender: Sender<BuilderChange>,
372}
373
374impl<TYPES: NodeType> BuilderTask<TYPES> for SimpleBuilderTask<TYPES> {
375    fn start(
376        mut self: Box<Self>,
377        mut stream: Box<dyn Stream<Item = Event<TYPES>> + std::marker::Unpin + Send + 'static>,
378    ) {
379        spawn(async move {
380            let mut should_build_blocks = true;
381            loop {
382                match stream.next().await {
383                    None => {
384                        break;
385                    },
386                    Some(evt) => match evt.event {
387                        EventType::ViewFinished { view_number } => {
388                            if let Some(change) = self.changes.remove(&view_number) {
389                                match change {
390                                    BuilderChange::Up => should_build_blocks = true,
391                                    BuilderChange::Down => {
392                                        should_build_blocks = false;
393                                        self.transactions.write().await.clear();
394                                        self.blocks.write().await.clear();
395                                    },
396                                    BuilderChange::FailClaims(value) => {
397                                        self.should_fail_claims.store(value, Ordering::Relaxed);
398                                    },
399                                }
400                                let _ = self.change_sender.broadcast(change).await;
401                            }
402                        },
403                        EventType::Decide { leaf_chain, .. } if should_build_blocks => {
404                            let mut queue = self.transactions.write().await;
405                            for leaf_info in leaf_chain.iter() {
406                                if let Some(ref payload) = leaf_info.leaf.block_payload() {
407                                    for txn in payload.transaction_commitments(
408                                        leaf_info.leaf.block_header().metadata(),
409                                    ) {
410                                        self.decided_transactions.put(txn, ());
411                                        queue.remove(&txn);
412                                    }
413                                }
414                            }
415                            self.blocks.write().await.clear();
416                        },
417                        EventType::DaProposal { proposal, .. } if should_build_blocks => {
418                            let payload = TYPES::BlockPayload::from_bytes(
419                                &proposal.data.encoded_transactions,
420                                &proposal.data.metadata,
421                            );
422                            let now = Instant::now();
423
424                            let mut queue = self.transactions.write().await;
425                            for commitment in
426                                payload.transaction_commitments(&proposal.data.metadata)
427                            {
428                                if let Some(txn) = queue.get_mut(&commitment) {
429                                    txn.claimed = Some(now);
430                                }
431                            }
432                        },
433                        EventType::Transactions { transactions } if should_build_blocks => {
434                            let mut queue = self.transactions.write().await;
435                            for transaction in transactions {
436                                if !self.decided_transactions.contains(&transaction.commit()) {
437                                    queue.insert(
438                                        transaction.commit(),
439                                        SubmittedTransaction {
440                                            claimed: None,
441                                            transaction: transaction.clone(),
442                                        },
443                                    );
444                                }
445                            }
446                        },
447                        _ => {},
448                    },
449                }
450            }
451        });
452    }
