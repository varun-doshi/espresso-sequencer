29/// An `EventTransformerState` that multiplies `QuorumProposalSend` events, incrementing the view number of the proposal
30pub struct BadProposalViewDos {
31    /// The number of times to duplicate a `QuorumProposalSend` event
32    pub multiplier: u64,
33    /// The view number increment each time it's duplicatedjust
34    pub increment: u64,
35}
36
37#[async_trait]
38impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> EventTransformerState<TYPES, I, V>
39    for BadProposalViewDos
40{
41    async fn recv_handler(&mut self, event: &HotShotEvent<TYPES>) -> Vec<HotShotEvent<TYPES>> {
42        vec![event.clone()]
43    }
44
45    async fn send_handler(
46        &mut self,
47        event: &HotShotEvent<TYPES>,
48        _public_key: &TYPES::SignatureKey,
49        _private_key: &<TYPES::SignatureKey as SignatureKey>::PrivateKey,
50        _upgrade_lock: &UpgradeLock<TYPES, V>,
51        consensus: Arc<RwLock<Consensus<TYPES>>>,
52    ) -> Vec<HotShotEvent<TYPES>> {
53        match event {
54            HotShotEvent::QuorumProposalSend(proposal, signature) => {
55                let mut result = Vec::new();
56
57                for n in 1..self.multiplier {
58                    let mut modified_proposal = proposal.clone();
59
60                    modified_proposal.data.proposal.view_number += n * self.increment;
61
62                    result.push(HotShotEvent::QuorumProposalSend(
63                        modified_proposal,
64                        signature.clone(),
65                    ));
66                }
67
68                consensus.write().await.reset_actions();
69                result
70            },
71            _ => vec![event.clone()],
72        }
73    }
74}

106/// An `EventHandlerState` that modifies justify_qc on `QuorumProposalSend` to that of a previous view to mock dishonest leader
107pub struct DishonestLeader<TYPES: NodeType> {
108    /// Store events from previous views
109    pub validated_proposals: Vec<QuorumProposalWrapper<TYPES>>,
110    /// How many times current node has been elected leader and sent proposal
111    pub total_proposals_from_node: u64,
112    /// Which proposals to be dishonest at
113    pub dishonest_at_proposal_numbers: HashSet<u64>,
114    /// How far back to look for a QC
115    pub view_look_back: usize,
116    /// Shared state of all view numbers we send bad proposal at
117    pub dishonest_proposal_view_numbers: Arc<RwLock<HashSet<TYPES::View>>>,
118}
119
120/// Add method that will handle `QuorumProposalSend` events
121/// If we have previous proposals stored and the total_proposals_from_node matches a value specified in dishonest_at_proposal_numbers
122/// Then send out the event with the modified proposal that has an older QC
123impl<TYPES: NodeType> DishonestLeader<TYPES> {
124    /// When a leader is sending a proposal this method will mock a dishonest leader
125    /// We accomplish this by looking back a number of specified views and using that cached proposals QC
126    async fn handle_proposal_send_event(
127        &self,
128        event: &HotShotEvent<TYPES>,
129        proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
130        sender: &TYPES::SignatureKey,
131    ) -> HotShotEvent<TYPES> {
132        let length = self.validated_proposals.len();
133        if !self
134            .dishonest_at_proposal_numbers
135            .contains(&self.total_proposals_from_node)
136            || length == 0
137        {
138            return event.clone();
139        }
140
141        // Grab proposal from specified view look back
142        let proposal_from_look_back = if length - 1 < self.view_look_back {
143            // If look back is too far just take the first proposal
144            self.validated_proposals[0].clone()
145        } else {
146            let index = (self.validated_proposals.len() - 1) - self.view_look_back;
147            self.validated_proposals[index].clone()
148        };
149
150        // Create a dishonest proposal by using the old proposals qc
151        let mut dishonest_proposal = proposal.clone();
152        dishonest_proposal.data.proposal.justify_qc = proposal_from_look_back.proposal.justify_qc;
153
154        // Save the view we sent the dishonest proposal on (used for coordination attacks with other byzantine replicas)
155        let mut dishonest_proposal_sent = self.dishonest_proposal_view_numbers.write().await;
156        dishonest_proposal_sent.insert(proposal.data.view_number());
157
158        HotShotEvent::QuorumProposalSend(dishonest_proposal, sender.clone())
159    }
160}

183                        .await,
184                ];
185            },
186            HotShotEvent::QuorumProposalValidated(proposal, _) => {
187                self.validated_proposals.push(proposal.data.clone());
188            },
189            _ => {},
190        }
191        vec![event.clone()]
192    }

196/// An `EventHandlerState` that modifies view number on the certificate of `DacSend` event to that of a future view
197pub struct DishonestDa {
198    /// How many times current node has been elected leader and sent Da Cert
199    pub total_da_certs_sent_from_node: u64,
200    /// Which proposals to be dishonest at
201    pub dishonest_at_da_cert_sent_numbers: HashSet<u64>,
202    /// When leader how many times we will send DacSend and increment view number
203    pub total_views_add_to_cert: u64,
204}

206#[async_trait]
207impl<TYPES: NodeType, I: NodeImplementation<TYPES> + std::fmt::Debug, V: Versions>
208    EventTransformerState<TYPES, I, V> for DishonestDa
209{
210    async fn recv_handler(&mut self, event: &HotShotEvent<TYPES>) -> Vec<HotShotEvent<TYPES>> {
211        vec![event.clone()]
212    }
213
214    async fn send_handler(
215        &mut self,
216        event: &HotShotEvent<TYPES>,
217        _public_key: &TYPES::SignatureKey,
218        _private_key: &<TYPES::SignatureKey as SignatureKey>::PrivateKey,
219        _upgrade_lock: &UpgradeLock<TYPES, V>,
220        _consensus: Arc<RwLock<Consensus<TYPES>>>,
221    ) -> Vec<HotShotEvent<TYPES>> {
222        if let HotShotEvent::DacSend(cert, sender) = event {
223            self.total_da_certs_sent_from_node += 1;
224            if self
225                .dishonest_at_da_cert_sent_numbers
226                .contains(&self.total_da_certs_sent_from_node)
227            {
228                let mut result = vec![HotShotEvent::DacSend(cert.clone(), sender.clone())];
229                for i in 1..=self.total_views_add_to_cert {
230                    let mut bad_cert = cert.clone();
231                    bad_cert.view_number = cert.view_number + i;
232                    result.push(HotShotEvent::DacSend(bad_cert, sender.clone()));
233                }
234                return result;
235            }
236        }
237        vec![event.clone()]
238    }

241/// View delay configuration
242#[derive(Debug)]
243pub struct ViewDelay<TYPES: NodeType> {
244    /// How many views the node will be delayed
245    pub number_of_views_to_delay: u64,
246    /// A map that is from view number to vector of events
247    pub events_for_view: HashMap<TYPES::View, Vec<HotShotEvent<TYPES>>>,
248    /// Specify which view number to stop delaying
249    pub stop_view_delay_at_view_number: u64,
250}
251
252#[async_trait]
253impl<TYPES: NodeType, I: NodeImplementation<TYPES> + std::fmt::Debug, V: Versions>
254    EventTransformerState<TYPES, I, V> for ViewDelay<TYPES>
255{
256    async fn recv_handler(&mut self, event: &HotShotEvent<TYPES>) -> Vec<HotShotEvent<TYPES>> {
257        let correct_event = vec![event.clone()];
258        if let Some(view_number) = event.view_number() {
259            if *view_number >= self.stop_view_delay_at_view_number {
260                return correct_event;
261            }
262
263            // add current view or push event to the map if view number has been added
264            let events_for_current_view = self.events_for_view.entry(view_number).or_default();
265            events_for_current_view.push(event.clone());
266
267            // ensure we are actually able to lookback enough views
268            let view_diff = (*view_number).saturating_sub(self.number_of_views_to_delay);
269            if view_diff > 0 {
270                return match self
271                    .events_for_view
272                    .remove(&<TYPES as NodeType>::View::new(view_diff))
273                {
274                    Some(lookback_events) => lookback_events.clone(),
275                    // we have already return all received events for this view
276                    None => vec![],
277                };
278            }
279        }
280
281        correct_event
282    }
283
284    async fn send_handler(
285        &mut self,
286        event: &HotShotEvent<TYPES>,
287        _public_key: &TYPES::SignatureKey,
288        _private_key: &<TYPES::SignatureKey as SignatureKey>::PrivateKey,
289        _upgrade_lock: &UpgradeLock<TYPES, V>,
290        _consensus: Arc<RwLock<Consensus<TYPES>>>,
291    ) -> Vec<HotShotEvent<TYPES>> {
292        vec![event.clone()]
293    }
294}

296/// An `EventHandlerState` that modifies view number on the vote of `QuorumVoteSend` event to that of a future view and correctly signs the vote
297pub struct DishonestVoting<TYPES: NodeType> {
298    /// Number added to the original vote's view number
299    pub view_increment: u64,
300    /// A function passed to `NetworkEventTaskStateModifier` to modify `NetworkEventTaskState` behaviour.
301    pub modifier: Arc<ModifierClosure<TYPES>>,
302}
303
304#[async_trait]
305impl<TYPES: NodeType, I: NodeImplementation<TYPES> + std::fmt::Debug, V: Versions>
306    EventTransformerState<TYPES, I, V> for DishonestVoting<TYPES>
307{
308    async fn recv_handler(&mut self, event: &HotShotEvent<TYPES>) -> Vec<HotShotEvent<TYPES>> {
309        vec![event.clone()]
310    }
311
312    async fn send_handler(
313        &mut self,
314        event: &HotShotEvent<TYPES>,
315        public_key: &TYPES::SignatureKey,
316        private_key: &<TYPES::SignatureKey as SignatureKey>::PrivateKey,
317        upgrade_lock: &UpgradeLock<TYPES, V>,
318        _consensus: Arc<RwLock<Consensus<TYPES>>>,
319    ) -> Vec<HotShotEvent<TYPES>> {
320        if let HotShotEvent::QuorumVoteSend(vote) = event {
321            let new_view = vote.view_number + self.view_increment;
322            let spoofed_vote = QuorumVote2::<TYPES>::create_signed_vote(
323                vote.data.clone(),
324                new_view,
325                public_key,
326                private_key,
327                upgrade_lock,
328            )
329            .await
330            .context("Failed to sign vote")
331            .unwrap();
332            tracing::debug!("Sending Quorum Vote for view: {new_view:?}");
333            return vec![HotShotEvent::QuorumVoteSend(spoofed_vote)];
334        }
335        vec![event.clone()]
336    }
337
338    fn add_network_event_task(
339        &self,
340        handle: &mut SystemContextHandle<TYPES, I, V>,
341        network: Arc<<I as NodeImplementation<TYPES>>::Network>,
342    ) {
343        let network_state: NetworkEventTaskState<_, V, _, _> = NetworkEventTaskState {
344            network,
345            view: TYPES::View::genesis(),
346            epoch: None,
347            membership_coordinator: handle.membership_coordinator.clone(),
348            storage: Arc::clone(&handle.storage()),
349            consensus: OuterConsensus::new(handle.consensus()),
350            upgrade_lock: handle.hotshot.upgrade_lock.clone(),
351            transmit_tasks: BTreeMap::new(),
352            epoch_height: handle.epoch_height,
353        };
354        let modified_network_state = NetworkEventTaskStateModifier {
355            network_event_task_state: network_state,
356            modifier: Arc::clone(&self.modifier),
357        };
358        handle.add_task(modified_network_state);
359    }
360}

371/// An `EventHandlerState` that will send a vote for a bad proposal
372pub struct DishonestVoter<TYPES: NodeType> {
373    /// Collect all votes the node sends
374    pub votes_sent: Vec<QuorumVote2<TYPES>>,
375    /// Shared state with views numbers that leaders were dishonest at
376    pub dishonest_proposal_view_numbers: Arc<RwLock<HashSet<TYPES::View>>>,
377}
378
379#[async_trait]
380impl<TYPES: NodeType, I: NodeImplementation<TYPES> + std::fmt::Debug, V: Versions>
381    EventTransformerState<TYPES, I, V> for DishonestVoter<TYPES>
382{
383    async fn recv_handler(&mut self, event: &HotShotEvent<TYPES>) -> Vec<HotShotEvent<TYPES>> {
384        vec![event.clone()]
385    }
386
387    async fn send_handler(
388        &mut self,
389        event: &HotShotEvent<TYPES>,
390        public_key: &TYPES::SignatureKey,
391        private_key: &<TYPES::SignatureKey as SignatureKey>::PrivateKey,
392        upgrade_lock: &UpgradeLock<TYPES, V>,
393        _consensus: Arc<RwLock<Consensus<TYPES>>>,
394    ) -> Vec<HotShotEvent<TYPES>> {
395        match event {
396            HotShotEvent::QuorumProposalRecv(proposal, _sender) => {
397                // Check if view is a dishonest proposal, if true send a vote
398                let dishonest_proposals = self.dishonest_proposal_view_numbers.read().await;
399                if dishonest_proposals.contains(&proposal.data.view_number()) {
400                    // Create a vote using data from most recent vote and the current event number
401                    // We wont update internal consensus state for this Byzantine replica but we are at least
402                    // Going to send a vote to the next honest leader
403                    let vote = QuorumVote2::<TYPES>::create_signed_vote(
404                        self.votes_sent.last().unwrap().data.clone(),
405                        event.view_number().unwrap(),
406                        public_key,
407                        private_key,
408                        upgrade_lock,
409                    )
410                    .await
411                    .context("Failed to sign vote")
412                    .unwrap();
413                    return vec![HotShotEvent::QuorumVoteSend(vote)];
414                }
415            },
416            HotShotEvent::TimeoutVoteSend(vote) => {
417                // Check if this view was a dishonest proposal view, if true dont send timeout
418                let dishonest_proposals = self.dishonest_proposal_view_numbers.read().await;
419                if dishonest_proposals.contains(&vote.view_number) {
420                    // We craft the vote upon `QuorumProposalRecv` and send out a vote.
421                    // So, dont send the timeout to the next leader from this byzantine replica
422                    return vec![];
423                }
424            },
425            HotShotEvent::QuorumVoteSend(vote) => {
426                self.votes_sent.push(vote.clone());
427            },
428            _ => {},
429        }
430        vec![event.clone()]
431    }
