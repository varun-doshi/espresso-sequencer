54/// Spinning task state
55pub struct SpinningTask<
56    TYPES: NodeType,
57    N: ConnectedNetwork<TYPES::SignatureKey>,
58    I: TestableNodeImplementation<TYPES>,
59    V: Versions,
60> {
61    /// epoch height
62    pub epoch_height: u64,
63    /// Epoch start block
64    pub epoch_start_block: u64,
65    /// Saved epoch information. This must be sorted ascending by epoch.
66    pub start_epoch_info: Vec<InitializerEpochInfo<TYPES>>,
67    /// handle to the nodes
68    pub(crate) handles: Arc<RwLock<Vec<Node<TYPES, I, V>>>>,
69    /// late start nodes
70    pub(crate) late_start: HashMap<u64, LateStartNode<TYPES, I, V>>,
71    /// time based changes
72    pub(crate) changes: BTreeMap<TYPES::View, Vec<ChangeNode>>,
73    /// most recent view seen by spinning task
74    pub(crate) latest_view: Option<TYPES::View>,
75    /// Last decided leaf that can be used as the anchor leaf to initialize the node.
76    pub(crate) last_decided_leaf: Leaf2<TYPES>,
77    /// Highest qc seen in the test for restarting nodes
78    pub(crate) high_qc: QuorumCertificate2<TYPES>,
79    /// Next epoch highest qc seen in the test for restarting nodes
80    pub(crate) next_epoch_high_qc: Option<NextEpochQuorumCertificate2<TYPES>>,
81    /// Add specified delay to async calls
82    pub(crate) async_delay_config: DelayConfig,
83    /// Context stored for nodes to be restarted with
84    pub(crate) restart_contexts: HashMap<usize, RestartContext<TYPES, N, I, V>>,
85    /// Generate network channel for restart nodes
86    pub(crate) channel_generator: AsyncGenerator<Network<TYPES, I>>,
87    /// The light client state update certificate
88    pub(crate) state_cert: LightClientStateUpdateCertificate<TYPES>,
89}

91#[async_trait]
92impl<
93        TYPES: NodeType<
94            InstanceState = TestInstanceState,
95            ValidatedState = TestValidatedState,
96            BlockHeader = TestBlockHeader,
97        >,
98        I: TestableNodeImplementation<TYPES>,
99        N: ConnectedNetwork<TYPES::SignatureKey>,
100        V: Versions,
101    > TestTaskState for SpinningTask<TYPES, N, I, V>
102where
103    I: TestableNodeImplementation<TYPES>,
104    I: NodeImplementation<
105        TYPES,
106        Network = N,
107        Storage = TestStorage<TYPES>,
108        AuctionResultsProvider = TestAuctionResultsProvider<TYPES>,
109    >,
110{
111    type Event = Event<TYPES>;
112    type Error = Error;
113
114    async fn handle_event(&mut self, (message, _id): (Self::Event, usize)) -> Result<()> {
115        let Event { view_number, event } = message;
116
117        if let EventType::Decide {
118            leaf_chain,
119            qc: _,
120            block_size: _,
121        } = event
122        {
123            let leaf = leaf_chain.first().unwrap().leaf.clone();
124            if leaf.view_number() > self.last_decided_leaf.view_number() {
125                self.last_decided_leaf = leaf;
126            }
127        } else if let EventType::QuorumProposal {
128            proposal,
129            sender: _,
130        } = event
131        {
132            if proposal.data.justify_qc().view_number() > self.high_qc.view_number() {
133                self.high_qc = proposal.data.justify_qc().clone();
134            }
135        }
136
137        let mut new_nodes = vec![];
138        let mut new_networks = vec![];
139        // if we have not seen this view before
140        if self.latest_view.is_none() || view_number > self.latest_view.unwrap() {
141            // perform operations on the nodes
142            if let Some(operations) = self.changes.remove(&view_number) {
143                for ChangeNode { idx, updown } in operations {
144                    match updown {
145                        NodeAction::Up => {
146                            let node_id = idx.try_into().unwrap();
147                            if let Some(node) = self.late_start.remove(&node_id) {
148                                tracing::error!("Node {} spinning up late", idx);
149                                let network = if let Some(network) = node.network {
150                                    network
151                                } else {
152                                    let generated_network = (self.channel_generator)(node_id).await;
153                                    generated_network.wait_for_ready().await;
154                                    generated_network
155                                };
156                                let node_id = idx.try_into().unwrap();
157                                let context = match node.context {
158                                    LateNodeContext::InitializedContext(context) => context,
159                                    // Node not initialized. Initialize it
160                                    // based on the received leaf.
161                                    LateNodeContext::UninitializedContext(late_context_params) => {
162                                        // We'll deconstruct the individual terms here.
163                                        let LateNodeContextParameters {
164                                            storage,
165                                            memberships,
166                                            config,
167                                            marketplace_config,
168                                        } = late_context_params;
169
170                                        let initializer = HotShotInitializer::<TYPES>::load(
171                                            TestInstanceState::new(self.async_delay_config.clone()),
172                                            self.epoch_height,
173                                            self.epoch_start_block,
174                                            self.start_epoch_info.clone(),
175                                            self.last_decided_leaf.clone(),
176                                            (
177                                                TYPES::View::genesis(),
178                                                genesis_epoch_from_version::<V, TYPES>(),
179                                            ),
180                                            (self.high_qc.clone(), self.next_epoch_high_qc.clone()),
181                                            BTreeMap::new(),
182                                            BTreeMap::new(),
183                                            None,
184                                            self.state_cert.clone(),
185                                        );
186                                        // We assign node's public key and stake value rather than read from config file since it's a test
187                                        let validator_config =
188                                            ValidatorConfig::generated_from_seed_indexed(
189                                                [0u8; 32],
190                                                node_id,
191                                                1,
192                                                // For tests, make the node DA based on its index
193                                                node_id < config.da_staked_committee_size as u64,
194                                            );
195
196                                        TestRunner::add_node_with_config(
197                                            node_id,
198                                            network.clone(),
199                                            memberships,
200                                            initializer,
201                                            config,
202                                            validator_config,
203                                            storage,
204                                            marketplace_config,
205                                        )
206                                        .await
207                                    },
208                                    LateNodeContext::Restart => {
209                                        panic!("Cannot spin up a node with Restart context")
210                                    },
211                                };
212
213                                let handle = context.run_tasks().await;
214
215                                // Create the node and add it to the state, so we can shut them
216                                // down properly later to avoid the overflow error in the overall
217                                // safety task.
218                                let node = Node {
219                                    node_id,
220                                    network,
221                                    handle,
222                                };
223                                node.handle.hotshot.start_consensus().await;
224
225                                self.handles.write().await.push(node);
226                            }
227                        },
228                        NodeAction::Down => {
229                            if let Some(node) = self.handles.write().await.get_mut(idx) {
230                                tracing::error!("Node {} shutting down", idx);
231                                node.handle.shut_down().await;
232                            }
233                        },
234                        NodeAction::RestartDown(delay_views) => {
235                            let node_id = idx.try_into().unwrap();
236                            if let Some(node) = self.handles.write().await.get_mut(idx) {
237                                tracing::error!("Node {} shutting down", idx);
238                                node.handle.shut_down().await;
239                                // For restarted nodes generate the network on correct view
240                                let generated_network = (self.channel_generator)(node_id).await;
241
242                                let Some(LateStartNode {
243                                    network: _,
244                                    context: LateNodeContext::Restart,
245                                }) = self.late_start.get(&node_id)
246                                else {
247                                    panic!("Restarted Nodes must have an uninitialized context");
248                                };
249
250                                let storage = node.handle.storage().clone();
251                                let memberships = node.handle.membership_coordinator.clone();
252                                let config = node.handle.hotshot.config.clone();
253                                let marketplace_config =
254                                    node.handle.hotshot.marketplace_config.clone();
255
256                                let read_storage = storage.read().await;
257                                let next_epoch_high_qc =
258                                    read_storage.next_epoch_high_qc_cloned().await;
259                                let start_view = read_storage.last_actioned_view().await;
260                                let start_epoch = read_storage.last_actioned_epoch().await;
261                                let high_qc = read_storage.high_qc_cloned().await.unwrap_or(
262                                    QuorumCertificate2::genesis::<V>(
263                                        &TestValidatedState::default(),
264                                        &TestInstanceState::default(),
265                                    )
266                                    .await,
267                                );
268                                let state_cert = read_storage
269                                    .state_cert_cloned()
270                                    .await
271                                    .unwrap_or(LightClientStateUpdateCertificate::genesis());
272                                let saved_proposals = read_storage.proposals_cloned().await;
273                                let mut vid_shares = BTreeMap::new();
274                                for (view, hash_map) in read_storage.vids_cloned().await {
275                                    let mut converted_hash_map = HashMap::new();
276                                    for (key, proposal) in hash_map {
277                                        converted_hash_map.insert(key, convert_proposal(proposal));
278                                    }
279                                    vid_shares.insert(view, converted_hash_map);
280                                }
281                                let decided_upgrade_certificate =
282                                    read_storage.decided_upgrade_certificate().await;
283
284                                let initializer = HotShotInitializer::<TYPES>::load(
285                                    TestInstanceState::new(self.async_delay_config.clone()),
286                                    self.epoch_height,
287                                    self.epoch_start_block,
288                                    self.start_epoch_info.clone(),
289                                    self.last_decided_leaf.clone(),
290                                    (start_view, start_epoch),
291                                    (high_qc, next_epoch_high_qc),
292                                    saved_proposals,
293                                    vid_shares,
294                                    decided_upgrade_certificate,
295                                    state_cert,
296                                );
297                                // We assign node's public key and stake value rather than read from config file since it's a test
298                                let validator_config = ValidatorConfig::generated_from_seed_indexed(
299                                    [0u8; 32],
300                                    node_id,
301                                    1,
302                                    // For tests, make the node DA based on its index
303                                    node_id < config.da_staked_committee_size as u64,
304                                );
305                                let internal_chan = broadcast(EVENT_CHANNEL_SIZE);
306                                let context =
307                                    TestRunner::<TYPES, I, V, N>::add_node_with_config_and_channels(
308                                        node_id,
309                                        generated_network.clone(),
310                                        Arc::clone(memberships.membership()),
311                                        initializer,
312                                        config,
313                                        validator_config,
314                                        (*read_storage).clone(),
315                                        marketplace_config.clone(),
316                                        internal_chan,
317                                        (
318                                            node.handle.external_channel_sender(),
319                                            node.handle.event_stream_known_impl().new_receiver(),
320                                        ),
321                                    )
322                                    .await;
323                                if delay_views == 0 {
324                                    new_nodes.push((context, idx));
325                                    new_networks.push(generated_network.clone());
326                                } else {
327                                    let up_view = view_number + delay_views;
328                                    let change = ChangeNode {
329                                        idx,
330                                        updown: NodeAction::RestartUp,
331                                    };
332                                    self.changes.entry(up_view).or_default().push(change);
333                                    let new_ctx = RestartContext {
334                                        context,
335                                        network: generated_network.clone(),
336                                    };
337                                    self.restart_contexts.insert(idx, new_ctx);
338                                }
339                            }
340                        },
341                        NodeAction::RestartUp => {
342                            if let Some(ctx) = self.restart_contexts.remove(&idx) {
343                                new_nodes.push((ctx.context, idx));
344                                new_networks.push(ctx.network.clone());
345                            }
346                        },
347                        NodeAction::NetworkUp => {
348                            if let Some(handle) = self.handles.write().await.get(idx) {
349                                tracing::error!("Node {} networks resuming", idx);
350                                handle.network.resume();
351                            }
352                        },
353                        NodeAction::NetworkDown => {
354                            if let Some(handle) = self.handles.write().await.get(idx) {
355                                tracing::error!("Node {} networks pausing", idx);
356                                handle.network.pause();
357                            }
358                        },
359                    }
360                }
361            }
362            let mut ready_futs = vec![];
363            while let Some(net) = new_networks.pop() {
364                ready_futs.push(async move {
365                    net.wait_for_ready().await;
366                });
367            }
368            join_all(ready_futs).await;
369
370            let mut start_futs = vec![];
371
372            while let Some((node, id)) = new_nodes.pop() {
373                let handles = self.handles.clone();
374                let fut = async move {
375                    tracing::info!("Starting node {} back up", id);
376                    let handle = node.run_tasks().await;
377
378                    // Create the node and add it to the state, so we can shut them
379                    // down properly later to avoid the overflow error in the overall
380                    // safety task.
381                    let node = Node {
382                        node_id: id.try_into().unwrap(),
383                        network: node.network.clone(),
384                        handle,
385                    };
386                    node.handle.hotshot.start_consensus().await;
387
388                    handles.write().await[id] = node;
389                };
390                start_futs.push(fut);
391            }
392            if !start_futs.is_empty() {
393                join_all(start_futs).await;
394                tracing::info!("Nodes all started");
395            }
396
397            // update our latest view
398            self.latest_view = Some(view_number);
399        }
400
401        Ok(())
402    }
403
404    async fn check(&self) -> TestResult {
405        TestResult::Pass
406    }
407}

420/// Spin the node up or down
421#[derive(Clone, Debug)]
422pub enum NodeAction {
423    /// spin the node up
424    Up,
425    /// spin the node down
426    Down,
427    /// spin the node's network up
428    NetworkUp,
429    /// spin the node's network down
430    NetworkDown,
431    /// Take a node down to be restarted after a number of views
432    RestartDown(u64),
433    /// Start a node up again after it's been shutdown for restart.  This
434    /// should only be created following a `RestartDown`
435    RestartUp,
436}

438/// denotes a change in node state
439#[derive(Clone, Debug)]
440pub struct ChangeNode {
441    /// the index of the node
442    pub idx: usize,
443    /// spin the node or node's network up or down
444    pub updown: NodeAction,
445}
