94#[allow(clippy::type_complexity)]
95pub fn gen_node_lists<TYPES: NodeType>(
96    num_staked_nodes: u64,
97    num_da_nodes: u64,
98) -> (Vec<PeerConfig<TYPES>>, Vec<PeerConfig<TYPES>>) {
99    let mut staked_nodes = Vec::new();
100    let mut da_nodes = Vec::new();
101
102    for n in 0..num_staked_nodes {
103        let validator_config: ValidatorConfig<TYPES> =
104            ValidatorConfig::generated_from_seed_indexed([0u8; 32], n, 1, n < num_da_nodes);
105
106        let peer_config = validator_config.public_config();
107        staked_nodes.push(peer_config.clone());
108
109        if n < num_da_nodes {
110            da_nodes.push(peer_config)
111        }
112    }
113
114    (staked_nodes, da_nodes)
115}

160pub fn nonempty_block_threshold(threshold: (u64, u64)) -> TransactionValidator {
161    Arc::new(move |transactions| {
162        if matches!(threshold, (0, _)) {
163            return Ok(());
164        }
165
166        let blocks: Vec<_> = transactions.iter().filter(|(view, _)| *view != 0).collect();
167
168        let num_blocks = blocks.len() as u64;
169        let mut num_nonempty_blocks = 0;
170
171        ensure!(num_blocks > 0, "Failed to commit any non-genesis blocks");
172
173        for (_, num_transactions) in blocks {
174            if *num_transactions > 0 {
175                num_nonempty_blocks += 1;
176            }
177        }
178
179        ensure!(
180          // i.e. num_nonempty_blocks / num_blocks >= threshold.0 / threshold.1
181          num_nonempty_blocks * threshold.1 >= threshold.0 * num_blocks,
182          "Failed to meet nonempty block threshold of {}/{}; got {num_nonempty_blocks} nonempty blocks out of a total of {num_blocks}", threshold.0, threshold.1
183        );
184
185        Ok(())
186    })
187}

189pub fn nonempty_block_limit(limit: (u64, u64)) -> TransactionValidator {
190    Arc::new(move |transactions| {
191        if matches!(limit, (_, 0)) {
192            return Ok(());
193        }
194
195        let blocks: Vec<_> = transactions.iter().filter(|(view, _)| *view != 0).collect();
196
197        let num_blocks = blocks.len() as u64;
198        let mut num_nonempty_blocks = 0;
199
200        ensure!(num_blocks > 0, "Failed to commit any non-genesis blocks");
201
202        for (_, num_transactions) in blocks {
203            if *num_transactions > 0 {
204                num_nonempty_blocks += 1;
205            }
206        }
207
208        ensure!(
209          // i.e. num_nonempty_blocks / num_blocks <= limit.0 / limit.1
210          num_nonempty_blocks * limit.1 <= limit.0 * num_blocks,
211          "Exceeded nonempty block limit of {}/{}; got {num_nonempty_blocks} nonempty blocks out of a total of {num_blocks}", limit.0, limit.1
212        );
213
214        Ok(())
215    })

225pub async fn create_test_handle<
226    TYPES: NodeType<InstanceState = TestInstanceState>,
227    I: NodeImplementation<TYPES>,
228    V: Versions,
229>(
230    metadata: TestDescription<TYPES, I, V>,
231    node_id: u64,
232    network: Network<TYPES, I>,
233    memberships: Arc<RwLock<TYPES::Membership>>,
234    config: HotShotConfig<TYPES>,
235    storage: I::Storage,
236    marketplace_config: MarketplaceConfig<TYPES, I>,
237) -> SystemContextHandle<TYPES, I, V> {
238    let initializer = HotShotInitializer::<TYPES>::from_genesis::<V>(
239        TestInstanceState::new(metadata.async_delay_config),
240        metadata.test_config.epoch_height,
241        metadata.test_config.epoch_start_block,
242        vec![],
243    )
244    .await
245    .unwrap();
246
247    // See whether or not we should be DA
248    let is_da = node_id < config.da_staked_committee_size as u64;
249
250    let validator_config: ValidatorConfig<TYPES> =
251        ValidatorConfig::generated_from_seed_indexed([0u8; 32], node_id, 1, is_da);
252
253    // Get key pair for certificate aggregation
254    let private_key = validator_config.private_key.clone();
255    let public_key = validator_config.public_key.clone();
256    let state_private_key = validator_config.state_private_key.clone();
257    let membership_coordinator = EpochMembershipCoordinator::new(memberships, config.epoch_height);
258
259    let behaviour = (metadata.behaviour)(node_id);
260    match behaviour {
261        Behaviour::ByzantineTwins(state) => {
262            let state = Box::leak(state);
263            let (left_handle, _right_handle) = state
264                .spawn_twin_handles(
265                    public_key,
266                    private_key,
267                    state_private_key,
268                    node_id,
269                    config,
270                    membership_coordinator,
271                    network,
272                    initializer,
273                    ConsensusMetricsValue::default(),
274                    storage,
275                    marketplace_config,
276                )
277                .await;
278
279            left_handle
280        },
281        Behaviour::Byzantine(state) => {
282            let state = Box::leak(state);
283            state
284                .spawn_handle(
285                    public_key,
286                    private_key,
287                    state_private_key,
288                    node_id,
289                    config,
290                    membership_coordinator,
291                    network,
292                    initializer,
293                    ConsensusMetricsValue::default(),
294                    storage,
295                    marketplace_config,
296                )
297                .await
298        },
299        Behaviour::Standard => {
300            let hotshot = SystemContext::<TYPES, I, V>::new(
301                public_key,
302                private_key,
303                state_private_key,
304                node_id,
305                config,
306                membership_coordinator,
307                network,
308                initializer,
309                ConsensusMetricsValue::default(),
310                storage,
311                marketplace_config,
312            )
313            .await;
314
315            hotshot.run_tasks().await
316        },
317    }
318}

522impl<
523        TYPES: NodeType<InstanceState = TestInstanceState>,
524        I: TestableNodeImplementation<TYPES>,
525        V: Versions,
526    > TestDescription<TYPES, I, V>
527where
528    I: NodeImplementation<TYPES, AuctionResultsProvider = TestAuctionResultsProvider<TYPES>>,
529{
530    /// turn a description of a test (e.g. a [`TestDescription`]) into
531    /// a [`TestLauncher`] that can be used to launch the test.
532    /// # Panics
533    /// if some of the configuration values are zero
534    pub fn gen_launcher(self) -> TestLauncher<TYPES, I, V> {
535        self.gen_launcher_with_tasks(vec![])
536    }
537
538    /// turn a description of a test (e.g. a [`TestDescription`]) into
539    /// a [`TestLauncher`] that can be used to launch the test, with
540    /// additional testing tasks to run in test harness
541    /// # Panics
542    /// if some of the configuration values are zero
543    #[must_use]
544    pub fn gen_launcher_with_tasks(
545        self,
546        additional_test_tasks: Vec<Box<dyn TestTaskStateSeed<TYPES, I, V>>>,
547    ) -> TestLauncher<TYPES, I, V> {
548        let TestDescription {
549            timing_data,
550            unreliable_network,
551            test_config,
552            ..
553        } = self.clone();
554
555        let num_nodes_with_stake = test_config.num_nodes_with_stake.into();
556        let num_bootstrap_nodes = test_config.num_bootstrap;
557        let da_staked_committee_size = test_config.da_staked_committee_size;
558
559        let validator_config = Rc::new(move |node_id| {
560            ValidatorConfig::<TYPES>::generated_from_seed_indexed(
561                [0u8; 32],
562                node_id,
563                1,
564                // This is the config for node 0
565                node_id < test_config.da_staked_committee_size as u64,
566            )
567        });
568
569        let hotshot_config = Rc::new(move |_| test_config.clone());
570        let TimingData {
571            next_view_timeout,
572            builder_timeout,
573            data_request_delay,
574            secondary_network_delay,
575            view_sync_timeout,
576        } = timing_data;
577        // TODO this should really be using the timing config struct
578        let mod_hotshot_config = move |hotshot_config: &mut HotShotConfig<TYPES>| {
579            hotshot_config.next_view_timeout = next_view_timeout;
580            hotshot_config.builder_timeout = builder_timeout;
581            hotshot_config.data_request_delay = data_request_delay;
582            hotshot_config.view_sync_timeout = view_sync_timeout;
583        };
584
585        let metadata = self.clone();
586        TestLauncher {
587            resource_generators: ResourceGenerators {
588                channel_generator: <I as TestableNodeImplementation<TYPES>>::gen_networks(
589                    num_nodes_with_stake,
590                    num_bootstrap_nodes,
591                    da_staked_committee_size,
592                    unreliable_network,
593                    secondary_network_delay,
594                ),
595                storage: Rc::new(move |_| {
596                    let mut storage = TestStorage::<TYPES>::default();
597                    // update storage impl to use settings delay option
598                    storage.delay_config = metadata.async_delay_config.clone();
599                    storage
600                }),
601                hotshot_config,
602                validator_config,
603                marketplace_config: Rc::new(|_| MarketplaceConfig::<TYPES, I> {
604                    auction_results_provider: TestAuctionResultsProvider::<TYPES>::default().into(),
605                    fallback_builder_url: Url::parse("http://localhost:9999").unwrap(),
606                }),
607            },
608            metadata: self,
609            additional_test_tasks,
610        }
611        .map_hotshot_config(mod_hotshot_config)
612    }
