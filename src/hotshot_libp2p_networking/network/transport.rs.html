27/// The maximum size of an authentication message. This is used to prevent
28/// DoS attacks by sending large messages.
29const MAX_AUTH_MESSAGE_SIZE: usize = 1024;
30
31/// The timeout for the authentication handshake. This is used to prevent
32/// attacks that keep connections open indefinitely by half-finishing the
33/// handshake.
34const AUTH_HANDSHAKE_TIMEOUT: std::time::Duration = std::time::Duration::from_secs(5);

36/// A wrapper for a `Transport` that bidirectionally authenticates connections
37/// by performing a handshake that checks if the remote peer is present in the
38/// stake table.
39#[pin_project]
40pub struct StakeTableAuthentication<T: Transport, Types: NodeType, C: StreamMuxer + Unpin> {
41    #[pin]
42    /// The underlying transport we are wrapping
43    pub inner: T,
44
45    /// The stake table we check against to authenticate connections
46    pub membership: Arc<Option<Arc<RwLock<Types::Membership>>>>,
47
48    /// A pre-signed message that we send to the remote peer for authentication
49    pub auth_message: Arc<Option<Vec<u8>>>,
50
51    /// Phantom data for the connection type
52    pd: std::marker::PhantomData<C>,
53}
54
55/// A type alias for the future that upgrades a connection to perform the authentication handshake
56type UpgradeFuture<T> =
57    Pin<Box<dyn Future<Output = Result<<T as Transport>::Output, <T as Transport>::Error>> + Send>>;
58
59impl<T: Transport, Types: NodeType, C: StreamMuxer + Unpin> StakeTableAuthentication<T, Types, C> {
60    /// Create a new `StakeTableAuthentication` transport that wraps the given transport
61    /// and authenticates connections against the stake table.
62    pub fn new(
63        inner: T,
64        membership: Option<Arc<RwLock<Types::Membership>>>,
65        auth_message: Option<Vec<u8>>,
66    ) -> Self {
67        Self {
68            inner,
69            membership: Arc::from(membership),
70            auth_message: Arc::from(auth_message),
71            pd: std::marker::PhantomData,
72        }
73    }
74
75    /// Prove to the remote peer that we are in the stake table by sending
76    /// them our authentication message.
77    ///
78    /// # Errors
79    /// - If we fail to write the message to the stream
80    pub async fn authenticate_with_remote_peer<W: AsyncWrite + Unpin>(
81        stream: &mut W,
82        auth_message: Arc<Option<Vec<u8>>>,
83    ) -> AnyhowResult<()> {
84        // If we have an auth message, send it to the remote peer, prefixed with
85        // the message length
86        if let Some(auth_message) = auth_message.as_ref() {
87            // Write the length-delimited message
88            write_length_delimited(stream, auth_message).await?;
89        }
90
91        Ok(())
92    }
93
94    /// Verify that the remote peer is:
95    /// - In the stake table
96    /// - Sending us a valid authentication message
97    /// - Sending us a valid signature
98    /// - Matching the peer ID we expect
99    ///
100    /// # Errors
101    /// If the peer fails verification. This can happen if:
102    /// - We fail to read the message from the stream
103    /// - The message is too large
104    /// - The message is invalid
105    /// - The peer is not in the stake table
106    /// - The signature is invalid
107    pub async fn verify_peer_authentication<R: AsyncReadExt + Unpin>(
108        stream: &mut R,
109        membership: Arc<Option<Arc<RwLock<Types::Membership>>>>,
110        required_peer_id: &PeerId,
111    ) -> AnyhowResult<()> {
112        // If we have a membership, read message and validate
113        if membership.is_some() {
114            // Read the length-delimited message from the remote peer
115            let message = read_length_delimited(stream, MAX_AUTH_MESSAGE_SIZE).await?;
116
117            // Deserialize the authentication message
118            let auth_message: AuthMessage<Types::SignatureKey> = bincode::deserialize(&message)
119                .with_context(|| "Failed to deserialize auth message")?;
120
121            // Verify the signature on the public keys
122            auth_message
123                .validate()
124                .with_context(|| "Failed to verify authentication message")?;
125
126            // Deserialize the `PeerId`
127            let peer_id = PeerId::from_bytes(&auth_message.peer_id_bytes)
128                .with_context(|| "Failed to deserialize peer ID")?;
129
130            // Verify that the peer ID is the same as the remote peer
131            if peer_id != *required_peer_id {
132                return Err(anyhow::anyhow!("Peer ID mismatch"));
133            }
134        }
135
136        Ok(())
137    }
138
139    /// Wrap the supplied future in an upgrade that performs the authentication handshake.
140    ///
141    /// `outgoing` is a boolean that indicates if the connection is incoming or outgoing.
142    /// This is needed because the flow of the handshake is different for each.
143    fn gen_handshake<F: Future<Output = Result<T::Output, T::Error>> + Send + 'static>(
144        original_future: F,
145        outgoing: bool,
146        membership: Arc<Option<Arc<RwLock<Types::Membership>>>>,
147        auth_message: Arc<Option<Vec<u8>>>,
148    ) -> UpgradeFuture<T>
149    where
150        T::Error: From<<C as StreamMuxer>::Error> + From<IoError>,
151        T::Output: AsOutput<C> + Send,
152
153        C::Substream: Unpin + Send,
154    {
155        // Create a new upgrade that performs the authentication handshake on top
156        Box::pin(async move {
157            // Wait for the original future to resolve
158            let mut stream = original_future.await?;
159
160            // Time out the authentication block
161            timeout(AUTH_HANDSHAKE_TIMEOUT, async {
162                // Open a substream for the handshake.
163                // The handshake order depends on whether the connection is incoming or outgoing.
164                let mut substream = if outgoing {
165                    poll_fn(|cx| stream.as_connection().poll_outbound_unpin(cx)).await?
166                } else {
167                    poll_fn(|cx| stream.as_connection().poll_inbound_unpin(cx)).await?
168                };
169
170                if outgoing {
171                    // If the connection is outgoing, authenticate with the remote peer first
172                    Self::authenticate_with_remote_peer(&mut substream, auth_message)
173                        .await
174                        .map_err(|e| {
175                            warn!("Failed to authenticate with remote peer: {:?}", e);
176                            IoError::new(IoErrorKind::Other, e)
177                        })?;
178
179                    // Verify the remote peer's authentication
180                    Self::verify_peer_authentication(
181                        &mut substream,
182                        membership,
183                        stream.as_peer_id(),
184                    )
185                    .await
186                    .map_err(|e| {
187                        warn!("Failed to verify remote peer: {:?}", e);
188                        IoError::new(IoErrorKind::Other, e)
189                    })?;
190                } else {
191                    // If it is incoming, verify the remote peer's authentication first
192                    Self::verify_peer_authentication(
193                        &mut substream,
194                        membership,
195                        stream.as_peer_id(),
196                    )
197                    .await
198                    .map_err(|e| {
199                        warn!("Failed to verify remote peer: {:?}", e);
200                        IoError::new(IoErrorKind::Other, e)
201                    })?;
202
203                    // Authenticate with the remote peer
204                    Self::authenticate_with_remote_peer(&mut substream, auth_message)
205                        .await
206                        .map_err(|e| {
207                            warn!("Failed to authenticate with remote peer: {:?}", e);
208                            IoError::new(IoErrorKind::Other, e)
209                        })?;
210                }
211
212                Ok(stream)
213            })
214            .await
215            .map_err(|e| {
216                warn!("Timed out performing authentication handshake: {:?}", e);
217                IoError::new(IoErrorKind::TimedOut, e)
218            })?
219        })
220    }
221}

223/// The deserialized form of an authentication message that is sent to the remote peer
224#[derive(Clone, Serialize, Deserialize)]
225struct AuthMessage<S: SignatureKey> {
226    /// The encoded (stake table) public key of the sender. This, along with the peer ID, is
227    /// signed. It is still encoded here to enable easy verification.
228    public_key_bytes: Vec<u8>,
229
230    /// The encoded peer ID of the sender. This is appended to the public key before signing.
231    /// It is still encoded here to enable easy verification.
232    peer_id_bytes: Vec<u8>,
233
234    /// The signature on the public key
235    signature: S::PureAssembledSignatureType,
236}
237
238impl<S: SignatureKey> AuthMessage<S> {
239    /// Validate the signature on the public key and return it if valid
240    pub fn validate(&self) -> AnyhowResult<S> {
241        // Deserialize the stake table public key
242        let public_key = S::from_bytes(&self.public_key_bytes)
243            .with_context(|| "Failed to deserialize public key")?;
244
245        // Reconstruct the signed message from the public key and peer ID
246        let mut signed_message = public_key.to_bytes();
247        signed_message.extend(self.peer_id_bytes.clone());
248
249        // Check if the signature is valid across both
250        if !public_key.validate(&self.signature, &signed_message) {
251            return Err(anyhow::anyhow!("Invalid signature"));
252        }
253
254        Ok(public_key)
255    }
256}

258/// Create an sign an authentication message to be sent to the remote peer
259///
260/// # Errors
261/// - If we fail to sign the public key
262/// - If we fail to serialize the authentication message
263pub fn construct_auth_message<S: SignatureKey + 'static>(
264    public_key: &S,
265    peer_id: &PeerId,
266    private_key: &S::PrivateKey,
267) -> AnyhowResult<Vec<u8>> {
268    // Serialize the stake table public key
269    let mut public_key_bytes = public_key.to_bytes();
270
271    // Serialize the peer ID and append it
272    let peer_id_bytes = peer_id.to_bytes();
273    public_key_bytes.extend_from_slice(&peer_id_bytes);
274
275    // Sign our public key
276    let signature =
277        S::sign(private_key, &public_key_bytes).with_context(|| "Failed to sign public key")?;
278
279    // Create the auth message
280    let auth_message = AuthMessage::<S> {
281        public_key_bytes,
282        peer_id_bytes,
283        signature,
284    };
285
286    // Serialize the auth message
287    bincode::serialize(&auth_message).with_context(|| "Failed to serialize auth message")
288}

290impl<T: Transport, Types: NodeType, C: StreamMuxer + Unpin> Transport
291    for StakeTableAuthentication<T, Types, C>
292where
293    T::Dial: Future<Output = Result<T::Output, T::Error>> + Send + 'static,
294    T::ListenerUpgrade: Send + 'static,
295    T::Output: AsOutput<C> + Send,
296    T::Error: From<<C as StreamMuxer>::Error> + From<IoError>,
297
298    C::Substream: Unpin + Send,
299{
300    // `Dial` is for connecting out, `ListenerUpgrade` is for accepting incoming connections
301    type Dial = Pin<Box<dyn Future<Output = Result<T::Output, T::Error>> + Send>>;
302    type ListenerUpgrade = Pin<Box<dyn Future<Output = Result<T::Output, T::Error>> + Send>>;
303
304    // These are just passed through
305    type Output = T::Output;
306    type Error = T::Error;
307
308    /// Dial a remote peer. This function is changed to perform an authentication handshake
309    /// on top.
310    fn dial(
311        &mut self,
312        addr: libp2p::Multiaddr,
313        opts: DialOpts,
314    ) -> Result<Self::Dial, libp2p::TransportError<Self::Error>> {
315        // Perform the inner dial
316        let res = self.inner.dial(addr, opts);
317
318        // Clone the necessary fields
319        let auth_message = Arc::clone(&self.auth_message);
320        let membership = Arc::clone(&self.membership);
321
322        // If the dial was successful, perform the authentication handshake on top
323        match res {
324            Ok(dial) => Ok(Self::gen_handshake(dial, true, membership, auth_message)),
325            Err(err) => Err(err),
326        }
327    }
328
329    /// This function is where we perform the authentication handshake for _incoming_ connections.
330    /// The flow in this case is the reverse of the `dial` function: we first verify the remote peer's
331    /// authentication, and then authenticate with them.
332    fn poll(
333        mut self: std::pin::Pin<&mut Self>,
334        cx: &mut std::task::Context<'_>,
335    ) -> std::task::Poll<libp2p::core::transport::TransportEvent<Self::ListenerUpgrade, Self::Error>>
336    {
337        match self.as_mut().project().inner.poll(cx) {
338            Poll::Ready(event) => Poll::Ready(match event {
339                // If we have an incoming connection, we need to perform the authentication handshake
340                TransportEvent::Incoming {
341                    listener_id,
342                    upgrade,
343                    local_addr,
344                    send_back_addr,
345                } => {
346                    // Clone the necessary fields
347                    let auth_message = Arc::clone(&self.auth_message);
348                    let membership = Arc::clone(&self.membership);
349
350                    // Generate the handshake upgrade future (inbound)
351                    let auth_upgrade =
352                        Self::gen_handshake(upgrade, false, membership, auth_message);
353
354                    // Return the new event
355                    TransportEvent::Incoming {
356                        listener_id,
357                        upgrade: auth_upgrade,
358                        local_addr,
359                        send_back_addr,
360                    }
361                },
362
363                // We need to re-map the other events because we changed the type of the upgrade
364                TransportEvent::AddressExpired {
365                    listener_id,
366                    listen_addr,
367                } => TransportEvent::AddressExpired {
368                    listener_id,
369                    listen_addr,
370                },
371                TransportEvent::ListenerClosed {
372                    listener_id,
373                    reason,
374                } => TransportEvent::ListenerClosed {
375                    listener_id,
376                    reason,
377                },
378                TransportEvent::ListenerError { listener_id, error } => {
379                    TransportEvent::ListenerError { listener_id, error }
380                },
381                TransportEvent::NewAddress {
382                    listener_id,
383                    listen_addr,
384                } => TransportEvent::NewAddress {
385                    listener_id,
386                    listen_addr,
387                },
388            }),
389
390            Poll::Pending => Poll::Pending,
391        }
392    }
393
394    /// The below functions just pass through to the inner transport, but we had
395    /// to define them
396    fn remove_listener(&mut self, id: libp2p::core::transport::ListenerId) -> bool {
397        self.inner.remove_listener(id)
398    }
399    fn listen_on(
400        &mut self,
401        id: libp2p::core::transport::ListenerId,
402        addr: libp2p::Multiaddr,
403    ) -> Result<(), libp2p::TransportError<Self::Error>> {
404        self.inner.listen_on(id, addr)
405    }
406}

410trait AsOutput<C: StreamMuxer + Unpin> {

432/// A helper function to read a length-delimited message from a stream. Takes into
433/// account the maximum message size.
434///
435/// # Errors
436/// - If the message is too big
437/// - If we fail to read from the stream
438pub async fn read_length_delimited<S: AsyncRead + Unpin>(
439    stream: &mut S,
440    max_size: usize,
441) -> AnyhowResult<Vec<u8>> {
442    // Receive the first 8 bytes of the message, which is the length
443    let mut len_bytes = [0u8; 4];
444    stream
445        .read_exact(&mut len_bytes)
446        .await
447        .with_context(|| "Failed to read message length")?;
448
449    // Parse the length of the message as a `u32`
450    let len = usize::try_from(u32::from_be_bytes(len_bytes))?;
451
452    // Quit if the message is too large
453    ensure!(len <= max_size, "Message too large");
454
455    // Read the actual message
456    let mut message = vec![0u8; len];
457    stream
458        .read_exact(&mut message)
459        .await
460        .with_context(|| "Failed to read message")?;
461
462    Ok(message)
463}
464
465/// A helper function to write a length-delimited message to a stream.
466///
467/// # Errors
468/// - If we fail to write to the stream
469pub async fn write_length_delimited<S: AsyncWrite + Unpin>(
470    stream: &mut S,
471    message: &[u8],
472) -> AnyhowResult<()> {
473    // Write the length of the message
474    stream
475        .write_all(&u32::try_from(message.len())?.to_be_bytes())
476        .await
477        .with_context(|| "Failed to write message length")?;
478
479    // Write the actual message
480    stream
481        .write_all(message)
482        .await
483        .with_context(|| "Failed to write message")?;
484
485    Ok(())
486}
