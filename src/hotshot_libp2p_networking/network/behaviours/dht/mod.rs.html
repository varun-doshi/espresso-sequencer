51/// Behaviour wrapping libp2p's kademlia
52/// included:
53/// - publishing API
54/// - Request API
55/// - bootstrapping into the network
56/// - peer discovery
57#[derive(Debug)]
58pub struct DHTBehaviour<K: SignatureKey + 'static, D: DhtPersistentStorage> {
59    /// in progress queries for nearby peers
60    pub in_progress_get_closest_peers: HashMap<QueryId, Sender<()>>,
61    /// List of in-progress get requests
62    in_progress_record_queries: HashMap<QueryId, KadGetQuery>,
63    /// The list of in-progress get requests by key
64    outstanding_dht_query_keys: HashMap<Vec<u8>, QueryId>,
65    /// List of in-progress put requests
66    in_progress_put_record_queries: HashMap<QueryId, KadPutQuery>,
67    /// State of bootstrapping
68    pub bootstrap_state: Bootstrap,
69    /// the peer id (useful only for debugging right now)
70    pub peer_id: PeerId,
71    /// replication factor
72    pub replication_factor: NonZeroUsize,
73    /// Sender to retry requests.
74    retry_tx: Option<UnboundedSender<ClientRequest>>,
75    /// Sender to the bootstrap task
76    bootstrap_tx: Option<mpsc::Sender<bootstrap::InputEvent>>,
77
78    /// Phantom type for the key and persistent storage
79    phantom: PhantomData<(K, D)>,
80}

141    /// print out the routing table to stderr
142    pub fn print_routing_table(
143        &mut self,
144        kadem: &mut KademliaBehaviour<PersistentStore<ValidatedStore<MemoryStore, K>, D>>,
145    ) {
146        let mut err = format!("KBUCKETS: PID: {:?}, ", self.peer_id);
147        let v = kadem.kbuckets().collect::<Vec<_>>();
148        for i in v {
149            for j in i.iter() {
150                let s = format!(
151                    "node: key: {:?}, val {:?}, status: {:?}",
152                    j.node.key, j.node.value, j.status
153                );
154                err.push_str(&s);
155            }
156        }
157        error!("{:?}", err);
158    }

172    /// Retrieve a value for a key from the DHT.
173    pub fn get_record(
174        &mut self,
175        key: Vec<u8>,
176        chans: Vec<Sender<Vec<u8>>>,
177        factor: NonZeroUsize,
178        backoff: ExponentialBackoff,
179        retry_count: u8,
180        kad: &mut KademliaBehaviour<PersistentStore<ValidatedStore<MemoryStore, K>, D>>,
181    ) {
182        // noop
183        if retry_count == 0 {
184            return;
185        }
186
187        // Check the cache before making the (expensive) query
188        if let Some(entry) = kad.store_mut().get(&key.clone().into()) {
189            // The key already exists in the cache, send the value to all channels
190            for chan in chans {
191                if chan.send(entry.value.clone()).is_err() {
192                    warn!("Get DHT: channel closed before get record request result could be sent");
193                }
194            }
195        } else {
196            // Check if the key is already being queried
197            if let Some(qid) = self.outstanding_dht_query_keys.get(&key) {
198                // The key was already being queried. Add the channel to the existing query
199                // Try to get the query from the query id
200                let Some(query) = self.in_progress_record_queries.get_mut(qid) else {
201                    warn!("Get DHT: outstanding query not found");
202                    return;
203                };
204
205                // Add the channel to the existing query
206                query.notify.extend(chans);
207            } else {
208                // The key was not already being queried and was not in the cache. Start a new query.
209                let qid = kad.get_record(key.clone().into());
210                let query = KadGetQuery {
211                    backoff,
212                    progress: DHTProgress::InProgress(qid),
213                    notify: chans,
214                    num_replicas: factor,
215                    key: key.clone(),
216                    retry_count: retry_count - 1,
217                    records: HashMap::default(),
218                };
219
220                // Add the key to the outstanding queries and in-progress queries
221                self.outstanding_dht_query_keys.insert(key, qid);
222                self.in_progress_record_queries.insert(qid, query);
223            }
224        }
225    }

227    /// Spawn a task which will retry the query after a backoff.
228    fn retry_get(&self, mut query: KadGetQuery) {
229        let Some(tx) = self.retry_tx.clone() else {
230            return;
231        };
232        let req = ClientRequest::GetDHT {
233            key: query.key,
234            notify: query.notify,
235            retry_count: query.retry_count,
236        };
237        let backoff = query.backoff.next_timeout(false);
238        spawn(async move {
239            sleep(backoff).await;
240            let _ = tx.send(req);
241        });
242    }
243
244    /// Spawn a task which will retry the query after a backoff.
245    fn retry_put(&self, mut query: KadPutQuery) {
246        let Some(tx) = self.retry_tx.clone() else {
247            return;
248        };
249        let req = ClientRequest::PutDHT {
250            key: query.key,
251            value: query.value,
252            notify: query.notify,
253        };
254        spawn(async move {
255            sleep(query.backoff.next_timeout(false)).await;
256            let _ = tx.send(req);
257        });
258    }

260    /// update state based on recv-ed get query
261    fn handle_get_query(
262        &mut self,
263        store: &mut PersistentStore<ValidatedStore<MemoryStore, K>, D>,
264        record_results: GetRecordResult,
265        id: QueryId,
266        mut last: bool,
267    ) {
268        let num = match self.in_progress_record_queries.get_mut(&id) {
269            Some(query) => match record_results {
270                Ok(results) => match results {
271                    GetRecordOk::FoundRecord(record) => {
272                        match query.records.entry(record.record.value) {
273                            std::collections::hash_map::Entry::Occupied(mut o) => {
274                                let num_entries = o.get_mut();
275                                *num_entries += 1;
276                                *num_entries
277                            },
278                            std::collections::hash_map::Entry::Vacant(v) => {
279                                v.insert(1);
280                                1
281                            },
282                        }
283                    },
284                    GetRecordOk::FinishedWithNoAdditionalRecord {
285                        cache_candidates: _,
286                    } => {
287                        tracing::debug!("GetRecord Finished with No Additional Record");
288                        last = true;
289                        0
290                    },
291                },
292                Err(err) => {
293                    warn!("Error in Kademlia query: {:?}", err);
294                    0
295                },
296            },
297            None => {
298                // We already finished the query (or it's been cancelled). Do nothing and exit the
299                // function.
300                return;
301            },
302        };
303
304        // if the query has completed and we need to retry
305        // or if the query has enough replicas to return to the client
306        // trigger retry or completion logic
307        if num >= NUM_REPLICATED_TO_TRUST || last {
308            if let Some(KadGetQuery {
309                backoff,
310                progress,
311                notify,
312                num_replicas,
313                key,
314                retry_count,
315                records,
316            }) = self.in_progress_record_queries.remove(&id)
317            {
318                // Remove the key from the outstanding queries so we are in sync
319                self.outstanding_dht_query_keys.remove(&key);
320
321                // `notify` is all channels that are still open
322                let notify = notify
323                    .into_iter()
324                    .filter(|n| !n.is_canceled())
325                    .collect::<Vec<_>>();
326
327                // If all are closed, we can exit
328                if notify.is_empty() {
329                    return;
330                }
331
332                // NOTE case where multiple nodes agree on different
333                // values is not handled because it can't be hit.
334                // We optimistically choose whichever record returns the most trusted entries first
335
336                // iterate through the records and find an value that has enough replicas
337                // to trust the value
338                if let Some((r, _)) = records
339                    .into_iter()
340                    .find(|(_, v)| *v >= NUM_REPLICATED_TO_TRUST)
341                {
342                    let record = Record {
343                        key: key.into(),
344                        value: r.clone(),
345                        publisher: None,
346                        expires: None,
347                    };
348
349                    // Only return the record if we can store it (validation passed)
350                    if store.put(record).is_ok() {
351                        // Send the record to all channels that are still open
352                        for n in notify {
353                            if n.send(r.clone()).is_err() {
354                                warn!("Get DHT: channel closed before get record request result could be sent");
355                            }
356                        }
357                    } else {
358                        error!("Failed to store record in local store");
359                    }
360                }
361                // disagreement => query more nodes
362                else {
363                    // there is some internal disagreement or not enough nodes returned
364                    // Initiate new query that hits more replicas
365                    if retry_count > 0 {
366                        let new_retry_count = retry_count - 1;
367                        warn!("Get DHT: Internal disagreement for get dht request {:?}! requerying with more nodes. {:?} retries left", progress, new_retry_count);
368                        let new_factor = NonZeroUsize::max(
369                            NonZeroUsize::new(num_replicas.get() + 1).unwrap_or(num_replicas),
370                            *MAX_DHT_QUERY_SIZE,
371                        );
372                        self.retry_get(KadGetQuery {
373                            backoff,
374                            progress: DHTProgress::NotStarted,
375                            notify,
376                            num_replicas: new_factor,
377                            key,
378                            retry_count: new_retry_count,
379                            records: HashMap::default(),
380                        });
381                    }
382                    warn!("Get DHT: Internal disagreement for get dht request {:?}! Giving up because out of retries. ", progress);
383                }
384            }
385        }
386    }

388    /// Update state based on put query
389    fn handle_put_query(&mut self, record_results: PutRecordResult, id: QueryId) {
390        if let Some(mut query) = self.in_progress_put_record_queries.remove(&id) {
391            // dropped so we handle further
392            if query.notify.is_canceled() {
393                return;
394            }
395
396            match record_results {
397                Ok(_) => {
398                    if query.notify.send(()).is_err() {
399                        warn!("Put DHT: client channel closed before put record request could be sent");
400                    }
401                },
402                Err(e) => {
403                    query.progress = DHTProgress::NotStarted;
404                    query.backoff.start_next(false);
405
406                    warn!(
407                        "Put DHT: error performing put: {:?}. Retrying on pid {:?}.",
408                        e, self.peer_id
409                    );
410                    // push back onto the queue
411                    self.retry_put(query);
412                },
413            }
414        } else {
415            warn!("Put DHT: completed DHT query that is no longer tracked.");
416        }
417    }

419    /// Send that the bootstrap succeeded
420    fn finish_bootstrap(&mut self) {
421        if let Some(mut tx) = self.bootstrap_tx.clone() {
422            spawn(async move { tx.send(bootstrap::InputEvent::BootstrapFinished).await });
423        }

425    #[allow(clippy::too_many_lines)]
426    /// handle a DHT event
427    pub fn dht_handle_event(
428        &mut self,
429        event: KademliaEvent,
430        store: &mut PersistentStore<ValidatedStore<MemoryStore, K>, D>,
431    ) -> Option<NetworkEvent> {
432        match event {
433            KademliaEvent::OutboundQueryProgressed {
434                result: QueryResult::PutRecord(record_results),
435                id,
436                step: ProgressStep { last, .. },
437                ..
438            } => {
439                if last {
440                    self.handle_put_query(record_results, id);
441                }
442            },
443            KademliaEvent::OutboundQueryProgressed {
444                result: QueryResult::GetClosestPeers(r),
445                id: query_id,
446                stats: _,
447                step: ProgressStep { last: true, .. },
448                ..
449            } => match r {
450                Ok(GetClosestPeersOk { key, peers: _ }) => {
451                    if let Some(chan) = self.in_progress_get_closest_peers.remove(&query_id) {
452                        if chan.send(()).is_err() {
453                            warn!("DHT: finished query but client was no longer interested");
454                        };
455                    };
456                    debug!("Successfully got closest peers for key {:?}", key);
457                },
458                Err(e) => {
459                    if let Some(chan) = self.in_progress_get_closest_peers.remove(&query_id) {
460                        let _: Result<_, _> = chan.send(());
461                    };
462                    warn!("Failed to get closest peers: {:?}", e);
463                },
464            },
465            KademliaEvent::OutboundQueryProgressed {
466                result: QueryResult::GetRecord(record_results),
467                id,
468                step: ProgressStep { last, .. },
469                ..
470            } => {
471                self.handle_get_query(store, record_results, id, last);
472            },
473            KademliaEvent::OutboundQueryProgressed {
474                result:
475                    QueryResult::Bootstrap(Ok(BootstrapOk {
476                        peer: _,
477                        num_remaining,
478                    })),
479                step: ProgressStep { last: true, .. },
480                ..
481            } => {
482                if num_remaining == 0 {
483                    self.finish_bootstrap();
484                } else {
485                    debug!("Bootstrap in progress, {} nodes remaining", num_remaining);
486                }
487                return Some(NetworkEvent::IsBootstrapped);
488            },
489            KademliaEvent::OutboundQueryProgressed {
490                result: QueryResult::Bootstrap(Err(e)),
491                ..
492            } => {
493                let BootstrapError::Timeout { num_remaining, .. } = e;
494                if num_remaining.is_none() {
495                    error!("Failed to bootstrap: {:?}", e);
496                }
497                self.finish_bootstrap();
498            },
499            KademliaEvent::RoutablePeer { peer, address: _ } => {
500                debug!("Found routable peer {:?}", peer);
501            },
502            KademliaEvent::PendingRoutablePeer { peer, address: _ } => {
503                debug!("Found pending routable peer {:?}", peer);
504            },
505            KademliaEvent::UnroutablePeer { peer } => {
506                debug!("Found unroutable peer {:?}", peer);
507            },
508            KademliaEvent::RoutingUpdated {
509                peer: _,
510                is_new_peer: _,
511                addresses: _,
512                bucket_range: _,
513                old_peer: _,
514            } => {
515                debug!("Routing table updated");
516            },
517            e @ KademliaEvent::OutboundQueryProgressed { .. } => {
518                debug!("Not handling dht event {:?}", e);
519            },
520            e => {
521                debug!("New unhandled swarm event: {e:?}");
522            },
523        }
524        None
525    }
526}

528/// Metadata holder for get query
529#[derive(Debug)]
530pub(crate) struct KadGetQuery {
531    /// Exponential retry backoff
532    pub(crate) backoff: ExponentialBackoff,
533    /// progress through DHT query
534    pub(crate) progress: DHTProgress,
535    /// The channels to notify of the result
536    pub(crate) notify: Vec<Sender<Vec<u8>>>,
537    /// number of replicas required to replicate over
538    pub(crate) num_replicas: NonZeroUsize,
539    /// the key to look up
540    pub(crate) key: Vec<u8>,
541    /// the number of remaining retries before giving up
542    pub(crate) retry_count: u8,
543    /// already received records
544    pub(crate) records: HashMap<Vec<u8>, usize>,
545}

547/// Metadata holder for get query
548#[derive(Debug)]
549pub struct KadPutQuery {

556    /// the key to put
557    pub(crate) key: Vec<u8>,
558    /// the value to put
559    pub(crate) value: Vec<u8>,
560}
