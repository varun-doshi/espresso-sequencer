20/// A trait that we use to save and load the DHT to a file on disk
21/// or other storage medium
22#[async_trait]
23pub trait DhtPersistentStorage: Send + Sync + 'static + Clone {
24    /// Save the DHT (as a list of serializable records) to the persistent storage
25    ///
26    /// # Errors
27    /// - If we fail to save the DHT to the persistent storage provider
28    async fn save(&self, _records: Vec<SerializableRecord>) -> anyhow::Result<()>;
29
30    /// Load the DHT (as a list of serializable records) from the persistent storage
31    ///
32    /// # Errors
33    /// - If we fail to load the DHT from the persistent storage provider
34    async fn load(&self) -> anyhow::Result<Vec<SerializableRecord>>;
35}
36
37/// A no-op `PersistentStorage` that does not persist the DHT
38#[derive(Clone)]
39pub struct DhtNoPersistence;

41#[async_trait]
42impl DhtPersistentStorage for DhtNoPersistence {
43    async fn save(&self, _records: Vec<SerializableRecord>) -> anyhow::Result<()> {
44        Ok(())
45    }
46
47    async fn load(&self) -> anyhow::Result<Vec<SerializableRecord>> {
48        Ok(vec![])
49    }
50}

68#[async_trait]
69impl DhtPersistentStorage for DhtFilePersistence {
70    /// Save the DHT to the file on disk
71    ///
72    /// # Errors
73    /// - If we fail to serialize the records
74    /// - If we fail to write the serialized records to the file
75    async fn save(&self, records: Vec<SerializableRecord>) -> anyhow::Result<()> {
76        // Bincode-serialize the records
77        let to_save =
78            bincode::serialize(&records).with_context(|| "Failed to serialize records")?;
79
80        // Write the serialized records to the file
81        std::fs::write(&self.path, to_save).with_context(|| "Failed to write records to file")?;
82
83        Ok(())
84    }
85
86    /// Load the DHT from the file on disk
87    ///
88    /// # Errors
89    /// - If we fail to read the file
90    /// - If we fail to deserialize the records
91    async fn load(&self) -> anyhow::Result<Vec<SerializableRecord>> {
92        // Read the contents of the file
93        let contents =
94            std::fs::read(&self.path).with_context(|| "Failed to read records from file")?;
95
96        // Deserialize the contents
97        let records: Vec<SerializableRecord> =
98            bincode::deserialize(&contents).with_context(|| "Failed to deserialize records")?;
99
100        Ok(records)
101    }
102}

104/// A `RecordStore` wrapper that occasionally saves the DHT to a persistent storage.
105pub struct PersistentStore<R: RecordStore, D: DhtPersistentStorage> {
106    /// The underlying record store
107    underlying_record_store: R,
108
109    /// The persistent storage
110    persistent_storage: D,
111
112    /// The semaphore for limiting the number of concurrent operations (to one)
113    semaphore: Arc<Semaphore>,
114
115    /// The maximum number of records that can be added to the store before the store is saved to the persistent storage
116    max_record_delta: u64,
117
118    /// The running delta between the records in the persistent storage and the records in the underlying store
119    record_delta: Arc<AtomicU64>,
120}

122/// A serializable version of a Libp2p `Record`
123#[derive(Serialize, Deserialize)]
124pub struct SerializableRecord {
125    /// The key of the record
126    pub key: libp2p::kad::RecordKey,
127    /// The value of the record
128    pub value: Vec<u8>,
129    /// The (original) publisher of the record.
130    pub publisher: Option<libp2p::PeerId>,
131    /// The record expiration time in seconds since the Unix epoch
132    ///
133    /// This is an approximation of the expiration time because we can't
134    /// serialize an `Instant` directly.
135    pub expires_unix_secs: Option<u64>,
136}

138/// Approximate an `Instant` to the number of seconds since the Unix epoch
139fn instant_to_unix_seconds(instant: Instant) -> anyhow::Result<u64> {
140    // Get the current instant and system time
141    let now_instant = Instant::now();
142    let now_system = SystemTime::now();
143
144    // Get the duration of time between the instant and now
145    if instant > now_instant {
146        Ok(now_system
147            .checked_add(instant - now_instant)
148            .with_context(|| "Overflow when approximating expiration time")?
149            .duration_since(UNIX_EPOCH)
150            .with_context(|| "Failed to get duration since Unix epoch")?
151            .as_secs())
152    } else {
153        Ok(now_system
154            .checked_sub(now_instant - instant)
155            .with_context(|| "Underflow when approximating expiration time")?
156            .duration_since(UNIX_EPOCH)
157            .with_context(|| "Failed to get duration since Unix epoch")?
158            .as_secs())
159    }
160}

162/// Convert a unix-second timestamp to an `Instant`
163fn unix_seconds_to_instant(unix_secs: u64) -> anyhow::Result<Instant> {
164    // Get the current instant and unix time
165    let now_instant = Instant::now();
166    let unix_secs_now = SystemTime::now()
167        .duration_since(UNIX_EPOCH)
168        .with_context(|| "Failed to get duration since Unix epoch")?
169        .as_secs();
170
171    if unix_secs > unix_secs_now {
172        // If the instant is in the future, add the duration to the current time
173        now_instant
174            .checked_add(Duration::from_secs(unix_secs - unix_secs_now))
175            .with_context(|| "Overflow when calculating future instant")
176    } else {
177        // If the instant is in the past, subtract the duration from the current time
178        now_instant
179            .checked_sub(Duration::from_secs(unix_secs_now - unix_secs))
180            .with_context(|| "Underflow when calculating past instant")
181    }
182}

184/// Allow conversion from a `libp2p::kad::Record` to a `SerializableRecord`
185impl TryFrom<libp2p::kad::Record> for SerializableRecord {
186    type Error = anyhow::Error;
187
188    fn try_from(record: libp2p::kad::Record) -> anyhow::Result<Self> {
189        Ok(SerializableRecord {
190            key: record.key,
191            value: record.value,
192            publisher: record.publisher,
193            expires_unix_secs: record.expires.map(instant_to_unix_seconds).transpose()?,
194        })
195    }
196}

198/// Allow conversion from a `SerializableRecord` to a `libp2p::kad::Record`
199impl TryFrom<SerializableRecord> for libp2p::kad::Record {
200    type Error = anyhow::Error;
201
202    fn try_from(record: SerializableRecord) -> anyhow::Result<Self> {
203        Ok(libp2p::kad::Record {
204            key: record.key,
205            value: record.value,
206            publisher: record.publisher,
207            expires: record
208                .expires_unix_secs
209                .map(unix_seconds_to_instant)
210                .transpose()?,
211        })
212    }

215impl<R: RecordStore, D: DhtPersistentStorage> PersistentStore<R, D> {
216    /// Create a new `PersistentStore` with the given underlying store and path.
217    /// On creation, the DHT is restored from the persistent storage if possible.
218    ///
219    /// `max_record_delta` is the maximum number of records that can be added to the store before
220    /// the store is saved to the persistent storage.
221    pub async fn new(
222        underlying_record_store: R,
223        persistent_storage: D,
224        max_record_delta: u64,
225    ) -> Self {
226        // Create the new store
227        let mut store = PersistentStore {
228            underlying_record_store,
229            persistent_storage,
230            max_record_delta,
231            record_delta: Arc::new(AtomicU64::new(0)),
232            semaphore: Arc::new(Semaphore::new(1)),
233        };
234
235        // Try to restore the DHT from the persistent store. If it fails, warn and start with an empty store
236        if let Err(err) = store.restore_from_persistent_storage().await {
237            warn!(
238                "Failed to restore DHT from persistent storage: {:?}. Starting with empty store",
239                err
240            );
241        }
242
243        // Return the new store
244        store
245    }
246
247    /// Try saving the DHT to persistent storage if a task is not already in progress.
248    ///
249    /// Returns `true` if the DHT was saved, `false` otherwise.
250    fn try_save_to_persistent_storage(&mut self) -> bool {
251        // Try to acquire the semaphore, warning if another save operation is already in progress
252        let Ok(permit) = Arc::clone(&self.semaphore).try_acquire_owned() else {
253            warn!("Skipping DHT save to persistent storage - another save operation is already in progress");
254            return false;
255        };
256
257        // Get all records and convert them to their serializable counterparts
258        let serializable_records: Vec<_> = self
259            .underlying_record_store
260            .records()
261            .filter_map(|record| {
262                SerializableRecord::try_from(record.into_owned())
263                    .map_err(|err| {
264                        warn!("Failed to convert record to serializable record: {:?}", err);
265                    })
266                    .ok()
267            })
268            .collect();
269
270        // Spawn a task to save the DHT to the persistent storage
271        let persistent_storage = self.persistent_storage.clone();
272        let record_delta = Arc::clone(&self.record_delta);
273        tokio::spawn(async move {
274            debug!("Saving DHT to persistent storage");
275
276            // Save the DHT to the persistent storage
277            match timeout(
278                Duration::from_secs(10),
279                persistent_storage.save(serializable_records),
280            )
281            .await
282            .map_err(|_| anyhow::anyhow!("save operation timed out"))
283            {
284                Ok(Ok(())) => {},
285                Ok(Err(error)) | Err(error) => {
286                    warn!("Failed to save DHT to persistent storage: {error}");
287                },
288            };
289
290            // Reset the record delta
291            record_delta.store(0, Ordering::Release);
292
293            drop(permit);
294
295            debug!("Saved DHT to persistent storage");
296        });
297
298        true
299    }
300
301    /// Attempt to restore the DHT to the underlying store from the persistent storage
302    ///
303    /// # Errors
304    /// - If we fail to load from the persistent storage
305    pub async fn restore_from_persistent_storage(&mut self) -> anyhow::Result<()> {
306        debug!("Restoring DHT from persistent storage");
307
308        // Read the contents of the persistent store
309        let serializable_records = self
310            .persistent_storage
311            .load()
312            .await
313            .with_context(|| "Failed to read DHT from persistent storage")?;
314
315        // Put all records into the new store
316        for serializable_record in serializable_records {
317            // Convert the serializable record back to a `libp2p::kad::Record`
318            match libp2p::kad::Record::try_from(serializable_record) {
319                Ok(record) => {
320                    // Put the record into the new store
321                    if let Err(err) = self.underlying_record_store.put(record) {
322                        warn!(
323                            "Failed to restore record from persistent storage: {:?}",
324                            err
325                        );
326                    }
327                },
328                Err(err) => {
329                    warn!("Failed to parse record from persistent storage: {:?}", err);
330                },
331            };
332        }
333
334        debug!("Restored DHT from persistent storage");
335
336        Ok(())
337    }
338}

340/// Implement the `RecordStore` trait for `PersistentStore`
341impl<R: RecordStore, D: DhtPersistentStorage> RecordStore for PersistentStore<R, D> {
342    type ProvidedIter<'a>
343        = R::ProvidedIter<'a>
344    where
345        R: 'a,
346        D: 'a;
347    type RecordsIter<'a>
348        = R::RecordsIter<'a>
349    where
350        R: 'a,
351        D: 'a;
352    // Delegate all `RecordStore` methods except `put` to the inner store
353    delegate! {
354        to self.underlying_record_store {
355            fn add_provider(&mut self, record: libp2p::kad::ProviderRecord) -> libp2p::kad::store::Result<()>;
356            fn get(&self, k: &libp2p::kad::RecordKey) -> Option<std::borrow::Cow<'_, libp2p::kad::Record>>;
357            fn provided(&self) -> Self::ProvidedIter<'_>;
358            fn providers(&self, key: &libp2p::kad::RecordKey) -> Vec<libp2p::kad::ProviderRecord>;
359            fn records(&self) -> Self::RecordsIter<'_>;
360            fn remove_provider(&mut self, k: &libp2p::kad::RecordKey, p: &libp2p::PeerId);
361        }
362    }
363
364    /// Overwrite the `put` method to potentially sync the DHT to the persistent store
365    fn put(&mut self, record: libp2p::kad::Record) -> Result<()> {
366        // Try to write to the underlying store
367        let result = self.underlying_record_store.put(record);
368
369        // If the record was successfully written,
370        if result.is_ok() {
371            // Update the record delta
372            self.record_delta.fetch_add(1, Ordering::Relaxed);
373
374            // Check if it's above the maximum record delta
375            if self.record_delta.load(Ordering::Relaxed) > self.max_record_delta {
376                // Try to save the DHT to persistent storage
377                self.try_save_to_persistent_storage();
378            }
379        }
380
381        result
382    }
383
384    /// Overwrite the `remove` method to potentially sync the DHT to the persistent store
385    fn remove(&mut self, k: &libp2p::kad::RecordKey) {
386        // Remove the record from the underlying store
387        self.underlying_record_store.remove(k);
388
389        // Update the record delta
390        self.record_delta.fetch_add(1, Ordering::Relaxed);
391
392        // Check if it's above the maximum record delta
393        if self.record_delta.load(Ordering::Relaxed) > self.max_record_delta {
394            // Try to save the DHT to persistent storage
395            self.try_save_to_persistent_storage();
396        }
397    }
398}

415            .try_init();
416
417        // Create a test store

533    #[test]
534    fn test_approximate_instant() {
535        // Create an expiry time in the future
536        let expiry_future = Instant::now() + Duration::from_secs(10);
537
538        // Approximate the expiry time
539        let approximate_expiry =
540            unix_seconds_to_instant(instant_to_unix_seconds(expiry_future).unwrap())
541                .unwrap()
542                .duration_since(Instant::now());
543
544        // Make sure it's close to 10 seconds in the future
545        assert!(approximate_expiry >= Duration::from_secs(9));
546        assert!(approximate_expiry <= Duration::from_secs(11));
547
548        // Create an expiry time in the past
549        let expiry_past = Instant::now().checked_sub(Duration::from_secs(10)).unwrap();
550
551        // Approximate the expiry time
552        let approximate_expiry =
553            unix_seconds_to_instant(instant_to_unix_seconds(expiry_past).unwrap()).unwrap();
554        let time_difference = approximate_expiry.elapsed();
555
556        // Make sure it's close to 10 seconds in the past
557        assert!(time_difference >= Duration::from_secs(9));
558        assert!(time_difference <= Duration::from_secs(11));
559    }
