115    /// starts the swarm listening on `listen_addr`
116    /// and optionally dials into peer `known_peer`
117    /// returns the address the swarm is listening upon
118    #[instrument(skip(self))]
119    pub async fn start_listen(
120        &mut self,
121        listen_addr: Multiaddr,
122    ) -> Result<Multiaddr, NetworkError> {
123        self.listener_id = Some(self.swarm.listen_on(listen_addr).map_err(|err| {
124            NetworkError::ListenError(format!("failed to listen for Libp2p: {err}"))
125        })?);
126        let addr = loop {
127            if let Some(SwarmEvent::NewListenAddr { address, .. }) = self.swarm.next().await {
128                break address;
129            }
130        };
131        info!("Libp2p listening on {:?}", addr);
132        Ok(addr)
133    }

135    /// initialize the DHT with known peers
136    /// add the peers to kademlia and then
137    /// the `spawn_listeners` function
138    /// will start connecting to peers
139    #[instrument(skip(self))]
140    pub fn add_known_peers(&mut self, known_peers: &[(PeerId, Multiaddr)]) {
141        debug!("Adding {} known peers", known_peers.len());
142        let behaviour = self.swarm.behaviour_mut();
143        let mut bs_nodes = HashMap::<PeerId, HashSet<Multiaddr>>::new();
144        let mut shuffled = known_peers.iter().collect::<Vec<_>>();
145        shuffled.shuffle(&mut thread_rng());
146        for (peer_id, addr) in shuffled {
147            if *peer_id != self.peer_id {
148                behaviour.dht.add_address(peer_id, addr.clone());
149                behaviour.autonat.add_server(*peer_id, Some(addr.clone()));
150                bs_nodes.insert(*peer_id, iter::once(addr.clone()).collect());
151            }
152        }
153    }

155    /// Creates a new `Network` with the given settings.
156    ///
157    /// Currently:
158    ///   * Generates a random key pair and associated [`PeerId`]
159    ///   * Launches a hopefully production ready transport:
160    ///       QUIC v1 (RFC 9000) + DNS
161    ///   * Generates a connection to the "broadcast" topic
162    ///   * Creates a swarm to manage peers and events
163    ///
164    /// # Errors
165    /// - If we fail to generate the transport or any of the behaviours
166    ///
167    /// # Panics
168    /// If 5 < 0
169    #[allow(clippy::too_many_lines)]
170    pub async fn new(
171        config: NetworkNodeConfig<T>,
172        dht_persistent_storage: D,
173    ) -> Result<Self, NetworkError> {
174        // Generate a random `KeyPair` if one is not specified
175        let keypair = config
176            .keypair
177            .clone()
178            .unwrap_or_else(Keypair::generate_ed25519);
179
180        // Get the `PeerId` from the `KeyPair`
181        let peer_id = PeerId::from(keypair.public());
182
183        // Generate the transport from the keypair, membership, and auth message
184        let transport: BoxedTransport = gen_transport::<T>(
185            keypair.clone(),
186            config.membership.clone(),
187            config.auth_message.clone(),
188        )
189        .await?;
190
191        // Generate the swarm
192        let mut swarm: Swarm<NetworkDef<T::SignatureKey, D>> = {
193            // Use the `Blake3` hash of the message's contents as the ID
194            let message_id_fn = |message: &GossipsubMessage| {
195                let hash = blake3::hash(&message.data);
196                MessageId::from(hash.as_bytes().to_vec())
197            };
198
199            // Derive a `Gossipsub` config from our gossip config
200            let gossipsub_config = GossipsubConfigBuilder::default()
201                .message_id_fn(message_id_fn) // Use the (blake3) hash of a message as its ID
202                .validation_mode(ValidationMode::Strict) // Force all messages to have valid signatures
203                .heartbeat_interval(config.gossip_config.heartbeat_interval) // Time between gossip heartbeats
204                .history_gossip(config.gossip_config.history_gossip) // Number of heartbeats to gossip about
205                .history_length(config.gossip_config.history_length) // Number of heartbeats to remember the full message for
206                .mesh_n(config.gossip_config.mesh_n) // Target number of mesh peers
207                .mesh_n_high(config.gossip_config.mesh_n_high) // Upper limit of mesh peers
208                .mesh_n_low(config.gossip_config.mesh_n_low) // Lower limit of mesh peers
209                .mesh_outbound_min(config.gossip_config.mesh_outbound_min) // Minimum number of outbound peers in mesh
210                .max_transmit_size(config.gossip_config.max_transmit_size) // Maximum size of a message
211                .max_ihave_length(config.gossip_config.max_ihave_length) // Maximum number of messages to include in an IHAVE message
212                .max_ihave_messages(config.gossip_config.max_ihave_messages) // Maximum number of IHAVE messages to accept from a peer within a heartbeat
213                .published_message_ids_cache_time(
214                    config.gossip_config.published_message_ids_cache_time,
215                ) // Cache duration for published message IDs
216                .iwant_followup_time(config.gossip_config.iwant_followup_time) // Time to wait for a message requested through IWANT following an IHAVE advertisement
217                .max_messages_per_rpc(config.gossip_config.max_messages_per_rpc) // The maximum number of messages we will process in a given RPC
218                .gossip_retransimission(config.gossip_config.gossip_retransmission) // Controls how many times we will allow a peer to request the same message id through IWANT gossip before we start ignoring them.
219                .flood_publish(config.gossip_config.flood_publish) // If enabled newly created messages will always be sent to all peers that are subscribed to the topic and have a good enough score.
220                .duplicate_cache_time(config.gossip_config.duplicate_cache_time) // The time period that messages are stored in the cache
221                .fanout_ttl(config.gossip_config.fanout_ttl) // Time to live for fanout peers
222                .heartbeat_initial_delay(config.gossip_config.heartbeat_initial_delay) // Initial delay in each heartbeat
223                .gossip_factor(config.gossip_config.gossip_factor) // Affects how many peers we will emit gossip to at each heartbeat
224                .gossip_lazy(config.gossip_config.gossip_lazy) // Minimum number of peers to emit gossip to during a heartbeat
225                .build()
226                .map_err(|err| {
227                    NetworkError::ConfigError(format!("error building gossipsub config: {err:?}"))
228                })?;
229
230            // - Build a gossipsub network behavior
231            let gossipsub: Gossipsub = Gossipsub::new(
232                MessageAuthenticity::Signed(keypair.clone()),
233                gossipsub_config,
234            )
235            .map_err(|err| {
236                NetworkError::ConfigError(format!("error building gossipsub behaviour: {err:?}"))
237            })?;
238
239            //   Build a identify network behavior needed for own
240            //   node connection information
241            //   E.g. this will answer the question: how are other nodes
242            //   seeing the peer from behind a NAT
243            let identify_cfg =
244                IdentifyConfig::new("HotShot/identify/1.0".to_string(), keypair.public());
245            let identify = IdentifyBehaviour::new(identify_cfg);
246
247            // - Build DHT needed for peer discovery
248            let mut kconfig = Config::new(StreamProtocol::new("/ipfs/kad/1.0.0"));
249            // 8 hours by default
250            let record_republication_interval = config
251                .republication_interval
252                .unwrap_or(Duration::from_secs(KAD_DEFAULT_REPUB_INTERVAL_SEC));
253            let ttl = Some(config.ttl.unwrap_or(16 * record_republication_interval));
254            kconfig
255                .set_parallelism(NonZeroUsize::new(5).unwrap())
256                .set_provider_publication_interval(Some(record_republication_interval))
257                .set_publication_interval(Some(record_republication_interval))
258                .set_record_ttl(ttl);
259
260            // allowing panic here because something is very wrong if this fales
261            #[allow(clippy::panic)]
262            if let Some(factor) = config.replication_factor {
263                kconfig.set_replication_factor(factor);
264            } else {
265                panic!("Replication factor not set");
266            }
267
268            // Create the DHT behaviour with the given persistent storage
269            let mut kadem = Behaviour::with_config(
270                peer_id,
271                PersistentStore::new(
272                    ValidatedStore::new(MemoryStore::new(peer_id)),
273                    dht_persistent_storage,
274                    5,
275                )
276                .await,
277                kconfig,
278            );
279            kadem.set_mode(Some(Mode::Server));
280
281            let rrconfig = Libp2pRequestResponseConfig::default();
282
283            // Create a new `cbor` codec with the given request and response sizes
284            let cbor = Cbor::new(
285                config.request_response_config.request_size_maximum,
286                config.request_response_config.response_size_maximum,
287            );
288
289            let direct_message: super::cbor::Behaviour<Vec<u8>, Vec<u8>> =
290                RequestResponse::with_codec(
291                    cbor,
292                    [(
293                        StreamProtocol::new("/HotShot/direct_message/1.0"),
294                        ProtocolSupport::Full,
295                    )]
296                    .into_iter(),
297                    rrconfig.clone(),
298                );
299
300            let autonat_config = autonat::Config {
301                only_global_ips: false,
302                ..Default::default()
303            };
304
305            let network = NetworkDef::new(
306                gossipsub,
307                kadem,
308                identify,
309                direct_message,
310                autonat::Behaviour::new(peer_id, autonat_config),
311            );
312
313            // build swarm
314            let swarm = SwarmBuilder::with_existing_identity(keypair.clone());
315            let swarm = swarm.with_tokio();
316
317            swarm
318                .with_other_transport(|_| transport)
319                .unwrap()
320                .with_behaviour(|_| network)
321                .unwrap()
322                .build()
323        };
324        for (peer, addr) in &config.to_connect_addrs {
325            if peer != swarm.local_peer_id() {
326                swarm.behaviour_mut().add_address(peer, addr.clone());
327            }
328        }
329
330        Ok(Self {
331            peer_id,
332            swarm,
333            listener_id: None,
334            direct_message_state: DMBehaviour::default(),
335            dht_handler: DHTBehaviour::new(
336                peer_id,
337                config
338                    .replication_factor
339                    .unwrap_or(NonZeroUsize::new(4).unwrap()),
340            ),
341            resend_tx: None,
342        })
343    }

345    /// Publish a key/value to the record store.
346    ///
347    /// # Panics
348    /// If the default replication factor is `None`
349    pub fn put_record(&mut self, mut query: KadPutQuery) {
350        let record = Record::new(query.key.clone(), query.value.clone());
351        match self.swarm.behaviour_mut().dht.put_record(
352            record,
353            libp2p::kad::Quorum::N(
354                NonZeroUsize::try_from(self.dht_handler.replication_factor().get() / 2)
355                    .expect("replication factor should be bigger than 0"),
356            ),
357        ) {
358            Err(e) => {
359                // failed try again later
360                query.progress = DHTProgress::NotStarted;
361                query.backoff.start_next(false);
362                error!("Error publishing to DHT: {e:?} for peer {:?}", self.peer_id);
363            },
364            Ok(qid) => {
365                debug!("Published record to DHT with qid {:?}", qid);
366                let query = KadPutQuery {
367                    progress: DHTProgress::InProgress(qid),
368                    ..query
369                };
370                self.dht_handler.put_record(qid, query);
371            },
372        }
373    }

375    /// event handler for client events
376    /// currently supported actions include
377    /// - shutting down the swarm
378    /// - gossipping a message to known peers on the `global` topic
379    /// - returning the id of the current peer
380    /// - subscribing to a topic
381    /// - unsubscribing from a toipc
382    /// - direct messaging a peer
383    #[instrument(skip(self))]
384    async fn handle_client_requests(
385        &mut self,
386        msg: Option<ClientRequest>,
387    ) -> Result<bool, NetworkError> {
388        let behaviour = self.swarm.behaviour_mut();
389        match msg {
390            Some(msg) => {
391                match msg {
392                    ClientRequest::BeginBootstrap => {
393                        debug!("Beginning Libp2p bootstrap");
394                        let _ = self.swarm.behaviour_mut().dht.bootstrap();
395                    },
396                    ClientRequest::LookupPeer(pid, chan) => {
397                        let id = self.swarm.behaviour_mut().dht.get_closest_peers(pid);
398                        self.dht_handler
399                            .in_progress_get_closest_peers
400                            .insert(id, chan);
401                    },
402                    ClientRequest::GetRoutingTable(chan) => {
403                        self.dht_handler
404                            .print_routing_table(&mut self.swarm.behaviour_mut().dht);
405                        if chan.send(()).is_err() {
406                            warn!("Tried to notify client but client not tracking anymore");
407                        }
408                    },
409                    ClientRequest::PutDHT { key, value, notify } => {
410                        let query = KadPutQuery {
411                            progress: DHTProgress::NotStarted,
412                            notify,
413                            key,
414                            value,
415                            backoff: ExponentialBackoff::default(),
416                        };
417                        self.put_record(query);
418                    },
419                    ClientRequest::GetConnectedPeerNum(s) => {
420                        if s.send(self.num_connected()).is_err() {
421                            error!("error sending peer number to client");
422                        }
423                    },
424                    ClientRequest::GetConnectedPeers(s) => {
425                        if s.send(self.connected_pids()).is_err() {
426                            error!("error sending peer set to client");
427                        }
428                    },
429                    ClientRequest::GetDHT {
430                        key,
431                        notify,
432                        retry_count,
433                    } => {
434                        self.dht_handler.get_record(
435                            key,
436                            notify,
437                            NonZeroUsize::new(NUM_REPLICATED_TO_TRUST).unwrap(),
438                            ExponentialBackoff::default(),
439                            retry_count,
440                            &mut self.swarm.behaviour_mut().dht,
441                        );
442                    },
443                    ClientRequest::IgnorePeers(_peers) => {
444                        // NOTE used by test with conductor only
445                    },
446                    ClientRequest::Shutdown => {
447                        if let Some(listener_id) = self.listener_id {
448                            self.swarm.remove_listener(listener_id);
449                        }
450
451                        return Ok(true);
452                    },
453                    ClientRequest::GossipMsg(topic, contents) => {
454                        behaviour.publish_gossip(Topic::new(topic.clone()), contents.clone());
455                    },
456                    ClientRequest::Subscribe(t, chan) => {
457                        behaviour.subscribe_gossip(&t);
458                        if let Some(chan) = chan {
459                            if chan.send(()).is_err() {
460                                error!("finished subscribing but response channel dropped");
461                            }
462                        }
463                    },
464                    ClientRequest::Unsubscribe(t, chan) => {
465                        behaviour.unsubscribe_gossip(&t);
466                        if let Some(chan) = chan {
467                            if chan.send(()).is_err() {
468                                error!("finished unsubscribing but response channel dropped");
469                            }
470                        }
471                    },
472                    ClientRequest::DirectRequest {
473                        pid,
474                        contents,
475                        retry_count,
476                    } => {
477                        debug!("Sending direct request to {:?}", pid);
478                        let id = behaviour.add_direct_request(pid, contents.clone());
479                        let req = DMRequest {
480                            peer_id: pid,
481                            data: contents,
482                            backoff: ExponentialBackoff::default(),
483                            retry_count,
484                        };
485                        self.direct_message_state.add_direct_request(req, id);
486                    },
487                    ClientRequest::DirectResponse(chan, msg) => {
488                        behaviour.add_direct_response(chan, msg);
489                    },
490                    ClientRequest::AddKnownPeers(peers) => {
491                        self.add_known_peers(&peers);
492                    },
493                    ClientRequest::Prune(pid) => {
494                        if self.swarm.disconnect_peer_id(pid).is_err() {
495                            warn!("Could not disconnect from {:?}", pid);
496                        }
497                    },
498                }
499            },
500            None => {
501                error!("Error receiving msg in main behaviour loop: channel closed");
502            },
503        }
504        Ok(false)
505    }

507    /// event handler for events emitted from the swarm
508    #[allow(clippy::type_complexity)]
509    #[instrument(skip(self))]
510    async fn handle_swarm_events(
511        &mut self,
512        event: SwarmEvent<NetworkEventInternal>,
513        send_to_client: &UnboundedSender<NetworkEvent>,
514    ) -> Result<(), NetworkError> {
515        // Make the match cleaner
516        debug!("Swarm event observed {:?}", event);
517
518        #[allow(deprecated)]
519        match event {
520            SwarmEvent::ConnectionEstablished {
521                connection_id: _,
522                peer_id,
523                endpoint,
524                num_established,
525                concurrent_dial_errors,
526                established_in: _established_in,
527            } => {
528                if num_established > ESTABLISHED_LIMIT {
529                    error!(
530                        "Num concurrent connections to a single peer exceeding {:?} at {:?}!",
531                        ESTABLISHED_LIMIT, num_established
532                    );
533                } else {
534                    debug!(
535                        "Connection established with {:?} at {:?} with {:?} concurrent dial errors",
536                        peer_id, endpoint, concurrent_dial_errors
537                    );
538                }
539
540                // Send the number of connected peers to the client
541                send_to_client
542                    .send(NetworkEvent::ConnectedPeersUpdate(self.num_connected()))
543                    .map_err(|err| NetworkError::ChannelSendError(err.to_string()))?;
544            },
545            SwarmEvent::ConnectionClosed {
546                connection_id: _,
547                peer_id,
548                endpoint,
549                num_established,
550                cause,
551            } => {
552                if num_established > ESTABLISHED_LIMIT_UNWR {
553                    error!(
554                        "Num concurrent connections to a single peer exceeding {:?} at {:?}!",
555                        ESTABLISHED_LIMIT, num_established
556                    );
557                } else {
558                    debug!(
559                        "Connection closed with {:?} at {:?} due to {:?}",
560                        peer_id, endpoint, cause
561                    );
562                }
563
564                // Send the number of connected peers to the client
565                send_to_client
566                    .send(NetworkEvent::ConnectedPeersUpdate(self.num_connected()))
567                    .map_err(|err| NetworkError::ChannelSendError(err.to_string()))?;
568            },
569            SwarmEvent::Dialing {
570                peer_id,
571                connection_id: _,
572            } => {
573                debug!("Attempting to dial {:?}", peer_id);
574            },
575            SwarmEvent::ListenerClosed {
576                listener_id: _,
577                addresses: _,
578                reason: _,
579            }
580            | SwarmEvent::NewListenAddr {
581                listener_id: _,
582                address: _,
583            }
584            | SwarmEvent::ExpiredListenAddr {
585                listener_id: _,
586                address: _,
587            }
588            | SwarmEvent::NewExternalAddrCandidate { .. }
589            | SwarmEvent::ExternalAddrExpired { .. }
590            | SwarmEvent::IncomingConnection {
591                connection_id: _,
592                local_addr: _,
593                send_back_addr: _,
594            } => {},
595            SwarmEvent::Behaviour(b) => {
596                let maybe_event = match b {
597                    NetworkEventInternal::DHTEvent(e) => self
598                        .dht_handler
599                        .dht_handle_event(e, self.swarm.behaviour_mut().dht.store_mut()),
600                    NetworkEventInternal::IdentifyEvent(e) => {
601                        // NOTE feed identified peers into kademlia's routing table for peer discovery.
602                        if let IdentifyEvent::Received {
603                            peer_id,
604                            info:
605                                IdentifyInfo {
606                                    listen_addrs,
607                                    protocols: _,
608                                    public_key: _,
609                                    protocol_version: _,
610                                    agent_version: _,
611                                    observed_addr: _,
612                                },
613                            connection_id: _,
614                        } = *e
615                        {
616                            let behaviour = self.swarm.behaviour_mut();
617
618                            // into hashset to delete duplicates (I checked: there are duplicates)
619                            for addr in listen_addrs.iter().collect::<HashSet<_>>() {
620                                behaviour.dht.add_address(&peer_id, addr.clone());
621                            }
622                        }
623                        None
624                    },
625                    NetworkEventInternal::GossipEvent(e) => match *e {
626                        GossipEvent::Message {
627                            propagation_source: _peer_id,
628                            message_id: _id,
629                            message,
630                        } => Some(NetworkEvent::GossipMsg(message.data)),
631                        GossipEvent::Subscribed { peer_id, topic } => {
632                            debug!("Peer {:?} subscribed to topic {:?}", peer_id, topic);
633                            None
634                        },
635                        GossipEvent::Unsubscribed { peer_id, topic } => {
636                            debug!("Peer {:?} unsubscribed from topic {:?}", peer_id, topic);
637                            None
638                        },
639                        GossipEvent::GossipsubNotSupported { peer_id } => {
640                            warn!("Peer {:?} does not support gossipsub", peer_id);
641                            None
642                        },
643                    },
644                    NetworkEventInternal::DMEvent(e) => self
645                        .direct_message_state
646                        .handle_dm_event(e, self.resend_tx.clone()),
647                    NetworkEventInternal::AutonatEvent(e) => {
648                        match e {
649                            autonat::Event::InboundProbe(_) => {},
650                            autonat::Event::OutboundProbe(e) => match e {
651                                autonat::OutboundProbeEvent::Request { .. }
652                                | autonat::OutboundProbeEvent::Response { .. } => {},
653                                autonat::OutboundProbeEvent::Error {
654                                    probe_id: _,
655                                    peer,
656                                    error,
657                                } => {
658                                    warn!(
659                                        "AutoNAT Probe failed to peer {:?} with error: {:?}",
660                                        peer, error
661                                    );
662                                },
663                            },
664                            autonat::Event::StatusChanged { old, new } => {
665                                debug!("AutoNAT Status changed. Old: {:?}, New: {:?}", old, new);
666                            },
667                        };
668                        None
669                    },
670                };
671
672                if let Some(event) = maybe_event {
673                    // forward messages directly to Client
674                    send_to_client
675                        .send(event)
676                        .map_err(|err| NetworkError::ChannelSendError(err.to_string()))?;
677                }
678            },
679            SwarmEvent::OutgoingConnectionError {
680                connection_id: _,
681                peer_id,
682                error,
683            } => {
684                warn!("Outgoing connection error to {:?}: {:?}", peer_id, error);
685            },
686            SwarmEvent::IncomingConnectionError {
687                connection_id: _,
688                local_addr: _,
689                send_back_addr: _,
690                error,
691            } => {
692                warn!("Incoming connection error: {:?}", error);
693            },
694            SwarmEvent::ListenerError {
695                listener_id: _,
696                error,
697            } => {
698                warn!("Listener error: {:?}", error);
699            },
700            SwarmEvent::ExternalAddrConfirmed { address } => {
701                let my_id = *self.swarm.local_peer_id();
702                self.swarm
703                    .behaviour_mut()
704                    .dht
705                    .add_address(&my_id, address.clone());
706            },
707            SwarmEvent::NewExternalAddrOfPeer { peer_id, address } => {
708                self.swarm
709                    .behaviour_mut()
710                    .dht
711                    .add_address(&peer_id, address.clone());
712            },
713            _ => {
714                debug!("Unhandled swarm event {:?}", event);
715            },
716        }
717        Ok(())
718    }

720    /// Spawn a task to listen for requests on the returned channel
721    /// as well as any events produced by libp2p
722    ///
723    /// # Errors
724    /// - If we fail to create the channels or the bootstrap channel
725    pub fn spawn_listeners(
726        mut self,
727    ) -> Result<
728        (
729            UnboundedSender<ClientRequest>,
730            UnboundedReceiver<NetworkEvent>,
731        ),
732        NetworkError,
733    > {
734        let (s_input, mut s_output) = unbounded_channel::<ClientRequest>();
735        let (r_input, r_output) = unbounded_channel::<NetworkEvent>();
736        let (mut bootstrap_tx, bootstrap_rx) = mpsc::channel(100);
737        self.resend_tx = Some(s_input.clone());
738        self.dht_handler.set_bootstrap_sender(bootstrap_tx.clone());
739
740        DHTBootstrapTask::run(bootstrap_rx, s_input.clone());
741        spawn(
742            async move {
743                loop {
744                    select! {
745                        event = self.swarm.next() => {
746                            debug!("peerid {:?}\t\thandling maybe event {:?}", self.peer_id, event);
747                            if let Some(event) = event {
748                                debug!("peerid {:?}\t\thandling event {:?}", self.peer_id, event);
749                                self.handle_swarm_events(event, &r_input).await?;
750                            }
751                        },
752                        msg = s_output.recv() => {
753                            debug!("peerid {:?}\t\thandling msg {:?}", self.peer_id, msg);
754                            let shutdown = self.handle_client_requests(msg).await?;
755                            if shutdown {
756                                let _ = bootstrap_tx.send(InputEvent::ShutdownBootstrap).await;
757                                break
758                            }
759                        }
760                    }
761                }
762                Ok::<(), NetworkError>(())
763            }
764            .instrument(info_span!("Libp2p NetworkBehaviour Handler")),
765        );
766        Ok((s_input, r_output))
767    }
