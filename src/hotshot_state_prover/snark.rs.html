28/// Given a SRS, returns the proving key and verifying key for state update
29/// # Errors
30/// Errors if unable to preprocess
31#[allow(clippy::cast_possible_truncation)]
32pub fn preprocess(
33    srs: &UniversalSrs,
34    stake_table_capacity: usize,
35) -> Result<(ProvingKey, VerifyingKey), PlonkError> {
36    let (circuit, _) = crate::circuit::build_for_preprocessing::<CircuitField, EdwardsConfig>(
37        stake_table_capacity,
38    )?;
39    PlonkKzgSnark::preprocess(srs, &circuit)
40}

42/// Given a proving key and
43/// - a list of stake table entries (`Vec<(BLSVerKey, Amount, SchnorrVerKey)>`)
44/// - a list of schnorr signatures of the updated states (`Vec<SchnorrSignature>`), default if the node doesn't sign the state
45/// - updated light client state (`(view_number, block_height, block_comm_root)`)
46/// - the static stake table state (containing 3 commitments to the 3 columns of the stake table and a threshold)
47/// - a bit vector indicates the signers
48///
49/// Returns error or a pair `(proof, public_inputs)` asserting that
50/// - the signer's accumulated weight exceeds the quorum threshold
51/// - the stake table corresponds to the one committed in the light client state
52/// - all schnorr signatures over the light client state are valid
53///
54/// # Errors
55/// Errors if unable to generate proof
56///
57/// # Panics
58/// if the stake table is not up to date
59#[allow(clippy::too_many_arguments)]
60pub fn generate_state_update_proof<STIter, R, BitIter, SigIter>(
61    rng: &mut R,
62    pk: &ProvingKey,
63    stake_table_entries: STIter,
64    signer_bit_vec: BitIter,
65    signatures: SigIter,
66    lightclient_state: &LightClientState,
67    stake_table_state: &StakeTableState,
68    stake_table_capacity: usize,
69) -> Result<(Proof, PublicInput), PlonkError>
70where
71    STIter: IntoIterator,
72    STIter::Item: Borrow<(StateVerKey, U256)>,
73    STIter::IntoIter: ExactSizeIterator,
74    R: CryptoRng + RngCore,
75    BitIter: IntoIterator,
76    BitIter::Item: Borrow<bool>,
77    BitIter::IntoIter: ExactSizeIterator,
78    SigIter: IntoIterator,
79    SigIter::Item: Borrow<Signature<EdwardsConfig>>,
80    SigIter::IntoIter: ExactSizeIterator,
81{
82    let signer_bit_vec = signer_bit_vec.into_iter().map(|b| {
83        if *b.borrow() {
84            CircuitField::from(1u64)
85        } else {
86            CircuitField::from(0u64)
87        }
88    });
89
90    let (circuit, public_inputs) = crate::circuit::build(
91        stake_table_entries,
92        signer_bit_vec,
93        signatures,
94        lightclient_state,
95        stake_table_state,
96        stake_table_capacity,
97    )?;
98    let proof = PlonkKzgSnark::<Bn254>::prove::<_, _, SolidityTranscript>(rng, &circuit, pk, None)?;
99    Ok((proof, public_inputs))
100}

136    // FIXME(Chengyu): see <https://github.com/EspressoSystems/jellyfish/issues/249>
137    #[allow(clippy::unnecessary_wraps)]
138    fn universal_setup_for_testing<R>(
139        max_degree: usize,
140        rng: &mut R,
141    ) -> anyhow::Result<UniversalSrs>
142    where
143        R: RngCore + CryptoRng,
144    {
145        use ark_ec::{scalar_mul::fixed_base::FixedBase, CurveGroup};
146        use ark_ff::PrimeField;
147        use ark_std::{end_timer, start_timer, UniformRand};
148
149        let setup_time = start_timer!(|| format!("KZG10::Setup with degree {}", max_degree));
150        let beta = <Bn254 as Pairing>::ScalarField::rand(rng);
151        let g = <Bn254 as Pairing>::G1::rand(rng);
152        let h = <Bn254 as Pairing>::G2::rand(rng);
153
154        let mut powers_of_beta = vec![<Bn254 as Pairing>::ScalarField::one()];
155
156        let mut cur = beta;
157        for _ in 0..max_degree {
158            powers_of_beta.push(cur);
159            cur *= &beta;
160        }
161
162        let window_size = FixedBase::get_mul_window_size(max_degree + 1);
163
164        let scalar_bits = <Bn254 as Pairing>::ScalarField::MODULUS_BIT_SIZE as usize;
165        let g_time = start_timer!(|| "Generating powers of G");
166        // TODO: parallelization
167        let g_table = FixedBase::get_window_table(scalar_bits, window_size, g);
168        let powers_of_g = FixedBase::msm::<<Bn254 as Pairing>::G1>(
169            scalar_bits,
170            window_size,
171            &g_table,
172            &powers_of_beta,
173        );
174        end_timer!(g_time);
175
176        let powers_of_g = <Bn254 as Pairing>::G1::normalize_batch(&powers_of_g);
177
178        let h = h.into_affine();
179        let beta_h = (h * beta).into_affine();
180
181        let pp = UniversalSrs {
182            powers_of_g,
183            h,
184            beta_h,
185            powers_of_h: vec![h, beta_h],
186        };
187        end_timer!(setup_time);
188        Ok(pp)
189    }

191    #[test]
192    fn test_proof_generation() {
193        let num_validators = 10;
194        let mut prng = test_rng();
195
196        let (bls_keys, schnorr_keys) = key_pairs_for_testing(num_validators, &mut prng);
197        let st = stake_table_for_testing(ST_CAPACITY, &bls_keys, &schnorr_keys);
198        let st_state = genesis_stake_table_state(&st);
199
200        let stake_table_entries = st
201            .try_iter(SnapshotVersion::LastEpochStart)
202            .unwrap()
203            .map(|(_, stake_amount, schnorr_key)| (schnorr_key, stake_amount))
204            .collect::<Vec<_>>();
205
206        let block_comm_root = VariableLengthRescueCRHF::<CircuitField, 1>::evaluate(vec![
207            CircuitField::from(1u32),
208            CircuitField::from(2u32),
209        ])
210        .unwrap()[0];
211
212        let lightclient_state = LightClientState {
213            view_number: 100,
214            block_height: 73,
215            block_comm_root,
216        };
217        let state_msg: [CircuitField; 3] = lightclient_state.clone().into();
218
219        let sigs = schnorr_keys
220            .iter()
221            .map(|(key, _)| SchnorrSignatureScheme::<Config>::sign(&(), key, state_msg, &mut prng))
222            .collect::<Result<Vec<_>, _>>()
223            .unwrap();
224
225        // bit vector with total weight 26
226        let bit_vec = [
227            true, true, true, false, true, true, false, false, true, false,
228        ];
229        let bit_masked_sigs = bit_vec
230            .iter()
231            .zip(sigs.iter())
232            .map(|(bit, sig)| {
233                if *bit {
234                    sig.clone()
235                } else {
236                    Signature::<Config>::default()
237                }
238            })
239            .collect::<Vec<_>>();
240
241        // good path
242        let num_gates =
243            build_for_preprocessing::<CircuitField, ark_ed_on_bn254::EdwardsConfig>(ST_CAPACITY)
244                .unwrap()
245                .0
246                .num_gates();
247        let test_srs = universal_setup_for_testing(num_gates + 2, &mut prng).unwrap();
248        ark_std::println!("Number of constraint in the circuit: {num_gates}");
249
250        let result = preprocess(&test_srs, ST_CAPACITY);
251        assert!(result.is_ok());
252        let (pk, vk) = result.unwrap();
253
254        let result = generate_state_update_proof::<_, _, _, _>(
255            &mut prng,
256            &pk,
257            &stake_table_entries,
258            &bit_vec,
259            &bit_masked_sigs,
260            &lightclient_state,
261            &st_state,
262            ST_CAPACITY,
263        );
264        assert!(result.is_ok());
265
266        let (proof, public_inputs) = result.unwrap();
267        assert!(PlonkKzgSnark::<Bn254>::verify::<SolidityTranscript>(
268            &vk,
269            public_inputs.as_ref(),
270            &proof,
271            None
272        )
273        .is_ok());
274
275        // minimum bad path, other bad cases are checked inside `circuit.rs`
276        let mut bad_st_state = st_state;
277        bad_st_state.threshold = CircuitField::from(100u32);
278        let result = generate_state_update_proof::<_, _, _, _>(
279            &mut prng,
280            &pk,
281            &stake_table_entries,
282            &bit_vec,
283            &bit_masked_sigs,
284            &lightclient_state,
285            &bad_st_state,
286            ST_CAPACITY,
287        );
288        assert!(result.is_err());
289    }
