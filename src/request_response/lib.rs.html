88/// The underlying configuration for the request-response protocol
89#[derive(Clone)]
90pub struct RequestResponseConfig {
91    /// The timeout for incoming requests. Do not respond to a request after this threshold
92    /// has passed.
93    pub incoming_request_ttl: Duration,
94    /// The maximum amount of time we will spend trying to both derive a response for a request and
95    /// send the response over the wire.
96    pub response_send_timeout: Duration,
97    /// The maximum amount of time we will spend trying to validate a response. This is used to prevent
98    /// an attack where a malicious participant sends us a bunch of requests that take a long time to
99    /// validate.
100    pub response_validate_timeout: Duration,
101    /// The batch size for outgoing requests. This is the number of request messages that we will
102    /// send out at a time for a single request before waiting for the [`request_batch_interval`].
103    pub request_batch_size: usize,
104    /// The time to wait (per request) between sending out batches of request messages
105    pub request_batch_interval: Duration,
106    /// The maximum (global) number of outgoing responses that can be in flight at any given time
107    pub max_outgoing_responses: usize,
108    /// The maximum (global) number of incoming responses that can be processed at any given time.
109    /// We need this because responses coming in need to be validated [asynchronously] that they
110    /// satisfy the request they are responding to
111    pub max_incoming_responses: usize,
112}

114/// A protocol that allows for request-response communication. Is cheaply cloneable, so there is no
115/// need to wrap it in an `Arc`
116#[derive(Deref)]
117pub struct RequestResponse<
118    S: Sender<K>,
119    R: Receiver,
120    Req: Request,
121    RS: RecipientSource<Req, K>,
122    DS: DataSource<Req>,
123    K: SignatureKey + 'static,
124> {
125    #[deref]
126    /// The inner implementation of the request-response protocol
127    inner: Arc<RequestResponseInner<S, R, Req, RS, DS, K>>,
128    /// A handle to the receiving task. This will automatically get cancelled when the protocol is dropped
129    _receiving_task_handle: Arc<AbortOnDropHandle<()>>,
130}
131
132/// We need to manually implement the `Clone` trait for this type because deriving
133/// `Deref` will cause an issue where it tries to clone the inner field instead
134impl<
135        S: Sender<K>,
136        R: Receiver,
137        Req: Request,
138        RS: RecipientSource<Req, K>,
139        DS: DataSource<Req>,
140        K: SignatureKey + 'static,
141    > Clone for RequestResponse<S, R, Req, RS, DS, K>
142{
143    fn clone(&self) -> Self {
144        Self {
145            inner: Arc::clone(&self.inner),
146            _receiving_task_handle: Arc::clone(&self._receiving_task_handle),
147        }
148    }
149}

151impl<
152        S: Sender<K>,
153        R: Receiver,
154        Req: Request,
155        RS: RecipientSource<Req, K>,
156        DS: DataSource<Req>,
157        K: SignatureKey + 'static,
158    > RequestResponse<S, R, Req, RS, DS, K>
159{
160    /// Create a new [`RequestResponseProtocol`]
161    pub fn new(
162        // The configuration for the protocol
163        config: RequestResponseConfig,
164        // The network sender that [`RequestResponseProtocol`] will use to send messages
165        sender: S,
166        // The network receiver that [`RequestResponseProtocol`] will use to receive messages
167        receiver: R,
168        // The recipient source that [`RequestResponseProtocol`] will use to get the recipients
169        // that a specific message should expect responses from
170        recipient_source: RS,
171        // The [response] data source that [`RequestResponseProtocol`] will use to derive the
172        // response data for a specific request
173        data_source: DS,
174    ) -> Self {
175        // Create the active requests map
176        let active_requests = ActiveRequestsMap::default();
177
178        // Create the inner implementation
179        let inner = Arc::new(RequestResponseInner {
180            config,
181            sender,
182            recipient_source,
183            data_source,
184            active_requests,
185            phantom_data: PhantomData,
186        });
187
188        // Start the task that receives messages and handles them. This will automatically get cancelled
189        // when the protocol is dropped
190        let inner_clone = Arc::clone(&inner);
191        let receive_task_handle =
192            AbortOnDropHandle::new(tokio::spawn(inner_clone.receiving_task(receiver)));
193
194        // Return the protocol
195        Self {
196            inner,
197            _receiving_task_handle: Arc::new(receive_task_handle),
198        }
199    }
200}

204    S: Sender<K>,
205    R: Receiver,
206    Req: Request,
207    RS: RecipientSource<Req, K>,
208    DS: DataSource<Req>,

212    config: RequestResponseConfig,
213    /// The sender to use for the protocol
214    pub sender: S,
215    /// The recipient source to use for the protocol
216    recipient_source: RS,
217    /// The data source to use for the protocol
218    data_source: DS,
219    /// The map of currently active requests
220    active_requests: ActiveRequestsMap<Req>,
221    /// Phantom data to help with type inference
222    phantom_data: PhantomData<(K, R, Req, DS)>,

224impl<
225        S: Sender<K>,
226        R: Receiver,
227        Req: Request,

233    /// Request something from the protocol indefinitely until we get a response
234    /// or there was a critical error (e.g. the request could not be signed)
235    ///
236    /// # Errors
237    /// - If the request was invalid
238    /// - If there was a critical error (e.g. the channel was closed)
239    pub async fn request_indefinitely(
240        self: &Arc<Self>,
241        public_key: &K,
242        private_key: &K::PrivateKey,
243        // The estimated TTL of other participants. This is used to decide when to
244        // stop making requests and sign a new one
245        estimated_request_ttl: Duration,
246        // The request to make
247        request: Req,
248    ) -> std::result::Result<Req::Response, RequestError> {
249        loop {
250            // Sign a request message
251            let request_message = RequestMessage::new_signed(public_key, private_key, &request)
252                .map_err(|e| {
253                    RequestError::InvalidRequest(anyhow::anyhow!(
254                        "failed to sign request message: {e}"
255                    ))
256                })?;
257
258            // Request the data, handling the errors appropriately
259            match self.request(request_message, estimated_request_ttl).await {
260                Ok(response) => return Ok(response),
261                Err(RequestError::Timeout) => continue,
262                Err(e) => return Err(e),
263            }
264        }
265    }

267    /// Request something from the protocol and wait for the response. This function
268    /// will join with an existing request for the same data (determined by `Blake3` hash),
269    /// however both will make requests until the timeout is reached
270    ///
271    /// # Errors
272    /// - If the request times out
273    /// - If the channel is closed (this is an internal error)
274    /// - If the request we sign is invalid
275    pub async fn request(
276        self: &Arc<Self>,
277        request_message: RequestMessage<Req, K>,
278        timeout_duration: Duration,
279    ) -> std::result::Result<Req::Response, RequestError> {
280        timeout(timeout_duration, async move {
281            // Calculate the hash of the request
282            let request_hash = blake3::hash(&request_message.request.to_bytes().map_err(|e| {
283                RequestError::InvalidRequest(anyhow::anyhow!(
284                    "failed to serialize request message: {e}"
285                ))
286            })?);
287
288            let request = {
289                // Get a write lock on the active requests map
290                let mut active_requests_write = self.active_requests.write();
291
292                // Conditionally get the active request, creating a new one if it doesn't exist or if
293                // the existing one has been dropped and not yet removed
294                if let Some(active_request) = active_requests_write
295                    .get(&request_hash)
296                    .and_then(Weak::upgrade)
297                {
298                    ActiveRequest(active_request)
299                } else {
300                    // Create a new broadcast channel for the response
301                    let (sender, receiver) = async_broadcast::broadcast(1);
302
303                    // Create a new active request
304                    let active_request = ActiveRequest(Arc::new(ActiveRequestInner {
305                        sender,
306                        receiver,
307                        request: request_message.request.clone(),
308                        active_requests: Arc::clone(&self.active_requests),
309                        request_hash,
310                    }));
311
312                    // Write the new active request to the map
313                    active_requests_write.insert(request_hash, Arc::downgrade(&active_request.0));
314
315                    // Return the new active request
316                    active_request
317                }
318            };
319
320            // Get the recipients that the request should expect responses from. Shuffle them so
321            // that we don't always send to the same recipients in the same order
322            let mut recipients = self
323                .recipient_source
324                .get_expected_responders(&request_message.request)
325                .await;
326            recipients.shuffle(&mut rand::thread_rng());
327
328            // Create a request message and serialize it
329            let message =
330                Bytes::from(Message::Request(request_message).to_bytes().map_err(|e| {
331                    RequestError::InvalidRequest(anyhow::anyhow!(
332                        "failed to serialize request message: {e}"
333                    ))
334                })?);
335
336            // Get the current time so we can check when the timeout has elapsed
337            let start_time = Instant::now();
338
339            // Spawn a task that sends out requests to the network
340            let self_clone = Arc::clone(self);
341            let _handle = AbortOnDropHandle::new(spawn(async move {
342                // Create a bounded queue for the outgoing requests. We use this to make sure
343                // we have less than [`config.request_batch_size`] requests in flight at any time.
344                //
345                // When newer requests are added, older ones are removed from the queue. Because we use
346                // `AbortOnDropHandle`, the older ones will automatically get cancelled
347                let mut outgoing_requests =
348                    BoundedVecDeque::new(self_clone.config.request_batch_size);
349
350                // While the timeout hasn't elapsed, send out requests to the network
351                while start_time.elapsed() < timeout_duration {
352                    // Send out requests to the network in their own separate tasks
353                    for recipient_batch in recipients.chunks(self_clone.config.request_batch_size) {
354                        for recipient in recipient_batch {
355                            // Clone ourselves, the message, and the recipient so they can be moved
356                            let self_clone = Arc::clone(&self_clone);
357                            let recipient_clone = recipient.clone();
358                            let message_clone = Arc::clone(&message);
359
360                            // Spawn the task that sends the request to the participant
361                            let individual_sending_task = spawn(async move {
362                                let _ = self_clone
363                                    .sender
364                                    .send_message(&message_clone, recipient_clone)
365                                    .await;
366                            });
367
368                            // Add the sending task to the queue
369                            outgoing_requests.push(AbortOnDropHandle::new(individual_sending_task));
370                        }
371
372                        // After we send the batch out, wait the [`config.request_batch_interval`]
373                        // before sending the next one
374                        sleep(self_clone.config.request_batch_interval).await;
375                    }
376                }
377            }));
378
379            // Wait for a response on the channel
380            request
381                .receiver
382                .clone()
383                .recv()
384                .await
385                .map_err(|_| RequestError::Other(anyhow!("channel was closed")))
386        })
387        .await
388        .map_err(|_| RequestError::Timeout)
389        .and_then(|result| result)
390    }

392    /// The task responsible for receiving messages from the receiver and handling them
393    async fn receiving_task(self: Arc<Self>, mut receiver: R) {
394        // Upper bound the number of outgoing and incoming responses
395        let mut outgoing_responses = BoundedVecDeque::new(self.config.max_outgoing_responses);
396        let mut incoming_responses = BoundedVecDeque::new(self.config.max_incoming_responses);
397
398        // While the receiver is open, we receive messages and handle them
399        loop {
400            // Try to receive a message
401            match receiver.receive_message().await {
402                Ok(message) => {
403                    // Deserialize the message, warning if it fails
404                    let message = match Message::from_bytes(&message) {
405                        Ok(message) => message,
406                        Err(e) => {
407                            warn!("Received invalid message: {e}");
408                            continue;
409                        },
410                    };
411
412                    // Handle the message based on its type
413                    match message {
414                        Message::Request(request_message) => {
415                            self.handle_request(request_message, &mut outgoing_responses);
416                        },
417                        Message::Response(response_message) => {
418                            self.handle_response(response_message, &mut incoming_responses);
419                        },
420                    }
421                },
422                // An error here means the receiver will _NEVER_ receive any more messages
423                Err(e) => {
424                    error!("Request/response receive task exited: {e}");
425                    return;
426                },
427            }
428        }
429    }

431    /// Handle a request sent to us
432    fn handle_request(
433        self: &Arc<Self>,
434        request_message: RequestMessage<Req, K>,
435        outgoing_responses: &mut OutgoingResponses,
436    ) {
437        // Spawn a task to:
438        // - Validate the request
439        // - Derive the response data (check if we have it)
440        // - Send the response to the requester
441        let self_clone = Arc::clone(self);
442        let response_task = AbortOnDropHandle::new(tokio::spawn(async move {
443            let result = timeout(self_clone.config.response_send_timeout, async move {
444                // Validate the request message. This includes:
445                // - Checking the signature and making sure it's valid
446                // - Checking the timestamp and making sure it's not too old
447                // - Calling the request's application-specific validation function
448                request_message
449                    .validate(self_clone.config.incoming_request_ttl)
450                    .await
451                    .with_context(|| "failed to validate request")?;
452
453                // Try to fetch the response data from the data source
454                let response = self_clone
455                    .data_source
456                    .derive_response_for(&request_message.request)
457                    .await
458                    .with_context(|| "failed to derive response for request")?;
459
460                // Create the response message and serialize it
461                let response = Bytes::from(
462                    Message::Response::<Req, K>(ResponseMessage {
463                        request_hash: blake3::hash(&request_message.request.to_bytes()?),
464                        response,
465                    })
466                    .to_bytes()
467                    .with_context(|| "failed to serialize response message")?,
468                );
469
470                // Send the response to the requester
471                self_clone
472                    .sender
473                    .send_message(&response, request_message.public_key)
474                    .await
475                    .with_context(|| "failed to send response to requester")?;
476
477                Ok::<(), anyhow::Error>(())
478            })
479            .await
480            .map_err(|_| anyhow::anyhow!("timed out while sending response"))
481            .and_then(|result| result);
482
483            if let Err(e) = result {
484                warn!("Failed to send response to requester: {e}");
485            }
486        }));
487
488        // Add the response task to the outgoing responses queue. This will automatically cancel an older task
489        // if there are more than [`config.max_outgoing_responses`] responses in flight.
490        outgoing_responses.push(response_task);
491    }
492
493    /// Handle a response sent to us
494    fn handle_response(
495        self: &Arc<Self>,
496        response: ResponseMessage<Req>,
497        incoming_responses: &mut IncomingResponses,
498    ) {
499        // Get the entry in the map, ignoring it if it doesn't exist
500        let Some(active_request) = self
501            .active_requests
502            .read()
503            .get(&response.request_hash)
504            .cloned()
505            .and_then(|r| r.upgrade())
506        else {
507            return;
508        };
509
510        // Spawn a task to validate the response and send it to the requester (us)
511        let response_validate_timeout = self.config.response_validate_timeout;
512        let response_task = AbortOnDropHandle::new(tokio::spawn(async move {
513            if timeout(response_validate_timeout, async move {
514                // Make sure the response is valid for the given request
515                if let Err(e) = response.response.validate(&active_request.request).await {
516                    warn!("Received invalid response: {e}");
517                    return;
518                }
519
520                // Send the response to the requester (the user of [`RequestResponse::request`])
521                let _ = active_request.sender.try_broadcast(response.response);
522            })
523            .await
524            .is_err()
525            {
526                warn!("Timed out while validating response");
527            }
528        }));
529
530        // Add the response task to the incoming responses queue. This will automatically cancel an older task
531        // if there are more than [`config.max_incoming_responses`] responses being processed
532        incoming_responses.push(response_task);
533    }

536/// An active request. This is what we use to track a request and its corresponding response
537/// in the protocol
538#[derive(Clone, Deref)]
539pub struct ActiveRequest<R: Request>(Arc<ActiveRequestInner<R>>);
540
541/// The inner implementation of an active request
542pub struct ActiveRequestInner<R: Request> {
543    /// The sender to use for the protocol
544    sender: async_broadcast::Sender<R::Response>,
545    /// The receiver to use for the protocol
546    receiver: async_broadcast::Receiver<R::Response>,
547    /// The request that we are waiting for a response to
548    request: R,
549
550    /// A copy of the map of currently active requests
551    active_requests: ActiveRequestsMap<R>,
552    /// The hash of the request. We need this so we can remove ourselves from the map
553    request_hash: RequestHash,
554}
555
556impl<R: Request> Drop for ActiveRequestInner<R> {
557    fn drop(&mut self) {
558        self.active_requests.write().remove(&self.request_hash);
559    }

687        /// Whether or not the data has been taken
688        taken: Arc<AtomicBool>,
689    }
690
691    #[async_trait]
692    impl DataSource<TestRequest> for TestDataSource {
693        async fn derive_response_for(&self, request: &TestRequest) -> Result<Vec<u8>> {
694            // Return a response if we hit the hit rate
695            if self.has_data && Instant::now() >= self.data_available_time {
696                if self.take_data && !self.taken.swap(true, std::sync::atomic::Ordering::Relaxed) {
697                    return Err(anyhow::anyhow!("data already taken"));
698                }
699                Ok(blake3::hash(&request.0).as_bytes().to_vec())
700            } else {
701                Err(anyhow::anyhow!("did not have the data"))
702            }
703        }
704    }

719    /// Create fully connected test networks with `num_participants` participants
720    fn create_participants(
721        num: usize,
722    ) -> Vec<(TestSender, mpsc::Receiver<Bytes>, (BLSPubKey, BLSPrivKey))> {
723        // The entire network
724        let mut network = HashMap::new();
725
726        // All receivers in the network
727        let mut receivers = Vec::new();
728
729        // All keypairs in the network
730        let mut keypairs = Vec::new();
731
732        // For each participant,
733        for i in 0..num {
734            // Create a unique `BLSPubKey`
735            let (public_key, private_key) =
736                BLSPubKey::generated_from_seed_indexed([2; 32], i.try_into().unwrap());
737
738            // Add the keypair to the list
739            keypairs.push((public_key, private_key));
740
741            // Create a channel for sending and receiving messages
742            let (sender, receiver) = mpsc::channel::<Bytes>(100);
743
744            // Add the participant to the network
745            network.insert(public_key, sender);
746
747            // Add the receiver to the list of receivers
748            receivers.push(receiver);
749        }
750
751        // Create a test sender from the network
752        let sender = TestSender {
753            network: Arc::new(network),
754        };
755
756        // Return all senders and receivers
757        receivers
758            .into_iter()
759            .zip(keypairs)
760            .map(|(r, k)| (sender.clone(), r, k))
761            .collect()
762    }

764    /// The configuration for an integration test
765    #[derive(Clone)]
766    struct IntegrationTestConfig {
767        /// The request response protocol configuration
768        request_response_config: RequestResponseConfig,
769        /// The number of participants in the network
770        num_participants: usize,
771        /// The number of participants that have the data
772        num_participants_with_data: usize,
773        /// The timeout for the requests
774        request_timeout: Duration,
775        /// The delay before the nodes have the data available
776        data_available_delay: Duration,
777    }
778
779    /// The result of an integration test
780    struct IntegrationTestResult {
781        /// The number of nodes that received a response
782        num_succeeded: usize,
783    }
784
785    /// Run an integration test with the given parameters
786    async fn run_integration_test(config: IntegrationTestConfig) -> IntegrationTestResult {
787        // Create a fully connected network with `num_participants` participants
788        let participants = create_participants(config.num_participants);
789
790        // Create a join set to wait for all the tasks to finish
791        let mut join_set = JoinSet::new();
792
793        // We need to keep these here so they don't get dropped
794        let handles = Arc::new(Mutex::new(Vec::new()));
795
796        // For each one, create a new [`RequestResponse`] protocol
797        for (i, (sender, receiver, (public_key, private_key))) in
798            participants.into_iter().enumerate()
799        {
800            let config_clone = config.request_response_config.clone();
801            let handles_clone = Arc::clone(&handles);
802            join_set.spawn(async move {
803                let protocol = RequestResponse::new(
804                    config_clone,
805                    sender.clone(),
806                    receiver,
807                    sender,
808                    TestDataSource {
809                        has_data: i < config.num_participants_with_data,
810                        data_available_time: Instant::now() + config.data_available_delay,
811                        take_data: false,
812                        taken: Arc::new(AtomicBool::new(false)),
813                    },
814                );
815
816                // Add the handle to the handles list so it doesn't get dropped and
817                // cancelled
818                #[allow(clippy::used_underscore_binding)]
819                handles_clone
820                    .lock()
821                    .unwrap()
822                    .push(Arc::clone(&protocol._receiving_task_handle));
823
824                // Create a random request
825                let request = TestRequest(vec![rand::thread_rng().gen(); 100]);
826
827                // Get the hash of the request
828                let request_hash = blake3::hash(&request.0).as_bytes().to_vec();
829
830                // Create a new request message
831                let request = RequestMessage::new_signed(&public_key, &private_key, &request)
832                    .expect("failed to create request message");
833
834                // Request the data from the protocol
835                let response = protocol.request(request, config.request_timeout).await?;
836
837                // Make sure the response is the hash of the request
838                assert_eq!(response, request_hash);
839
840                Ok::<(), anyhow::Error>(())
841            });
842        }
843
844        // Wait for all the tasks to finish
845        let mut num_succeeded = config.num_participants;
846        while let Some(result) = join_set.join_next().await {
847            if result.is_err() || result.unwrap().is_err() {
848                num_succeeded -= 1;
849            }
850        }
851
852        IntegrationTestResult { num_succeeded }
853    }

912    /// Test that we can join an existing request for the same data and get the same (single) response
913    #[tokio::test(flavor = "multi_thread")]
914    async fn test_join_existing_request() {
915        // Build a config
916        let config = default_protocol_config();
917
918        // Create two participants
919        let mut participants = Vec::new();
920
921        for (sender, receiver, (public_key, private_key)) in create_participants(2) {
922            // For each, create a new [`RequestResponse`] protocol
923            let protocol = RequestResponse::new(
924                config.clone(),
925                sender.clone(),
926                receiver,
927                sender,
928                TestDataSource {
929                    take_data: true,
930                    has_data: true,
931                    data_available_time: Instant::now() + Duration::from_secs(2),
932                    taken: Arc::new(AtomicBool::new(false)),
933                },
934            );
935
936            // Add the participants to the list
937            participants.push((protocol, public_key, private_key));
938        }
939
940        // Take the first participant
941        let one = Arc::new(participants.remove(0));
942
943        // Create the request that they should all be able to join on
944        let request = TestRequest(vec![rand::thread_rng().gen(); 100]);
945
946        // Create a join set to wait for all the tasks to finish
947        let mut join_set = JoinSet::new();
948
949        // Make 10 requests with the same hash
950        for _ in 0..10 {
951            // Clone the first participant
952            let one_clone = Arc::clone(&one);
953
954            // Clone the request
955            let request_clone = request.clone();
956
957            // Spawn a task to request the data
958            join_set.spawn(async move {
959                // Create a new, signed request message
960                let request_message =
961                    RequestMessage::new_signed(&one_clone.1, &one_clone.2, &request_clone)?;
962
963                // Start requesting it
964                one_clone
965                    .0
966                    .request(request_message, Duration::from_secs(20))
967                    .await?;
968
969                Ok::<(), anyhow::Error>(())
970            });
971        }
972
973        // Wait for all the tasks to finish, making sure they all succeed
974        while let Some(result) = join_set.join_next().await {
975            result
976                .expect("failed to join task")
977                .expect("failed to request data");
978        }
979    }
