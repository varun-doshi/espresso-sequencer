1//! Adversarial Espresso client.
2//!
3//! This program runs a randomized client for the Espresso query service. It is designed to hit
4//! methods and usage patterns that a normal client, such as a rollup, would not do, so that we can
5//! discover edge cases in the performance of the query service.
6//!
7//! The client works by repeatedly generating random "actions" to execute. Actions include both
8//! one-off queries and maintenance of long-lived connections. The client may keep many connections
9//! open at one time, which has been a source of performance problems for the server in the past.
10//!
11//! The program also runs a web server to provide some visibility into its state. The web server
12//! provides a healthcheck endpoint as well as a prometheus endpoint which provides metrics like the
13//! count of various types of actions performed and the number of open streams.
14

495    /// Retry a fallible operation several times before giving up.
496    ///
497    /// Some queries are allowed to fail occasionally, but should heal themselves quickly. For
498    /// example, we may query the block height from one node and then get routed to another node to
499    /// query an object at that height. The second node may be lagging and return 404, but it should
500    /// catch up quickly and the query should start to succeed. Or, we may query for an object which
501    /// is missing, but the server should quickly heal by fetching the missing object from a peer.
502    ///
503    /// This function will retry a fallible operation a configurable number of times. If the
504    /// operation ever fails, a warning will be logged, but it is not considered a failed action
505    /// unless the operation fails several times in a row.
506    ///
507    /// If the operation fails even once but eventually succeeds, we will retry the operation
508    /// several times to check that all servers are "healed": at this point, the operation should no
509    /// longer fail even once.
510    async fn retry<F: TryFuture<Error = anyhow::Error>>(
511        &self,
512        span: tracing::Span,
513        f: impl Fn() -> F,
514    ) -> anyhow::Result<F::Ok> {
515        let _enter = span.enter();
516        tracing::debug!("starting fallible operation");
517
518        let mut i = 0;
519        loop {
520            match f().into_future().await {
521                Ok(res) if i == 0 => {
522                    // Succeeded on the first try, get on with it.
523                    return Ok(res);
524                },
525                Ok(res) => {
526                    // Succeeded after at least one failure; retry a number of additional times to
527                    // be sure the endpoint is healed.
528                    tracing::info!("succeeded after {i} retries; checking health");
529                    for _ in 0..self.cfg.min_retries {
530                        f().into_future().await.context(
531                            "operation is flaky; succeeded on retry but not fully healed",
532                        )?;
533                    }
534                    return Ok(res);
535                },
536                Err(err) if i < self.cfg.max_retries => {
537                    tracing::warn!("failed, will retry: {err:#}");
538                    i += 1;
539                },
540                Err(err) => {
541                    return Err(err).context("failed too many times");
542                },
543            }
544        }
545    }

547    /// Send an HTTP GET request and deserialize the response.
548    ///
549    /// This method is a wrapper around `self.client.get()`, which adds instrumentation and metrics
550    /// for request latency.
551    async fn get<R: DeserializeOwned>(&self, path: impl Into<String>) -> anyhow::Result<R> {
552        let path = path.into();
553        tracing::debug!("-> GET {path}");
554
555        let start = Instant::now();
556        let res = self.client.get::<R>(&path).send().await;
557        let elapsed = start.elapsed();
558
559        let status = match &res {
560            Ok(_) => StatusCode::OK,
561            Err(err) => err.status(),
562        };
563        tracing::debug!("<- GET {path} {} ({elapsed:?})", u16::from(status));
564
565        self.metrics
566            .request_latency
567            .add_point((elapsed.as_millis() as f64) / 1000.);
568        if elapsed >= self.cfg.http_timeout_warning {
569            self.metrics.slow_requests.add(1);
570            tracing::warn!(%path, ?elapsed, "slow request");
571        }
572
573        res.context(format!("GET {path}"))
574    }

576    async fn query(&self, at: u64) -> anyhow::Result<()> {
577        let at = self.adjust_index(at).await?;
578        let obj = self
579            .retry(
580                info_span!("query", resource = Self::singular(), at),
581                || async {
582                    self.get::<T>(format!("availability/{}/{at}", Self::singular()))
583                        .await
584                },
585            )
586            .await?;
587
588        // Query by hash and check consistency.
589        let hash = obj.hash();
590        let by_hash = self
591            .retry(
592                info_span!("query by hash", resource = Self::singular(), at, hash),
593                || async {
594                    self.get(format!(
595                        "availability/{}/{}/{hash}",
596                        Self::singular(),
597                        T::HASH_URL_SEGMENT,
598                    ))
599                    .await
600                },
601            )
602            .await?;
603        ensure!(
604            obj == by_hash,
605            format!(
606                "query for {} {at} by hash {hash} is not consistent",
607                Self::singular()
608            )
609        );
610
611        // Query by payload hash and check consistency.
612        if let Some(segment) = T::PAYLOAD_HASH_URL_SEGMENT {
613            let payload_hash = obj.payload_hash();
614            let by_payload_hash = self
615                .retry(
616                    info_span!(
617                        "query by payload hash",
618                        resource = Self::singular(),
619                        at,
620                        payload_hash
621                    ),
622                    || async {
623                        self.get::<T>(format!(
624                            "availability/{}/{segment}/{payload_hash}",
625                            Self::singular(),
626                        ))
627                        .await
628                    },
629                )
630                .await?;
631            // We might not get the exact object this time, due to non-uniqueness of payloads, but we
632            // should get an object with the same payload.
633            ensure!(
634                payload_hash == by_payload_hash.payload_hash(),
635                format!(
636                    "query for {} {at} by payload hash {payload_hash} is not consistent",
637                    Self::singular()
638                )
639            );
640        }
641
642        self.metrics.query_actions[&T::RESOURCE].add(1);
643        Ok(())
644    }

646    async fn query_range(&mut self, from: u64, len: u16) -> anyhow::Result<()> {
647        let from = self.adjust_index(from).await? as usize;
648        let limits = self
649            .get::<availability::Limits>("availability/limits")
650            .await?;
651        let limit = if T::IS_LARGE_OBJECT {
652            limits.large_object_range_limit
653        } else {
654            limits.small_object_range_limit
655        };
656
657        // Adjust `len`, 10% of the time query above the limit for this type (so the query fails);
658        // the rest of the time query a valid range.
659        let max_len = limit * 11 / 10;
660        let to = self
661            .adjust_index(from as u64 + (len as u64) % (max_len as u64))
662            .await? as usize;
663        let to = max(from, to);
664        match self
665            .get::<Vec<T>>(format!("availability/{}/{from}/{to}", Self::singular()))
666            .await
667        {
668            Ok(range) => {
669                ensure!(to - from <= limit, "range endpoint succeeded and returned {} results for request over limit; limit: {limit} from: {from} to: {to}", range.len());
670                ensure!(range.len() == to - from, "range endpoint returned wrong number of results; from: {from} to: {to} results: {}", range.len());
671                for (i, obj) in range.iter().enumerate() {
672                    ensure!(
673                        obj.height() == from + i,
674                        "object in range has wrong height; from: {from} to: {to} i: {i} height: {}",
675                        obj.height()
676                    );
677                }
678            },
679            Err(_) if to - from > limit => {
680                tracing::info!(
681                    limit,
682                    from,
683                    to,
684                    "range query exceeding limit failed as expected"
685                );
686            },
687            Err(err) => {
688                return Err(err).context("error in range query");
689            },
690        }
691
692        self.metrics.query_range_actions[&T::RESOURCE].add(1);
693        Ok(())
694    }

696    async fn open_stream(&mut self, from: u64) -> anyhow::Result<()> {
697        if self.open_streams.len() >= self.cfg.max_open_streams {
698            tracing::info!(
699                num = self.open_streams.len(),
700                "not opening stream, number of open streams exceeds maximum"
701            );
702            return Ok(());
703        }
704
705        let from = self.adjust_index(from).await?;
706        let stream = self
707            .client
708            .socket(&format!("availability/stream/{}/{from}", Self::plural()))
709            .subscribe()
710            .await
711            .context(format!("subscribing to {} from {from}", Self::plural()))?;
712        let id = self.next_stream_id;
713        self.next_stream_id += 1;
714        tracing::info!("opened {} stream {id} at {from}", Self::singular());
715        self.open_streams.insert(
716            id,
717            Subscription {
718                stream: Box::pin(stream.peekable()),
719                position: from,
720                refreshed: Instant::now(),
721            },
722        );
723
724        self.metrics.open_streams[&T::RESOURCE].update(1);
725        self.metrics.open_stream_actions[&T::RESOURCE].add(1);
726        Ok(())
727    }

729    async fn close_stream(&mut self, index: usize) {
730        if self.open_streams.is_empty() {
731            tracing::debug!("not closing {} stream; no open streams", Self::singular());
732            return;
733        }
734        let id = *self
735            .open_streams
736            .keys()
737            .nth(index % self.open_streams.len())
738            .unwrap();
739        tracing::info!("closing {} stream {id}", Self::singular());
740        self.open_streams.remove(&id);
741        self.metrics.open_streams[&T::RESOURCE].update(-1);
742        self.metrics.close_stream_actions[&T::RESOURCE].add(1);
743    }
744
745    async fn poll_stream(&mut self, index: usize, amount: u8) -> anyhow::Result<()> {
746        if self.open_streams.is_empty() {
747            tracing::debug!("not polling {} stream; no open streams", Self::singular());
748            return Ok(());
749        }
750        self.metrics.poll_stream_actions[&T::RESOURCE].add(1);
751
752        let index = index % self.open_streams.len();
753        let mut blocking = 0;
754        for _ in 0..amount {
755            let (id, stream) = self.open_streams.iter_mut().nth(index).unwrap();
756
757            // Check if the next item is immediately available or if we're going to block.
758            let will_block = stream.stream.as_mut().peek().now_or_never().is_none();
759            if will_block {
760                blocking += 1;
761                if blocking > self.cfg.max_blocking_polls {
762                    tracing::info!("aborting poll_stream action; exceeded maximum blocking polls");
763                    return Ok(());
764                }
765            }
766
767            let pos = stream.position;
768            let refreshed = stream.refreshed;
769            stream.position += 1;
770            let span = info_span!(
771                "polling stream",
772                resource = Self::singular(),
773                id,
774                pos,
775                ?refreshed,
776            );
777            let _enter = span.enter();
778            let obj = loop {
779                let Some(res) = stream.stream.next().await else {
780                    let id = *id;
781                    self.open_streams.remove(&id);
782                    self.metrics.open_streams[&T::RESOURCE].update(-1);
783
784                    // All of our streams are supposed to be indefinite.
785                    bail!("{} stream {id} ended", Self::singular());
786                };
787                match res {
788                    Ok(obj) => {
789                        if will_block {
790                            // Successfully reading from a WebSockets stream should reset the idle
791                            // conenection timeout, so we don't expect errors from this connection
792                            // in the near future. Note that this applies only to reads which
793                            // actually block. Reads which don't block may come directly from the
794                            // local TCP buffer, and thus not generate any traffic on the idle TCP
795                            // connection.
796                            stream.refreshed = Instant::now();
797                            tracing::debug!(
798                                refreshed = ?stream.refreshed,
799                                "{} stream refreshed due to blocking read",
800                                Self::singular(),
801                            );
802                        }
803                        break obj;
804                    },
805                    Err(err) if refreshed.elapsed() >= self.cfg.web_socket_timeout => {
806                        // Streams are allowed to fail if the connection is too old. Warn about it,
807                        // but refresh the connection and try again.
808                        tracing::warn!("error in old connection, refreshing connection: {err:#}");
809                        let conn = self
810                            .client
811                            .socket(&format!("availability/stream/{}/{pos}", Self::plural()))
812                            .subscribe()
813                            .await
814                            .context(format!("subscribing to {} from {pos}", Self::plural()))?;
815                        stream.stream = Box::pin(conn.peekable());
816                        stream.refreshed = Instant::now();
817                        tracing::info!(
818                            refreshed = ?stream.refreshed,
819                            "{} stream refreshed due to connection reset",
820                            Self::singular(),
821                        );
822                    },
823                    Err(err) => {
824                        // Errors on a relatively fresh connection are not allowed. Close the stream
825                        // since it is apparently in a bad state, and return an error.
826                        let id = *id;
827                        self.open_streams.remove(&id);
828                        self.metrics.open_streams[&T::RESOURCE].update(-1);
829                        return Err(err).context(format!(
830                            "polling {} stream {id} at {pos}, last refreshed {:?} ago",
831                            Self::singular(),
832                            refreshed.elapsed()
833                        ));
834                    },
835                }
836            };
837
838            // Check consistency against a regular query.
839            let id = *id;
840            let expected = self
841                .retry(info_span!("fetching expected object"), || async {
842                    self.get(format!("availability/{}/{pos}", Self::singular()))
843                        .await
844                })
845                .await?;
846            ensure!(
847                obj == expected,
848                format!(
849                    "{} stream {id} is not consistent with query at {pos}",
850                    Self::singular()
851                ),
852            );
853        }
854
855        Ok(())
856    }
857
858    async fn adjust_index(&self, at: u64) -> anyhow::Result<u64> {
859        let block_height = loop {
860            let block_height: u64 = self.get("status/block-height").await?;
861            if block_height == 0 {
862                // None of our tests work with an empty history, but if we just wait briefly there
863                // should be some blocks produced soon.
864                tracing::info!("waiting for block height");
865                sleep(Duration::from_secs(1)).await;
866                continue;
867            }
868            break block_height;
869        };
870        Ok(at % block_height)
871    }

874impl ResourceManager<Header> {
875    async fn query_window(&self, from: u64, duration: u16) -> anyhow::Result<()> {
876        let now = OffsetDateTime::now_utc().unix_timestamp() as u64;
877        let start = from % now;
878        let end = start + duration as u64;
879
880        let window = self
881            .retry(
882                info_span!("timestamp window", resource = Self::singular(), start, end),
883                || async {
884                    self.get::<TimeWindowQueryData<Header>>(format!(
885                        "node/header/window/{start}/{end}"
886                    ))
887                    .await
888                },
889            )
890            .await?;
891
892        // Sanity check the window: prev and next should be correct bookends.
893        if let Some(prev) = &window.prev {
894            ensure!(
895                prev.timestamp() < start,
896                format!("prev header {} is later than {start}", prev.height())
897            );
898        }
899        if let Some(next) = &window.next {
900            ensure!(
901                next.timestamp() >= end,
902                format!("next header {} is earlier than {end}", next.height())
903            );
904        }
905        // Each header in the window proper should have an appropriate timestamp.
906        let mut prev = window.prev;
907        for header in window.window {
908            ensure!(
909                header.timestamp() >= start && header.timestamp() < end,
910                format!(
911                    "header {} with timestamp {} is not in window [{start}, {end})",
912                    header.height(),
913                    header.timestamp()
914                )
915            );
916
917            if let Some(prev) = prev {
918                ensure!(
919                    prev.height() + 1 == header.height(),
920                    format!(
921                        "headers in window from {start} to {end} are not consecutive (prev = {}, curr = {})",
922                        prev.height(),
923                        header.height(),
924                    ),
925                );
926                ensure!(
927                    prev.timestamp() <= header.timestamp(),
928                    format!(
929                        "headers in window from {start} to {end} have decreasing timestamps (prev = {}, curr = {})",
930                        prev.timestamp(),
931                        header.timestamp(),
932                    ),
933                );
934            }
935            prev = Some(header);
936        }
937
938        self.metrics.query_window_actions.add(1);
939        Ok(())
940    }
941
942    async fn query_block_state(&self, block: u64, index: u64) -> anyhow::Result<()> {
943        let (block, index) = match self.adjust_index(block).await? {
944            0 | 1 => {
945                // The block state at height 0 is empty, so to have a valid query just adjust to
946                // querying at height 1. At height 1, the only valid index to query is 0.
947                (1, 0)
948            },
949            block => {
950                // At any other height, all indices between 0 and `block - 1` are valid to query.
951                (block, index % (block - 1))
952            },
953        };
954
955        // Get the header of the state snapshot we're going to query and the block commitment we're
956        // going to look up from the Merkle tree, so we can later verify our results.
957        let block_header = self
958            .retry(info_span!("get block header", block), || async {
959                self.get::<Header>(format!("availability/header/{block}"))
960                    .await
961            })
962            .await?;
963        let index_header = self
964            .retry(info_span!("get index header", index), || async {
965                self.get::<Header>(format!("availability/header/{index}"))
966                    .await
967            })
968            .await?;
969
970        // Get a Merkle proof for the block commitment at position `index` from state `block`.
971        let proof = self
972            .retry(info_span!("get block proof", block, index), || async {
973                self.get::<<BlockMerkleTree as MerkleTreeScheme>::MembershipProof>(format!(
974                    "block-state/{block}/{index}"
975                ))
976                .await
977            })
978            .await?;
979
980        // Check that the proof proves inclusion of `index_header` at position `index` relative to
981        // `block_header`.
982        BlockMerkleTree::verify(
983            block_header.block_merkle_tree_root().digest(),
984            index,
985            &proof,
986        )
987        .context("malformed merkle proof")?
988        .or_else(|_| bail!("invalid merkle proof"))?;
989        ensure!(
990            proof.elem() == Some(&index_header.commit()),
991            "merkle proof is for wrong element: {:?} != {:?}",
992            proof.elem(),
993            index_header.commit()
994        );
995
996        // Look up the proof a different way, by state commitment, and check that we get the same
997        // proof.
998        let proof2 = self
999            .retry(
1000                info_span!(
1001                    "get block proof by state commitment",
1002                    block,
1003                    index,
1004                    commitment = %block_header.block_merkle_tree_root(),
1005                ),
1006                || async {
1007                    self.get::<<BlockMerkleTree as MerkleTreeScheme>::MembershipProof>(format!(
1008                        "block-state/commit/{}/{index}",
1009                        block_header.block_merkle_tree_root(),
1010                    ))
1011                    .await
1012                },
1013            )
1014            .await?;
1015        ensure!(
1016            proof2 == proof,
1017            "got a different proof when querying by commitment, {proof2:?} != {proof:?}"
1018        );
1019
1020        self.metrics.query_block_state_actions.add(1);
1021        Ok(())
1022    }

1024    async fn query_fee_state(&self, block: u64, builder: u64) -> anyhow::Result<()> {
1025        let block = self.adjust_index(block).await?;
1026        let builder = if block == 0 { 0 } else { builder % block };
1027
1028        // Get the header of block `builder` so we can get an address (the builder account) to
1029        // query.
1030        let builder_header = self
1031            .retry(info_span!("get builder header", builder), || async {
1032                self.get::<Header>(format!("availability/header/{builder}"))
1033                    .await
1034            })
1035            .await?;
1036
1037        // Since we have multiple fee accounts, we need to select one.
1038        let accounts = builder_header.fee_info().accounts();
1039        let builder_address = accounts.first().unwrap();
1040
1041        // Get the header of the state snapshot we're going to query so we can later verify our
1042        // results.
1043        let block_header = self
1044            .retry(info_span!("get block header", block), || async {
1045                self.get::<Header>(format!("availability/header/{block}"))
1046                    .await
1047            })
1048            .await?;
1049
1050        // Get a Merkle proof for the fee state of `builder_address` from state `block`.
1051        let proof = self
1052            .retry(
1053                info_span!("get account proof", block, %builder_address),
1054                || async {
1055                    self.get::<<FeeMerkleTree as MerkleTreeScheme>::MembershipProof>(&format!(
1056                        "fee-state/{block}/{builder_address}"
1057                    ))
1058                    .await
1059                },
1060            )
1061            .await?;
1062
1063        // Check that the proof is valid relative to `builder_header`.
1064        if proof.elem().is_some() {
1065            FeeMerkleTree::verify(
1066                block_header.fee_merkle_tree_root().digest(),
1067                builder_address,
1068                &proof,
1069            )
1070            .context("malformed membership proof")?
1071            .or_else(|_| bail!("invalid membership proof"))?;
1072        } else {
1073            ensure!(
1074                FeeMerkleTree::from_commitment(block_header.fee_merkle_tree_root())
1075                    .non_membership_verify(builder_address, &proof)
1076                    .context("malformed non-membership proof")?,
1077                "invalid non-membership proof"
1078            );
1079        }
1080
1081        // Look up the proof a different way, by state commitment, and check that we get the same
1082        // proof.
1083        let proof2 = self
1084            .retry(
1085                info_span!(
1086                    "get account proof by state commitment",
1087                    block,
1088                    %builder_address,
1089                    commitment = %block_header.fee_merkle_tree_root(),
1090                ),
1091                || async {
1092                    self.get::<<FeeMerkleTree as MerkleTreeScheme>::MembershipProof>(format!(
1093                        "fee-state/commit/{}/{builder_address}",
1094                        block_header.fee_merkle_tree_root(),
1095                    ))
1096                    .await
1097                },
1098            )
1099            .await?;
1100        ensure!(
1101            proof2 == proof,
1102            "got a different proof when querying by commitment, {proof2:?} != {proof:?}"
1103        );
1104
1105        self.metrics.query_fee_state_actions.add(1);
1106        Ok(())
1107    }
1108}

1110impl ResourceManager<BlockQueryData<SeqTypes>> {
1111    async fn query_namespace(&self, block: u64, index: usize) -> anyhow::Result<()> {
1112        let block = self.adjust_index(block).await?;
1113        let span = info_span!("query namespace", resource = Self::singular(), block, index);
1114        let _enter = span.enter();
1115
1116        // Download the header so we can translate the `namespace` index to a namespace ID using
1117        // the namespace table.
1118        let header: Header = self
1119            .retry(info_span!("fetch header"), || async {
1120                self.get(format!("availability/header/{block}")).await
1121            })
1122            .await?;
1123        let num_namespaces = header.ns_table().iter().count();
1124        if num_namespaces == 0 {
1125            tracing::info!("not fetching namespace because block {block} is empty");
1126            return Ok(());
1127        }
1128        let ns_index = header
1129            .ns_table()
1130            .iter()
1131            .nth(index % num_namespaces)
1132            .unwrap();
1133        let ns = header.ns_table().read_ns_id(&ns_index).unwrap();
1134
1135        let ns_proof: ADVZNamespaceProofQueryData = self
1136            .retry(info_span!("fetch namespace", %ns), || async {
1137                self.get(format!("availability/block/{block}/namespace/{ns}"))
1138                    .await
1139            })
1140            .await?;
1141
1142        // Verify proof.
1143        let vid_common: ADVZCommonQueryData<SeqTypes> = self
1144            .retry(info_span!("fetch VID common"), || async {
1145                self.get(format!("availability/vid/common/{block}")).await
1146            })
1147            .await?;
1148        ensure!(
1149            ns_proof.proof.is_some(),
1150            format!("missing namespace proof for {block}:{ns}")
1151        );
1152        ensure!(
1153            ns_proof
1154                .proof
1155                .unwrap()
1156                .verify(
1157                    header.ns_table(),
1158                    &header.payload_commitment(),
1159                    vid_common.common(),
1160                )
1161                .is_some(),
1162            format!("failure to verify namespace proof for {block}:{ns}")
1163        );
1164
1165        self.metrics.query_namespace_actions.add(1);
1166        Ok(())
1167    }

1202#[derive(Clone, Copy, Debug, EnumDiscriminants)]
1203#[strum_discriminants(derive(VariantArray))]
1204enum Action {
1205    Query {
1206        resource: Resource,
1207        at: u64,
1208    },
1209    QueryRange {
1210        resource: Resource,
1211        from: u64,
1212        len: u16,
1213    },
1214    OpenStream {
1215        resource: Resource,
1216        from: u64,
1217    },
1218    CloseStream {
1219        resource: Resource,
1220        id: usize,
1221    },
1222    PollStream {
1223        resource: Resource,
1224        id: usize,
1225        amount: u8,
1226    },
1227    QueryWindow {
1228        from: u64,
1229        duration: u16,
1230    },
1231    QueryNamespace {
1232        block: u64,
1233        namespace: usize,
1234    },
1235    QueryBlockState {
1236        block: u64,
1237        index: u64,
1238    },
1239    QueryFeeState {
1240        block: u64,
1241        // The index of the block whose builder address should be looked up. This leads to more
1242        // realistic queries than just randomly generating addresses.
1243        builder: u64,
1244    },
1245}
1246
1247impl Action {
1248    fn random(rng: &mut impl RngCore, distribution: &ActionDistribution) -> Self {
1249        match ActionDiscriminants::VARIANTS
1250            .choose_weighted(rng, |action| distribution.weight(*action))
1251            .unwrap()
1252        {
1253            ActionDiscriminants::Query => Self::Query {
1254                resource: Resource::random(rng),
1255                at: rng.next_u64(),
1256            },
1257            ActionDiscriminants::QueryRange => Self::QueryRange {
1258                resource: Resource::random(rng),
1259                from: rng.next_u64(),
1260                len: (rng.next_u32() % u16::MAX as u32) as u16,
1261            },
1262            ActionDiscriminants::OpenStream => Self::OpenStream {
1263                resource: Resource::random(rng),
1264                from: rng.next_u64(),
1265            },
1266            ActionDiscriminants::CloseStream => Self::CloseStream {
1267                resource: Resource::random(rng),
1268                id: rng.next_u32() as usize,
1269            },
1270            ActionDiscriminants::PollStream => Self::PollStream {
1271                resource: Resource::random(rng),
1272                id: rng.next_u32() as usize,
1273                amount: (rng.next_u32() % u8::MAX as u32) as u8,
1274            },
1275            ActionDiscriminants::QueryWindow => Self::QueryWindow {
1276                from: rng.next_u64(),
1277                duration: (rng.next_u32() % u16::MAX as u32) as u16,
1278            },
1279            ActionDiscriminants::QueryNamespace => Self::QueryNamespace {
1280                block: rng.next_u64(),
1281                namespace: rng.next_u32() as usize,
1282            },
1283            ActionDiscriminants::QueryBlockState => Self::QueryBlockState {
1284                block: rng.next_u64(),
1285                index: rng.next_u64(),
1286            },
1287            ActionDiscriminants::QueryFeeState => Self::QueryFeeState {
1288                block: rng.next_u64(),
1289                builder: rng.next_u64(),
1290            },
1291        }
1292    }
1293}

1295#[derive(Debug)]
1296struct Client {
1297    blocks: ResourceManager<BlockQueryData<SeqTypes>>,
1298    leaves: ResourceManager<LeafQueryData<SeqTypes>>,
1299    headers: ResourceManager<Header>,
1300    payloads: ResourceManager<PayloadQueryData<SeqTypes>>,
1301}
1302
1303impl Client {
1304    fn new(opt: &Options, registry: &PrometheusMetrics) -> Self {
1305        let metrics = Arc::new(Metrics::new(registry));
1306        Self {
1307            blocks: ResourceManager::new(opt, metrics.clone()),
1308            leaves: ResourceManager::new(opt, metrics.clone()),
1309            headers: ResourceManager::new(opt, metrics.clone()),
1310            payloads: ResourceManager::new(opt, metrics.clone()),
1311        }
1312    }
1313
1314    async fn run(&mut self, action: Action) -> anyhow::Result<()> {
1315        tracing::trace!(?action, "execute action");
1316
1317        match action {
1318            Action::Query { resource, at } => match resource {
1319                Resource::Blocks => self.blocks.query(at).await,
1320                Resource::Leaves => self.leaves.query(at).await,
1321                Resource::Headers => self.headers.query(at).await,
1322                Resource::Payloads => self.payloads.query(at).await,
1323            },
1324            Action::QueryRange {
1325                resource,
1326                from,
1327                len,
1328            } => match resource {
1329                Resource::Blocks => self.blocks.query_range(from, len).await,
1330                Resource::Leaves => self.leaves.query_range(from, len).await,
1331                Resource::Headers => self.headers.query_range(from, len).await,
1332                Resource::Payloads => self.payloads.query_range(from, len).await,
1333            },
1334            Action::OpenStream { resource, from } => match resource {
1335                Resource::Blocks => self.blocks.open_stream(from).await,
1336                Resource::Leaves => self.leaves.open_stream(from).await,
1337                Resource::Headers => self.headers.open_stream(from).await,
1338                Resource::Payloads => self.payloads.open_stream(from).await,
1339            },
1340            Action::CloseStream { resource, id } => {
1341                match resource {
1342                    Resource::Blocks => self.blocks.close_stream(id).await,
1343                    Resource::Leaves => self.leaves.close_stream(id).await,
1344                    Resource::Headers => self.headers.close_stream(id).await,
1345                    Resource::Payloads => self.payloads.close_stream(id).await,
1346                };
1347                Ok(())
1348            },
1349            Action::PollStream {
1350                resource,
1351                id,
1352                amount,
1353            } => match resource {
1354                Resource::Blocks => self.blocks.poll_stream(id, amount).await,
1355                Resource::Leaves => self.leaves.poll_stream(id, amount).await,
1356                Resource::Headers => self.headers.poll_stream(id, amount).await,
1357                Resource::Payloads => self.payloads.poll_stream(id, amount).await,
1358            },
1359            Action::QueryWindow { from, duration } => {
1360                self.headers.query_window(from, duration).await
1361            },
1362            Action::QueryNamespace { block, namespace } => {
1363                self.blocks.query_namespace(block, namespace).await
1364            },
1365            Action::QueryBlockState { block, index } => {
1366                self.headers.query_block_state(block, index).await
1367            },
1368            Action::QueryFeeState { block, builder } => {
1369                self.headers.query_fee_state(block, builder).await
1370            },
1371        }
1372    }
1373}

1375async fn serve(port: u16, metrics: PrometheusMetrics) {
1376    let api = toml! {
1377        [route.metrics]
1378        PATH = ["/metrics"]
1379        METHOD = "METRICS"
1380    };
1381    let mut app = App::<_, ServerError>::with_state(RwLock::new(metrics));
1382    app.module::<ServerError, SequencerApiVersion>("status", api)
1383        .unwrap()
1384        .metrics("metrics", |_req, state| {
1385            async move { Ok(Cow::Borrowed(state)) }.boxed()
1386        })
1387        .unwrap();
1388    if let Err(err) = app
1389        .serve(format!("0.0.0.0:{port}"), SequencerApiVersion::instance())
1390        .await
1391    {
1392        tracing::error!("web server exited unexpectedly: {err:#}");
1393    }
1394}

1396#[tokio::main]
1397async fn main() {
1398    let opt = Options::parse();
1399    opt.logging.init();
1400
1401    let metrics = PrometheusMetrics::default();
1402    let total_actions = metrics.create_counter("total_actions".into(), None);
1403    let failed_actions = metrics.create_counter("failed_actions".into(), None);
1404    metrics
1405        .subgroup("http".into())
1406        .create_gauge("slow_request_threshold".into(), Some("s".into()))
1407        .set(opt.client_config.http_timeout_warning.as_secs() as usize);
1408    let mut client = Client::new(&opt, &metrics);
1409    let mut rng = rand::thread_rng();
1410
1411    spawn(serve(opt.port, metrics));
1412
1413    loop {
1414        if let Err(err) = client
1415            .run(Action::random(&mut rng, &opt.distribution))
1416            .await
1417        {
1418            failed_actions.add(1);
1419            tracing::error!("action failed: {err:#}");
1420        }
1421        total_actions.add(1);
1422    }
1423}
