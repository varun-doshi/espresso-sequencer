22/// A stake table config stored in a file
23#[derive(serde::Serialize, serde::Deserialize, Debug, Clone)]
24#[serde(bound(deserialize = ""))]
25pub struct PermissionedStakeTableConfig<TYPES: NodeType> {
26    /// The list of public keys that are initially inserted into the
27    /// permissioned stake table contract.
28    #[serde(default)]
29    pub public_keys: Vec<PeerConfigKeys<TYPES>>,
30}
31
32impl<TYPES> PermissionedStakeTableConfig<TYPES>
33where
34    TYPES: NodeType<SignatureKey = BLSPubKey, StateSignatureKey = SchnorrPubKey>,
35{
36    pub fn from_toml_file(path: &Path) -> anyhow::Result<Self> {
37        let config_file_as_string: String = fs::read_to_string(path)
38            .unwrap_or_else(|_| panic!("Could not read config file located at {}", path.display()));
39
40        Ok(
41            toml::from_str::<Self>(&config_file_as_string).unwrap_or_else(|err| {
42                panic!(
43                    "Unable to convert config file {} to TOML: {err}",
44                    path.display()
45                )
46            }),
47        )
48    }
49}

51impl<TYPES> From<PermissionedStakeTableConfig<TYPES>> for Vec<NodeInfo>
52where
53    TYPES: NodeType<SignatureKey = BLSPubKey, StateSignatureKey = SchnorrPubKey>,
54{
55    fn from(value: PermissionedStakeTableConfig<TYPES>) -> Self {
56        value
57            .public_keys
58            .into_iter()
59            .map(|peer_config| {
60                let node_info: NodeInfoJf = peer_config.clone().into();

62            })
63            .collect()
64    }

78/// Information to add and remove stakers in the permissioned stake table contract.
79#[derive(serde::Serialize, serde::Deserialize, Debug, Clone)]
80#[serde(bound(deserialize = ""))]
81pub struct PermissionedStakeTableUpdate<TYPES: NodeType> {
82    #[serde(default)]
83    stakers_to_remove: Vec<StakerIdentity>,
84    #[serde(default)]
85    new_stakers: Vec<PeerConfigKeys<TYPES>>,
86}
87
88impl<TYPES> PermissionedStakeTableUpdate<TYPES>
89where
90    TYPES: NodeType<SignatureKey = BLSPubKey, StateSignatureKey = SchnorrPubKey>,
91{
92    pub fn from_toml_file(path: &Path) -> anyhow::Result<Self> {
93        let config_file_as_string: String = fs::read_to_string(path)
94            .unwrap_or_else(|_| panic!("Could not read config file located at {}", path.display()));
95
96        Ok(
97            toml::from_str::<Self>(&config_file_as_string).unwrap_or_else(|err| {
98                panic!(
99                    "Unable to convert config file {} to TOML: {err}",
100                    path.display()
101                )
102            }),
103        )
104    }
105
106    fn stakers_to_remove(&self) -> Vec<G2Point> {
107        self.stakers_to_remove
108            .iter()
109            .map(|v| bls_jf_to_sol(v.clone().into()))
110            .collect()
111    }
112
113    fn new_stakers(&self) -> Vec<NodeInfo> {
114        self.new_stakers
115            .iter()
116            .map(|peer_config| {
117                let node_info: NodeInfoJf = peer_config.clone().into();
118                node_info.into()
119            })
120            .collect()
121    }
122}

124pub async fn update_stake_table<TYPES>(
125    l1url: Url,
126    l1_interval: Duration,
127    mnemonic: String,
128    account_index: u32,
129    contract_address: Address,
130    update: PermissionedStakeTableUpdate<TYPES>,
131) -> anyhow::Result<()>
132where
133    TYPES: NodeType<SignatureKey = BLSPubKey, StateSignatureKey = SchnorrPubKey>,
134{
135    let provider = Provider::<Http>::try_from(l1url.to_string())?.interval(l1_interval);
136    let chain_id = provider.get_chainid().await?.as_u64();
137    let wallet = MnemonicBuilder::<English>::default()
138        .phrase(mnemonic.as_str())
139        .index(account_index)?
140        .build()?
141        .with_chain_id(chain_id);
142    let l1 = Arc::new(SignerMiddleware::new(provider.clone(), wallet));
143
144    let contract = PermissionedStakeTable::new(contract_address, l1);
145
146    tracing::info!("sending stake table update transaction");
147
148    let tx_receipt = contract
149        .update(update.stakers_to_remove(), update.new_stakers())
150        .send()
151        .await?
152        .await?;
153    tracing::info!("Transaction receipt: {:?}", tx_receipt);
154    Ok(())
155}

172    fn assert_peer_config_eq<TYPES: NodeType>(
173        p1: &PeerConfigKeys<TYPES>,
174        p2: &PeerConfigKeys<TYPES>,
175    ) {
176        assert_eq!(p1.stake_table_key, p2.stake_table_key);

182    fn mk_keys<TYPES: NodeType<SignatureKey = BLSPubKey, StateSignatureKey = SchnorrPubKey>>(
183    ) -> Vec<PeerConfigKeys<TYPES>> {
184        let mut keys = Vec::new();
185        for i in 0..3 {
186            let (pubkey, _) = BLSPubKey::generated_from_seed_indexed([0; 32], i);
187            let state_kp = StateKeyPair::generate_from_seed_indexed([0; 32], i).0;
188            let ver_key = state_kp.ver_key();
189            keys.push(PeerConfigKeys::<TYPES> {
190                stake_table_key: pubkey,
191                state_ver_key: ver_key,
192                stake: i + 1,
193                da: i == 0,
194            });
195        }
196        keys
197    }
