68    pub async fn spawn(self) -> Anvil {
69        let state_dir = TempDir::new().unwrap();
70        let (child, url) = Anvil::spawn_server(&self, Some(state_dir.path())).await;
71        let anvil: Anvil = Anvil {
72            child,
73            url,
74            state_dir,
75            opt: self,
76        };
77
78        // When we are running a local Anvil node, as in tests, some endpoints (e.g. eth_feeHistory)
79        // do not work until at least one block has been mined.
80        while let Err(err) = anvil
81            .provider()
82            .fee_history(1, BlockNumber::Latest, &[])
83            .await
84        {
85            tracing::warn!("RPC is not ready: {err}");
86            sleep(Duration::from_millis(200)).await;
87        }
88
89        anvil
90    }

104    async fn spawn_server(opt: &AnvilOptions, state_dir: Option<&Path>) -> (Child, Url) {
105        let port = opt
106            .port
107            .unwrap_or_else(|| portpicker::pick_unused_port().unwrap());
108
109        let mut command = Command::new("anvil");
110        command.args([
111            "--silent",
112            "--port",
113            &port.to_string(),
114            "--accounts",
115            &opt.accounts.unwrap_or(20).to_string(),
116        ]);
117
118        if let Some(state_dir) = state_dir {
119            command.args(["--dump-state", &state_dir.display().to_string()]);
120        }
121        if let Some(block_time) = opt.block_time {
122            command.args(["-b", &block_time.as_secs().to_string()]);
123        }
124        if let Some(load_state) = &opt.load_state {
125            command.args(["--load-state", &load_state.display().to_string()]);
126        }
127        if let Some(chain_id) = opt.chain_id {
128            command.args(["--chain-id", &chain_id.to_string()]);
129        }
130
131        tracing::info!("Starting Anvil: {:?}", &command);
132
133        let child = command.spawn().unwrap();
134
135        let url = Url::parse(&format!("http://localhost:{port}")).unwrap();
136        wait_for_rpc(&url, Duration::from_millis(200), 20)
137            .await
138            .unwrap();
139
140        (child, url)
141    }

166    /// Restart the server, possibly with different options.
167    pub async fn restart(&mut self, mut opt: AnvilOptions) {
168        // Stop the server and wait for it to dump its state.
169        self.shutdown_gracefully();
170
171        // If `opt` does not explicitly override the URL, use the current one.
172        if opt.port.is_none() {
173            opt.port = self.url.port();
174        }
175        // If `opt` does not explicitly override the chain ID, use the current one.
176        if opt.chain_id.is_none() {
177            opt.chain_id = self.opt.chain_id;
178        }
179
180        // Load state from the file where we just dumped state.
181        opt = opt.load_state(self.state_dir.path().join("state.json"));
182
183        // Restart the server with the new options, loading state from disk.
184        let (child, url) = Self::spawn_server(&opt, Some(self.state_dir.path())).await;
185        self.child = child;
186        self.url = url;
187        self.opt = opt;
188    }

190    /// Force a reorg in the L1.
191    ///
192    /// This function will block until the reorg has completed; that is, a block has been added to
193    /// the chain at the same height as a different, earlier block.
194    ///
195    /// Due to limitations in Anvil, the common ancestor of the reorg will always be the L1 genesis.
196    /// However, after the reorg, the genesis state of the EVM will be the same as the state when
197    /// this function was originally called. The recommended way to use this for testing reorg
198    /// handling, then, is to perform initialization (like deploying contracts) and then start some
199    /// service in the background and immediately call this function. This function will let the L1
200    /// chain run until it reaches block height at least `min_depth` and then reset it to block
201    /// height 0. Then it will let the L1 chain run again until it reaches block height `min_depth`.
202    ///
203    /// From the perspective of the service under test, this looks like an L1 chain with some
204    /// initial state (e.g. contracts deployed) and a block height which happens to be 0 but, as far
205    /// as the service cares, could be arbitrary. The block height grows somewhat before getting
206    /// reorged back to an earlier height (0) and then growing again.
207    ///
208    /// The L1 node will restart several times during this process.
209    pub async fn reorg(&mut self, min_depth: u64) {
210        // Stop the server and wait for it to dump its state, to obtain a snapshot of the current,
211        // pre-reorg state.
212        self.shutdown_gracefully();
213
214        // Ensure we restart on the same port and load the state snapshot.
215        let opt = self
216            .opt
217            .clone()
218            .port(self.url.port().unwrap())
219            .load_state(self.state_dir.path().into());
220
221        // Restart the server, loading state from disk but not dumping it on the next exit.
222        let (child, url) = Self::spawn_server(&opt, None).await;
223        self.child = child;
224        self.url = url;
225
226        // Wait for enough blocks to be produced.
227        let client = Provider::try_from(self.url.to_string()).unwrap();
228        while client.get_block_number().await.unwrap().as_u64() < min_depth {
229            sleep(Duration::from_secs(1)).await;
230        }
231
232        // Restart again, loading from the previous state checkpoint.
233        self.shutdown_gracefully();
234        let (child, url) = Self::spawn_server(&opt, Some(self.state_dir.path())).await;
235        self.child = child;
236        self.url = url;
237
238        // Wait for the chain to reach its former height again.
239        let client = Provider::try_from(self.url.to_string()).unwrap();
240        while client.get_block_number().await.unwrap().as_u64() < min_depth {
241            sleep(Duration::from_secs(1)).await;
242        }
243    }
244}

252/// Prepare a `SignerMiddleware` by connecting to a provider and initiating a local wallet.
253/// Returns a signer/client that can sign and send transactions to network.
254pub async fn init_signer(provider: &Url, mnemonic: &str, index: u32) -> Option<Signer> {
255    let provider = match Provider::try_from(provider.to_string()) {
256        Ok(provider) => provider,
257        Err(err) => {
258            tracing::error!("error connecting to RPC {}: {}", provider, err);
259            return None;
260        },
261    };
262    let chain_id = match provider.get_chainid().await {
263        Ok(id) => id.as_u64(),
264        Err(err) => {
265            tracing::error!("error getting chain ID: {}", err);
266            return None;
267        },
268    };
269    let mnemonic = match MnemonicBuilder::<English>::default()
270        .phrase(mnemonic)
271        .index(index)
272    {
273        Ok(mnemonic) => mnemonic,
274        Err(err) => {
275            tracing::error!("error building wallet: {}", err);
276            return None;
277        },
278    };
279    let wallet = match mnemonic.build() {
280        Ok(wallet) => wallet,
281        Err(err) => {
282            tracing::error!("error opening wallet: {}", err);
283            return None;
284        },
285    };
286    let wallet = wallet.with_chain_id(chain_id);
287    Some(SignerMiddleware::new(provider, wallet))
288}

290pub async fn wait_for_http(
291    url: &Url,
292    interval: Duration,
293    max_retries: usize,
294) -> Result<usize, String> {
295    for i in 0..(max_retries + 1) {
296        let res = surf::get(url).await;
297        if res.is_ok() {
298            tracing::debug!("Connected to {url}");
299            return Ok(i);
300        }
301        tracing::debug!("Waiting for {url}, retrying in {interval:?}");
302        sleep(interval).await;
303    }
304    Err(format!("Url {url:?} not available."))
305}

307pub async fn wait_for_rpc(
308    url: &Url,
309    interval: Duration,
310    max_retries: usize,
311) -> Result<usize, String> {
312    let retries = wait_for_http(url, interval, max_retries).await?;
313    let client = Provider::new(Http::new(url.clone()));
314    for i in retries..(max_retries + 1) {
315        if client.get_block_number().await.is_ok() {
316            tracing::debug!("JSON-RPC ready at {url}");
317            return Ok(i);
318        }
319        tracing::debug!("Waiting for JSON-RPC at {url}, retrying in {interval:?}");
320        sleep(interval).await;
321    }
322
323    Err(format!("No JSON-RPC at {url}"))
324}

326/// converting a keccak256-based structured commitment (32 bytes) into type `U256`
327pub fn commitment_to_u256<T: Committable>(comm: Commitment<T>) -> U256 {
328    let mut buf = vec![];
329    comm.serialize_uncompressed(&mut buf).unwrap();
330    let state_comm: [u8; 32] = buf.try_into().unwrap();
331    U256::from_little_endian(&state_comm)
332}
333
334/// converting a `U256` value into a keccak256-based structured commitment (32 bytes)
335pub fn u256_to_commitment<T: Committable>(comm: U256) -> Result<Commitment<T>, SerializationError> {
336    let mut commit_bytes = [0; 32];
337    comm.to_little_endian(&mut commit_bytes);
338    Commitment::deserialize_uncompressed_unchecked(&*commit_bytes.to_vec())
339}

355/// send a transaction and wait for confirmation before returning the tx receipt and block included.
356pub async fn contract_send<M: Middleware, T: Detokenize, E>(
357    call: &ContractCall<M, T>,
358) -> Result<(TransactionReceipt, u64), anyhow::Error>
359where
360    M::Provider: Clone,
361    E: ContractRevert + Debug,
362{
363    let pending = match call.send().await {
364        Ok(pending) => pending,
365        Err(err) => {
366            if let Some(e) = err.decode_contract_revert::<E>() {
367                tracing::error!("contract revert: {:?}", e);
368            }
369            return Err(anyhow!("error sending transaction: {:?}", err));
370        },
371    };
372
373    let hash = pending.tx_hash();
374    let provider = pending.provider();
375    tracing::debug!("submitted contract call {:x}", hash);
376
377    if !wait_for_transaction_to_be_mined(&provider, hash).await {
378        return Err(anyhow!("transaction not mined"));
379    }
380
381    let receipt = match provider.get_transaction_receipt(hash).await {
382        Ok(Some(receipt)) => receipt,
383        Ok(None) => {
384            return Err(anyhow!("contract call {hash:x}: no receipt"));
385        },
386        Err(err) => {
387            return Err(anyhow!(
388                "contract call {hash:x}: error getting transaction receipt: {err}"
389            ))
390        },
391    };
392    if receipt.status != Some(1.into()) {
393        return Err(anyhow!("contract call {hash:x}: transaction reverted"));
394    }
395
396    // If a transaction is mined and we get a receipt for it, the block number should _always_ be
397    // set. If it is not, something has gone horribly wrong with the RPC.
398    let block_number = receipt
399        .block_number
400        .expect("transaction mined but block number not set");
401    Ok((receipt, block_number.as_u64()))
402}

404async fn wait_for_transaction_to_be_mined<P: JsonRpcClient>(
405    provider: &Provider<P>,
406    hash: H256,
407) -> bool {
408    let retries = 10;
409    // It's common to have to try a few times before the transactions is mined. It is too noisy if
410    // we log every retry. However, if it is not mined after several retries, something might be
411    // wrong, and it would be useful to start logging.
412    let log_retries = 5;
413
414    let interval = provider.get_interval();
415    for i in 0..retries {
416        match provider.get_transaction(hash).await {
417            Err(err) => {
418                if i >= log_retries {
419                    tracing::warn!("contract call {hash:?} (retry {i}/{retries}): error getting transaction status: {err}");
420                }
421            },
422            Ok(None) => {
423                if i >= log_retries {
424                    tracing::warn!(
425                        "contract call {hash:?} (retry {i}/{retries}): missing from mempool"
426                    );
427                }
428            },
429            Ok(Some(tx)) if tx.block_number.is_none() => {
430                if i >= log_retries {
431                    tracing::warn!("contract call {hash:?} (retry {i}/{retries}): pending");
432                }
433            },
434            Ok(Some(_)) => return true,
435        }
436
437        sleep(interval).await;
438    }
439
440    tracing::error!("contract call {hash:?}: not mined after {retries} retries");
441    false
442}
