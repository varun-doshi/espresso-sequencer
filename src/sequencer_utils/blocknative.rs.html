16/// A client over HTTP for the [BlockNative](https://www.blocknative.com/gas-estimator) gas tracker API
17/// that implements the `GasOracle` trait.
18#[derive(Clone, Debug)]
19#[must_use]
20pub struct BlockNative {

22    url: Url,
23    api_key: Option<String>,
24    gas_category: GasCategory,

63impl Response {
64    #[inline]
65    pub fn estimate_from_category(&self, gas_category: &GasCategory) -> Result<GasEstimate> {
66        let confidence = gas_category_to_confidence(gas_category);
67        let price = self
68            .block_prices
69            .first()
70            .ok_or(GasOracleError::InvalidResponse)?
71            .estimated_prices
72            .iter()
73            .find(|p| p.confidence == confidence)
74            .ok_or(GasOracleError::GasCategoryNotSupported)?;
75        Ok(*price)
76    }
77}

85#[cfg_attr(target_arch = "wasm32", async_trait(?Send))]
86#[cfg_attr(not(target_arch = "wasm32"), async_trait)]
87impl GasOracle for BlockNative {
88    async fn fetch(&self) -> Result<U256> {
89        let estimate = self
90            .query()
91            .await?
92            .estimate_from_category(&self.gas_category)?;
93        Ok(from_gwei_f64(estimate.price))
94    }
95
96    async fn estimate_eip1559_fees(&self) -> Result<(U256, U256)> {
97        let estimate = self
98            .query()
99            .await?
100            .estimate_from_category(&self.gas_category)?;
101        let max = from_gwei_f64(estimate.max_fee_per_gas);
102        let prio = from_gwei_f64(estimate.max_priority_fee_per_gas);
103        Ok((max, prio))
104    }
105}

107impl BlockNative {
108    /// Creates a new [BlockNative](https://www.blocknative.com/gas-estimator) gas oracle.
109    pub fn new(api_key: Option<String>) -> Self {
110        Self::with_client(Client::new(), api_key)
111    }
112
113    /// Same as [`Self::new`] but with a custom [`Client`].
114    pub fn with_client(client: Client, api_key: Option<String>) -> Self {
115        let url = Url::parse(URL).unwrap();
116        Self {
117            client,
118            api_key,
119            url,
120            gas_category: GasCategory::Standard,
121        }
122    }
123
124    /// Sets the gas price category to be used when fetching the gas price.
125    pub fn category(mut self, gas_category: GasCategory) -> Self {
126        self.gas_category = gas_category;
127        self
128    }
129
130    /// Perform a request to the gas price API and deserialize the response.
131    pub async fn query(&self) -> Result<Response, GasOracleError> {
132        let mut request = self.client.get(self.url.clone());
133        if let Some(api_key) = self.api_key.as_ref() {
134            request = request.header(AUTHORIZATION, api_key);
135        }
136        let response = request.send().await?.error_for_status()?.json().await?;
137        Ok(response)
138    }
139}

141#[inline]
142fn gas_category_to_confidence(gas_category: &GasCategory) -> u64 {
143    match gas_category {
144        GasCategory::SafeLow => 80,
145        GasCategory::Standard => 90,
146        GasCategory::Fast => 95,
147        GasCategory::Fastest => 99,
148    }
