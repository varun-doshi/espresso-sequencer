47/// [SubmitPublicUrlsToScrapeTask] is a task that is capable of submitting
48/// public urls to a url sender at a regular interval.  This task will
49/// submit the provided urls to the url sender every 5 minutes.
50pub struct SubmitPublicUrlsToScrapeTask {
51    pub task_handle: Option<JoinHandle<()>>,
52}
53
54const PUBLIC_URL_RESUBMIT_INTERVAL: Duration = Duration::from_secs(300);
55
56impl SubmitPublicUrlsToScrapeTask {
57    pub fn new<S>(url_sender: S, urls: Vec<Url>) -> Self
58    where
59        S: Sink<Url, Error = SendError> + Send + Unpin + 'static,
60    {
61        let task_handle = spawn(Self::submit_urls(url_sender, urls));
62
63        Self {
64            task_handle: Some(task_handle),
65        }
66    }
67
68    pub async fn submit_urls<S>(url_sender: S, urls: Vec<Url>)
69    where
70        S: Sink<Url, Error = SendError> + Unpin + 'static,
71    {
72        if urls.is_empty() {
73            tracing::warn!("no urls to send to url sender");
74            return;
75        }
76
77        let mut url_sender = url_sender;
78        tracing::debug!("sending initial urls to url sender to process node identity");
79        loop {
80            for url in urls.iter() {
81                let send_result = url_sender.send(url.clone()).await;
82                if let Err(err) = send_result {
83                    tracing::error!("url sender closed: {}", err);
84                    panic!("SubmitPublicUrlsToScrapeTask url sender is closed, unrecoverable, the node state will stagnate.");
85                }
86            }
87
88            // Sleep for 5 minutes before sending the urls again
89            tokio::time::sleep(PUBLIC_URL_RESUBMIT_INTERVAL).await;
90        }
91    }
92}

94/**
95 * create_node_validator_processing is a function that creates a node validator
96 * processing environment.  This function will create a number of tasks that
97 * will be responsible for processing the data streams that are coming in from
98 * the various sources.  This function will also create the data state that
99 * will be used to store the state of the network.
100 */
101pub async fn create_node_validator_processing(
102    config: NodeValidatorConfig,
103    internal_client_message_receiver: Receiver<InternalClientMessage<Sender<ServerMessage>>>,
104    leaf_and_block_pair_receiver: Receiver<LeafAndBlock<SeqTypes>>,
105) -> Result<NodeValidatorAPI<Sender<Url>>, CreateNodeValidatorProcessingError> {
106    let client_thread_state = ClientThreadState::<Sender<ServerMessage>>::new(
107        Default::default(),
108        Default::default(),
109        Default::default(),
110        Default::default(),
111        ClientId::from_count(1),
112    );
113
114    let client_stake_table = surf_disco::Client::new(config.stake_table_url_base.clone());
115
116    let stake_table = get_stake_table_from_sequencer(client_stake_table)
117        .await
118        .map_err(CreateNodeValidatorProcessingError::FailedToGetStakeTable)?;
119
120    let data_state = DataState::new(Default::default(), Default::default(), stake_table);
121
122    let data_state = Arc::new(RwLock::new(data_state));
123    let client_thread_state = Arc::new(RwLock::new(client_thread_state));
124    let (block_detail_sender, block_detail_receiver) = mpsc::channel(32);
125    let (node_identity_sender_1, node_identity_receiver_1) = mpsc::channel(32);
126    let (node_identity_sender_2, node_identity_receiver_2) = mpsc::channel(32);
127    let (voters_sender, voters_receiver) = mpsc::channel(32);
128    let (url_sender, url_receiver) = mpsc::channel(32);
129
130    let process_internal_client_message_handle = InternalClientMessageProcessingTask::new(
131        internal_client_message_receiver,
132        data_state.clone(),
133        client_thread_state.clone(),
134    );
135
136    let process_distribute_block_detail_handle = ProcessDistributeBlockDetailHandlingTask::new(
137        client_thread_state.clone(),
138        block_detail_receiver,
139    );
140
141    let process_distribute_node_identity_handle = ProcessDistributeNodeIdentityHandlingTask::new(
142        client_thread_state.clone(),
143        node_identity_receiver_2,
144    );
145
146    let process_distribute_voters_handle =
147        ProcessDistributeVotersHandlingTask::new(client_thread_state.clone(), voters_receiver);
148
149    let process_leaf_stream_handle = ProcessLeafAndBlockPairStreamTask::new(
150        leaf_and_block_pair_receiver,
151        data_state.clone(),
152        block_detail_sender,
153        voters_sender,
154    );
155
156    let process_node_identity_stream_handle = ProcessNodeIdentityStreamTask::new(
157        node_identity_receiver_1,
158        data_state.clone(),
159        node_identity_sender_2,
160    );
161
162    let process_url_stream_handle =
163        ProcessNodeIdentityUrlStreamTask::new(url_receiver, node_identity_sender_1);
164
165    // Send any initial URLS to the url sender for immediate processing.
166    // These urls are supplied by the configuration of this function
167    let submit_public_urls_handle = SubmitPublicUrlsToScrapeTask::new(
168        url_sender.clone(),
169        config.initial_node_public_base_urls.clone(),
170    );
171
172    Ok(NodeValidatorAPI {
173        process_internal_client_message_handle: Some(process_internal_client_message_handle),
174        process_distribute_block_detail_handle: Some(process_distribute_block_detail_handle),
175        process_distribute_node_identity_handle: Some(process_distribute_node_identity_handle),
176        process_distribute_voters_handle: Some(process_distribute_voters_handle),
177        process_leaf_stream_handle: Some(process_leaf_stream_handle),
178        process_node_identity_stream_handle: Some(process_node_identity_stream_handle),
179        process_url_stream_handle: Some(process_url_stream_handle),
180        submit_public_urls_handle: Some(submit_public_urls_handle),
181        url_sender,
182    })
183}
