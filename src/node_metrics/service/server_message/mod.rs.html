10/// [ServerMessage] represents the messages that the server can send to the
11/// client for a response.
12#[derive(Debug, Serialize, Deserialize)]
13pub enum ServerMessage {
14    /// This allows the client to know what client_id they have been assigned
15    YouAre(ClientId),
16
17    /// LatestBlock is a message that is meant to show the most recent block
18    /// that has arrived.
19    LatestBlock(Arc<BlockDetail<SeqTypes>>),
20
21    /// LatestNodeIdentity is a message that is meant to show the most recent
22    /// node identity that has arrived.
23    LatestNodeIdentity(Arc<NodeIdentity>),
24
25    /// LatestVoters is a message that is meant to show the most recent
26    /// voters that have arrived.
27    LatestVoters(BitVec<u16>),
28
29    /// BlocksSnapshot is a message that is sent in response to a request for
30    /// the snapshot of block information that is available.
31    BlocksSnapshot(Arc<Vec<BlockDetail<SeqTypes>>>),
32
33    /// NodeIdentitySnapshot is a message that is sent in response to a request
34    /// for the snapshot of the current node identity information.
35    NodeIdentitySnapshot(Arc<Vec<NodeIdentity>>),
36
37    /// HistogramSnapshot is a message that is sent in response to a request
38    /// for the snapshot of the current histogram information.
39    HistogramSnapshot(Arc<ExplorerHistograms>),
40
41    /// VotersSnapshot is a message that is sent in response to a request for
42    /// the snapshot of the current voters information.
43    VotersSnapshot(Arc<Vec<BitVec<u16>>>),
44}
45
46impl PartialEq for ServerMessage {
47    fn eq(&self, other: &Self) -> bool {
48        match (self, other) {
49            (Self::YouAre(lhs), Self::YouAre(rhg)) => lhs == rhg,
50            (Self::LatestBlock(lhs), Self::LatestBlock(rhs)) => lhs == rhs,
51            (Self::LatestNodeIdentity(lhs), Self::LatestNodeIdentity(rhs)) => lhs == rhs,
52            (Self::LatestVoters(lhs), Self::LatestVoters(rhs)) => lhs == rhs,
53            (Self::BlocksSnapshot(lhs), Self::BlocksSnapshot(rhs)) => lhs == rhs,
54            (Self::NodeIdentitySnapshot(lhs), Self::NodeIdentitySnapshot(rhs)) => lhs == rhs,
55            (Self::HistogramSnapshot(_), Self::HistogramSnapshot(_)) => false,
56            (Self::VotersSnapshot(lhs), Self::VotersSnapshot(rhs)) => lhs == rhs,
57            _ => false,
58        }
59    }
60}
