7//! Trait for stake table data structures
8
9use ark_std::{rand::SeedableRng, vec::Vec};
10use digest::crypto_common::rand_core::CryptoRngCore;
11use displaydoc::Display;
12
13/// Snapshots of the stake table
14pub enum SnapshotVersion {
15    /// the latest "Head" where all new changes are applied to
16    Head,
17    /// marks the snapshot at the beginning of the current epoch
18    EpochStart,
19    /// marks the beginning of the last epoch
20    LastEpochStart,
21    /// at arbitrary block height
22    BlockNum(u64),
23}
24
25/// Common interfaces required for a stake table used in `HotShot` System.
26/// APIs that doesn't take `version: SnapshotVersion` as an input by default works on the head/latest version.
27pub trait StakeTableScheme {
28    /// type for stake key
29    type Key: Clone;
30    /// type for the staked amount
31    type Amount: Clone + Copy;
32    /// type for the commitment to the current stake table
33    type Commitment;
34    /// type for the proof associated with the lookup result (if any)
35    type LookupProof;
36    /// type for the iterator over (key, value) entries
37    type IntoIter: Iterator<Item = (Self::Key, Self::Amount, Self::Aux)>;
38    /// Auxiliary information associated with the key
39    type Aux: Clone;
40
41    /// Register a new key into the stake table.
42    ///
43    /// # Errors
44    ///
45    /// Return err if key is already registered.
46    fn register(
47        &mut self,
48        new_key: Self::Key,
49        amount: Self::Amount,
50        aux: Self::Aux,
51    ) -> Result<(), StakeTableError>;
52
53    /// Batch register a list of new keys. A default implementation is provided
54    /// w/o batch optimization.
55    ///
56    /// # Errors
57    ///
58    /// Return err if any of `new_keys` fails to register.
59    fn batch_register<I, J, K>(
60        &mut self,
61        new_keys: I,
62        amounts: J,
63        auxs: K,
64    ) -> Result<(), StakeTableError>
65    where
66        I: IntoIterator<Item = Self::Key>,
67        J: IntoIterator<Item = Self::Amount>,
68        K: IntoIterator<Item = Self::Aux>,
69    {
70        let _ = new_keys
71            .into_iter()
72            .zip(amounts)
73            .zip(auxs)
74            .try_for_each(|((key, amount), aux)| Self::register(self, key, amount, aux));
75        Ok(())
76    }
77
78    /// Deregister an existing key from the stake table.
79    /// Returns error if some keys are not found.
80    ///
81    /// # Errors
82    /// Return err if `existing_key` wasn't registered.
83    fn deregister(&mut self, existing_key: &Self::Key) -> Result<(), StakeTableError>;
84
85    /// Batch deregister a list of keys. A default implementation is provided
86    /// w/o batch optimization.
87    ///
88    /// # Errors
89    /// Return err if any of `existing_keys` fail to deregister.
90    fn batch_deregister<'a, I>(&mut self, existing_keys: I) -> Result<(), StakeTableError>
91    where
92        I: IntoIterator<Item = &'a <Self as StakeTableScheme>::Key>,
93        <Self as StakeTableScheme>::Key: 'a,
94    {
95        let _ = existing_keys
96            .into_iter()
97            .try_for_each(|key| Self::deregister(self, key));
98        Ok(())
99    }
100
101    /// Returns the commitment to the `version` of stake table.
102    ///
103    /// # Errors
104    /// Return err if the `version` is not supported.
105    fn commitment(&self, version: SnapshotVersion) -> Result<Self::Commitment, StakeTableError>;
106
107    /// Returns the accumulated stakes of all registered keys of the `version`
108    /// of stake table.
109    ///
110    /// # Errors
111    /// Return err if the `version` is not supported.
112    fn total_stake(&self, version: SnapshotVersion) -> Result<Self::Amount, StakeTableError>;
113
114    /// Returns the number of keys in the `version` of the table.
115    ///
116    /// # Errors
117    /// Return err if the `version` is not supported.
118    fn len(&self, version: SnapshotVersion) -> Result<usize, StakeTableError>;
119
120    /// Returns true if `key` is currently registered, else returns false.
121    fn contains_key(&self, key: &Self::Key) -> bool;
122
123    /// Returns the stakes withhelded by a public key.
124    ///
125    /// # Errors
126    /// Return err if the `version` is not supported or `key` doesn't exist.
127    fn lookup(
128        &self,
129        version: SnapshotVersion,
130        key: &Self::Key,
131    ) -> Result<Self::Amount, StakeTableError>;
132
133    /// Returns the stakes withhelded by a public key along with a membership proof.
134    ///
135    /// # Errors
136    /// Return err if the `version` is not supported or `key` doesn't exist.
137    fn lookup_with_proof(
138        &self,
139        version: SnapshotVersion,
140        key: &Self::Key,
141    ) -> Result<(Self::Amount, Self::LookupProof), StakeTableError>;
142
143    /// Return the associated stake amount and auxiliary information of a public key,
144    /// along with a membership proof.
145    ///
146    /// # Errors
147    /// Return err if the `version` is not supported or `key` doesn't exist.
148    #[allow(clippy::type_complexity)]
149    fn lookup_with_aux_and_proof(
150        &self,
151        version: SnapshotVersion,
152        key: &Self::Key,
153    ) -> Result<(Self::Amount, Self::Aux, Self::LookupProof), StakeTableError>;
154
155    /// Update the stake of the `key` with `(negative ? -1 : 1) * delta`.
156    /// Return the updated stake or error.
157    ///
158    /// # Errors
159    /// Return err if the `key` doesn't exist of if the update overflow/underflow.
160    fn update(
161        &mut self,
162        key: &Self::Key,
163        delta: Self::Amount,
164        negative: bool,
165    ) -> Result<Self::Amount, StakeTableError>;
166
167    /// Batch update the stake balance of `keys`. Read documentation about
168    /// [`Self::update()`]. By default, we call `Self::update()` on each
169    /// (key, amount, negative) tuple.
170    ///
171    /// # Errors
172    /// Return err if any one of the `update` failed.
173    fn batch_update(
174        &mut self,
175        keys: &[Self::Key],
176        amounts: &[Self::Amount],
177        negative_flags: Vec<bool>,
178    ) -> Result<Vec<Self::Amount>, StakeTableError> {
179        let updated_amounts = keys
180            .iter()
181            .zip(amounts.iter())
182            .zip(negative_flags.iter())
183            .map(|((key, &amount), negative)| Self::update(self, key, amount, *negative))
184            .collect::<Result<Vec<_>, _>>()?;
185
186        Ok(updated_amounts)
187    }
188
189    /// Randomly sample a (key, stake amount) pair proportional to the stake distributions,
190    /// given a fixed seed for `rng`, this sampling should be deterministic.
191    fn sample(
192        &self,
193        rng: &mut (impl SeedableRng + CryptoRngCore),
194    ) -> Option<(&Self::Key, &Self::Amount)>;
195
196    /// Returns an iterator over all (key, value) entries of the `version` of the table
197    ///
198    /// # Errors
199    /// Return err if the `version` is not supported.
200    fn try_iter(&self, version: SnapshotVersion) -> Result<Self::IntoIter, StakeTableError>;
201}
202
203/// Error type for [`StakeTableScheme`]
204#[derive(Debug, Display)]
205pub enum StakeTableError {
206    /// Internal error caused by Rescue
207    RescueError,
208    /// Key mismatched
209    MismatchedKey,
210    /// Key not found
211    KeyNotFound,
212    /// Key already exists
213    ExistingKey,
214    /// Malformed Merkle proof
215    MalformedProof,
216    /// Verification Error
217    VerificationError,
218    /// Insufficient fund: the number of stake cannot be negative
219    InsufficientFund,
220    /// The number of stake exceed U256
221    StakeOverflow,
222    /// The historical snapshot requested is not supported.
223    SnapshotUnsupported,
224}
225
226impl ark_std::error::Error for StakeTableError {}
