74    /// Get a Membership for a given Epoch, which is guaranteed to have a randomized stake
75    /// table for the given Epoch
76    pub async fn membership_for_epoch(
77        &self,
78        maybe_epoch: Option<TYPES::Epoch>,
79    ) -> Result<EpochMembership<TYPES>> {
80        let ret_val = EpochMembership {
81            epoch: maybe_epoch,
82            coordinator: self.clone(),
83        };
84        let Some(epoch) = maybe_epoch else {
85            return Ok(ret_val);
86        };
87        if self
88            .membership
89            .read()
90            .await
91            .has_randomized_stake_table(epoch)
92        {
93            return Ok(ret_val);
94        }
95        if self.catchup_map.lock().await.contains_key(&epoch) {
96            return Err(warn!(
97                "Randomized stake table for epoch {:?} unavailable. Catchup already in progress",
98                epoch
99            ));
100        }
101        let coordinator = self.clone();
102        spawn_catchup(coordinator, epoch);
103
104        Err(warn!(
105            "Randomized stake table for epoch {:?} unavailable. Starting catchup",
106            epoch
107        ))
108    }

110    /// Get a Membership for a given Epoch, which is guaranteed to have a stake
111    /// table for the given Epoch
112    pub async fn stake_table_for_epoch(
113        &self,
114        maybe_epoch: Option<TYPES::Epoch>,
115    ) -> Result<EpochMembership<TYPES>> {
116        let ret_val = EpochMembership {
117            epoch: maybe_epoch,
118            coordinator: self.clone(),
119        };
120        let Some(epoch) = maybe_epoch else {
121            return Ok(ret_val);
122        };
123        if self.membership.read().await.has_stake_table(epoch) {
124            return Ok(ret_val);
125        }
126        if self.catchup_map.lock().await.contains_key(&epoch) {
127            return Err(warn!(
128                "Stake table for Epoch {:?} Unavailable. Catch up already in Progress",
129                epoch
130            ));
131        }
132        let coordinator = self.clone();
133        spawn_catchup(coordinator, epoch);
134
135        Err(warn!(
136            "Stake table for Epoch {:?} Unavailable. Starting catchup",
137            epoch
138        ))
139    }

141    /// Catches the membership up to the epoch passed as an argument.  
142    /// To do this try to get the stake table for the epoch containing this epoch's root
143    /// if the root does not exist recursively catchup until you've found it
144    ///
145    /// If there is another catchup in progress this will not duplicate efforts
146    /// e.g. if we start with only epoch 0 stake table and call catchup for epoch 10, then call catchup for epoch 20
147    /// the first caller will actually do the work for to catchup to epoch 10 then the second caller will continue
148    /// catching up to epoch 20
149    async fn catchup(self, epoch: TYPES::Epoch) -> Result<EpochMembership<TYPES>> {
150        // recursively catchup until we have a stake table for the epoch containing our root
151        ensure!(
152            *epoch != 0 && *epoch != 1,
153            "We are trying to catchup to epoch 0! This means the initial stake table is missing!"
154        );
155        let root_epoch = TYPES::Epoch::new(*epoch - 2);
156
157        let root_membership = if self.membership.read().await.has_stake_table(root_epoch) {
158            EpochMembership {
159                epoch: Some(root_epoch),
160                coordinator: self.clone(),
161            }
162        } else {
163            Box::pin(self.wait_for_catchup(root_epoch)).await?
164        };
165
166        // Get the epoch root headers and update our membership with them, finally sync them
167        // Verification of the root is handled in get_epoch_root_and_drb
168        let Ok(header) = root_membership
169            .get_epoch_root(root_block_in_epoch(*root_epoch, self.epoch_height))
170            .await
171        else {
172            anytrace::bail!("get epoch root failed for epoch {:?}", root_epoch);
173        };
174
175        let updater = self
176            .membership
177            .read()
178            .await
179            .add_epoch_root(epoch, header)
180            .await
181            .ok_or(anytrace::warn!("add epoch root failed"))?;
182        updater(&mut *(self.membership.write().await));
183
184        let drb_membership = match root_membership.next_epoch_stake_table().await {
185            Ok(drb_membership) => drb_membership,
186            Err(_) => Box::pin(self.wait_for_catchup(root_epoch + 1)).await?,
187        };
188
189        // get the DRB from the last block of the epoch right before the one we're catching up to
190        let Ok(drb) = drb_membership
191            .get_epoch_drb(transition_block_for_epoch(
192                *(root_epoch + 1),
193                self.epoch_height,
194            ))
195            .await
196        else {
197            return Err(anytrace::warn!(
198                "get epoch drb failed for in epoch {:?}",
199                root_epoch + 1
200            ));
201        };
202
203        self.membership.write().await.add_drb_result(epoch, drb);
204        Ok(EpochMembership {
205            epoch: Some(epoch),
206            coordinator: self.clone(),
207        })
208    }

210    pub async fn wait_for_catchup(&self, epoch: TYPES::Epoch) -> Result<EpochMembership<TYPES>> {
211        let Some(mut rx) = self
212            .catchup_map
213            .lock()
214            .await
215            .get(&epoch)
216            .map(InactiveReceiver::activate_cloned)
217        else {
218            return self.clone().catchup(epoch).await;
219        };
220        let Ok(Ok(mem)) = rx.recv_direct().await else {
221            return self.clone().catchup(epoch).await;
222        };
223        Ok(mem)
224    }
225}

227fn spawn_catchup<T: NodeType>(coordinator: EpochMembershipCoordinator<T>, epoch: T::Epoch) {
228    tokio::spawn(async move {
229        let tx = {
230            let mut map = coordinator.catchup_map.lock().await;
231            if map.contains_key(&epoch) {
232                return;
233            }
234            let (tx, rx) = broadcast(1);
235            map.insert(epoch, rx.deactivate());
236            tx
237        };
238        // do catchup
239        let ret = coordinator.catchup(epoch).await;
240        let _ = tx.broadcast_direct(ret).await;
241    });
242}
