51/// Compute the DRB result for the leader rotation.
52///
53/// This is to be started two epochs in advance and spawned in a non-blocking thread.
54///
55/// # Arguments
56/// * `drb_seed_input` - Serialized QC signature.
57#[must_use]
58pub fn compute_drb_result<TYPES: NodeType>(drb_seed_input: DrbSeedInput) -> DrbResult {
59    let mut hash = drb_seed_input.to_vec();
60    for _iter in 0..DIFFICULTY_LEVEL {
61        // TODO: This may be optimized to avoid memcopies after we bench the hash time.
62        // <https://github.com/EspressoSystems/HotShot/issues/3880>
63        hash = Sha256::digest(hash).to_vec();
64    }
65
66    // Convert the hash to the DRB result.
67    let mut drb_result = [0u8; 32];
68    drb_result.copy_from_slice(&hash);
69    drb_result
70}

72/// Seeds for DRB computation and computed results.
73#[derive(Clone, Debug)]
74pub struct DrbResults<TYPES: NodeType> {
75    /// Stored results from computations
76    pub results: BTreeMap<TYPES::Epoch, DrbResult>,
77}

79impl<TYPES: NodeType> DrbResults<TYPES> {
80    #[must_use]
81    /// Constructor with initial values for epochs 1 and 2.
82    pub fn new() -> Self {
83        Self {
84            results: BTreeMap::from([
85                (TYPES::Epoch::new(1), INITIAL_DRB_RESULT),
86                (TYPES::Epoch::new(2), INITIAL_DRB_RESULT),
87            ]),
88        }
89    }
90
91    pub fn store_result(&mut self, epoch: TYPES::Epoch, result: DrbResult) {
92        self.results.insert(epoch, result);
93    }
94
95    /// Garbage collects internal data structures
96    pub fn garbage_collect(&mut self, epoch: TYPES::Epoch) {
97        if epoch.u64() < KEEP_PREVIOUS_RESULT_COUNT {
98            return;
99        }
100
101        let retain_epoch = epoch - KEEP_PREVIOUS_RESULT_COUNT;
102        // N.B. x.split_off(y) returns the part of the map where key >= y
103
104        // Remove result entries older than EPOCH
105        self.results = self.results.split_off(&retain_epoch);
106    }
107}
108
109impl<TYPES: NodeType> Default for DrbResults<TYPES> {
110    fn default() -> Self {
111        Self::new()
112    }
113}

115/// Functions for leader selection based on the DRB.
116///
117/// The algorithm we use is:
118///
119/// Initialization:
120/// - obtain `drb: [u8; 32]` from the DRB calculation
121/// - sort the stake table for a given epoch by `xor(drb, public_key)`
122/// - generate a cdf of the cumulative stake using this newly-sorted table,
123///   along with a hash of the stake table entries
124///
125/// Selecting a leader:
126/// - calculate the SHA512 hash of the `drb_result`, `view_number` and `stake_table_hash`
127/// - find the first index in the cdf for which the remainder of this hash modulo the `total_stake`
128///   is strictly smaller than the cdf entry
129/// - return the corresponding node as the leader for that view
130pub mod election {
131    use primitive_types::{U256, U512};
132    use sha2::{Digest, Sha256, Sha512};
133
134    use crate::traits::signature_key::{SignatureKey, StakeTableEntryType};
135
136    /// Calculate `xor(drb.cycle(), public_key)`, returning the result as a vector of bytes
137    fn cyclic_xor(drb: [u8; 32], public_key: Vec<u8>) -> Vec<u8> {
138        let drb: Vec<u8> = drb.to_vec();
139
140        let mut result: Vec<u8> = vec![];
141
142        for (drb_byte, public_key_byte) in public_key.iter().zip(drb.iter().cycle()) {
143            result.push(drb_byte ^ public_key_byte);
144        }
145
146        result
147    }
148
149    /// Generate the stake table CDF, as well as a hash of the resulting stake table
150    pub fn generate_stake_cdf<Key: SignatureKey, Entry: StakeTableEntryType<Key>>(
151        mut stake_table: Vec<Entry>,
152        drb: [u8; 32],
153    ) -> RandomizedCommittee<Entry> {
154        // sort by xor(public_key, drb_result)
155        stake_table.sort_by(|a, b| {
156            cyclic_xor(drb, a.public_key().to_bytes())
157                .cmp(&cyclic_xor(drb, b.public_key().to_bytes()))
158        });
159
160        let mut hasher = Sha256::new();
161
162        let mut cumulative_stake = U256::from(0);
163        let mut cdf = vec![];
164
165        for entry in stake_table {
166            cumulative_stake += entry.stake();
167            hasher.update(entry.public_key().to_bytes());
168
169            cdf.push((entry, cumulative_stake));
170        }
171
172        RandomizedCommittee {
173            cdf,
174            stake_table_hash: hasher.finalize().into(),
175            drb,
176        }
177    }
178
179    /// select the leader for a view
180    ///
181    /// # Panics
182    /// Panics if `cdf` is empty. Results in undefined behaviour if `cdf` is not ordered.
183    ///
184    /// Note that we try to downcast a U512 to a U256,
185    /// but this should never panic because the U512 should be strictly smaller than U256::MAX by construction.
186    pub fn select_randomized_leader<
187        SignatureKey,
188        Entry: StakeTableEntryType<SignatureKey> + Clone,
189    >(
190        randomized_committee: &RandomizedCommittee<Entry>,
191        view: u64,
192    ) -> Entry {
193        let RandomizedCommittee {
194            cdf,
195            stake_table_hash,
196            drb,
197        } = randomized_committee;
198        // We hash the concatenated drb, view and stake table hash.
199        let mut hasher = Sha512::new();
200        hasher.update(drb);
201        hasher.update(view.to_le_bytes());
202        hasher.update(stake_table_hash);
203        let raw_breakpoint: [u8; 64] = hasher.finalize().into();
204
205        // then calculate the remainder modulo the total stake as a U512
206        let remainder: U512 =
207            U512::from_little_endian(&raw_breakpoint) % U512::from(cdf.last().unwrap().1);
208
209        // and drop the top 32 bytes, downcasting to a U256
210        let breakpoint: U256 = U256::try_from(remainder).unwrap();
211
212        // now find the first index where the breakpoint is strictly smaller than the cdf
213        //
214        // in principle, this may result in an index larger than `cdf.len()`.
215        // however, we have ensured by construction that `breakpoint < total_stake`
216        // and so the largest index we can actually return is `cdf.len() - 1`
217        let index = cdf.partition_point(|(_, cumulative_stake)| breakpoint >= *cumulative_stake);
218
219        // and return the corresponding entry
220        cdf[index].0.clone()
221    }
222
223    #[derive(Clone, Debug)]
224    pub struct RandomizedCommittee<Entry> {
225        /// cdf of nodes by cumulative stake
226        cdf: Vec<(Entry, U256)>,
227        /// Hash of the stake table
228        stake_table_hash: [u8; 32],
229        /// DRB result
230        drb: [u8; 32],
231    }
