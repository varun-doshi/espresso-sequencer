22/// Holds configuration for a `HotShot`
23#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
24#[serde(bound(deserialize = ""))]
25pub struct HotShotConfigFile<TYPES: NodeType> {
26    /// The proportion of nodes required before the orchestrator issues the ready signal,
27    /// expressed as (numerator, denominator)
28    pub start_threshold: (u64, u64),
29    /// Total number of staked nodes in the network
30    pub num_nodes_with_stake: NonZeroUsize,
31    #[serde(skip)]
32    /// The known nodes' public key and stake value
33    pub known_nodes_with_stake: Vec<PeerConfig<TYPES>>,
34    #[serde(skip)]
35    /// The known DA nodes' public key and stake values
36    pub known_da_nodes: Vec<PeerConfig<TYPES>>,
37    /// Number of staking DA nodes
38    pub staked_da_nodes: usize,
39    /// Number of fixed leaders for GPU VID
40    pub fixed_leader_for_gpuvid: usize,
41    /// Base duration for next-view timeout, in milliseconds
42    pub next_view_timeout: u64,
43    /// Duration for view sync round timeout
44    pub view_sync_timeout: Duration,
45    /// Number of network bootstrap nodes
46    pub num_bootstrap: usize,
47    /// The maximum amount of time a leader can wait to get a block from a builder
48    pub builder_timeout: Duration,
49    /// Time to wait until we request data associated with a proposal
50    pub data_request_delay: Option<Duration>,
51    /// Builder API base URL
52    #[serde(default = "default_builder_urls")]
53    pub builder_urls: Vec1<Url>,
54    /// Upgrade config
55    pub upgrade: UpgradeConfig,
56    /// Number of blocks in an epoch, zero means there are no epochs
57    pub epoch_height: u64,
58    /// Epoch start block
59    pub epoch_start_block: u64,
60}

62impl<TYPES: NodeType> From<HotShotConfigFile<TYPES>> for HotShotConfig<TYPES> {
63    fn from(val: HotShotConfigFile<TYPES>) -> Self {
64        HotShotConfig {
65            start_threshold: val.start_threshold,
66            num_nodes_with_stake: val.num_nodes_with_stake,
67            known_da_nodes: val.known_da_nodes,
68            known_nodes_with_stake: val.known_nodes_with_stake,
69            da_staked_committee_size: val.staked_da_nodes,
70            fixed_leader_for_gpuvid: val.fixed_leader_for_gpuvid,
71            next_view_timeout: val.next_view_timeout,
72            view_sync_timeout: val.view_sync_timeout,
73            num_bootstrap: val.num_bootstrap,
74            builder_timeout: val.builder_timeout,
75            data_request_delay: val
76                .data_request_delay
77                .unwrap_or(Duration::from_millis(REQUEST_DATA_DELAY)),
78            builder_urls: val.builder_urls,
79            start_proposing_view: val.upgrade.start_proposing_view,
80            stop_proposing_view: val.upgrade.stop_proposing_view,
81            start_voting_view: val.upgrade.start_voting_view,
82            stop_voting_view: val.upgrade.stop_voting_view,
83            start_proposing_time: val.upgrade.start_proposing_time,
84            stop_proposing_time: val.upgrade.stop_proposing_time,
85            start_voting_time: val.upgrade.start_voting_time,
86            stop_voting_time: val.upgrade.stop_voting_time,
87            epoch_height: val.epoch_height,
88            epoch_start_block: val.epoch_start_block,
89        }
90    }
91}

93impl<TYPES: NodeType> HotShotConfigFile<TYPES> {
94    /// Creates a new `HotShotConfigFile` with 5 nodes and 10 DA nodes.
95    ///
96    /// # Panics
97    ///
98    /// Cannot panic, but will if `NonZeroUsize` is somehow an error.
99    #[must_use]
100    pub fn hotshot_config_5_nodes_10_da() -> Self {
101        let staked_da_nodes: usize = 5;
102
103        let mut known_da_nodes = Vec::new();
104
105        let gen_known_nodes_with_stake = (0..10)
106            .map(|node_id| {
107                let mut cur_validator_config: ValidatorConfig<TYPES> =
108                    ValidatorConfig::generated_from_seed_indexed([0u8; 32], node_id, 1, false);
109
110                if node_id < staked_da_nodes as u64 {
111                    known_da_nodes.push(cur_validator_config.public_config());
112                    cur_validator_config.is_da = true;
113                }
114
115                cur_validator_config.public_config()
116            })
117            .collect();
118
119        Self {
120            num_nodes_with_stake: NonZeroUsize::new(10).unwrap(),
121            start_threshold: (1, 1),
122            known_nodes_with_stake: gen_known_nodes_with_stake,
123            staked_da_nodes,
124            known_da_nodes,
125            fixed_leader_for_gpuvid: 1,
126            next_view_timeout: 10000,
127            view_sync_timeout: Duration::from_millis(1000),
128            num_bootstrap: 5,
129            builder_timeout: Duration::from_secs(10),
130            data_request_delay: Some(Duration::from_millis(REQUEST_DATA_DELAY)),
131            builder_urls: default_builder_urls(),
132            upgrade: UpgradeConfig::default(),
133            epoch_height: 0,
134            epoch_start_block: 0,
135        }
136    }
137}
