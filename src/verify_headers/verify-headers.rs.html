16/// Utility program to verify properties of headers sequenced by HotShot.
17#[derive(Clone, Debug, Parser)]
18struct Options {
19    /// Start counting from block FROM.
20    #[clap(long, name = "FROM")]
21    from: Option<usize>,
22
23    /// Stop counting at block TO.
24    #[clap(long, name = "TO")]
25    to: Option<usize>,
26
27    /// Skip verifying header timestamps.
28    #[clap(long)]
29    no_timestamps: bool,
30
31    /// Skip verifying L1 head numbers.
32    #[clap(long)]
33    no_l1_heads: bool,
34
35    /// Skip verifying L1 finalized references.
36    #[clap(long)]
37    no_l1_finalized: bool,
38
39    /// L1 RPC URL
40    ///
41    /// This will be used to verify the hashes and timestamps of L1 finalized references. If not
42    /// provided, only block numbers will be verified.
43    #[clap(long)]
44    l1: Option<Url>,
45
46    /// Number of parallel tasks to run.
47    #[clap(short, long, default_value = "1")]
48    jobs: usize,
49
50    /// URL of the HotShot query service.
51    url: Url,
52
53    #[clap(flatten)]
54    logging: logging::Config,
55}
56
57type SequencerClient<ApiVer> = surf_disco::Client<hotshot_query_service::Error, ApiVer>;
58
59async fn verify_header<ApiVer: StaticVersionType>(
60    opt: &Options,
61    seq: &SequencerClient<ApiVer>,
62    l1: Option<&Provider<Http>>,
63    parent: Option<Header>,
64    height: usize,
65) -> (Header, bool) {
66    tracing::debug!("checking header {height}");
67    let header = get_header(seq, height).await;
68
69    // Check invariants relative to parent. If this is the genesis header, there is no parent and
70    // nothing to do.
71    if height == 0 {
72        return (header, true);
73    }
74
75    let parent = match parent {
76        Some(parent) => parent,
77        None => get_header(seq, height - 1).await,
78    };
79
80    let mut ok = true;
81
82    if !opt.no_timestamps && header.timestamp() < parent.timestamp() {
83        tracing::error!(
84            "header {height} has decreasing timestamp: {} -> {}",
85            parent.timestamp(),
86            header.timestamp()
87        );
88        ok = false;
89    }
90    if !opt.no_l1_heads && header.l1_head() < parent.l1_head() {
91        tracing::error!(
92            "header {height} has decreasing L1 head: {} -> {}",
93            parent.l1_head(),
94            header.l1_head()
95        );
96        ok = false;
97    }
98    if !opt.no_l1_finalized && header.l1_finalized() < parent.l1_finalized() {
99        tracing::error!(
100            "header {height} has decreasing L1 finalized: {:?} -> {:?}",
101            parent.l1_finalized(),
102            header.l1_finalized()
103        );
104        ok = false;
105
106        if let (Some(l1), Some(l1_finalized)) = (l1, &header.l1_finalized()) {
107            let l1_block = get_l1_block(l1, l1_finalized.number).await;
108            if *l1_finalized != l1_block {
109                tracing::error!(
110                    "header {height} has wrong L1 finalized info: {l1_finalized:?}, expected {l1_block:?}"
111                );
112                ok = false;
113            }
114        }
115    }
116
117    (header, ok)
118}

120async fn get_header<ApiVer: StaticVersionType>(
121    seq: &SequencerClient<ApiVer>,
122    height: usize,
123) -> Header {
124    loop {
125        match seq
126            .get(&format!("availability/header/{height}"))
127            .send()
128            .await
129        {
130            Ok(header) => break header,
131            Err(err) => {
132                tracing::warn!("error fetching header {height}: {err}");
133
134                // Back off a bit and then retry.
135                sleep(Duration::from_millis(100)).await;
136            },
137        }
138    }
139}
140
141async fn get_l1_block(l1: &Provider<Http>, height: u64) -> L1BlockInfo {
142    loop {
143        let block = match l1.get_block(height).await {
144            Ok(Some(block)) => block,
145            Ok(None) => {
146                tracing::warn!("L1 block {height} not yet available");
147                sleep(Duration::from_secs(1)).await;
148                continue;
149            },
150            Err(err) => {
151                tracing::warn!("error fetching L1 block {height}: {err}");
152                sleep(Duration::from_millis(100)).await;
153                continue;
154            },
155        };
156
157        let Some(hash) = block.hash else {
158            tracing::warn!("L1 block {height} has no hash, might not be finalized yet");
159            sleep(Duration::from_secs(1)).await;
160            continue;
161        };
162
163        return L1BlockInfo {
164            number: height,
165            hash,
166            timestamp: block.timestamp,
167        };
168    }
169}

171#[tokio::main]
172async fn main() {
173    let opt = Arc::new(Options::parse());
174    opt.logging.init();
175
176    let seq = Arc::new(SequencerClient::<SequencerApiVersion>::new(opt.url.clone()));
177
178    let block_height: usize = seq.get("status/latest_block_height").send().await.unwrap();
179    let from = opt.from.unwrap_or(0);
180    let to = max(opt.to.unwrap_or(block_height), from + 1);
181
182    tracing::info!("checking {} headers in [{from}, {to})", to - from);
183    let ok = join_all(
184        (from..to)
185            .chunks((to - from) / opt.jobs)
186            .into_iter()
187            .map(|heights| {
188                let opt = opt.clone();
189                let seq = seq.clone();
190                let l1 = opt
191                    .l1
192                    .as_ref()
193                    .map(|url| Provider::try_from(url.to_string()).unwrap());
194                async move {
195                    let mut ok = true;
196
197                    let mut parent = None;
198                    for height in heights {
199                        let (header, header_ok) =
200                            verify_header(&opt, &seq, l1.as_ref(), parent, height).await;
201                        ok &= header_ok;
202                        parent = Some(header);
203                    }
204
205                    ok
206                }
207            }),
208    )
209    .await
210    .into_iter()
211    .all(|ok| ok);
212
213    if ok {
214        tracing::info!("all headers in [{from}, {to}) ok")
215    } else {
216        tracing::error!("some headers were invalid");
217        exit(1);
218    }
219}
