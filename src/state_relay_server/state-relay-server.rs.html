1use clap::Parser;
2use ethers::types::U256;
3use hotshot_state_prover::service::one_honest_threshold;
4use sequencer::{state_signature::relay_server::run_relay_server, SequencerApiVersion};
5use sequencer_utils::logging;
6use vbs::version::StaticVersionType;
7
8#[derive(Parser)]
9struct Args {
10    /// Port to run the server on.
11    #[clap(
12        short,
13        long,
14        env = "ESPRESSO_STATE_RELAY_SERVER_PORT",
15        default_value = "8083"
16    )]
17    port: u16,
18
19    /// Total amount of stake.
20    /// WARNING: this is a temporary flag, should remove after integrating with stake table.
21    /// Related issue: [https://github.com/EspressoSystems/espresso-sequencer/issues/1022]
22    #[clap(
23        long,
24        env = "ESPRESSO_STATE_SIGNATURE_TOTAL_STAKE",
25        default_value = "5"
26    )]
27    total_stake: u64,
28
29    #[clap(flatten)]
30    logging: logging::Config,
31}
32
33#[tokio::main]
34async fn main() {
35    let args = Args::parse();
36    args.logging.init();
37
38    let threshold = one_honest_threshold(U256::from(args.total_stake));
39
40    tracing::info!(
41        port = args.port,
42        "starting state relay server, quorum threshold: {threshold}"
43    );
44    run_relay_server(
45        None,
46        threshold,
47        format!("http://0.0.0.0:{}", args.port).parse().unwrap(),
48        SequencerApiVersion::instance(),
49    )
50    .await
51    .unwrap();
52}
