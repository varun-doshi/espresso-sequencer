19#[derive(Clone, Copy, Debug, Display, Default, ValueEnum)]
20enum Scheme {
21    #[default]
22    #[display("all")]
23    All,
24    #[display("bls")]
25    Bls,
26    #[display("schnorr")]
27    Schnorr,
28}
29
30impl Scheme {
31    fn gen(self, seed: [u8; 32], index: u64, env_file: &mut impl Write) -> anyhow::Result<()> {
32        match self {
33            Self::All => {
34                Self::Bls.gen(seed, index, env_file)?;
35                Self::Schnorr.gen(seed, index, env_file)?;
36            },
37            Self::Bls => {
38                let (pub_key, priv_key) = BLSPubKey::generated_from_seed_indexed(seed, index);
39                let priv_key = priv_key.to_tagged_base64()?;
40                writeln!(env_file, "ESPRESSO_SEQUENCER_PUBLIC_STAKING_KEY={pub_key}")?;
41                writeln!(
42                    env_file,
43                    "ESPRESSO_SEQUENCER_PRIVATE_STAKING_KEY={priv_key}"
44                )?;
45                tracing::info!(%pub_key, "generated staking key")
46            },
47            Self::Schnorr => {
48                let key_pair = StateKeyPair::generate_from_seed_indexed(seed, index);
49                let priv_key = key_pair.sign_key_ref().to_tagged_base64()?;
50                writeln!(
51                    env_file,
52                    "ESPRESSO_SEQUENCER_PUBLIC_STATE_KEY={}",
53                    key_pair.ver_key()
54                )?;
55                writeln!(env_file, "ESPRESSO_SEQUENCER_PRIVATE_STATE_KEY={priv_key}")?;
56                tracing::info!(pub_key = %key_pair.ver_key(), "generated state key");
57            },
58        }
59        Ok(())
60    }
61}

63/// Utility program to generate keypairs
64///
65/// With no options, this program generates the keys needed to run a single instance of the Espresso
66/// sequencer. Options can be given to control the number or type of keys generated.
67///
68/// Generated secret keys are written to a file in .env format, which can directly be used to
69/// configure a sequencer node. Public information about the generated keys is printed to stdout.
70#[derive(Clone, Debug, Parser)]
71struct Options {
72    /// Seed for generating keys.
73    ///
74    /// If not provided, a random seed will be generated using system entropy.
75    #[clap(long, short = 's', value_parser = parse_seed)]
76    seed: Option<[u8; 32]>,
77
78    /// Signature scheme to generate.
79    ///
80    /// Sequencer nodes require both a BLS key (called the staking key) and a Schnorr key (called
81    /// the state key). By default, this program generates these keys in pairs, to make it easy to
82    /// configure sequencer nodes, but this option can be specified to generate keys for only one of
83    /// the signature schemes.
84    #[clap(long, default_value = "all")]
85    scheme: Scheme,
86
87    /// Number of setups to generate.
88    ///
89    /// Default is 1.
90    #[clap(long, short = 'n', name = "N", default_value = "1")]
91    num: usize,
92
93    /// Write private keys to .env files under DIR.
94    ///
95    /// DIR must be a directory. If it does not exist, one will be created. Private key setups will
96    /// be written to files immediately under DIR, with names like 0.env, 1.env, etc. for 0 through
97    /// N - 1. The random seed used to generate the keys will also be written to a file in DIR
98    /// called .seed.
99    #[clap(short, long, name = "OUT")]
100    out: PathBuf,
101
102    #[clap(flatten)]
103    logging: logging::Config,
104}
105
106fn parse_seed(s: &str) -> Result<[u8; 32], anyhow::Error> {
107    let bytes = hex::decode(s)?;
108    bytes
109        .try_into()
110        .map_err(|bytes: Vec<u8>| anyhow!("invalid seed length: {} (expected 32)", bytes.len()))
111}
112
113fn gen_default_seed() -> [u8; 32] {
114    let mut seed = [0u8; 32];
115    let mut rng = rand_chacha::ChaChaRng::from_entropy();
116    rng.fill_bytes(&mut seed);
117
118    seed
119}
120
121fn main() -> anyhow::Result<()> {
122    let opts = Options::parse();
123    opts.logging.init();
124
125    tracing::debug!(
126        "Generating {} keypairs with scheme {}",
127        opts.num,
128        opts.scheme
129    );
130
131    // Create output dir if necessary.
132    fs::create_dir_all(&opts.out)?;
133
134    let seed = opts.seed.unwrap_or_else(|| {
135        tracing::debug!("No seed provided, generating a random seed");
136        gen_default_seed()
137    });
138    fs::write(opts.out.join(".seed"), hex::encode(seed))?;
139
140    for index in 0..opts.num {
141        let span = info_span!("gen", index);
142        let _enter = span.enter();
143        tracing::info!("generating new key set");
144
145        let path = opts.out.join(format!("{index}.env"));
146        let mut file = File::options()
147            .write(true)
148            .create(true)
149            .truncate(true)
150            .open(&path)?;
151        opts.scheme.gen(seed, index as u64, &mut file)?;
152
153        tracing::info!("private keys written to {}", path.display());
154    }
155
156    Ok(())
157}
