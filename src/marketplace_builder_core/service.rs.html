48/// Configuration to initialize the builder
49#[derive(Debug, Clone)]
50pub struct BuilderConfig<Types: NodeType> {
51    /// Keys that this builder will use to sign responses
52    pub builder_keys: BuilderKeys<Types>,
53    /// Maximum time allotted for the builder to respond to an API call.
54    /// If the response isn't ready by this time, an error will be returned
55    /// to the caller.
56    pub api_timeout: Duration,
57    /// Time the builder will wait for new transactions before answering an
58    /// `available_blocks` API call if the builder doesn't have any transactions at the moment
59    /// of the call. Should be less than [`Self::api_timeout`]
60    pub tx_capture_timeout: Duration,
61    /// (Approximate) duration over which included transaction hashes will be stored
62    /// by the builder for deduplication of incoming transactions.
63    pub txn_garbage_collect_duration: Duration,
64    /// Channel capacity for incoming transactions for a single builder state.
65    pub txn_channel_capacity: usize,
66    /// Capacity of cache storing information for transaction status API
67    pub tx_status_cache_capacity: usize,
68    /// Base fee; the sequencing fee for a bundle is calculated as bundle size Ã— base fee
69    pub base_fee: u64,
70}

162    /// Spawns an event loop handling HotShot events from the provided stream.
163    /// Returns a handle for the spawned task.
164    pub fn start_event_loop(
165        &self,
166        event_stream: impl Stream<Item = Event<Types>> + Unpin + Send + 'static,
167    ) -> JoinHandle<anyhow::Result<()>> {

175    /// Internal implementation of the event loop, drives the underlying coordinator
176    /// and runs hooks
177    async fn event_loop(
178        coordinator: Arc<BuilderStateCoordinator<Types>>,
179        hooks: Arc<Hooks>,
180        mut event_stream: impl Stream<Item = Event<Types>> + Unpin + Send + 'static,
181    ) -> anyhow::Result<()> {
182        loop {
183            let Some(event) = event_stream.next().await else {
184                anyhow::bail!("Event stream ended");
185            };
186
187            hooks.handle_hotshot_event(&event).await;
188
189            match event.event {
190                EventType::Error { error } => {
191                    tracing::error!("Error event in HotShot: {:?}", error);
192                },
193                EventType::Transactions { transactions } => {
194                    let hooks = Arc::clone(&hooks);
195                    let coordinator = Arc::clone(&coordinator);
196                    spawn(async move {
197                        let transactions = hooks.process_transactions(transactions).await;
198
199                        let _ = transactions
200                            .into_iter()
201                            .map(|txn| {
202                                coordinator.handle_transaction(ReceivedTransaction::new(
203                                    txn,
204                                    TransactionSource::Public,
205                                ))
206                            })
207                            .collect::<FuturesUnordered<_>>()
208                            .collect::<Vec<_>>()
209                            .await;
210                    });
211                },
212                EventType::Decide { leaf_chain, .. } => {
213                    let coordinator = Arc::clone(&coordinator);
214                    spawn(async move { coordinator.handle_decide(leaf_chain).await });
215                },
216                EventType::DaProposal { proposal, .. } => {
217                    let coordinator = Arc::clone(&coordinator);
218                    spawn(async move { coordinator.handle_da_proposal(proposal.data).await });
219                },
220                EventType::QuorumProposal { proposal, .. } => {
221                    let coordinator = Arc::clone(&coordinator);
222                    spawn(async move { coordinator.handle_quorum_proposal(proposal.data).await });
223                },
224                _ => {},
225            }
226        }
227    }

229    /// Collect transactions to include in the bundle. Will wait until we have
230    /// at least one transaction or up to the configured `tx_capture_timeout` duration elapses.
231    #[tracing::instrument(skip_all, fields(builder_parent_block_references = %state.parent_block_references))]
232    async fn collect_transactions(
233        &self,
234        state: &Arc<BuilderState<Types>>,
235    ) -> Option<Vec<Types::Transaction>> {
236        // collect all the transactions from the near future
237        let timeout_after = Instant::now() + self.tx_capture_timeout;
238        let sleep_interval = self.tx_capture_timeout / 10;
239        while Instant::now() <= timeout_after {
240            let queue_populated = state.collect_txns(timeout_after).await;
241
242            if queue_populated || Instant::now() + sleep_interval > timeout_after {
243                // we don't have time for another iteration
244                break;
245            }
246
247            sleep(sleep_interval).await
248        }
249
250        let transactions = state
251            .txn_queue
252            .read()
253            .await
254            .iter()
255            .map(|txn| txn.transaction.clone())
256            .collect();
257
258        Some(transactions)
259    }

261    /// Assembles a [`Bundle`] for a certain view from a list of transactions by adding fee and signature
262    async fn assemble_bundle(
263        &self,
264        transactions: Vec<Types::Transaction>,
265        view_number: u64,
266    ) -> Result<Bundle<Types>, BuildError> {
267        let bundle_size: u64 = transactions
268            .iter()
269            .map(|txn| txn.minimum_block_size())
270            .sum();
271        let offered_fee = self.base_fee * bundle_size;
272
273        let fee_signature =
274            <Types::BuilderSignatureKey as BuilderSignatureKey>::sign_sequencing_fee_marketplace(
275                &self.builder_keys.1,
276                offered_fee,
277                view_number,
278            )
279            .map_err(|e| BuildError::Error(e.to_string()))?;
280
281        let sequencing_fee: BuilderFee<Types> = BuilderFee {
282            fee_amount: offered_fee,
283            fee_account: self.builder_keys.0.clone(),
284            fee_signature,
285        };
286
287        let commitments = transactions
288            .iter()
289            .flat_map(|txn| <[u8; 32]>::from(txn.commit()))
290            .collect::<Vec<u8>>();
291
292        let signature = <Types::BuilderSignatureKey as BuilderSignatureKey>::sign_builder_message(
293            &self.builder_keys.1,
294            &commitments,
295        )
296        .map_err(|e| BuildError::Error(e.to_string()))?;
297
298        Ok(Bundle {
299            sequencing_fee,
300            transactions,
301            signature,
302        })
303    }

317#[async_trait]
318impl<Types, Hooks> BuilderDataSource<Types> for ProxyGlobalState<Types, Hooks>
319where
320    Types: NodeType,
321    Hooks: BuilderHooks<Types>,
322    for<'a> <<Types::SignatureKey as SignatureKey>::PureAssembledSignatureType as TryFrom<
323        &'a TaggedBase64,
324    >>::Error: Display,
325    for<'a> <Types::SignatureKey as TryFrom<&'a TaggedBase64>>::Error: Display,
326{
327    #[tracing::instrument(
328        skip(self),
329        err(level = Level::INFO)
330        ret(level = Level::TRACE)
331    )]
332    async fn bundle(
333        &self,
334        parent_view: u64,
335        parent_hash: &VidCommitment,
336        view_number: u64,
337    ) -> Result<Bundle<Types>, BuildError> {
338        let start = Instant::now();
339
340        let parent_view = Types::View::new(parent_view);
341        let state_id = BuilderStateId {
342            parent_view,
343            parent_commitment: *parent_hash,
344        };
345
346        loop {
347            // Couldn't serve a bundle in time
348            if start.elapsed() > self.api_timeout {
349                tracing::warn!("Timeout while trying to serve a bundle");
350                return Err(BuildError::NotFound);
351            };
352
353            let builder_state = match self.coordinator.lookup_builder_state(&state_id).await {
354                BuilderStateLookup::Found(builder_state_entry) => builder_state_entry,
355                BuilderStateLookup::NotFound => {
356                    // If we couldn't find the state because it hasn't yet been created, try again
357                    sleep(self.api_timeout / 10).await;
358                    continue;
359                },
360                BuilderStateLookup::Decided => {
361                    // If we couldn't find the state because the view has already been decided, we can just return an error
362                    tracing::warn!("Requested a bundle for view we already GCd as decided",);
363                    return Err(BuildError::Error(
364                        "Request for a bundle for a view that has already been decided.".to_owned(),
365                    ));
366                },
367            };
368
369            tracing::info!(
370                "Request handled by builder with view {}@{:?} for (view_num: {:?})",
371                builder_state.parent_block_references.vid_commitment,
372                builder_state.parent_block_references.view_number,
373                parent_view
374            );
375
376            let Some(transactions) = self.collect_transactions(&builder_state).await else {
377                tracing::debug!("No response to send");
378                return Err(BuildError::NotFound);
379            };
380
381            let bundle = self.assemble_bundle(transactions, view_number).await?;
382
383            tracing::info!("Serving bundle");
384
385            return Ok(bundle);
386        }
387    }

396#[async_trait]
397impl<Types, Hooks> AcceptsTxnSubmits<Types> for ProxyGlobalState<Types, Hooks>
398where
399    Hooks: BuilderHooks<Types>,
400    Types: NodeType,
401{
402    async fn submit_txns(
403        &self,
404        txns: Vec<<Types as NodeType>::Transaction>,
405    ) -> Result<Vec<Commitment<<Types as NodeType>::Transaction>>, BuildError> {
406        let txns = self.hooks.process_transactions(txns).await;
407
408        txns.into_iter()
409            .map(|txn| ReceivedTransaction::new(txn, TransactionSource::Private))
410            .map(|txn| async {
411                let commit = txn.commit;
412                self.coordinator.handle_transaction(txn).await?;
413                Ok(commit)
414            })
415            .collect::<FuturesOrdered<_>>()
416            .try_collect()
417            .await
418    }
419
420    async fn txn_status(
421        &self,
422        txn_hash: Commitment<<Types as NodeType>::Transaction>,
423    ) -> Result<TransactionStatus, BuildError> {
424        Ok(self.coordinator.tx_status(&txn_hash))
425    }
