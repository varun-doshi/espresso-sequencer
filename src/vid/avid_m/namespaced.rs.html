14/// Dummy struct for namespaced AvidM scheme

17/// Namespaced commitment type
18pub type NsAvidMCommit = super::AvidMCommit;
19/// Namespaced parameter type

22/// Namespaced share for each storage node
23#[derive(Clone, Debug, Hash, Serialize, Deserialize, Eq, PartialEq, Default)]
24pub struct NsAvidMShare {
25    /// Index number of the given share.
26    index: u32,
27    /// The list of all namespace commitments
28    ns_commits: Vec<AvidMCommit>,
29    /// The size of each namespace
30    ns_lens: Vec<usize>,
31    /// Actual share content
32    content: Vec<RawAvidMShare>,
33}
34
35impl NsAvidMShare {
36    fn inner_ns_share(&self, ns_id: usize) -> AvidMShare {
37        AvidMShare {
38            index: self.index,
39            payload_byte_len: self.ns_lens[ns_id],
40            content: self.content[ns_id].clone(),
41        }
42    }
43
44    /// Return the length of underlying payload in bytes
45    pub fn payload_byte_len(&self) -> usize {
46        self.ns_lens.iter().sum()
47    }
48}
49
50impl NsAvidMScheme {
51    /// Setup an instance for AVID-M scheme
52    pub fn setup(recovery_threshold: usize, total_weights: usize) -> VidResult<NsAvidMParam> {
53        NsAvidMParam::new(recovery_threshold, total_weights)
54    }
55
56    /// Commit to a payload given namespace table.
57    pub fn commit(
58        param: &NsAvidMParam,
59        payload: &[u8],
60        ns_table: impl IntoIterator<Item = Range<usize>>,
61    ) -> VidResult<NsAvidMCommit> {
62        let ns_commits = ns_table
63            .into_iter()
64            .map(|ns_range| {
65                AvidMScheme::commit(param, &payload[ns_range]).map(|commit| commit.commit)
66            })
67            .collect::<Result<Vec<_>, _>>()?;
68        Ok(NsAvidMCommit {
69            commit: MerkleTree::from_elems(None, &ns_commits)
70                .map_err(|err| VidError::Internal(err.into()))?
71                .commitment(),
72        })
73    }
74
75    /// Disperse a payload according to a distribution table and a namespace
76    /// table.
77    /// WARN: it assumes that the namespace table is well formed, i.e. ranges
78    /// are non-overlapping and cover the whole payload.
79    pub fn ns_disperse(
80        param: &NsAvidMParam,
81        distribution: &[u32],
82        payload: &[u8],
83        ns_table: impl IntoIterator<Item = Range<usize>>,
84    ) -> VidResult<(NsAvidMCommit, Vec<NsAvidMShare>)> {
85        let mut ns_commits = vec![];
86        let mut disperses = vec![];
87        let mut ns_lens = vec![];
88        for ns_range in ns_table {
89            ns_lens.push(ns_range.len());
90            let (commit, shares) = AvidMScheme::disperse(param, distribution, &payload[ns_range])?;
91            ns_commits.push(commit.commit);
92            disperses.push(shares);
93        }
94        let commit = NsAvidMCommit {
95            commit: MerkleTree::from_elems(None, &ns_commits)
96                .map_err(|err| VidError::Internal(err.into()))?
97                .commitment(),
98        };
99        let ns_commits: Vec<_> = ns_commits
100            .into_iter()
101            .map(|comm| AvidMCommit { commit: comm })
102            .collect();
103        let mut shares = vec![NsAvidMShare::default(); disperses[0].len()];
104        shares.iter_mut().for_each(|share| {
105            share.index = disperses[0][0].index;
106            share.ns_commits = ns_commits.clone();
107            share.ns_lens = ns_lens.clone();
108        });
109        disperses.into_iter().for_each(|ns_disperse| {
110            shares
111                .iter_mut()
112                .zip(ns_disperse)
113                .for_each(|(share, ns_share)| share.content.push(ns_share.content))
114        });
115        Ok((commit, shares))
116    }
117
118    /// Verify a namespaced share
119    pub fn verify_share(
120        param: &NsAvidMParam,
121        commit: &NsAvidMCommit,
122        share: &NsAvidMShare,
123    ) -> VidResult<crate::VerificationResult> {
124        if !(share.ns_commits.len() == share.ns_lens.len()
125            && share.ns_commits.len() == share.content.len())
126        {
127            return Err(VidError::InvalidShare);
128        }
129        // Verify the share for each namespace
130        for (commit, content) in share.ns_commits.iter().zip(share.content.iter()) {
131            if AvidMScheme::verify_internal(param, commit, content)?.is_err() {
132                return Ok(Err(()));
133            }
134        }
135        // Verify the namespace MT commitment
136        let expected_commit = NsAvidMCommit {
137            commit: MerkleTree::from_elems(
138                None,
139                share.ns_commits.iter().map(|commit| commit.commit),
140            )
141            .map_err(|err| VidError::Internal(err.into()))?
142            .commitment(),
143        };
144        Ok(if &expected_commit == commit {
145            Ok(())
146        } else {
147            Err(())
148        })
149    }
150
151    /// Recover the entire payload from enough share
152    pub fn recover(param: &NsAvidMParam, shares: &[NsAvidMShare]) -> VidResult<Vec<u8>> {
153        if shares.is_empty() {
154            return Err(VidError::InsufficientShares);
155        }
156        let mut result = vec![];
157        for ns_id in 0..shares[0].ns_lens.len() {
158            result.append(&mut Self::ns_recover(param, ns_id, shares)?)
159        }
160        Ok(result)
161    }
162
163    /// Recover the payload for a given namespace
164    pub fn ns_recover(
165        param: &NsAvidMParam,
166        ns_id: usize,
167        shares: &[NsAvidMShare],
168    ) -> VidResult<Vec<u8>> {
169        if shares.is_empty() {
170            return Err(VidError::InsufficientShares);
171        }
172        let ns_commit = shares[0].ns_commits[ns_id];
173        let shares: Vec<_> = shares
174            .iter()
175            .map(|share| share.inner_ns_share(ns_id))
176            .collect();
177        AvidMScheme::recover(param, &ns_commit, &shares)
178    }
179}

181/// Unit tests
182#[cfg(test)]
183pub mod tests {
184    use rand::{seq::SliceRandom, RngCore};
185
186    use crate::avid_m::namespaced::NsAvidMScheme;
187
188    #[test]
189    fn round_trip() {
190        // play with these items
191        let num_storage_nodes = 9;
192        let recovery_threshold = 3;
193        let ns_lens = [15, 33];
194        let ns_table = [(0usize..15), (15..48)];
195        let payload_byte_len = ns_lens.iter().sum();
196
197        // more items as a function of the above
198
199        let mut rng = jf_utils::test_rng();
200
201        let weights: Vec<u32> = (0..num_storage_nodes)
202            .map(|_| rng.next_u32() % 5 + 1)
203            .collect();
204        let total_weights: u32 = weights.iter().sum();
205        let params = NsAvidMScheme::setup(recovery_threshold, total_weights as usize).unwrap();
206
207        println!(
208            "recovery_threshold:: {} num_storage_nodes: {} payload_byte_len: {}",
209            recovery_threshold, num_storage_nodes, payload_byte_len
210        );
211        println!("weights: {:?}", weights);
212
213        let payload = {
214            let mut bytes_random = vec![0u8; payload_byte_len];
215            rng.fill_bytes(&mut bytes_random);
216            bytes_random
217        };
218
219        let (commit, mut shares) =
220            NsAvidMScheme::ns_disperse(&params, &weights, &payload, ns_table.iter().cloned())
221                .unwrap();
222
223        assert_eq!(shares.len(), num_storage_nodes);
224
225        assert_eq!(
226            commit,
227            NsAvidMScheme::commit(&params, &payload, ns_table.iter().cloned()).unwrap()
228        );
229
230        // verify shares
231        shares.iter().for_each(|share| {
232            assert!(NsAvidMScheme::verify_share(&params, &commit, share).is_ok_and(|r| r.is_ok()))
233        });
234
235        // test payload recovery on a random subset of shares
236        shares.shuffle(&mut rng);
237        let mut cumulated_weights = 0;
238        let mut cut_index = 0;
239        while cumulated_weights <= recovery_threshold {
240            cumulated_weights += shares[cut_index].content[0].range.len();
241            cut_index += 1;
242        }
243        let ns0_payload_recovered =
244            NsAvidMScheme::ns_recover(&params, 0, &shares[..cut_index]).unwrap();
245        assert_eq!(ns0_payload_recovered[..], payload[ns_table[0].clone()]);
246        let ns1_payload_recovered =
247            NsAvidMScheme::ns_recover(&params, 1, &shares[..cut_index]).unwrap();
248        assert_eq!(ns1_payload_recovered[..], payload[ns_table[1].clone()]);
249        let payload_recovered = NsAvidMScheme::recover(&params, &shares[..cut_index]).unwrap();
250        assert_eq!(payload_recovered, payload);
251    }
