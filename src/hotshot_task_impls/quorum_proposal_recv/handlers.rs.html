51/// Spawn a task which will fire a request to get a proposal, and store it.
52#[allow(clippy::too_many_arguments)]
53fn spawn_fetch_proposal<TYPES: NodeType, V: Versions>(
54    view: TYPES::View,
55    event_sender: Sender<Arc<HotShotEvent<TYPES>>>,
56    event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
57    membership: EpochMembershipCoordinator<TYPES>,
58    consensus: OuterConsensus<TYPES>,
59    sender_public_key: TYPES::SignatureKey,
60    sender_private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
61    upgrade_lock: UpgradeLock<TYPES, V>,
62    epoch_height: u64,
63) {
64    spawn(async move {
65        let lock = upgrade_lock;
66
67        let _ = fetch_proposal(
68            view,
69            event_sender,
70            event_receiver,
71            membership,
72            consensus,
73            sender_public_key,
74            sender_private_key,
75            &lock,
76            epoch_height,
77        )
78        .await;
79    });
80}

82/// Update states in the event that the parent state is not found for a given `proposal`.
83#[instrument(skip_all)]
84pub async fn validate_proposal_liveness<
85    TYPES: NodeType,
86    I: NodeImplementation<TYPES>,
87    V: Versions,
88>(
89    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
90    validation_info: &ValidationInfo<TYPES, I, V>,
91) -> Result<()> {
92    let mut valid_epoch_transition = false;
93    if validation_info
94        .upgrade_lock
95        .version(proposal.data.view_number())
96        .await
97        .is_ok_and(|v| v >= V::Epochs::VERSION)
98    {
99        let Some(block_number) = proposal.data.justify_qc().data.block_number else {
100            bail!("Quorum Proposal has no block number but it's after the epoch upgrade");
101        };
102        if is_epoch_transition(block_number, validation_info.epoch_height) {
103            validate_epoch_transition_qc(proposal, validation_info).await?;
104            valid_epoch_transition = true;
105        }
106    }
107    let mut consensus_writer = validation_info.consensus.write().await;
108
109    let leaf = Leaf2::from_quorum_proposal(&proposal.data);
110
111    let state = Arc::new(
112        <TYPES::ValidatedState as ValidatedState<TYPES>>::from_header(proposal.data.block_header()),
113    );
114
115    if let Err(e) = consensus_writer.update_leaf(leaf.clone(), state, None) {
116        tracing::trace!("{e:?}");
117    }
118
119    let liveness_check = proposal.data.justify_qc().view_number() > consensus_writer.locked_view();
120    // if we are using HS2 we update our locked view for any QC from a leader greater than our current lock
121    if liveness_check
122        && validation_info
123            .upgrade_lock
124            .version(leaf.view_number())
125            .await
126            .is_ok_and(|v| v >= V::Epochs::VERSION)
127    {
128        consensus_writer.update_locked_view(proposal.data.justify_qc().view_number())?;
129    }
130
131    drop(consensus_writer);
132
133    if !liveness_check && !valid_epoch_transition {
134        bail!("Quorum Proposal failed the liveness check");
135    }
136
137    Ok(())
138}

140async fn validate_epoch_transition_block<
141    TYPES: NodeType,
142    I: NodeImplementation<TYPES>,
143    V: Versions,
144>(
145    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
146    validation_info: &ValidationInfo<TYPES, I, V>,
147) -> Result<()> {
148    if !validation_info
149        .upgrade_lock
150        .epochs_enabled(proposal.data.view_number())
151        .await
152    {
153        return Ok(());
154    }
155    if !is_epoch_transition(
156        proposal.data.block_header().block_number(),
157        validation_info.epoch_height,
158    ) {
159        return Ok(());
160    }
161    // transition block does not have to be empty
162    if is_transition_block(
163        proposal.data.block_header().block_number(),
164        validation_info.epoch_height,
165    ) {
166        return Ok(());
167    }
168    // TODO: Is this the best way to do this?
169    let (empty_payload, metadata) = <TYPES as NodeType>::BlockPayload::empty();
170    let header = proposal.data.block_header();
171    ensure!(
172        empty_payload.builder_commitment(&metadata) == header.builder_commitment()
173            && &metadata == header.metadata(),
174        "Block is not empty"
175    );
176    Ok(())
177}

179async fn validate_current_epoch<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions>(
180    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
181    validation_info: &ValidationInfo<TYPES, I, V>,
182) -> Result<()> {
183    if !validation_info
184        .upgrade_lock
185        .epochs_enabled(proposal.data.view_number())
186        .await
187        || proposal.data.justify_qc().view_number()
188            <= validation_info
189                .upgrade_lock
190                .upgrade_view()
191                .await
192                .unwrap_or(TYPES::View::new(0))
193    {
194        return Ok(());
195    }
196
197    let block_number = proposal.data.block_header().block_number();
198
199    let Some(high_block_number) = validation_info
200        .consensus
201        .read()
202        .await
203        .high_qc()
204        .data
205        .block_number
206    else {
207        bail!("High QC has no block number");
208    };
209
210    ensure!(
211        epoch_from_block_number(block_number, validation_info.epoch_height)
212            >= epoch_from_block_number(high_block_number + 1, validation_info.epoch_height),
213        "Quorum proposal has an inconsistent epoch"
214    );
215
216    Ok(())
217}

219/// Validate that the proposal's block height is one greater than the justification QC's block height.
220async fn validate_block_height<TYPES: NodeType>(
221    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
222) -> Result<()> {
223    let Some(qc_block_number) = proposal.data.justify_qc().data.block_number else {
224        return Ok(());
225    };
226    ensure!(
227        qc_block_number + 1 == proposal.data.block_header().block_number(),
228        "Quorum proposal has an inconsistent block height"
229    );
230    Ok(())
231}

233/// Handles the `QuorumProposalRecv` event by first validating the cert itself for the view, and then
234/// updating the states, which runs when the proposal cannot be found in the internal state map.
235///
236/// This code can fail when:
237/// - The justify qc is invalid.
238/// - The task is internally inconsistent.
239/// - The sequencer storage update fails.
240#[allow(clippy::too_many_lines)]
241#[instrument(skip_all)]
242pub(crate) async fn handle_quorum_proposal_recv<
243    TYPES: NodeType,
244    I: NodeImplementation<TYPES>,
245    V: Versions,
246>(
247    proposal: &Proposal<TYPES, QuorumProposalWrapper<TYPES>>,
248    quorum_proposal_sender_key: &TYPES::SignatureKey,
249    event_sender: &Sender<Arc<HotShotEvent<TYPES>>>,
250    event_receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
251    validation_info: ValidationInfo<TYPES, I, V>,
252) -> Result<()> {
253    proposal
254        .data
255        .validate_epoch(&validation_info.upgrade_lock, validation_info.epoch_height)
256        .await?;
257    // validate the proposal's epoch matches ours
258    validate_current_epoch(proposal, &validation_info).await?;
259    let quorum_proposal_sender_key = quorum_proposal_sender_key.clone();
260
261    validate_proposal_view_and_certs(proposal, &validation_info)
262        .await
263        .context(warn!("Failed to validate proposal view or attached certs"))?;
264
265    validate_block_height(proposal).await?;
266
267    let view_number = proposal.data.view_number();
268
269    let justify_qc = proposal.data.justify_qc().clone();
270    let maybe_next_epoch_justify_qc = proposal.data.next_epoch_justify_qc().clone();
271
272    let proposal_block_number = proposal.data.block_header().block_number();
273    let proposal_epoch = option_epoch_from_block_number::<TYPES>(
274        proposal.data.epoch().is_some(),
275        proposal_block_number,
276        validation_info.epoch_height,
277    );
278
279    validate_epoch_transition_block(proposal, &validation_info).await?;
280
281    validate_qc_and_next_epoch_qc(
282        &justify_qc,
283        maybe_next_epoch_justify_qc.as_ref(),
284        &validation_info.consensus,
285        &validation_info.membership.coordinator,
286        &validation_info.upgrade_lock,
287        validation_info.epoch_height,
288    )
289    .await?;
290
291    broadcast_event(
292        Arc::new(HotShotEvent::QuorumProposalPreliminarilyValidated(
293            proposal.clone(),
294        )),
295        event_sender,
296    )
297    .await;
298
299    // Get the parent leaf and state.
300    let parent_leaf = validation_info
301        .consensus
302        .read()
303        .await
304        .saved_leaves()
305        .get(&justify_qc.data.leaf_commit)
306        .cloned();
307
308    if parent_leaf.is_none() {
309        spawn_fetch_proposal(
310            justify_qc.view_number(),
311            event_sender.clone(),
312            event_receiver.clone(),
313            validation_info.membership.coordinator.clone(),
314            OuterConsensus::new(Arc::clone(&validation_info.consensus.inner_consensus)),
315            // Note that we explicitly use the node key here instead of the provided key in the signature.
316            // This is because the key that we receive is for the prior leader, so the payload would be routed
317            // incorrectly.
318            validation_info.public_key.clone(),
319            validation_info.private_key.clone(),
320            validation_info.upgrade_lock.clone(),
321            validation_info.epoch_height,
322        );
323    }
324    let consensus_reader = validation_info.consensus.read().await;
325
326    let parent = match parent_leaf {
327        Some(leaf) => {
328            if let (Some(state), _) = consensus_reader.state_and_delta(leaf.view_number()) {
329                Some((leaf, Arc::clone(&state)))
330            } else {
331                bail!("Parent state not found! Consensus internally inconsistent");
332            }
333        },
334        None => None,
335    };
336    drop(consensus_reader);
337    if justify_qc.view_number()
338        > validation_info
339            .consensus
340            .read()
341            .await
342            .high_qc()
343            .view_number()
344    {
345        update_high_qc(proposal, &validation_info).await?;
346    }
347
348    let Some((parent_leaf, _parent_state)) = parent else {
349        tracing::warn!(
350            "Proposal's parent missing from storage with commitment: {:?}",
351            justify_qc.data.leaf_commit
352        );
353        validate_proposal_liveness(proposal, &validation_info).await?;
354        tracing::trace!(
355            "Sending ViewChange for view {} and epoch {:?}",
356            view_number,
357            proposal_epoch
358        );
359        validation_info
360            .consensus
361            .write()
362            .await
363            .update_highest_block(proposal_block_number);
364        broadcast_event(
365            Arc::new(HotShotEvent::ViewChange(view_number, proposal_epoch)),
366            event_sender,
367        )
368        .await;
369        return Ok(());
370    };
371
372    // Validate the proposal
373    validate_proposal_safety_and_liveness::<TYPES, I, V>(
374        proposal.clone(),
375        parent_leaf,
376        &validation_info,
377        event_sender.clone(),
378        quorum_proposal_sender_key,
379    )
380    .await?;
381
382    tracing::trace!(
383        "Sending ViewChange for view {} and epoch {:?}",
384        view_number,
385        proposal_epoch
386    );
387    validation_info
388        .consensus
389        .write()
390        .await
391        .update_highest_block(proposal_block_number);
392    {
393        validation_info.consensus.write().await.highest_block = proposal_block_number;
394    }
395    broadcast_event(
396        Arc::new(HotShotEvent::ViewChange(view_number, proposal_epoch)),
397        event_sender,
398    )
399    .await;
400
401    Ok(())
