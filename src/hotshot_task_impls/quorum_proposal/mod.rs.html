40/// The state for the quorum proposal task.
41pub struct QuorumProposalTaskState<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> {
42    /// Latest view number that has been proposed for.
43    pub latest_proposed_view: TYPES::View,
44
45    /// Current epoch
46    pub cur_epoch: Option<TYPES::Epoch>,
47
48    /// Table for the in-progress proposal dependency tasks.
49    pub proposal_dependencies: BTreeMap<TYPES::View, JoinHandle<()>>,
50
51    /// Formed QCs
52    pub formed_quorum_certificates: BTreeMap<TYPES::View, QuorumCertificate2<TYPES>>,
53
54    /// Formed QCs for the next epoch
55    pub formed_next_epoch_quorum_certificates:
56        BTreeMap<TYPES::View, NextEpochQuorumCertificate2<TYPES>>,
57
58    /// Immutable instance state
59    pub instance_state: Arc<TYPES::InstanceState>,
60
61    /// Membership for Quorum Certs/votes
62    pub membership_coordinator: EpochMembershipCoordinator<TYPES>,
63
64    /// Our public key
65    pub public_key: TYPES::SignatureKey,
66
67    /// Our Private Key
68    pub private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
69
70    /// View timeout from config.
71    pub timeout: u64,
72
73    /// This node's storage ref
74    pub storage: Arc<RwLock<I::Storage>>,
75
76    /// Shared consensus task state
77    pub consensus: OuterConsensus<TYPES>,
78
79    /// The node's id
80    pub id: u64,
81
82    /// The most recent upgrade certificate this node formed.
83    /// Note: this is ONLY for certificates that have been formed internally,
84    /// so that we can propose with them.
85    ///
86    /// Certificates received from other nodes will get reattached regardless of this fields,
87    /// since they will be present in the leaf we propose off of.
88    pub formed_upgrade_certificate: Option<UpgradeCertificate<TYPES>>,
89
90    /// Lock for a decided upgrade
91    pub upgrade_lock: UpgradeLock<TYPES, V>,
92
93    /// Number of blocks in an epoch, zero means there are no epochs
94    pub epoch_height: u64,
95}

100    /// Create an event dependency
101    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view), name = "Create event dependency", level = "info")]
102    fn create_event_dependency(
103        &self,
104        dependency_type: ProposalDependency,
105        view_number: TYPES::View,
106        event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
107    ) -> EventDependency<Arc<HotShotEvent<TYPES>>> {
108        let id = self.id;
109        EventDependency::new(
110            event_receiver,
111            Box::new(move |event| {
112                let event = event.as_ref();
113                let event_view = match dependency_type {
114                    ProposalDependency::Qc => {
115                        if let HotShotEvent::Qc2Formed(either::Left(qc)) = event {
116                            qc.view_number() + 1
117                        } else {
118                            return false;
119                        }
120                    },
121                    ProposalDependency::TimeoutCert => {
122                        if let HotShotEvent::Qc2Formed(either::Right(timeout)) = event {
123                            timeout.view_number() + 1
124                        } else {
125                            return false;
126                        }
127                    },
128                    ProposalDependency::ViewSyncCert => {
129                        if let HotShotEvent::ViewSyncFinalizeCertificateRecv(view_sync_cert) = event
130                        {
131                            view_sync_cert.view_number()
132                        } else {
133                            return false;
134                        }
135                    },
136                    ProposalDependency::Proposal => {
137                        if let HotShotEvent::QuorumProposalPreliminarilyValidated(proposal) = event
138                        {
139                            proposal.data.view_number() + 1
140                        } else {
141                            return false;
142                        }
143                    },
144                    ProposalDependency::PayloadAndMetadata => {
145                        if let HotShotEvent::SendPayloadCommitmentAndMetadata(
146                            _payload_commitment,
147                            _builder_commitment,
148                            _metadata,
149                            view_number,
150                            _fee,
151                            _auction_result,
152                        ) = event
153                        {
154                            *view_number
155                        } else {
156                            return false;
157                        }
158                    },
159                    ProposalDependency::VidShare => {
160                        if let HotShotEvent::VidDisperseSend(vid_disperse, _) = event {
161                            vid_disperse.data.view_number()
162                        } else {
163                            return false;
164                        }
165                    },
166                };
167                let valid = event_view == view_number;
168                if valid {
169                    tracing::debug!(
170                        "Dependency {dependency_type:?} is complete for view {event_view:?}, my id is {id:?}!",
171                    );
172                }
173                valid
174            }),
175        )
176    }

178    /// Creates the requisite dependencies for the Quorum Proposal task. It also handles any event forwarding.
179    fn create_and_complete_dependencies(
180        &self,
181        view_number: TYPES::View,
182        event_receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
183        event: Arc<HotShotEvent<TYPES>>,
184    ) -> AndDependency<Vec<Vec<Arc<HotShotEvent<TYPES>>>>> {
185        let mut proposal_dependency = self.create_event_dependency(
186            ProposalDependency::Proposal,
187            view_number,
188            event_receiver.clone(),
189        );
190
191        let mut qc_dependency = self.create_event_dependency(
192            ProposalDependency::Qc,
193            view_number,
194            event_receiver.clone(),
195        );
196
197        let mut view_sync_dependency = self.create_event_dependency(
198            ProposalDependency::ViewSyncCert,
199            view_number,
200            event_receiver.clone(),
201        );
202
203        let mut timeout_dependency = self.create_event_dependency(
204            ProposalDependency::TimeoutCert,
205            view_number,
206            event_receiver.clone(),
207        );
208
209        let mut payload_commitment_dependency = self.create_event_dependency(
210            ProposalDependency::PayloadAndMetadata,
211            view_number,
212            event_receiver.clone(),
213        );
214
215        let mut vid_share_dependency = self.create_event_dependency(
216            ProposalDependency::VidShare,
217            view_number,
218            event_receiver.clone(),
219        );
220
221        let epoch_height = self.epoch_height;
222
223        // Next epoch QC dependency is fulfilled if we get the next epoch QC or
224        // form a current qc that isn't during transition
225        let mut next_epoch_qc_dependency = EventDependency::new(
226            event_receiver.clone(),
227            Box::new(move |event| {
228                if let HotShotEvent::NextEpochQc2Formed(Either::Left(next_epoch_qc)) =
229                    event.as_ref()
230                {
231                    return next_epoch_qc.view_number() + 1 == view_number;
232                }
233                if let HotShotEvent::Qc2Formed(Either::Left(qc)) = event.as_ref() {
234                    if qc.view_number() + 1 == view_number {
235                        return qc
236                            .data
237                            .block_number
238                            .is_none_or(|bn| !is_epoch_transition(bn, epoch_height));
239                    }
240                }
241                false
242            }),
243        );
244
245        match event.as_ref() {
246            HotShotEvent::SendPayloadCommitmentAndMetadata(..) => {
247                payload_commitment_dependency.mark_as_completed(Arc::clone(&event));
248            },
249            HotShotEvent::QuorumProposalPreliminarilyValidated(..) => {
250                proposal_dependency.mark_as_completed(event);
251            },
252            HotShotEvent::Qc2Formed(quorum_certificate) => match quorum_certificate {
253                Either::Right(_) => {
254                    timeout_dependency.mark_as_completed(event);
255                },
256                Either::Left(qc) => {
257                    if qc
258                        .data
259                        .block_number
260                        .is_none_or(|bn| !is_epoch_transition(bn, epoch_height))
261                    {
262                        next_epoch_qc_dependency.mark_as_completed(event.clone());
263                    }
264                    qc_dependency.mark_as_completed(event);
265                },
266            },
267            HotShotEvent::ViewSyncFinalizeCertificateRecv(_) => {
268                view_sync_dependency.mark_as_completed(event);
269            },
270            HotShotEvent::VidDisperseSend(..) => {
271                vid_share_dependency.mark_as_completed(event);
272            },
273            HotShotEvent::NextEpochQc2Formed(Either::Left(_)) => {
274                next_epoch_qc_dependency.mark_as_completed(event);
275            },
276            _ => {},
277        };
278
279        // We have three cases to consider:
280        let mut secondary_deps = vec![
281            // 1. A timeout cert was received
282            AndDependency::from_deps(vec![timeout_dependency]),
283            // 2. A view sync cert was received.
284            AndDependency::from_deps(vec![view_sync_dependency]),
285        ];
286        // 3. A `Qc2Formed`` event (and `QuorumProposalRecv` event)
287        if *view_number > 1 {
288            secondary_deps.push(AndDependency::from_deps(vec![
289                qc_dependency,
290                proposal_dependency,
291                next_epoch_qc_dependency,
292            ]));
293        } else {
294            secondary_deps.push(AndDependency::from_deps(vec![qc_dependency]));
295        }
296
297        let primary_deps = vec![payload_commitment_dependency, vid_share_dependency];
298
299        AndDependency::from_deps(vec![OrDependency::from_deps(vec![
300            AndDependency::from_deps(vec![
301                OrDependency::from_deps(vec![AndDependency::from_deps(primary_deps)]),
302                OrDependency::from_deps(secondary_deps),
303            ]),
304        ])])
305    }

307    /// Create and store an [`AndDependency`] combining [`EventDependency`]s associated with the
308    /// given view number if it doesn't exist. Also takes in the received `event` to seed a
309    /// dependency as already completed. This allows for the task to receive a proposable event
310    /// without losing the data that it received, as the dependency task would otherwise have no
311    /// ability to receive the event and, thus, would never propose.
312    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view), name = "Create dependency task", level = "error")]
313    async fn create_dependency_task_if_new(
314        &mut self,
315        view_number: TYPES::View,
316        epoch_number: Option<TYPES::Epoch>,
317        event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
318        event_sender: Sender<Arc<HotShotEvent<TYPES>>>,
319        event: Arc<HotShotEvent<TYPES>>,
320        epoch_transition_indicator: EpochTransitionIndicator,
321    ) -> Result<()> {
322        let epoch_membership = self
323            .membership_coordinator
324            .membership_for_epoch(epoch_number)
325            .await?;
326        let leader_in_current_epoch =
327            epoch_membership.leader(view_number).await? == self.public_key;
328        // If we are in the epoch transition and we are the leader in the next epoch,
329        // we might want to start collecting dependencies for our next epoch proposal.
330
331        let leader_in_next_epoch = epoch_number.is_some()
332            && matches!(
333                epoch_transition_indicator,
334                EpochTransitionIndicator::InTransition
335            )
336            && epoch_membership
337                .next_epoch()
338                .await
339                .context(warn!(
340                    "Missing the randomized stake table for epoch {:?}",
341                    epoch_number.unwrap() + 1
342                ))?
343                .leader(view_number)
344                .await?
345                == self.public_key;
346
347        // Don't even bother making the task if we are not entitled to propose anyway.
348        ensure!(
349            leader_in_current_epoch || leader_in_next_epoch,
350            debug!("We are not the leader of the next view")
351        );
352
353        // Don't try to propose twice for the same view.
354        ensure!(
355            view_number > self.latest_proposed_view,
356            "We have already proposed for this view"
357        );
358
359        tracing::debug!(
360            "Attempting to make dependency task for view {view_number:?} and event {event:?}"
361        );
362
363        ensure!(
364            !self.proposal_dependencies.contains_key(&view_number),
365            "Task already exists"
366        );
367
368        let dependency_chain =
369            self.create_and_complete_dependencies(view_number, &event_receiver, event);
370
371        let dependency_task = DependencyTask::new(
372            dependency_chain,
373            ProposalDependencyHandle {
374                latest_proposed_view: self.latest_proposed_view,
375                view_number,
376                sender: event_sender,
377                receiver: event_receiver,
378                membership: epoch_membership,
379                public_key: self.public_key.clone(),
380                private_key: self.private_key.clone(),
381                instance_state: Arc::clone(&self.instance_state),
382                consensus: OuterConsensus::new(Arc::clone(&self.consensus.inner_consensus)),
383                timeout: self.timeout,
384                formed_upgrade_certificate: self.formed_upgrade_certificate.clone(),
385                upgrade_lock: self.upgrade_lock.clone(),
386                id: self.id,
387                view_start_time: Instant::now(),
388                epoch_height: self.epoch_height,
389            },
390        );
391        self.proposal_dependencies
392            .insert(view_number, dependency_task.run());
393
394        Ok(())
395    }

397    /// Update the latest proposed view number.
398    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view), name = "Update latest proposed view", level = "error")]
399    async fn update_latest_proposed_view(&mut self, new_view: TYPES::View) -> bool {
400        if *self.latest_proposed_view < *new_view {
401            tracing::debug!(
402                "Updating latest proposed view from {} to {}",
403                *self.latest_proposed_view,
404                *new_view
405            );
406
407            // Cancel the old dependency tasks.
408            for view in (*self.latest_proposed_view + 1)..=(*new_view) {
409                if let Some(dependency) = self.proposal_dependencies.remove(&TYPES::View::new(view))
410                {
411                    dependency.abort();
412                }
413            }
414
415            self.latest_proposed_view = new_view;
416
417            return true;
418        }
419        false
420    }

422    /// Handles a consensus event received on the event stream
423    #[instrument(skip_all, fields(id = self.id, latest_proposed_view = *self.latest_proposed_view, epoch = self.cur_epoch.map(|x| *x)), name = "handle method", level = "error", target = "QuorumProposalTaskState")]
424    pub async fn handle(
425        &mut self,
426        event: Arc<HotShotEvent<TYPES>>,
427        event_receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
428        event_sender: Sender<Arc<HotShotEvent<TYPES>>>,
429    ) -> Result<()> {
430        let epoch_number = self.cur_epoch;
431        let epoch_transition_indicator = if self.consensus.read().await.is_high_qc_for_last_block()
432        {
433            EpochTransitionIndicator::InTransition
434        } else {
435            EpochTransitionIndicator::NotInTransition
436        };
437        match event.as_ref() {
438            HotShotEvent::UpgradeCertificateFormed(cert) => {
439                tracing::debug!(
440                    "Upgrade certificate received for view {}!",
441                    *cert.view_number
442                );
443                // Update our current upgrade_cert as long as we still have a chance of reaching a decide on it in time.
444                if cert.data.decide_by >= self.latest_proposed_view + 3 {
445                    tracing::debug!("Updating current formed_upgrade_certificate");
446
447                    self.formed_upgrade_certificate = Some(cert.clone());
448                }
449            },
450            HotShotEvent::Qc2Formed(cert) => match cert.clone() {
451                either::Right(timeout_cert) => {
452                    let view_number = timeout_cert.view_number + 1;
453                    self.create_dependency_task_if_new(
454                        view_number,
455                        epoch_number,
456                        event_receiver,
457                        event_sender,
458                        Arc::clone(&event),
459                        epoch_transition_indicator,
460                    )
461                    .await?;
462                },
463                either::Left(qc) => {
464                    // Only update if the qc is from a newer view
465                    if qc.view_number <= self.consensus.read().await.high_qc().view_number {
466                        tracing::trace!(
467                            "Received a QC for a view that was not > than our current high QC"
468                        );
469                    }
470
471                    self.formed_quorum_certificates
472                        .insert(qc.view_number(), qc.clone());
473
474                    handle_eqc_formed(
475                        qc.view_number(),
476                        qc.data.leaf_commit,
477                        qc.data.block_number,
478                        self,
479                        &event_sender,
480                    )
481                    .await;
482
483                    let view_number = qc.view_number() + 1;
484                    self.create_dependency_task_if_new(
485                        view_number,
486                        epoch_number,
487                        event_receiver,
488                        event_sender,
489                        Arc::clone(&event),
490                        epoch_transition_indicator,
491                    )
492                    .await?;
493                },
494            },
495            HotShotEvent::SendPayloadCommitmentAndMetadata(
496                _payload_commitment,
497                _builder_commitment,
498                _metadata,
499                view_number,
500                _fee,
501                _auction_result,
502            ) => {
503                let view_number = *view_number;
504
505                self.create_dependency_task_if_new(
506                    view_number,
507                    epoch_number,
508                    event_receiver,
509                    event_sender,
510                    Arc::clone(&event),
511                    EpochTransitionIndicator::NotInTransition,
512                )
513                .await?;
514            },
515            HotShotEvent::ViewSyncFinalizeCertificateRecv(certificate) => {
516                let epoch_number = certificate.data.epoch;
517                let epoch_membership = self
518                    .membership_coordinator
519                    .stake_table_for_epoch(epoch_number)
520                    .await
521                    .context(warn!("No Stake Table for Epoch = {:?}", epoch_number))?;
522
523                let membership_stake_table = epoch_membership.stake_table().await;
524                let membership_success_threshold = epoch_membership.success_threshold().await;
525
526                certificate
527                    .is_valid_cert(
528                        StakeTableEntries::<TYPES>::from(membership_stake_table).0,
529                        membership_success_threshold,
530                        &self.upgrade_lock,
531                    )
532                    .await
533                    .context(|e| {
534                        warn!(
535                            "View Sync Finalize certificate {:?} was invalid: {}",
536                            certificate.data(),
537                            e
538                        )
539                    })?;
540
541                let view_number = certificate.view_number;
542
543                self.create_dependency_task_if_new(
544                    view_number,
545                    epoch_number,
546                    event_receiver,
547                    event_sender,
548                    event,
549                    EpochTransitionIndicator::NotInTransition,
550                )
551                .await?;
552            },
553            HotShotEvent::QuorumProposalPreliminarilyValidated(proposal) => {
554                let view_number = proposal.data.view_number();
555                // All nodes get the latest proposed view as a proxy of `cur_view` of old.
556                if !self.update_latest_proposed_view(view_number).await {
557                    tracing::trace!("Failed to update latest proposed view");
558                }
559
560                self.create_dependency_task_if_new(
561                    view_number + 1,
562                    epoch_number,
563                    event_receiver,
564                    event_sender,
565                    Arc::clone(&event),
566                    epoch_transition_indicator,
567                )
568                .await?;
569            },
570            HotShotEvent::QuorumProposalSend(proposal, _) => {
571                let view = proposal.data.view_number();
572
573                ensure!(
574                    self.update_latest_proposed_view(view).await,
575                    "Failed to update latest proposed view"
576                );
577            },
578            HotShotEvent::VidDisperseSend(vid_disperse, _) => {
579                let view_number = vid_disperse.data.view_number();
580                self.create_dependency_task_if_new(
581                    view_number,
582                    epoch_number,
583                    event_receiver,
584                    event_sender,
585                    Arc::clone(&event),
586                    EpochTransitionIndicator::NotInTransition,
587                )
588                .await?;
589            },
590            HotShotEvent::ViewChange(view, epoch) => {
591                if epoch > &self.cur_epoch {
592                    self.cur_epoch = *epoch;
593                }
594                let keep_view = TYPES::View::new(view.saturating_sub(1));
595                self.cancel_tasks(keep_view);
596            },
597            HotShotEvent::Timeout(view, ..) => {
598                let keep_view = TYPES::View::new(view.saturating_sub(1));
599                self.cancel_tasks(keep_view);
600            },
601            HotShotEvent::NextEpochQc2Formed(Either::Left(next_epoch_qc)) => {
602                // Only update if the qc is from a newer view
603                let current_next_epoch_qc =
604                    self.consensus.read().await.next_epoch_high_qc().cloned();
605                ensure!(current_next_epoch_qc.is_none() ||
606                    next_epoch_qc.view_number > current_next_epoch_qc.unwrap().view_number,
607                    debug!("Received a next epoch QC for a view that was not > than our current next epoch high QC")
608                );
609
610                self.formed_next_epoch_quorum_certificates
611                    .insert(next_epoch_qc.view_number(), next_epoch_qc.clone());
612
613                handle_eqc_formed(
614                    next_epoch_qc.view_number(),
615                    next_epoch_qc.data.leaf_commit,
616                    next_epoch_qc.data.block_number,
617                    self,
618                    &event_sender,
619                )
620                .await;
621
622                let view_number = next_epoch_qc.view_number() + 1;
623                self.create_dependency_task_if_new(
624                    view_number,
625                    epoch_number,
626                    event_receiver,
627                    event_sender,
628                    Arc::clone(&event),
629                    epoch_transition_indicator,
630                )
631                .await?;
632            },
633            _ => {},
634        }
635        Ok(())
636    }
637
638    /// Cancel all tasks the consensus tasks has spawned before the given view
639    pub fn cancel_tasks(&mut self, view: TYPES::View) {
640        let keep = self.proposal_dependencies.split_off(&view);
641        while let Some((_, task)) = self.proposal_dependencies.pop_first() {
642            task.abort();
643        }
644        self.proposal_dependencies = keep;
645    }
646}

648#[async_trait]
649impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> TaskState
650    for QuorumProposalTaskState<TYPES, I, V>
651{
652    type Event = HotShotEvent<TYPES>;
653
654    async fn handle_event(
655        &mut self,
656        event: Arc<Self::Event>,
657        sender: &Sender<Arc<Self::Event>>,
658        receiver: &Receiver<Arc<Self::Event>>,
659    ) -> Result<()> {
660        self.handle(event, receiver.clone(), sender.clone()).await
661    }
662
663    fn cancel_subtasks(&mut self) {
664        while let Some((_, handle)) = self.proposal_dependencies.pop_first() {
665            handle.abort();
666        }
667    }
