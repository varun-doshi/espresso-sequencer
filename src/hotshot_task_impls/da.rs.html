42/// Tracks state of a DA task
43pub struct DaTaskState<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> {
44    /// Output events to application
45    pub output_event_stream: async_broadcast::Sender<Event<TYPES>>,
46
47    /// View number this view is executing in.
48    pub cur_view: TYPES::View,
49
50    /// Epoch number this node is executing in.
51    pub cur_epoch: Option<TYPES::Epoch>,
52
53    /// Reference to consensus. Leader will require a read lock on this.
54    pub consensus: OuterConsensus<TYPES>,
55
56    /// Membership for the DA committee and quorum committee.
57    /// We need the latter only for calculating the proper VID scheme
58    /// from the number of nodes in the quorum.
59    pub membership_coordinator: EpochMembershipCoordinator<TYPES>,
60
61    /// The underlying network
62    pub network: Arc<I::Network>,
63
64    /// A map of `DaVote` collector tasks.
65    pub vote_collectors: VoteCollectorsMap<TYPES, DaVote2<TYPES>, DaCertificate2<TYPES>, V>,
66
67    /// This Nodes public key
68    pub public_key: TYPES::SignatureKey,
69
70    /// This Nodes private key
71    pub private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
72
73    /// This state's ID
74    pub id: u64,
75
76    /// This node's storage ref
77    pub storage: Arc<RwLock<I::Storage>>,
78
79    /// Lock for a decided upgrade
80    pub upgrade_lock: UpgradeLock<TYPES, V>,
81}

83impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> DaTaskState<TYPES, I, V> {
84    /// main task event handler
85    #[instrument(skip_all, fields(id = self.id, view = *self.cur_view, epoch = self.cur_epoch.map(|x| *x)), name = "DA Main Task", level = "error", target = "DaTaskState")]
86    pub async fn handle(
87        &mut self,
88        event: Arc<HotShotEvent<TYPES>>,
89        event_stream: Sender<Arc<HotShotEvent<TYPES>>>,
90    ) -> Result<()> {
91        match event.as_ref() {
92            HotShotEvent::DaProposalRecv(proposal, sender) => {
93                let sender = sender.clone();
94                tracing::debug!(
95                    "DA proposal received for view: {:?}",
96                    proposal.data.view_number()
97                );
98                // ED NOTE: Assuming that the next view leader is the one who sends DA proposal for this view
99                let view = proposal.data.view_number();
100
101                // Allow a DA proposal that is one view older, in case we have voted on a quorum
102                // proposal and updated the view.
103                //
104                // Anything older is discarded because it is no longer relevant.
105                ensure!(
106                    self.cur_view <= view + 1,
107                    "Throwing away DA proposal that is more than one view older"
108                );
109
110                if let Some(entry) = self.consensus.read().await.saved_payloads().get(&view) {
111                    ensure!(entry.payload.encode() == proposal.data.encoded_transactions, error!(
112                      "Received DA proposal for view {:?} but we already have a payload for that view and they are not identical.  Throwing it away",
113                      view)
114                    );
115                }
116
117                let encoded_transactions_hash = Sha256::digest(&proposal.data.encoded_transactions);
118                let view_leader_key = self
119                    .membership_coordinator
120                    .membership_for_epoch(proposal.data.epoch)
121                    .await
122                    .context(warn!("No stake table for epoch"))?
123                    .leader(view)
124                    .await?;
125                ensure!(
126                    view_leader_key == sender,
127                    warn!(
128                      "DA proposal doesn't have expected leader key for view {} \n DA proposal is: {:?}",
129                      *view,
130                      proposal.data.clone()
131                    )
132                );
133
134                ensure!(
135                    view_leader_key.validate(&proposal.signature, &encoded_transactions_hash),
136                    warn!("Could not verify proposal.")
137                );
138
139                broadcast_event(
140                    Arc::new(HotShotEvent::DaProposalValidated(proposal.clone(), sender)),
141                    &event_stream,
142                )
143                .await;
144            },
145            HotShotEvent::DaProposalValidated(proposal, sender) => {
146                let cur_view = self.consensus.read().await.cur_view();
147                let view_number = proposal.data.view_number();
148                let epoch_number = proposal.data.epoch;
149                let membership = self
150                    .membership_coordinator
151                    .stake_table_for_epoch(epoch_number)
152                    .await
153                    .context(warn!("No stake table for epoch"))?;
154
155                ensure!(
156                  cur_view <= view_number + 1,
157                  debug!(
158                    "Validated DA proposal for prior view but it's too old now Current view {:?}, DA Proposal view {:?}", 
159                    cur_view,
160                    proposal.data.view_number()
161                  )
162                );
163
164                // Proposal is fresh and valid, notify the application layer
165                broadcast_event(
166                    Event {
167                        view_number,
168                        event: EventType::DaProposal {
169                            proposal: proposal.clone(),
170                            sender: sender.clone(),
171                        },
172                    },
173                    &self.output_event_stream,
174                )
175                .await;
176
177                ensure!(
178                    membership.has_da_stake(&self.public_key).await,
179                    debug!(
180                        "We were not chosen for consensus committee for view {:?} in epoch {:?}",
181                        view_number, epoch_number
182                    )
183                );
184                let total_weight =
185                    vid_total_weight::<TYPES>(membership.stake_table().await, epoch_number);
186
187                let mut next_epoch_total_weight = total_weight;
188                if epoch_number.is_some() {
189                    next_epoch_total_weight = vid_total_weight::<TYPES>(
190                        membership
191                            .next_epoch_stake_table()
192                            .await?
193                            .stake_table()
194                            .await,
195                        epoch_number.map(|epoch| epoch + 1),
196                    );
197                }
198
199                let version = self.upgrade_lock.version_infallible(view_number).await;
200
201                let txns = Arc::clone(&proposal.data.encoded_transactions);
202                let txns_clone = Arc::clone(&txns);
203                let metadata = proposal.data.metadata.encode();
204                let metadata_clone = metadata.clone();
205                let payload_commitment = spawn_blocking(move || {
206                    vid_commitment::<V>(&txns, &metadata, total_weight, version)
207                })
208                .await;
209                let payload_commitment = payload_commitment.unwrap();
210                let next_epoch_payload_commitment = if matches!(
211                    proposal.data.epoch_transition_indicator,
212                    EpochTransitionIndicator::InTransition
213                ) && self
214                    .upgrade_lock
215                    .epochs_enabled(proposal.data.view_number())
216                    .await
217                {
218                    let commit_result = spawn_blocking(move || {
219                        vid_commitment::<V>(
220                            &txns_clone,
221                            &metadata_clone,
222                            next_epoch_total_weight,
223                            version,
224                        )
225                    })
226                    .await;
227                    Some(commit_result.unwrap())
228                } else {
229                    None
230                };
231
232                self.storage
233                    .write()
234                    .await
235                    .append_da2(proposal, payload_commitment)
236                    .await
237                    .wrap()
238                    .context(error!("Failed to append DA proposal to storage"))?;
239                // Generate and send vote
240                let vote = DaVote2::create_signed_vote(
241                    DaData2 {
242                        payload_commit: payload_commitment,
243                        next_epoch_payload_commit: next_epoch_payload_commitment,
244                        epoch: epoch_number,
245                    },
246                    view_number,
247                    &self.public_key,
248                    &self.private_key,
249                    &self.upgrade_lock,
250                )
251                .await?;
252
253                tracing::debug!("Sending vote to the DA leader {:?}", vote.view_number());
254
255                broadcast_event(Arc::new(HotShotEvent::DaVoteSend(vote)), &event_stream).await;
256                let mut consensus_writer = self.consensus.write().await;
257
258                // Ensure this view is in the view map for garbage collection.
259
260                if let Err(e) =
261                    consensus_writer.update_da_view(view_number, epoch_number, payload_commitment)
262                {
263                    tracing::trace!("{e:?}");
264                }
265
266                let payload_with_metadata = Arc::new(PayloadWithMetadata {
267                    payload: TYPES::BlockPayload::from_bytes(
268                        proposal.data.encoded_transactions.as_ref(),
269                        &proposal.data.metadata,
270                    ),
271                    metadata: proposal.data.metadata.clone(),
272                });
273
274                // Record the payload we have promised to make available.
275                if let Err(e) =
276                    consensus_writer.update_saved_payloads(view_number, payload_with_metadata)
277                {
278                    tracing::trace!("{e:?}");
279                }
280                drop(consensus_writer);
281
282                // Optimistically calculate and update VID if we know that the primary network is down.
283                if self.network.is_primary_down() {
284                    let consensus =
285                        OuterConsensus::new(Arc::clone(&self.consensus.inner_consensus));
286                    let pk = self.private_key.clone();
287                    let public_key = self.public_key.clone();
288                    let chan = event_stream.clone();
289                    let upgrade_lock = self.upgrade_lock.clone();
290                    let next_epoch = epoch_number.map(|epoch| epoch + 1);
291
292                    let target_epoch = if membership.has_stake(&public_key).await {
293                        epoch_number
294                    } else if membership
295                        .next_epoch_stake_table()
296                        .await?
297                        .has_stake(&public_key)
298                        .await
299                    {
300                        next_epoch
301                    } else {
302                        bail!("Not calculating VID, the node doesn't belong to the current epoch or the next epoch.");
303                    };
304
305                    let membership = membership.clone();
306                    spawn(async move {
307                        Consensus::calculate_and_update_vid::<V>(
308                            OuterConsensus::new(Arc::clone(&consensus.inner_consensus)),
309                            view_number,
310                            target_epoch,
311                            membership.coordinator.clone(),
312                            &pk,
313                            &upgrade_lock,
314                        )
315                        .await;
316                        if let Some(Some(vid_share)) = consensus
317                            .read()
318                            .await
319                            .vid_shares()
320                            .get(&view_number)
321                            .map(|shares| shares.get(&public_key).cloned())
322                        {
323                            broadcast_event(
324                                Arc::new(HotShotEvent::VidShareRecv(
325                                    public_key.clone(),
326                                    vid_share.clone(),
327                                )),
328                                &chan,
329                            )
330                            .await;
331                        }
332                    });
333                }
334            },
335            HotShotEvent::DaVoteRecv(ref vote) => {
336                tracing::debug!("DA vote recv, Main Task {:?}", vote.view_number());
337                // Check if we are the leader and the vote is from the sender.
338                let view = vote.view_number();
339                let epoch = vote.data.epoch;
340                let membership = self
341                    .membership_coordinator
342                    .membership_for_epoch(epoch)
343                    .await
344                    .context(warn!("No stake table for epoch"))?;
345
346                ensure!(
347                    membership.leader(view).await? == self.public_key,
348                    debug!(
349                      "We are not the DA committee leader for view {} are we leader for next view? {}",
350                      *view,
351                      membership.leader(view + 1).await? == self.public_key
352                    )
353                );
354
355                handle_vote(
356                    &mut self.vote_collectors,
357                    vote,
358                    self.public_key.clone(),
359                    &membership,
360                    self.id,
361                    &event,
362                    &event_stream,
363                    &self.upgrade_lock,
364                    EpochTransitionIndicator::NotInTransition,
365                )
366                .await?;
367            },
368            HotShotEvent::ViewChange(view, epoch) => {
369                if *epoch > self.cur_epoch {
370                    self.cur_epoch = *epoch;
371                }
372
373                let view = *view;
374                ensure!(
375                    *self.cur_view < *view,
376                    info!("Received a view change to an older view.")
377                );
378
379                if *view - *self.cur_view > 1 {
380                    tracing::info!("View changed by more than 1 going to view {:?}", view);
381                }
382                self.cur_view = view;
383            },
384            HotShotEvent::BlockRecv(packed_bundle) => {
385                let PackedBundle::<TYPES> {
386                    encoded_transactions,
387                    metadata,
388                    view_number,
389                    ..
390                } = packed_bundle;
391                let view_number = *view_number;
392
393                // quick hash the encoded txns with sha256
394                let encoded_transactions_hash = Sha256::digest(encoded_transactions);
395
396                // sign the encoded transactions as opposed to the VID commitment
397                let signature =
398                    TYPES::SignatureKey::sign(&self.private_key, &encoded_transactions_hash)
399                        .wrap()?;
400
401                let epoch = self.cur_epoch;
402                let leader = self
403                    .membership_coordinator
404                    .membership_for_epoch(epoch)
405                    .await
406                    .context(warn!("No stake table for epoch"))?
407                    .leader(view_number)
408                    .await?;
409                if leader != self.public_key {
410                    tracing::debug!(
411                        "We are not the leader in the current epoch. Do not send the DA proposal"
412                    );
413                    return Ok(());
414                }
415                let epoch_transition_indicator =
416                    if self.consensus.read().await.is_high_qc_ge_root_block() {
417                        EpochTransitionIndicator::InTransition
418                    } else {
419                        EpochTransitionIndicator::NotInTransition
420                    };
421                let data: DaProposal2<TYPES> = DaProposal2 {
422                    encoded_transactions: Arc::clone(encoded_transactions),
423                    metadata: metadata.clone(),
424                    // Upon entering a new view we want to send a DA Proposal for the next view -> Is it always the case that this is cur_view + 1?
425                    view_number,
426                    epoch,
427                    epoch_transition_indicator,
428                };
429
430                let message = Proposal {
431                    data,
432                    signature,
433                    _pd: PhantomData,
434                };
435
436                broadcast_event(
437                    Arc::new(HotShotEvent::DaProposalSend(
438                        message.clone(),
439                        self.public_key.clone(),
440                    )),
441                    &event_stream,
442                )
443                .await;
444                let payload_with_metadata = Arc::new(PayloadWithMetadata {
445                    payload: TYPES::BlockPayload::from_bytes(
446                        encoded_transactions.as_ref(),
447                        metadata,
448                    ),
449                    metadata: metadata.clone(),
450                });
451                // Save the payload early because we might need it to calculate VID for the next epoch nodes.
452                if let Err(e) = self
453                    .consensus
454                    .write()
455                    .await
456                    .update_saved_payloads(view_number, payload_with_metadata)
457                {
458                    tracing::trace!("{e:?}");
459                }
460            },
461            _ => {},
462        }
463        Ok(())
464    }
465}
466
467#[async_trait]
468/// task state implementation for DA Task
469impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> TaskState
470    for DaTaskState<TYPES, I, V>
471{
472    type Event = HotShotEvent<TYPES>;
473
474    async fn handle_event(
475        &mut self,
476        event: Arc<Self::Event>,
477        sender: &Sender<Arc<Self::Event>>,
478        _receiver: &Receiver<Arc<Self::Event>>,
479    ) -> Result<()> {
480        self.handle(event, sender.clone()).await
481    }
482
483    fn cancel_subtasks(&mut self) {}
484}
