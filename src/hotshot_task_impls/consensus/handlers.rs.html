31/// Handle a `QuorumVoteRecv` event.
32pub(crate) async fn handle_quorum_vote_recv<
33    TYPES: NodeType,
34    I: NodeImplementation<TYPES>,
35    V: Versions,
36>(
37    vote: &QuorumVote2<TYPES>,
38    event: Arc<HotShotEvent<TYPES>>,
39    sender: &Sender<Arc<HotShotEvent<TYPES>>>,
40    task_state: &mut ConsensusTaskState<TYPES, I, V>,
41) -> Result<()> {
42    let in_transition = task_state
43        .consensus
44        .read()
45        .await
46        .is_high_qc_for_last_block();
47    let epoch_membership = task_state
48        .membership_coordinator
49        .membership_for_epoch(vote.data.epoch)
50        .await
51        .context(warn!("No stake table for epoch"))?;
52
53    let we_are_leader =
54        epoch_membership.leader(vote.view_number() + 1).await? == task_state.public_key;
55    ensure!(
56        in_transition || we_are_leader,
57        info!(
58            "We are not the leader for view {:?} and we are not in the epoch transition",
59            vote.view_number() + 1
60        )
61    );
62
63    let transition_indicator = if in_transition {
64        EpochTransitionIndicator::InTransition
65    } else {
66        EpochTransitionIndicator::NotInTransition
67    };
68    handle_vote(
69        &mut task_state.vote_collectors,
70        vote,
71        task_state.public_key.clone(),
72        &epoch_membership,
73        task_state.id,
74        &event,
75        sender,
76        &task_state.upgrade_lock,
77        transition_indicator.clone(),
78    )
79    .await?;
80
81    if vote.epoch().is_some()
82        && vote
83            .data
84            .block_number
85            .is_some_and(|b| is_epoch_transition(b, task_state.epoch_height))
86    {
87        // If the vote sender belongs to the next epoch, collect it separately to form the second QC
88        let has_stake = epoch_membership
89            .next_epoch_stake_table()
90            .await?
91            .has_stake(&vote.signing_key())
92            .await;
93        if has_stake {
94            handle_vote(
95                &mut task_state.next_epoch_vote_collectors,
96                &vote.clone().into(),
97                task_state.public_key.clone(),
98                // We eventually verify in `handle_vote` that we are the leader before assembling the certificate here,
99                // so we must request the full randomized stake table.
100                //
101                // I'm not sure this is really necessary, but I've opted not to modify the logic.
102                &epoch_membership.next_epoch().await?.clone(),
103                task_state.id,
104                &event,
105                sender,
106                &task_state.upgrade_lock,
107                transition_indicator,
108            )
109            .await?;
110        }
111    }
112
113    Ok(())
114}

116/// Handle a `TimeoutVoteRecv` event.
117pub(crate) async fn handle_timeout_vote_recv<
118    TYPES: NodeType,
119    I: NodeImplementation<TYPES>,
120    V: Versions,
121>(
122    vote: &TimeoutVote2<TYPES>,
123    event: Arc<HotShotEvent<TYPES>>,
124    sender: &Sender<Arc<HotShotEvent<TYPES>>>,
125    task_state: &mut ConsensusTaskState<TYPES, I, V>,
126) -> Result<()> {
127    let epoch_membership = task_state
128        .membership_coordinator
129        .membership_for_epoch(task_state.cur_epoch)
130        .await
131        .context(warn!("No stake table for epoch"))?;
132    // Are we the leader for this view?
133    ensure!(
134        epoch_membership.leader(vote.view_number() + 1).await? == task_state.public_key,
135        info!(
136            "We are not the leader for view {:?}",
137            vote.view_number() + 1
138        )
139    );
140
141    handle_vote(
142        &mut task_state.timeout_vote_collectors,
143        vote,
144        task_state.public_key.clone(),
145        &task_state
146            .membership_coordinator
147            .membership_for_epoch(vote.data.epoch)
148            .await?,
149        task_state.id,
150        &event,
151        sender,
152        &task_state.upgrade_lock,
153        EpochTransitionIndicator::NotInTransition,
154    )
155    .await?;
156
157    Ok(())
158}

160/// Send an event to the next leader containing the highest QC we have
161/// This is a necessary part of HotStuff 2 but not the original HotStuff
162///
163/// #Errors
164/// Returns and error if we can't get the version or the version doesn't
165/// yet support HS 2
166pub async fn send_high_qc<TYPES: NodeType, V: Versions, I: NodeImplementation<TYPES>>(
167    new_view_number: TYPES::View,
168    sender: &Sender<Arc<HotShotEvent<TYPES>>>,
169    receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
170    task_state: &mut ConsensusTaskState<TYPES, I, V>,
171) -> Result<()> {
172    let version = task_state.upgrade_lock.version(new_view_number).await?;
173    ensure!(
174        version >= V::Epochs::VERSION,
175        debug!("HotStuff 2 upgrade not yet in effect")
176    );
177
178    let consensus_reader = task_state.consensus.read().await;
179    let high_qc = consensus_reader.high_qc().clone();
180    let is_eqc = high_qc
181        .data
182        .block_number
183        .is_some_and(|b| is_last_block(b, task_state.epoch_height));
184    drop(consensus_reader);
185
186    if is_eqc {
187        let Some(next_epoch_high_qc) = wait_for_next_epoch_qc(
188            &high_qc,
189            &task_state.consensus,
190            task_state.timeout,
191            task_state.view_start_time,
192            receiver,
193        )
194        .await
195        else {
196            bail!("We've seen an extended QC but we don't have a corresponding next epoch extended QC");
197        };
198        tracing::debug!(
199            "Broadcasting Extended QC for view {:?} and epoch {:?}, my id {:?}.",
200            high_qc.view_number(),
201            high_qc.epoch(),
202            task_state.id
203        );
204        broadcast_event(
205            Arc::new(HotShotEvent::ExtendedQcSend(
206                high_qc,
207                next_epoch_high_qc,
208                task_state.public_key.clone(),
209            )),
210            sender,
211        )
212        .await;
213    } else {
214        let leader = task_state
215            .membership_coordinator
216            .membership_for_epoch(task_state.cur_epoch)
217            .await?
218            .leader(new_view_number)
219            .await?;
220
221        let (high_qc, maybe_next_epoch_qc) = if high_qc
222            .data
223            .block_number
224            .is_some_and(|b| is_epoch_transition(b, task_state.epoch_height))
225        {
226            let Some((qc, next_epoch_qc)) =
227                task_state.consensus.read().await.transition_qc().cloned()
228            else {
229                bail!("We don't have a transition QC");
230            };
231            validate_qc_and_next_epoch_qc(
232                &high_qc,
233                Some(&next_epoch_qc),
234                &task_state.consensus,
235                &task_state.membership_coordinator,
236                &task_state.upgrade_lock,
237                task_state.epoch_height,
238            )
239            .await?;
240            (qc, Some(next_epoch_qc))
241        } else {
242            (high_qc, None)
243        };
244        validate_qc_and_next_epoch_qc(
245            &high_qc,
246            maybe_next_epoch_qc.as_ref(),
247            &task_state.consensus,
248            &task_state.membership_coordinator,
249            &task_state.upgrade_lock,
250            task_state.epoch_height,
251        )
252        .await?;
253        tracing::trace!(
254            "Sending high QC for view {:?}, height {:?}",
255            high_qc.view_number(),
256            high_qc.data.block_number
257        );
258        broadcast_event(
259            Arc::new(HotShotEvent::HighQcSend(
260                high_qc,
261                maybe_next_epoch_qc,
262                leader,
263                task_state.public_key.clone(),
264            )),
265            sender,
266        )
267        .await;
268    }
269    Ok(())
270}

272/// Handle a `ViewChange` event.
273#[instrument(skip_all)]
274pub(crate) async fn handle_view_change<
275    TYPES: NodeType,
276    I: NodeImplementation<TYPES>,
277    V: Versions,
278>(
279    new_view_number: TYPES::View,
280    epoch_number: Option<TYPES::Epoch>,
281    sender: &Sender<Arc<HotShotEvent<TYPES>>>,
282    receiver: &Receiver<Arc<HotShotEvent<TYPES>>>,
283    task_state: &mut ConsensusTaskState<TYPES, I, V>,
284) -> Result<()> {
285    if epoch_number > task_state.cur_epoch {
286        task_state.cur_epoch = epoch_number;
287        if let Some(new_epoch) = epoch_number {
288            let _ = task_state.consensus.write().await.update_epoch(new_epoch);
289            tracing::info!("Progress: entered epoch {:>6}", *new_epoch);
290        }
291    }
292
293    ensure!(
294        new_view_number > task_state.cur_view,
295        "New view is not larger than the current view"
296    );
297
298    let old_view_number = task_state.cur_view;
299    tracing::debug!("Updating view from {old_view_number:?} to {new_view_number:?}");
300
301    if *old_view_number / 100 != *new_view_number / 100 {
302        tracing::info!("Progress: entered view {:>6}", *new_view_number);
303    }
304
305    // Send our high qc to the next leader immediately upon finishing a view.
306    // Part of HotStuff 2
307    let _ = send_high_qc(new_view_number, sender, receiver, task_state)
308        .await
309        .inspect_err(|e| {
310            tracing::debug!("High QC sending failed with error: {:?}", e);
311        });
312
313    // Move this node to the next view
314    task_state.cur_view = new_view_number;
315    task_state
316        .consensus
317        .write()
318        .await
319        .update_view(new_view_number)?;
320
321    // If we have a decided upgrade certificate, the protocol version may also have been upgraded.
322    let decided_upgrade_certificate_read = task_state
323        .upgrade_lock
324        .decided_upgrade_certificate
325        .read()
326        .await
327        .clone();
328    if let Some(cert) = decided_upgrade_certificate_read {
329        if new_view_number == cert.data.new_version_first_view {
330            tracing::error!(
331                "Version upgraded based on a decided upgrade cert: {:?}",
332                cert
333            );
334        }
335    }
336
337    // Spawn a timeout task if we did actually update view
338    let timeout = task_state.timeout;
339    let new_timeout_task = spawn({
340        let stream = sender.clone();
341        let view_number = new_view_number;
342        async move {
343            sleep(Duration::from_millis(timeout)).await;
344            broadcast_event(
345                Arc::new(HotShotEvent::Timeout(
346                    TYPES::View::new(*view_number),
347                    epoch_number,
348                )),
349                &stream,
350            )
351            .await;
352        }
353    });
354
355    // Cancel the old timeout task
356    std::mem::replace(&mut task_state.timeout_task, new_timeout_task).abort();
357
358    let old_view_leader_key = task_state
359        .membership_coordinator
360        .membership_for_epoch(task_state.cur_epoch)
361        .await
362        .context(warn!("No stake table for epoch"))?
363        .leader(old_view_number)
364        .await?;
365
366    let consensus_reader = task_state.consensus.read().await;
367    consensus_reader
368        .metrics
369        .current_view
370        .set(usize::try_from(task_state.cur_view.u64()).unwrap());
371    let cur_view_time = Utc::now().timestamp();
372    if old_view_leader_key == task_state.public_key {
373        #[allow(clippy::cast_precision_loss)]
374        consensus_reader
375            .metrics
376            .view_duration_as_leader
377            .add_point((cur_view_time - task_state.cur_view_time) as f64);
378    }
379    task_state.cur_view_time = cur_view_time;
380
381    // Do the comparison before the subtraction to avoid potential overflow, since
382    // `last_decided_view` may be greater than `cur_view` if the node is catching up.
383    if usize::try_from(task_state.cur_view.u64()).unwrap()
384        > usize::try_from(consensus_reader.last_decided_view().u64()).unwrap()
385    {
386        consensus_reader
387            .metrics
388            .number_of_views_since_last_decide
389            .set(
390                usize::try_from(task_state.cur_view.u64()).unwrap()
391                    - usize::try_from(consensus_reader.last_decided_view().u64()).unwrap(),
392            );
393    }
394
395    broadcast_event(
396        Event {
397            view_number: old_view_number,
398            event: EventType::ViewFinished {
399                view_number: old_view_number,
400            },
401        },
402        &task_state.output_event_stream,
403    )
404    .await;
405    Ok(())
406}

408/// Handle a `Timeout` event.
409#[instrument(skip_all)]
410pub(crate) async fn handle_timeout<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions>(
411    view_number: TYPES::View,
412    epoch: Option<TYPES::Epoch>,
413    sender: &Sender<Arc<HotShotEvent<TYPES>>>,
414    task_state: &mut ConsensusTaskState<TYPES, I, V>,
415) -> Result<()> {
416    ensure!(
417        task_state.cur_view <= view_number,
418        "Timeout event is for an old view"
419    );
420
421    ensure!(
422        task_state
423            .membership_coordinator
424            .stake_table_for_epoch(epoch)
425            .await
426            .context(warn!("No stake table for epoch"))?
427            .has_stake(&task_state.public_key)
428            .await,
429        debug!(
430            "We were not chosen for the consensus committee for view {:?}",
431            view_number
432        )
433    );
434
435    let vote = TimeoutVote2::create_signed_vote(
436        TimeoutData2::<TYPES> {
437            view: view_number,
438            epoch,
439        },
440        view_number,
441        &task_state.public_key,
442        &task_state.private_key,
443        &task_state.upgrade_lock,
444    )
445    .await
446    .wrap()
447    .context(error!("Failed to sign TimeoutData"))?;
448
449    broadcast_event(Arc::new(HotShotEvent::TimeoutVoteSend(vote)), sender).await;
450    broadcast_event(
451        Event {
452            view_number,
453            event: EventType::ViewTimeout { view_number },
454        },
455        &task_state.output_event_stream,
456    )
457    .await;
458
459    tracing::error!(
460        "We did not receive evidence for view {} in time, sending timeout vote for that view!",
461        *view_number
462    );
463
464    broadcast_event(
465        Event {
466            view_number,
467            event: EventType::ReplicaViewTimeout { view_number },
468        },
469        &task_state.output_event_stream,
470    )
471    .await;
472
473    let leader = task_state
474        .membership_coordinator
475        .membership_for_epoch(task_state.cur_epoch)
476        .await
477        .context(warn!("No stake table for epoch"))?
478        .leader(view_number)
479        .await;
480
481    let consensus_reader = task_state.consensus.read().await;
482    consensus_reader.metrics.number_of_timeouts.add(1);
483    if leader? == task_state.public_key {
484        consensus_reader.metrics.number_of_timeouts_as_leader.add(1);
485    }
486
487    Ok(())
