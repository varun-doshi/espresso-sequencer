44/// Task state for the Consensus task.
45pub struct ConsensusTaskState<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> {
46    /// Our public key
47    pub public_key: TYPES::SignatureKey,
48
49    /// Our Private Key
50    pub private_key: <TYPES::SignatureKey as SignatureKey>::PrivateKey,
51
52    /// Immutable instance state
53    pub instance_state: Arc<TYPES::InstanceState>,
54
55    /// The underlying network
56    pub network: Arc<I::Network>,
57
58    /// Membership for Quorum Certs/votes
59    pub membership_coordinator: EpochMembershipCoordinator<TYPES>,
60
61    /// A map of `QuorumVote` collector tasks.
62    pub vote_collectors: VoteCollectorsMap<TYPES, QuorumVote2<TYPES>, QuorumCertificate2<TYPES>, V>,
63
64    /// A map of `QuorumVote` collector tasks. They collect votes from the nodes in the next epoch.
65    pub next_epoch_vote_collectors: VoteCollectorsMap<
66        TYPES,
67        NextEpochQuorumVote2<TYPES>,
68        NextEpochQuorumCertificate2<TYPES>,
69        V,
70    >,
71
72    /// A map of `TimeoutVote` collector tasks.
73    pub timeout_vote_collectors:
74        VoteCollectorsMap<TYPES, TimeoutVote2<TYPES>, TimeoutCertificate2<TYPES>, V>,
75
76    /// The view number that this node is currently executing in.
77    pub cur_view: TYPES::View,
78
79    /// Timestamp this view starts at.
80    pub cur_view_time: i64,
81
82    /// The epoch number that this node is currently executing in.
83    pub cur_epoch: Option<TYPES::Epoch>,
84
85    /// Output events to application
86    pub output_event_stream: async_broadcast::Sender<Event<TYPES>>,
87
88    /// Timeout task handle
89    pub timeout_task: JoinHandle<()>,
90
91    /// View timeout from config.
92    pub timeout: u64,
93
94    /// A reference to the metrics trait.
95    pub consensus: OuterConsensus<TYPES>,
96
97    /// A reference to the storage trait.
98    pub storage: Arc<RwLock<I::Storage>>,
99
100    /// The node's id
101    pub id: u64,
102
103    /// Lock for a decided upgrade
104    pub upgrade_lock: UpgradeLock<TYPES, V>,
105
106    /// Number of blocks in an epoch, zero means there are no epochs
107    pub epoch_height: u64,
108
109    /// The time this view started
110    pub view_start_time: Instant,
111}

113impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> ConsensusTaskState<TYPES, I, V> {
114    /// Handles a consensus event received on the event stream
115    #[instrument(skip_all, fields(id = self.id, cur_view = *self.cur_view, cur_epoch = self.cur_epoch.map(|x| *x)), name = "Consensus replica task", level = "error", target = "ConsensusTaskState")]
116    pub async fn handle(
117        &mut self,
118        event: Arc<HotShotEvent<TYPES>>,
119        sender: Sender<Arc<HotShotEvent<TYPES>>>,
120        receiver: Receiver<Arc<HotShotEvent<TYPES>>>,
121    ) -> Result<()> {
122        match event.as_ref() {
123            HotShotEvent::QuorumVoteRecv(ref vote) => {
124                if let Err(e) =
125                    handle_quorum_vote_recv(vote, Arc::clone(&event), &sender, self).await
126                {
127                    tracing::debug!("Failed to handle QuorumVoteRecv event; error = {e}");
128                }
129            },
130            HotShotEvent::TimeoutVoteRecv(ref vote) => {
131                if let Err(e) =
132                    handle_timeout_vote_recv(vote, Arc::clone(&event), &sender, self).await
133                {
134                    tracing::debug!("Failed to handle TimeoutVoteRecv event; error = {e}");
135                }
136            },
137            HotShotEvent::ViewChange(new_view_number, epoch_number) => {
138                if let Err(e) =
139                    handle_view_change(*new_view_number, *epoch_number, &sender, &receiver, self)
140                        .await
141                {
142                    tracing::trace!("Failed to handle ViewChange event; error = {e}");
143                }
144                self.view_start_time = Instant::now();
145            },
146            HotShotEvent::Timeout(view_number, epoch) => {
147                if let Err(e) = handle_timeout(*view_number, *epoch, &sender, self).await {
148                    tracing::debug!("Failed to handle Timeout event; error = {e}");
149                }
150            },
151            HotShotEvent::ExtendedQc2Formed(eqc) => {
152                let cert_view = eqc.view_number();
153                let Some(cert_block_number) = eqc.data.block_number else {
154                    tracing::error!("Received extended QC but no block number");
155                    return Ok(());
156                };
157                let cert_epoch = epoch_from_block_number(cert_block_number, self.epoch_height);
158                tracing::error!(
159                    "Formed Extended QC for view {:?} and epoch {:?}.",
160                    cert_view,
161                    cert_epoch
162                );
163                // Transition to the new epoch by sending ViewChange
164                let next_epoch = TYPES::Epoch::new(cert_epoch + 1);
165                tracing::info!("Entering new epoch: {:?}", next_epoch);
166                broadcast_event(
167                    Arc::new(HotShotEvent::ViewChange(cert_view + 1, Some(next_epoch))),
168                    &sender,
169                )
170                .await;
171            },
172            HotShotEvent::ExtendedQcRecv(high_qc, next_epoch_high_qc, _) => {
173                if !high_qc
174                    .data
175                    .block_number
176                    .is_some_and(|bn| is_last_block(bn, self.epoch_height))
177                {
178                    tracing::warn!("Received extended QC but we can't verify the leaf is extended");
179                    return Ok(());
180                }
181                if let Err(e) = validate_qc_and_next_epoch_qc(
182                    high_qc,
183                    Some(next_epoch_high_qc),
184                    &self.consensus,
185                    &self.membership_coordinator,
186                    &self.upgrade_lock,
187                    self.epoch_height,
188                )
189                .await
190                {
191                    tracing::error!("Received invalid extended QC: {}", e);
192                    return Ok(());
193                }
194
195                let mut consensus_writer = self.consensus.write().await;
196                let high_qc_updated = consensus_writer.update_high_qc(high_qc.clone()).is_ok();
197                let next_high_qc_updated = consensus_writer
198                    .update_next_epoch_high_qc(next_epoch_high_qc.clone())
199                    .is_ok();
200                drop(consensus_writer);
201
202                self.storage
203                    .write()
204                    .await
205                    .update_high_qc2(high_qc.clone())
206                    .await
207                    .map_err(|_| warn!("Failed to update high QC"))?;
208                self.storage
209                    .write()
210                    .await
211                    .update_next_epoch_high_qc2(next_epoch_high_qc.clone())
212                    .await
213                    .map_err(|_| warn!("Failed to update next epoch high QC"))?;
214
215                tracing::debug!(
216                    "Received Extended QC for view {:?} and epoch {:?}.",
217                    high_qc.view_number(),
218                    high_qc.epoch()
219                );
220                if high_qc_updated || next_high_qc_updated {
221                    // Send ViewChange indicating new view and new epoch.
222                    let next_epoch = high_qc.data.epoch().map(|x| x + 1);
223                    tracing::info!("Entering new epoch: {:?}", next_epoch);
224                    broadcast_event(
225                        Arc::new(HotShotEvent::ViewChange(
226                            high_qc.view_number() + 1,
227                            next_epoch,
228                        )),
229                        &sender,
230                    )
231                    .await;
232                }
233            },
234            _ => {},
235        }
236
237        Ok(())
238    }
239}

241#[async_trait]
242impl<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> TaskState
243    for ConsensusTaskState<TYPES, I, V>
244{
245    type Event = HotShotEvent<TYPES>;
246
247    async fn handle_event(
248        &mut self,
249        event: Arc<Self::Event>,
250        sender: &Sender<Arc<Self::Event>>,
251        receiver: &Receiver<Arc<Self::Event>>,
252    ) -> Result<()> {
253        self.handle(event, sender.clone(), receiver.clone()).await
254    }
255
256    /// Joins all subtasks.
257    fn cancel_subtasks(&mut self) {
258        // Cancel the old timeout task
259        std::mem::replace(&mut self.timeout_task, tokio::spawn(async {})).abort();
260    }
261}
