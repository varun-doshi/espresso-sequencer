65/// Returns `OpenKeys` for KZG10 over BN254 curve from Aztec's SRS
66pub fn open_key() -> OpenKey<Bn254> {
67    let g = G1Affine::new_unchecked(MontFp!("1"), MontFp!("2"));
68    let h = G2Affine::new(
69        Fp2::new(
70            Fq::from(BigUint::from_str_radix(H[0], 16).unwrap()),
71            Fq::from(BigUint::from_str_radix(H[1], 16).unwrap()),
72        ),
73        Fp2::new(
74            Fq::from(BigUint::from_str_radix(H[2], 16).unwrap()),
75            Fq::from(BigUint::from_str_radix(H[3], 16).unwrap()),
76        ),
77    );
78    let beta_h = G2Affine::new(
79        Fp2::new(
80            Fq::from(BigUint::from_str_radix(BETA_H[0], 16).unwrap()),
81            Fq::from(BigUint::from_str_radix(BETA_H[1], 16).unwrap()),
82        ),
83        Fp2::new(
84            Fq::from(BigUint::from_str_radix(BETA_H[2], 16).unwrap()),
85            Fq::from(BigUint::from_str_radix(BETA_H[3], 16).unwrap()),
86        ),
87    );
88
89    OpenKey {
90        g,
91        h,
92        beta_h,
93        powers_of_g: vec![g],
94        powers_of_h: vec![h, beta_h],
95    }
96}

98/// an intermediate representation of the transcript parsed from abi.encode(transcript) from Solidity.
99#[derive(Clone, EthAbiType, EthAbiCodec)]
100pub struct ParsedTranscript {
101    pub(crate) state: H256,
102    pub(crate) transcript: Bytes,
103}
104
105impl FromStr for ParsedTranscript {
106    type Err = AbiError;
107    fn from_str(s: &str) -> Result<Self, Self::Err> {
108        let parsed: (ParsedTranscript,) = AbiDecode::decode_hex(s)?;
109        Ok(parsed.0)
110    }
111}

113impl From<SolidityTranscript> for ParsedTranscript {
114    fn from(t: SolidityTranscript) -> Self {
115        let (state, transcript) = t.internal();
116        Self {
117            state: H256::from_slice(&state),
118            transcript: transcript.into(),
119        }
120    }
121}

170        let mut g2_lsb = [0u8; 32];
171        let mut g2_msb = [0u8; 32];
172        g2_lsb.copy_from_slice(&g2_bytes[..32]);
173        g2_msb.copy_from_slice(&g2_bytes[32..]);
174
175        // since G2 point from the Aztec's SRS we use is fixed
176        // remove these sanity check if using other SRS

178        // ```rust
179        // let srs = ark_srs::kzg10::aztec20::setup(2u64.pow(6) as usize + 2).expect("Aztec SRS fail to load");
180        // println!("{}", hex::encode(jf_utils::to_bytes!(&srs.beta_h).unwrap()));
181        // ```
182        assert_eq!(
183            g2_lsb.encode_hex::<String>(),
184            String::from("b0838893ec1f237e8b07323b0744599f4e97b598b3b589bcc2bc37b8d5c41801")
185        );
186        assert_eq!(
187            g2_msb.encode_hex::<String>(),
188            String::from("c18393c0fa30fe4e8b038e357ad851eae8de9107584effe7c7f1f651b2010e26")
189        );
190
191        Self {
192            domain_size: U256::from(vk.domain_size),
193            num_inputs: U256::from(vk.num_inputs),
194            sigma_0: vk.sigma_comms[0].0.into(),
195            sigma_1: vk.sigma_comms[1].0.into(),

378// NOTE: this extra "middle-man" conversion is the unfortunate result of type in rust binding is auto-generated,
379// and are mutually oblivious of the types in Jellyfish, thus no From can be implemented at either side.
380// Since we cannot implement foreign traits on foreign types, we can only transit via our own `ParsedPlonkProof`.
381impl From<ParsedPlonkProof> for contract_bindings_ethers::i_plonk_verifier::PlonkProof {
382    fn from(p: ParsedPlonkProof) -> Self {
383        // parsed_proof is our own defined type, which share exactly the same structure and field types
384        // as the auto-generated rust binding types, thus, we can safely do mem::transmute()
385        unsafe { std::mem::transmute(p) }
386    }
387}
