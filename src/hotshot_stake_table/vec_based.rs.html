24/// a snapshot of the stake table
25#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
26struct StakeTableSnapshot<K1, K2> {
27    /// bls keys
28    pub bls_keys: Vec<K1>,
29    /// schnorr
30    pub schnorr_keys: Vec<K2>,
31    /// amount of stake
32    pub stake_amount: Vec<U256>,
33}
34
35impl<K1, K2> Default for StakeTableSnapshot<K1, K2> {
36    fn default() -> Self {
37        Self {
38            bls_keys: vec![],
39            schnorr_keys: vec![],
40            stake_amount: vec![],
41        }
42    }
43}

45/// Locally maintained stake table, generic over public key type `K`.
46/// Whose commitment is a rescue hash of all key-value pairs over field `F`.
47/// NOTE: the commitment is only available for the finalized versions, and is
48/// computed only once when it's finalized.
49#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
50pub struct StakeTable<K1, K2, F>
51where
52    K1: Eq + Hash + Clone + ToFields<F>,
53    K2: Eq + Hash + Clone + Default + ToFields<F>,
54    F: RescueParameter,
55{
56    /// upper bound on table size
57    capacity: usize,
58    /// The most up-to-date stake table, where the incoming transactions shall be performed on.
59    head: StakeTableSnapshot<K1, K2>,
60    /// The snapshot of stake table at the beginning of the current epoch
61    epoch_start: StakeTableSnapshot<K1, K2>,
62    /// The stake table used for leader election.
63    last_epoch_start: StakeTableSnapshot<K1, K2>,
64
65    /// Total stakes in the most update-to-date stake table
66    head_total_stake: U256,
67    /// Total stakes in the snapshot version `EpochStart`
68    epoch_start_total_stake: U256,
69    /// Total stakes in the snapshot version `LastEpochStart`
70    last_epoch_start_total_stake: U256,
71
72    /// Commitment of the stake table snapshot version `EpochStart`
73    /// We only support committing the finalized versions.
74    /// Commitment for a finalized version is a triple where
75    ///  - First item is the rescue hash of the bls keys
76    ///  - Second item is the rescue hash of the Schnorr keys
77    ///  - Third item is the rescue hash of all the stake amounts
78    epoch_start_comm: (F, F, F),
79
80    /// Commitment of the stake table snapshot version `LastEpochStart`
81    last_epoch_start_comm: (F, F, F),
82
83    /// The mapping from public keys to their location in the Merkle tree.
84    #[serde(skip)]
85    bls_mapping: HashMap<K1, usize>,
86}
87
88impl<K1, K2, F> StakeTableScheme for StakeTable<K1, K2, F>
89where
90    K1: Eq + Hash + Clone + ToFields<F>,
91    K2: Eq + Hash + Clone + Default + ToFields<F>,
92    F: RescueParameter,
93{
94    /// The stake table is indexed by BLS key
95    type Key = K1;
96    /// The auxiliary information is the associated Schnorr key
97    type Aux = K2;
98    type Amount = U256;
99    type Commitment = (F, F, F);
100    type LookupProof = ();
101    // TODO(Chengyu): Can we make it references?
102    type IntoIter = <Vec<(K1, U256, K2)> as ark_std::iter::IntoIterator>::IntoIter;
103
104    fn register(
105        &mut self,
106        new_key: Self::Key,
107        amount: Self::Amount,
108        aux: Self::Aux,
109    ) -> Result<(), StakeTableError> {
110        if self.bls_mapping.contains_key(&new_key) {
111            Err(StakeTableError::ExistingKey)
112        } else {
113            let pos = self.bls_mapping.len();
114            self.head.bls_keys.push(new_key.clone());
115            self.head.schnorr_keys.push(aux);
116            self.head.stake_amount.push(amount);
117            self.head_total_stake += amount;
118            self.bls_mapping.insert(new_key, pos);
119            Ok(())
120        }
121    }
122
123    fn deregister(&mut self, existing_key: &Self::Key) -> Result<(), StakeTableError> {
124        match self.bls_mapping.get(existing_key) {
125            Some(pos) => {
126                self.head_total_stake -= self.head.stake_amount[*pos];
127                self.head.stake_amount[*pos] = U256::zero();
128                Ok(())
129            },
130            None => Err(StakeTableError::KeyNotFound),
131        }
132    }
133
134    fn commitment(&self, version: SnapshotVersion) -> Result<Self::Commitment, StakeTableError> {
135        match version {
136            // IMPORTANT: we don't support committing the head version b/c it's not finalized.
137            SnapshotVersion::EpochStart => Ok(self.epoch_start_comm),
138            SnapshotVersion::LastEpochStart => Ok(self.last_epoch_start_comm),
139            _ => Err(StakeTableError::SnapshotUnsupported),
140        }
141    }
142
143    fn total_stake(&self, version: SnapshotVersion) -> Result<Self::Amount, StakeTableError> {
144        match version {
145            SnapshotVersion::Head => Ok(self.head_total_stake),
146            SnapshotVersion::EpochStart => Ok(self.epoch_start_total_stake),
147            SnapshotVersion::LastEpochStart => Ok(self.last_epoch_start_total_stake),
148            SnapshotVersion::BlockNum(_) => Err(StakeTableError::SnapshotUnsupported),
149        }
150    }
151
152    fn len(&self, version: SnapshotVersion) -> Result<usize, StakeTableError> {
153        Ok(self.version(&version)?.bls_keys.len())
154    }
155
156    fn contains_key(&self, key: &Self::Key) -> bool {
157        self.bls_mapping.contains_key(key)
158    }
159
160    fn lookup(
161        &self,
162        version: SnapshotVersion,
163        key: &Self::Key,
164    ) -> Result<Self::Amount, StakeTableError> {
165        let table = self.version(&version)?;
166        let pos = self.lookup_pos(key)?;
167        if pos >= table.bls_keys.len() {
168            Err(StakeTableError::KeyNotFound)
169        } else {
170            Ok(table.stake_amount[pos])
171        }
172    }
173
174    fn lookup_with_proof(
175        &self,
176        version: SnapshotVersion,
177        key: &Self::Key,
178    ) -> Result<(Self::Amount, Self::LookupProof), StakeTableError> {
179        let amount = self.lookup(version, key)?;
180        Ok((amount, ()))
181    }
182
183    fn lookup_with_aux_and_proof(
184        &self,
185        version: SnapshotVersion,
186        key: &Self::Key,
187    ) -> Result<(Self::Amount, Self::Aux, Self::LookupProof), StakeTableError> {
188        let table = self.version(&version)?;
189        let pos = self.lookup_pos(key)?;
190        if pos >= table.bls_keys.len() {
191            Err(StakeTableError::KeyNotFound)
192        } else {
193            Ok((table.stake_amount[pos], table.schnorr_keys[pos].clone(), ()))
194        }
195    }
196
197    fn update(
198        &mut self,
199        key: &Self::Key,
200        delta: Self::Amount,
201        negative: bool,
202    ) -> Result<Self::Amount, StakeTableError> {
203        let pos = self.lookup_pos(key)?;
204        if negative {
205            if delta > self.head.stake_amount[pos] {
206                return Err(StakeTableError::InsufficientFund);
207            }
208            self.head_total_stake -= delta;
209            self.head.stake_amount[pos] -= delta;
210        } else {
211            self.head_total_stake += delta;
212            self.head.stake_amount[pos] += delta;
213        }
214        Ok(self.head.stake_amount[pos])
215    }
216
217    fn sample(
218        &self,
219        rng: &mut (impl SeedableRng + CryptoRngCore),
220    ) -> Option<(&Self::Key, &Self::Amount)> {
221        let mut bytes = [0u8; 64];
222        rng.fill_bytes(&mut bytes);
223        let r = U512::from_big_endian(&bytes);
224        let m = U512::from(self.last_epoch_start_total_stake);
225        let mut pos: U256 = (r % m).try_into().unwrap(); // won't fail
226        let idx = 0;
227        while pos > self.last_epoch_start.stake_amount[idx] {
228            pos -= self.last_epoch_start.stake_amount[idx];
229        }
230        Some((
231            &self.last_epoch_start.bls_keys[idx],
232            &self.last_epoch_start.stake_amount[idx],
233        ))
234    }
235
236    fn try_iter(&self, version: SnapshotVersion) -> Result<Self::IntoIter, StakeTableError> {
237        let table = self.version(&version)?;
238        let owned = (0..table.bls_keys.len())
239            .map(|i| {
240                (
241                    table.bls_keys[i].clone(),
242                    table.stake_amount[i],
243                    table.schnorr_keys[i].clone(),
244                )
245            })
246            .collect::<Vec<_>>();
247        Ok(owned.into_iter())
248    }
249}

251impl<K1, K2, F> StakeTable<K1, K2, F>
252where
253    K1: Eq + Hash + Clone + ToFields<F>,
254    K2: Eq + Hash + Clone + Default + ToFields<F>,
255    F: RescueParameter,
256{
257    /// Initiating an empty stake table.
258    /// # Panics
259    /// If unable to evaluate a preimage
260    #[must_use]
261    pub fn new(capacity: usize) -> Self {
262        let bls_comm_preimage = vec![F::default(); capacity * <K1 as ToFields<F>>::SIZE];
263        let default_bls_comm =
264            VariableLengthRescueCRHF::<F, 1>::evaluate(&bls_comm_preimage).unwrap()[0];
265        let schnorr_comm_preimage = vec![F::default(); capacity * <K2 as ToFields<F>>::SIZE];
266        let default_schnorr_comm =
267            VariableLengthRescueCRHF::<F, 1>::evaluate(&schnorr_comm_preimage).unwrap()[0];
268        let stake_comm_preimage = vec![F::default(); capacity];
269        let default_stake_comm =
270            VariableLengthRescueCRHF::<F, 1>::evaluate(&stake_comm_preimage).unwrap()[0];
271        let default_comm = (default_bls_comm, default_schnorr_comm, default_stake_comm);
272        Self {
273            capacity,
274            head: StakeTableSnapshot::default(),
275            epoch_start: StakeTableSnapshot::default(),
276            last_epoch_start: StakeTableSnapshot::default(),
277            head_total_stake: U256::zero(),
278            epoch_start_total_stake: U256::zero(),
279            last_epoch_start_total_stake: U256::zero(),
280            bls_mapping: HashMap::new(),
281            epoch_start_comm: default_comm,
282            last_epoch_start_comm: default_comm,
283        }
284    }
285
286    /// Update the stake table when the epoch number advances, should be manually called.
287    pub fn advance(&mut self) {
288        // Could we avoid this `clone()`?
289        self.last_epoch_start = self.epoch_start.clone();
290        self.last_epoch_start_total_stake = self.epoch_start_total_stake;
291        self.last_epoch_start_comm = self.epoch_start_comm;
292        self.epoch_start = self.head.clone();
293        self.epoch_start_total_stake = self.head_total_stake;
294        self.epoch_start_comm = self.compute_head_comm();
295    }
296
297    /// Set the stake withheld by `key` to be `value`.
298    /// Return the previous stake if succeed.
299    /// # Errors
300    /// Errors if key is not in the stake table
301    pub fn set_value(&mut self, key: &K1, value: U256) -> Result<U256, StakeTableError> {
302        match self.bls_mapping.get(key) {
303            Some(pos) => {
304                let old_value = self.head.stake_amount[*pos];
305                self.head.stake_amount[*pos] = value;
306                self.head_total_stake -= old_value;
307                self.head_total_stake += value;
308                Ok(old_value)
309            },
310            None => Err(StakeTableError::KeyNotFound),
311        }
312    }
313
314    /// Helper function to recompute the stake table commitment for head version
315    /// Commitment of a stake table is a triple `(bls_keys_comm, schnorr_keys_comm, stake_amount_comm)`
316    /// TODO(Chengyu): The BLS verification keys doesn't implement Default. Thus we directly pad with `F::default()`.
317    fn compute_head_comm(&mut self) -> (F, F, F) {
318        let padding_len = self.capacity - self.head.bls_keys.len();
319        // Compute rescue hash for bls keys
320        let mut bls_comm_preimage = self
321            .head
322            .bls_keys
323            .iter()
324            .flat_map(ToFields::to_fields)
325            .collect::<Vec<_>>();
326        bls_comm_preimage.resize(self.capacity * <K1 as ToFields<F>>::SIZE, F::default());
327        let bls_comm = VariableLengthRescueCRHF::<F, 1>::evaluate(bls_comm_preimage).unwrap()[0];
328
329        // Compute rescue hash for Schnorr keys
330        let schnorr_comm_preimage = self
331            .head
332            .schnorr_keys
333            .iter()
334            .chain(ark_std::iter::repeat(&K2::default()).take(padding_len))
335            .flat_map(ToFields::to_fields)
336            .collect::<Vec<_>>();
337        let schnorr_comm =
338            VariableLengthRescueCRHF::<F, 1>::evaluate(schnorr_comm_preimage).unwrap()[0];
339
340        // Compute rescue hash for stake amounts
341        let mut stake_comm_preimage = self
342            .head
343            .stake_amount
344            .iter()
345            .map(|x| u256_to_field(x))
346            .collect::<Vec<_>>();
347        stake_comm_preimage.resize(self.capacity, F::default());
348        let stake_comm =
349            VariableLengthRescueCRHF::<F, 1>::evaluate(stake_comm_preimage).unwrap()[0];
350        (bls_comm, schnorr_comm, stake_comm)
351    }
352
353    /// Return the index of a given key.
354    /// Err if the key doesn't exists
355    fn lookup_pos(&self, key: &K1) -> Result<usize, StakeTableError> {
356        match self.bls_mapping.get(key) {
357            Some(pos) => Ok(*pos),
358            None => Err(StakeTableError::KeyNotFound),
359        }
360    }
361
362    /// returns the snapshot version
363    fn version(
364        &self,
365        version: &SnapshotVersion,
366    ) -> Result<&StakeTableSnapshot<K1, K2>, StakeTableError> {
367        match version {
368            SnapshotVersion::Head => Ok(&self.head),
369            SnapshotVersion::EpochStart => Ok(&self.epoch_start),
370            SnapshotVersion::LastEpochStart => Ok(&self.last_epoch_start),
371            SnapshotVersion::BlockNum(_) => Err(StakeTableError::SnapshotUnsupported),
372        }
373    }
374}

387#[cfg(test)]
388mod tests {
389    use ark_std::{rand::SeedableRng, vec::Vec};
390    use hotshot_types::traits::stake_table::{SnapshotVersion, StakeTableError, StakeTableScheme};
391    use jf_signature::{
392        bls_over_bn254::BLSOverBN254CurveSignatureScheme, schnorr::SchnorrSignatureScheme,
393        SignatureScheme,
394    };
395    use primitive_types::U256;
396
397    use super::{
398        config::{FieldType as F, QCVerKey, StateVerKey},
399        StakeTable,
400    };
401
402    #[test]
403    fn crypto_test_stake_table() -> Result<(), StakeTableError> {
404        let mut st = StakeTable::<QCVerKey, StateVerKey, F>::default();
405        let mut pseudo_rng = jf_utils::test_rng();
406        let keys = (0..10)
407            .map(|_| {
408                (
409                    BLSOverBN254CurveSignatureScheme::key_gen(&(), &mut pseudo_rng)
410                        .unwrap()
411                        .1,
412                    SchnorrSignatureScheme::key_gen(&(), &mut pseudo_rng)
413                        .unwrap()
414                        .1,
415                )
416            })
417            .collect::<Vec<_>>();
418        assert_eq!(st.total_stake(SnapshotVersion::Head)?, U256::from(0));
419
420        // Registering keys
421        keys.iter()
422            .take(4)
423            .for_each(|key| st.register(key.0, U256::from(100), key.1.clone()).unwrap());
424        assert_eq!(st.total_stake(SnapshotVersion::Head)?, U256::from(400));
425        assert_eq!(st.total_stake(SnapshotVersion::EpochStart)?, U256::from(0));
426        assert_eq!(
427            st.total_stake(SnapshotVersion::LastEpochStart)?,
428            U256::from(0)
429        );
430        // set to zero for further sampling test
431        assert_eq!(
432            st.set_value(&keys[1].0, U256::from(0)).unwrap(),
433            U256::from(100)
434        );
435        st.advance();
436        keys.iter()
437            .skip(4)
438            .take(3)
439            .for_each(|key| st.register(key.0, U256::from(100), key.1.clone()).unwrap());
440        assert_eq!(st.total_stake(SnapshotVersion::Head)?, U256::from(600));
441        assert_eq!(
442            st.total_stake(SnapshotVersion::EpochStart)?,
443            U256::from(300)
444        );
445        assert_eq!(
446            st.total_stake(SnapshotVersion::LastEpochStart)?,
447            U256::from(0)
448        );
449        st.advance();
450        keys.iter()
451            .skip(7)
452            .for_each(|key| st.register(key.0, U256::from(100), key.1.clone()).unwrap());
453        assert_eq!(st.total_stake(SnapshotVersion::Head)?, U256::from(900));
454        assert_eq!(
455            st.total_stake(SnapshotVersion::EpochStart)?,
456            U256::from(600)
457        );
458        assert_eq!(
459            st.total_stake(SnapshotVersion::LastEpochStart)?,
460            U256::from(300)
461        );
462
463        // No duplicate register
464        assert!(st
465            .register(keys[0].0, U256::from(100), keys[0].1.clone())
466            .is_err());
467        // The 9-th key is still in head stake table
468        assert!(st.lookup(SnapshotVersion::EpochStart, &keys[9].0).is_err());
469        assert!(st.lookup(SnapshotVersion::EpochStart, &keys[5].0).is_ok());
470        // The 6-th key is still frozen
471        assert!(st
472            .lookup(SnapshotVersion::LastEpochStart, &keys[6].0)
473            .is_err());
474        assert!(st
475            .lookup(SnapshotVersion::LastEpochStart, &keys[2].0)
476            .is_ok());
477
478        // Set value shall return the old value
479        assert_eq!(
480            st.set_value(&keys[0].0, U256::from(101)).unwrap(),
481            U256::from(100)
482        );
483        assert_eq!(st.total_stake(SnapshotVersion::Head)?, U256::from(901));
484        assert_eq!(
485            st.total_stake(SnapshotVersion::EpochStart)?,
486            U256::from(600)
487        );
488
489        // Update that results in a negative stake
490        assert!(st.update(&keys[0].0, U256::from(1000), true).is_err());
491        // Update should return the updated stake
492        assert_eq!(
493            st.update(&keys[0].0, U256::from(1), true).unwrap(),
494            U256::from(100)
495        );
496        assert_eq!(
497            st.update(&keys[0].0, U256::from(100), false).unwrap(),
498            U256::from(200)
499        );
500
501        // Commitment test
502        assert!(st.commitment(SnapshotVersion::Head).is_err());
503        assert!(st.commitment(SnapshotVersion::EpochStart).is_ok());
504        assert!(st.commitment(SnapshotVersion::LastEpochStart).is_ok());
505
506        // Random test for sampling keys
507        let mut rng = rand_chacha::ChaCha20Rng::seed_from_u64(41u64);
508        for _ in 0..100 {
509            let (_key, value) = st.sample(&mut rng).unwrap();
510            // Sampled keys should have positive stake
511            assert!(value > &U256::from(0));
512        }
513
514        // Test for try_iter
515        for (i, (k1, _, k2)) in st.try_iter(SnapshotVersion::Head).unwrap().enumerate() {
516            assert_eq!((k1, k2), keys[i]);
517        }
518
519        Ok(())
520    }
