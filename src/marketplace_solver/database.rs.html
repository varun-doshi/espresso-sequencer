11// PgPool is wrapped in an Arc internally so cloning here increments the reference count
12#[derive(Clone)]
13pub struct PostgresClient(PgPool);
14
15impl PostgresClient {
16    pub async fn connect(opts: DatabaseOptions) -> anyhow::Result<Self> {
17        let DatabaseOptions {
18            url,
19            host,
20            port,
21            db_name,
22            username,
23            password,
24            max_connections,
25            acquire_timeout,
26            require_ssl,
27            migrations,
28            reset,
29        } = opts;
30
31        let mut options = PgPoolOptions::new();
32
33        let postgres_url: Url = match url {
34            Some(url) => url.parse()?,
35            None => {
36                let host = host.context("host not provided")?;
37
38                let mut connect_opts = PgConnectOptions::new()
39                    .host(&host)
40                    .ssl_mode(PgSslMode::Allow);
41
42                if let Some(port) = port {
43                    connect_opts = connect_opts.port(port);
44                }
45
46                if let Some(username) = username {
47                    connect_opts = connect_opts.username(&username);
48                }
49
50                if let Some(password) = password {
51                    connect_opts = connect_opts.password(&password);
52                }
53
54                if let Some(db_name) = db_name {
55                    connect_opts = connect_opts.database(&db_name);
56                }
57
58                if require_ssl {
59                    connect_opts = connect_opts.ssl_mode(PgSslMode::Require)
60                }
61
62                connect_opts.to_url_lossy()
63            },
64        };
65
66        if let Some(max_connections) = max_connections {
67            options = options.max_connections(max_connections);
68        }
69
70        if let Some(acquire_timeout) = acquire_timeout {
71            options = options.acquire_timeout(acquire_timeout);
72        }
73
74        let connection = options.connect(postgres_url.as_str()).await?;
75
76        if reset {
77            sqlx::query("DROP SCHEMA public CASCADE;")
78                .execute(&connection)
79                .await
80                .unwrap();
81            sqlx::query("CREATE SCHEMA public;")
82                .execute(&connection)
83                .await
84                .unwrap();
85        }
86
87        if migrations {
88            sqlx::migrate!("./migrations").run(&connection).await?;
89        }
90
91        Ok(Self(connection))
92    }
93
94    pub fn pool(&self) -> &PgPool {
95        &self.0
96    }
97
98    pub async fn acquire(&self) -> Result<PoolConnection<Postgres>, Error> {
99        self.0.acquire().await
100    }
101}
