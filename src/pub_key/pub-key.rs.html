1use std::str::FromStr;

3use anyhow::bail;
4use clap::Parser;
5use espresso_types::{PrivKey, PubKey};
6use hotshot::{traits::implementations::derive_libp2p_peer_id, types::BLSPubKey};
7use hotshot_types::{
8    light_client::{StateKeyPair, StateSignKey},
9    traits::signature_key::SignatureKey,

11use tagged_base64::TaggedBase64;

13#[derive(Clone, Debug)]
14enum PrivateKey {
15    Bls(PrivKey),
16    Schnorr(StateSignKey),
17}
18
19impl FromStr for PrivateKey {
20    type Err = anyhow::Error;
21
22    fn from_str(s: &str) -> Result<Self, Self::Err> {
23        if let Ok(key) = TaggedBase64::parse(s)?.try_into() {
24            Ok(Self::Bls(key))
25        } else if let Ok(key) = TaggedBase64::parse(s)?.try_into() {
26            Ok(Self::Schnorr(key))
27        } else {
28            bail!("unrecognized key type")
29        }
30    }
31}
32
33/// Get the public key corresponding to a private key.
34#[derive(Clone, Debug, Parser)]
35struct Options {
36    /// The private key to get the public key for.
37    key: PrivateKey,
38
39    // Whether or not to derive the libp2p peer ID from the private key.
40    #[clap(long, short)]
41    libp2p: bool,
42}
43
44fn main() {
45    let opt = Options::parse();
46
47    match (opt.libp2p, opt.key) {
48        // Non-libp2p
49        (false, PrivateKey::Bls(key)) => println!("{}", PubKey::from_private(&key)),
50        (false, PrivateKey::Schnorr(key)) => {
51            println!("{}", StateKeyPair::from_sign_key(key).ver_key())
52        },
53
54        // Libp2p
55        (true, PrivateKey::Bls(key)) => {
56            println!(
57                "{}",
58                derive_libp2p_peer_id::<BLSPubKey>(&key).expect("Failed to derive libp2p peer ID")
59            );
60        },
61        (true, _) => {
62            eprintln!("Key type unsupported for libp2p peer ID derivation");
63        },
64    }
65}
