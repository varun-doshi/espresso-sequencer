57    async fn create(opt: Self::Options, provider: Provider, reset: bool) -> anyhow::Result<Self> {
58        let fetch_limit = opt.fetch_rate_limit;
59        let active_fetch_delay = opt.active_fetch_delay;
60        let chunk_fetch_delay = opt.chunk_fetch_delay;
61        let mut cfg = Config::try_from(&opt)?;
62
63        if reset {
64            cfg = cfg.reset_schema();
65        }
66
67        let mut builder = cfg.builder(provider).await?;
68
69        if let Some(limit) = fetch_limit {
70            builder = builder.with_rate_limit(limit);
71        }
72
73        if opt.lightweight {
74            tracing::warn!("enabling light weight mode..");
75            builder = builder.leaf_only();
76        }
77
78        if let Some(delay) = active_fetch_delay {
79            builder = builder.with_active_fetch_delay(delay);
80        }
81        if let Some(delay) = chunk_fetch_delay {
82            builder = builder.with_chunk_fetch_delay(delay);
83        }
84
85        builder.build().await
86    }
87}

169        &self,
170        instance: &NodeState,
171        height: u64,
172        view: ViewNumber,
173    ) -> anyhow::Result<BlocksFrontier> {
174        let mut tx = self.read().await.context(format!(
175            "opening transaction to fetch frontier at height {height}"
176        ))?;

187        // directly.
188        if height < block_height {
189            load_frontier(&mut tx, height).await
190        } else {
191            // If we do not have the exact snapshot we need, we can try going back to the last

195            match state.block_merkle_tree.lookup(height - 1) {
196                LookupResult::Ok(_, proof) => Ok(proof),
197                _ => {
198                    bail!("state snapshot {view:?},{height} was found but does not contain frontier at height {}; this should not be possible", height - 1);

214    async fn get_leaf_chain(&self, height: u64) -> anyhow::Result<Vec<Leaf2>> {
215        let mut tx = self
216            .read()
217            .await
218            .context(format!("opening transaction to fetch leaf at {height}"))?;
219        let leaf = tx
220            .get_leaf((height as usize).into())
221            .await
222            .context(format!("leaf {height} not available"))?;
223        let mut last_leaf: Leaf2 = leaf.leaf().clone();
224        let mut chain = vec![last_leaf.clone()];
225        let mut h = height + 1;
226
227        let mut found = false;
228        while !found {
229            for leaf in tx.get_leaves(h).await? {
230                let leaf = leaf.leaf();
231
232                if leaf.justify_qc().view_number() == last_leaf.view_number() {
233                    chain.push(leaf.clone());
234                } else {
235                    h += 1;
236                    continue;
237                }
238
239                // just one away from deciding
240                if leaf.view_number() == last_leaf.view_number() + 1 {
241                    last_leaf = leaf.clone();
242                    h += 1;
243                    found = true;
244                    break;
245                }
246                h += 1;
247                last_leaf = leaf.clone();
248            }
249        }
250
251        found = false;
252        while !found {
253            for leaf in tx.get_leaves(h).await? {
254                let leaf = leaf.leaf();
255                if leaf.justify_qc().view_number() == last_leaf.view_number() {
256                    chain.push(leaf.clone());
257                    found = true;
258                    break;
259                }
260                h += 1;
261            }
262        }
263
264        Ok(chain)
265    }
266}

342async fn load_reward_accounts<Mode: TransactionMode>(
343    tx: &mut Transaction<Mode>,
344    height: u64,
345    accounts: &[RewardAccount],
346) -> anyhow::Result<(RewardMerkleTree, Leaf2)> {
347    let leaf = tx
348        .get_leaf(LeafId::<SeqTypes>::from(height as usize))
349        .await
350        .context(format!("leaf {height} not available"))?;
351    let header = leaf.header();
352
353    let Some(merkle_root) = header.reward_merkle_tree_root() else {
354        bail!("reward merkle tree root not available");
355    };
356
357    let mut snapshot = RewardMerkleTree::from_commitment(merkle_root);
358    for account in accounts {
359        let proof = tx
360            .get_path(
361                Snapshot::<SeqTypes, RewardMerkleTree, { RewardMerkleTree::ARITY }>::Index(
362                    header.height(),
363                ),
364                *account,
365            )
366            .await
367            .context(format!(
368                "fetching reward account {account}; height {}",
369                header.height()
370            ))?;
371        match proof.proof.first().context(format!(
372            "empty proof for reward account {account}; height {}",
373            header.height()
374        ))? {
375            MerkleNode::Leaf { pos, elem, .. } => {
376                snapshot.remember(*pos, *elem, proof)?;
377            },
378            MerkleNode::Empty => {
379                snapshot.non_membership_remember(*account, proof)?;
380            },
381            _ => {
382                bail!("Invalid proof");
383            },
384        }
385    }
386
387    Ok((snapshot, leaf.leaf().clone()))
388}

447#[tracing::instrument(skip(instance, tx))]
448async fn reconstruct_state<Mode: TransactionMode>(
449    instance: &NodeState,
450    tx: &mut Transaction<Mode>,
451    from_height: u64,
452    to_view: ViewNumber,
453    fee_accounts: Option<&[FeeAccount]>,
454    reward_accounts: Option<&[RewardAccount]>,
455) -> anyhow::Result<(ValidatedState, Leaf2)> {
456    tracing::info!("attempting to reconstruct fee state");
457    let from_leaf = tx
458        .get_leaf((from_height as usize).into())
459        .await
460        .context(format!("leaf {from_height} not available"))?;
461    let from_leaf: Leaf2 = from_leaf.leaf().clone();
462    ensure!(
463        from_leaf.view_number() < to_view,
464        "state reconstruction: starting state {:?} must be before ending state {to_view:?}",
465        from_leaf.view_number(),
466    );
467
468    // Get the sequence of headers we will be applying to compute the latest state.
469    let mut leaves = VecDeque::new();
470    let to_leaf = get_leaf_from_proposal(tx, "view = $1", &(to_view.u64() as i64))
471        .await
472        .context(format!(
473            "unable to reconstruct state because leaf {to_view:?} is not available"
474        ))?;
475    let mut parent = to_leaf.parent_commitment();
476    tracing::debug!(?to_leaf, ?parent, view = ?to_view, "have required leaf");
477    leaves.push_front(to_leaf.clone());
478    while parent != Committable::commit(&from_leaf) {
479        let leaf = get_leaf_from_proposal(tx, "leaf_hash = $1", &parent.to_string())
480            .await
481            .context(format!(
482                "unable to reconstruct state because leaf {parent} is not available"
483            ))?;
484        parent = leaf.parent_commitment();
485        tracing::debug!(?leaf, ?parent, "have required leaf");
486        leaves.push_front(leaf);
487    }
488
489    // Get the initial state.
490    let mut parent = from_leaf;
491    let mut state = ValidatedState::from_header(parent.block_header());
492
493    // Pre-load the state with the accounts we care about to ensure they will be present in the
494    // final state.
495    let mut catchup = NullStateCatchup::default();
496    if let Some(accounts) = fee_accounts {
497        let mut accounts = accounts.iter().copied().collect::<HashSet<_>>();
498        // Add in all the accounts we will need to replay any of the headers, to ensure that we don't
499        // need to do catchup recursively.
500        let dependencies =
501            fee_header_dependencies(&mut catchup, tx, instance, &parent, &leaves).await?;
502        accounts.extend(dependencies);
503        let accounts = accounts.into_iter().collect::<Vec<_>>();
504        state.fee_merkle_tree = load_accounts(tx, from_height, &accounts)
505            .await
506            .context("unable to reconstruct state because accounts are not available at origin")?
507            .0;
508        ensure!(
509            state.fee_merkle_tree.commitment() == parent.block_header().fee_merkle_tree_root(),
510            "loaded fee state does not match parent header"
511        );
512    }
513
514    if let Some(accounts) = reward_accounts {
515        // TODO: don't think this is needed?
516        // What if we are reconstructing a state from the leaf which is on pre-epoch version?
517        // that is okie?
518        // if parent.block_header().reward_merkle_tree_root().is_none() {
519        //     bail!("reward merkle tree root not set");
520        // }
521
522        let mut accounts = accounts.iter().copied().collect::<HashSet<_>>();
523
524        let dependencies =
525            reward_header_dependencies(&mut catchup, tx, instance, &parent, &leaves).await?;
526        accounts.extend(dependencies);
527        let accounts = accounts.into_iter().collect::<Vec<_>>();
528        state.reward_merkle_tree = load_reward_accounts(tx, from_height, &accounts)
529            .await
530            .context(
531                "unable to reconstruct state because reward accounts are not available at origin",
532            )?
533            .0;
534        ensure!(
535            Some(state.reward_merkle_tree.commitment())
536                == parent.block_header().reward_merkle_tree_root(),
537            "loaded reward state does not match parent header"
538        );
539    }
540
541    // We need the blocks frontier as well, to apply the STF.
542    let frontier = load_frontier(tx, from_height)
543        .await
544        .context("unable to reconstruct state because frontier is not available at origin")?;
545    match frontier
546        .proof
547        .first()
548        .context("empty proof for frontier at origin")?
549    {
550        MerkleNode::Leaf { pos, elem, .. } => state
551            .block_merkle_tree
552            .remember(*pos, *elem, frontier)
553            .context("failed to remember frontier")?,
554        _ => bail!("invalid frontier proof"),
555    }
556
557    // Apply subsequent headers to compute the later state.
558    for proposal in leaves {
559        state = compute_state_update(&state, instance, &catchup, &parent, &proposal)
560            .await
561            .context(format!(
562                "unable to reconstruct state because state update {} failed",
563                proposal.height(),
564            ))?
565            .0;
566        parent = proposal;
567    }
568
569    tracing::info!(from_height, ?to_view, "successfully reconstructed state");
570    Ok((state, to_leaf))
571}

573/// Get the dependencies needed to apply the STF to the given list of headers.
574///
575/// Returns
576/// * A state catchup implementation seeded with all the chain configs required to apply the headers
577///   in `leaves`
578/// * The set of accounts that must be preloaded to apply these headers
579async fn fee_header_dependencies<Mode: TransactionMode>(
580    catchup: &mut NullStateCatchup,
581    tx: &mut Transaction<Mode>,
582    instance: &NodeState,
583    mut parent: &Leaf2,
584    leaves: impl IntoIterator<Item = &Leaf2>,
585) -> anyhow::Result<HashSet<FeeAccount>> {
586    let mut accounts = HashSet::default();
587
588    for proposal in leaves {
589        let header = proposal.block_header();
590        let height = header.height();
591        let view = proposal.view_number();
592        tracing::debug!(height, ?view, "fetching dependencies for proposal");
593
594        let header_cf = header.chain_config();
595        let chain_config = if header_cf.commit() == instance.chain_config.commit() {
596            instance.chain_config
597        } else {
598            match header_cf.resolve() {
599                Some(cf) => cf,
600                None => {
601                    tracing::info!(
602                        height,
603                        ?view,
604                        commit = %header_cf.commit(),
605                        "chain config not available, attempting to load from storage",
606                    );
607                    let cf = load_chain_config(tx, header_cf.commit())
608                        .await
609                        .context(format!(
610                            "loading chain config {} for header {},{:?}",
611                            header_cf.commit(),
612                            header.height(),
613                            proposal.view_number()
614                        ))?;
615
616                    // If we had to fetch a chain config now, store it in the catchup implementation
617                    // so the STF will be able to look it up later.
618                    catchup.add_chain_config(cf);
619                    cf
620                },
621            }
622        };
623
624        accounts.insert(chain_config.fee_recipient);
625        accounts.extend(
626            get_l1_deposits(instance, header, parent, chain_config.fee_contract)
627                .await
628                .into_iter()
629                .map(|fee| fee.account()),
630        );
631        accounts.extend(header.fee_info().accounts());
632        parent = proposal;
633    }
634    Ok(accounts)
635}

637async fn reward_header_dependencies<Mode: TransactionMode>(
638    catchup: &mut NullStateCatchup,
639    tx: &mut Transaction<Mode>,
640    instance: &NodeState,
641    parent: &Leaf2,
642    leaves: impl IntoIterator<Item = &Leaf2>,
643) -> anyhow::Result<HashSet<RewardAccount>> {
644    let mut reward_accounts = HashSet::default();
645    let epoch_height = instance.epoch_height;
646
647    let Some(epoch_height) = epoch_height else {
648        bail!("epoch height not set");
649    };
650
651    // we apply reward function in each block
652    // however, the stake table can only be changed every epoch
653    // so in order to get all the reward accounts that need to be loaded, we can call membership for every epoch
654    let leaves: Vec<_> = leaves.into_iter().collect();
655
656    if leaves.is_empty() {
657        bail!(format!("no leaves found for parent={parent:?}"));
658    }
659
660    let last_leaf = leaves.last().unwrap();
661
662    let from_epoch = epoch_from_block_number(parent.height(), epoch_height);
663    let to_epoch = epoch_from_block_number(last_leaf.height(), epoch_height);
664
665    let coordinator = instance.coordinator.clone();
666    for epoch in from_epoch..=to_epoch {
667        let epoch_membership = coordinator
668            .membership_for_epoch(Some(EpochNumber::new(epoch)))
669            .await?;
670        let membership = epoch_membership.coordinator.membership().read().await;
671        let validators = membership.validators(&EpochNumber::new(epoch))?;
672
673        for (validator, config) in validators {
674            let delegators: Vec<RewardAccount> = config
675                .delegators
676                .keys()
677                .map(|d| RewardAccount(d.to_ethers()))
678                .collect();
679            reward_accounts.insert(RewardAccount(validator.to_ethers()));
680            reward_accounts.extend(delegators);
681        }
682    }
683
684    // add all the chain configs needed to apply STF to headers to the catchup
685    for proposal in leaves {
686        let header = proposal.block_header();
687        let height = header.height();
688        let view = proposal.view_number();
689        tracing::debug!(height, ?view, "fetching dependencies for proposal");
690
691        let header_cf = header.chain_config();
692        if header_cf.commit() != instance.chain_config.commit() && header_cf.resolve().is_none() {
693            tracing::info!(
694                height,
695                ?view,
696                commit = %header_cf.commit(),
697                "chain config not available, attempting to load from storage",
698            );
699            let cf = load_chain_config(tx, header_cf.commit())
700                .await
701                .context(format!(
702                    "loading chain config {} for header {},{:?}",
703                    header_cf.commit(),
704                    header.height(),
705                    proposal.view_number()
706                ))?;
707
708            catchup.add_chain_config(cf);
709        };
710    }
711    Ok(reward_accounts)
712}

714async fn get_leaf_from_proposal<Mode, P>(
715    tx: &mut Transaction<Mode>,
716    where_clause: &str,
717    param: P,
718) -> anyhow::Result<Leaf2>
719where
720    P: Type<Db> + for<'q> Encode<'q, Db>,
721{
722    let (data,) = query_as::<(Vec<u8>,)>(&format!(
723        "SELECT data FROM quorum_proposals2 WHERE {where_clause} LIMIT 1",
724    ))
725    .bind(param)
726    .fetch_one(tx.as_mut())
727    .await?;
728    let proposal: Proposal<SeqTypes, QuorumProposalWrapper<SeqTypes>> =
729        bincode::deserialize(&data)?;
730    Ok(Leaf2::from_quorum_proposal(&proposal.data))
731}
