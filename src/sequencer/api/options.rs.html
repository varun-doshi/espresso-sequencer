140    /// Start the server.
141    ///
142    /// The function `init_context` is used to create a sequencer context from a metrics object and
143    /// optional saved consensus state. The metrics object is created from the API data source, so
144    /// that consensus will populuate metrics that can then be read and served by the API.
145    pub async fn serve<N, P, F, V: Versions + 'static>(
146        mut self,
147        init_context: F,
148    ) -> anyhow::Result<SequencerContext<N, P, V>>
149    where
150        N: ConnectedNetwork<PubKey>,
151        P: SequencerPersistence,
152        F: FnOnce(
153            Box<dyn Metrics>,
154            Box<dyn EventConsumer>,
155        ) -> BoxFuture<'static, anyhow::Result<SequencerContext<N, P, V>>>,
156    {
157        // Create a channel to send the context to the web server after it is initialized. This
158        // allows the web server to start before initialization can complete, since initialization
159        // can take a long time (and is dependent on other nodes).
160        let (send_ctx, recv_ctx) = oneshot::channel();
161        let state = ApiState::new(async move {
162            recv_ctx
163                .await
164                .expect("context initialized and sent over channel")
165        });
166        let mut tasks = TaskList::default();
167
168        // The server state type depends on whether we are running a query or status API or not, so
169        // we handle the two cases differently.
170        let (metrics, consumer): (Box<dyn Metrics>, Box<dyn EventConsumer>) =
171            if let Some(query_opt) = self.query.take() {
172                if let Some(opt) = self.storage_sql.take() {
173                    self.init_with_query_module_sql(
174                        query_opt,
175                        opt,
176                        state,
177                        &mut tasks,
178                        SequencerApiVersion::instance(),
179                    )
180                    .await?
181                } else if let Some(opt) = self.storage_fs.take() {
182                    self.init_with_query_module_fs(
183                        query_opt,
184                        opt,
185                        state,
186                        &mut tasks,
187                        SequencerApiVersion::instance(),
188                    )
189                    .await?
190                } else {
191                    bail!("query module requested but not storage provided");
192                }
193            } else if self.status.is_some() {
194                // If a status API is requested but no availability API, we use the
195                // `MetricsDataSource`, which allows us to run the status API with no persistent
196                // storage.
197                let ds = MetricsDataSource::default();
198                let metrics = ds.populate_metrics();
199                let mut app = App::<_, Error>::with_state(AppState::from(
200                    ExtensibleDataSource::new(ds, state.clone()),
201                ));
202
203                // Initialize status API.
204                let status_api =
205                    status::define_api(&Default::default(), SequencerApiVersion::instance())?;
206                app.register_module("status", status_api)?;
207
208                self.init_hotshot_modules(&mut app)?;
209
210                if self.hotshot_events.is_some() {
211                    self.init_and_spawn_hotshot_event_streaming_module(state, &mut tasks)?;
212                }
213
214                tasks.spawn(
215                    "API server",
216                    self.listen(self.http.port, app, SequencerApiVersion::instance()),
217                );
218
219                (metrics, Box::new(NullEventConsumer))
220            } else {
221                // If no status or availability API is requested, we don't need metrics or a query
222                // service data source. The only app state is the HotShot handle, which we use to
223                // submit transactions.
224                //
225                // If we have no availability API, we cannot load a saved leaf from local storage,
226                // so we better have been provided the leaf ahead of time if we want it at all.
227                let mut app = App::<_, Error>::with_state(AppState::from(state.clone()));
228
229                self.init_hotshot_modules(&mut app)?;
230
231                if self.hotshot_events.is_some() {
232                    self.init_and_spawn_hotshot_event_streaming_module(state, &mut tasks)?;
233                }
234
235                tasks.spawn(
236                    "API server",
237                    self.listen(self.http.port, app, SequencerApiVersion::instance()),
238                );
239
240                (Box::new(NoMetrics), Box::new(NullEventConsumer))
241            };
242
243        let ctx = init_context(metrics, consumer).await?;
244        send_ctx
245            .send(super::ConsensusState::from(&ctx))
246            .ok()
247            .context("API server exited without receiving context")?;
248        Ok(ctx.with_task_list(tasks))
249    }

251    async fn init_app_modules<N, P, D, V: Versions>(
252        &self,
253        ds: D,
254        state: ApiState<N, P, V>,
255        bind_version: SequencerApiVersion,
256    ) -> anyhow::Result<(
257        Box<dyn Metrics>,
258        Arc<StorageState<N, P, D, V>>,
259        App<AppState<StorageState<N, P, D, V>>, Error>,
260    )>
261    where
262        N: ConnectedNetwork<PubKey>,
263        P: SequencerPersistence,
264        D: SequencerDataSource + CatchupStorage + Send + Sync + 'static,
265    {
266        let metrics = ds.populate_metrics();
267        let ds = Arc::new(ExtensibleDataSource::new(ds, state.clone()));
268        let api_state: endpoints::AvailState<N, P, D, V> = ds.clone().into();
269        let mut app = App::<_, Error>::with_state(api_state);
270
271        // Initialize status API
272        let status_api = status::define_api::<endpoints::AvailState<N, P, D, _>, _>(
273            &Default::default(),
274            bind_version,
275        )?;
276        app.register_module("status", status_api)?;
277
278        // Initialize availability and node APIs (these both use the same data source).
279
280        // Note: We initialize two versions of the availability module: `availability/v0` and `availability/v1`.
281        // - `availability/v0/leaf/0` returns the old `Leaf1` type for backward compatibility.
282        // - `availability/v1/leaf/0` returns the new `Leaf2` type
283
284        // initialize the availability module for API version V0.
285        // This ensures compatibility for nodes that expect `Leaf1` for leaf endpoints
286        app.register_module(
287            "availability",
288            endpoints::availability("0.0.1".parse().unwrap())?,
289        )?;
290
291        // initialize the availability module for API version V1.
292        // This enables support for the new `Leaf2` type
293        app.register_module(
294            "availability",
295            endpoints::availability("1.0.0".parse().unwrap())?,
296        )?;
297
298        app.register_module("node", endpoints::node()?)?;
299
300        // Initialize submit API
301        if self.submit.is_some() {
302            app.register_module(
303                "submit",
304                endpoints::submit::<_, _, _, SequencerApiVersion>()?,
305            )?;
306        }
307
308        tracing::info!("initializing catchup API");
309        app.register_module("catchup", endpoints::catchup(bind_version)?)?;
310
311        app.register_module("state-signature", endpoints::state_signature(bind_version)?)?;
312
313        if self.config.is_some() {
314            app.register_module("config", endpoints::config(bind_version)?)?;
315        }
316        Ok((metrics, ds, app))
317    }

319    async fn init_with_query_module_fs<N, P, V: Versions + 'static>(
320        &self,
321        query_opt: Query,
322        mod_opt: persistence::fs::Options,
323        state: ApiState<N, P, V>,
324        tasks: &mut TaskList,
325        bind_version: SequencerApiVersion,
326    ) -> anyhow::Result<(Box<dyn Metrics>, Box<dyn EventConsumer>)>
327    where
328        N: ConnectedNetwork<PubKey>,
329        P: SequencerPersistence,
330    {
331        let ds = <fs::DataSource as SequencerDataSource>::create(
332            mod_opt,
333            provider::<V>(query_opt.peers, bind_version),
334            false,
335        )
336        .await?;
337
338        let (metrics, ds, app) = self
339            .init_app_modules(ds, state.clone(), bind_version)
340            .await?;
341
342        if self.hotshot_events.is_some() {
343            self.init_and_spawn_hotshot_event_streaming_module(state, tasks)?;
344        }
345
346        tasks.spawn("API server", self.listen(self.http.port, app, bind_version));
347        Ok((metrics, Box::new(ApiEventConsumer::from(ds))))
348    }

350    async fn init_with_query_module_sql<N, P, V: Versions + 'static>(
351        self,
352        query_opt: Query,
353        mod_opt: persistence::sql::Options,
354        state: ApiState<N, P, V>,
355        tasks: &mut TaskList,
356        bind_version: SequencerApiVersion,
357    ) -> anyhow::Result<(Box<dyn Metrics>, Box<dyn EventConsumer>)>
358    where
359        N: ConnectedNetwork<PubKey>,
360        P: SequencerPersistence,
361    {
362        let mut provider = Provider::default();
363
364        // Use the database itself as a fetching provider: sometimes we can fetch data that is
365        // missing from the query service from ephemeral consensus storage.
366        provider = provider.with_provider(mod_opt.clone().create().await?);
367        // If that fails, fetch missing data from peers.
368        for peer in query_opt.peers {
369            tracing::info!("will fetch missing data from {peer}");
370            provider = provider.with_provider(QueryServiceProvider::new(peer, bind_version));
371        }
372
373        let ds = sql::DataSource::create(mod_opt.clone(), provider, false).await?;
374        let (metrics, ds, mut app) = self
375            .init_app_modules(ds, state.clone(), bind_version)
376            .await?;
377
378        if self.explorer.is_some() {
379            app.register_module("explorer", endpoints::explorer()?)?;
380        }
381
382        // Initialize merklized state module for block merkle tree
383        app.register_module(
384            "block-state",
385            endpoints::merklized_state::<N, P, _, BlockMerkleTree, _, 3>()?,
386        )?;
387        // Initialize merklized state module for fee merkle tree
388        app.register_module(
389            "fee-state",
390            endpoints::get_balance::<_, SequencerApiVersion>()?,
391        )?;
392
393        app.register_module(
394            "reward-state",
395            endpoints::merklized_state::<N, P, _, RewardMerkleTree, _, 256>()?,
396        )?;
397
398        let get_node_state = {
399            let state = state.clone();
400            async move { state.node_state().await.clone() }
401        };
402        tasks.spawn(
403            "merklized state storage update loop",
404            update_state_storage_loop(ds.clone(), get_node_state),
405        );
406        if self.hotshot_events.is_some() {
407            self.init_and_spawn_hotshot_event_streaming_module(state, tasks)?;
408        }
409
410        tasks.spawn(
411            "API server",
412            self.listen(self.http.port, app, SequencerApiVersion::instance()),
413        );
414        Ok((metrics, Box::new(ApiEventConsumer::from(ds))))
415    }

417    /// Initialize the modules for interacting with HotShot.
418    ///
419    /// This function adds the `submit`, `state`, and `state_signature` API modules to the given
420    /// app. These modules only require a HotShot handle as state, and thus they work with any data
421    /// source, so initialization is the same no matter what mode the service is running in.
422    fn init_hotshot_modules<N, P, S>(&self, app: &mut App<S, Error>) -> anyhow::Result<()>
423    where
424        S: 'static + Send + Sync + ReadState,
425        P: SequencerPersistence,
426        S::State: Send
427            + Sync
428            + SubmitDataSource<N, P>
429            + StateSignatureDataSource<N>
430            + NodeStateDataSource
431            + CatchupDataSource
432            + HotShotConfigDataSource,
433        N: ConnectedNetwork<PubKey>,
434    {
435        let bind_version = SequencerApiVersion::instance();
436        // Initialize submit API
437        if self.submit.is_some() {
438            let submit_api = endpoints::submit::<_, _, _, SequencerApiVersion>()?;
439            app.register_module("submit", submit_api)?;
440        }
441
442        // Initialize state API.
443        if self.catchup.is_some() {
444            tracing::info!("initializing state API");
445            let catchup_api = endpoints::catchup(bind_version)?;
446            app.register_module("catchup", catchup_api)?;
447        }
448
449        let state_signature_api = endpoints::state_signature(bind_version)?;
450        app.register_module("state-signature", state_signature_api)?;
451
452        if self.config.is_some() {
453            app.register_module("config", endpoints::config(bind_version)?)?;
454        }
455
456        Ok(())
457    }

459    // Enable the events streaming api module
460    fn init_and_spawn_hotshot_event_streaming_module<
461        N,
462        P: SequencerPersistence,
463        V: Versions + 'static,
464    >(
465        &self,
466        state: ApiState<N, P, V>,
467        tasks: &mut TaskList,
468    ) -> anyhow::Result<()>
469    where
470        N: ConnectedNetwork<PubKey>,
471    {
472        // Start the event streaming API server if it is enabled.
473        // It runs to different port and app because State and Extensible Data source needs to support required
474        // EventsSource trait, which is currently intended not to implement to separate hotshot-query-service crate, and
475        // hotshot-events-service crate.
476
477        let mut app = App::<_, EventStreamingError>::with_state(AppState::from(state));
478
479        tracing::info!("initializing hotshot events API");
480        let hotshot_events_api = hotshot_events_service::events::define_api(
481            &hotshot_events_service::events::Options::default(),
482        )?;
483
484        app.register_module::<_, SequencerApiVersion>("hotshot-events", hotshot_events_api)?;
485
486        tasks.spawn(
487            "Hotshot Events Streaming API server",
488            self.listen(
489                self.hotshot_events.unwrap().events_service_port,
490                app,
491                SequencerApiVersion::instance(),
492            ),
493        );
494
495        Ok(())
496    }
497
498    fn listen<S, E, ApiVer>(
499        &self,
500        port: u16,
501        app: App<S, E>,
502        bind_version: ApiVer,
503    ) -> impl Future<Output = anyhow::Result<()>>
504    where
505        S: Send + Sync + 'static,
506        E: Send + Sync + tide_disco::Error,
507        ApiVer: StaticVersionType + 'static,
508    {
509        let max_connections = self.http.max_connections;
510
511        async move {
512            if let Some(limit) = max_connections {
513                app.serve(RateLimitListener::with_port(port, limit), bind_version)
514                    .await?;
515            } else {
516                app.serve(format!("0.0.0.0:{}", port), bind_version).await?;
517            }
518            Ok(())
519        }
520    }

523/// The minimal HTTP API.
524///
525/// The API automatically includes health and version endpoints. Additional API modules can be
526/// added by including the query-api or submit-api modules.
527#[derive(Parser, Clone, Copy, Debug)]
528pub struct Http {
529    /// Port that the HTTP API will use.
530    #[clap(long, env = "ESPRESSO_SEQUENCER_API_PORT", default_value = "8080")]
531    pub port: u16,
532
533    /// Maximum number of concurrent HTTP connections the server will allow.
534    ///
535    /// Connections exceeding this will receive and immediate 429 response and be closed.
536    ///
537    /// Leave unset for no connection limit.
538    #[clap(long, env = "ESPRESSO_SEQUENCER_MAX_CONNECTIONS")]
539    pub max_connections: Option<usize>,
540}
541
542impl Http {
543    /// Default options for running a web server on the given port.
544    pub fn with_port(port: u16) -> Self {
545        Self {
546            port,
547            max_connections: None,
548        }
549    }
550}
