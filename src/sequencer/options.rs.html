26// This options struct is a bit unconventional. The sequencer has multiple optional modules which
27// can be added, in any combination, to the service. These include, for example, the API server.
28// Each of these modules has its own options, which are all required if the module is added but can
29// be omitted otherwise. Clap doesn't have a good way to handle "grouped" arguments like this (they
30// have something called an argument group, but it's different). Sub-commands do exactly this, but
31// you can't have multiple sub-commands in a single command.
32//
33// What we do, then, is take the optional modules as if they were sub-commands, but we use a Clap
34// `raw` argument to collect all the module commands and their options into a single string. This
35// string is then parsed manually (using a secondary Clap `Parser`, the `SequencerModule` type) when
36// the user calls `modules()`.
37//
38// One slightly unfortunate consequence of this is that the auto-generated help documentation for
39// `SequencerModule` is not included in the help for this top-level type. Users can still get at the
40// help for individual modules by passing `help` as a subcommand, as in
41// `sequencer [options] -- help` or `sequencer [options] -- help <module>`. This means that IT IS
42// BEST NOT TO ADD REQUIRED ARGUMENTS TO THIS TYPE, since the required arguments will be required
43// even if the user is only asking for help on a module. Try to give every argument on this type a
44// default value, even if it is a bit arbitrary.

387impl Options {
388    pub fn modules(&self) -> Modules {
389        ModuleArgs(self.modules.clone()).parse()
390    }
391
392    pub fn private_keys(&self) -> anyhow::Result<(BLSPrivKey, StateSignKey)> {
393        if let Some(path) = &self.key_file {
394            let vars = dotenvy::from_path_iter(path)?.collect::<Result<HashMap<_, _>, _>>()?;
395            let staking = TaggedBase64::parse(
396                vars.get("ESPRESSO_SEQUENCER_PRIVATE_STAKING_KEY")
397                    .context("key file missing ESPRESSO_SEQUENCER_PRIVATE_STAKING_KEY")?,
398            )?
399            .try_into()?;
400
401            let state = TaggedBase64::parse(
402                vars.get("ESPRESSO_SEQUENCER_PRIVATE_STATE_KEY")
403                    .context("key file missing ESPRESSO_SEQUENCER_PRIVATE_STATE_KEY")?,
404            )?
405            .try_into()?;
406
407            Ok((staking, state))
408        } else if let (Some(staking), Some(state)) = (
409            self.private_staking_key.clone(),
410            self.private_state_key.clone(),
411        ) {
412            let staking = bls_over_bn254::SignKey::try_from(staking)?;
413            let state = schnorr::SignKey::try_from(state)?;
414
415            Ok((staking, state))
416        } else {
417            bail!("neither key file nor full set of private keys was provided")
418        }
419    }
420}

459    write!(
460        fmt,
461        "{:?}",
462        v.iter().map(|i| i.to_string()).collect::<Vec<_>>()

466fn fmt_opt_urls(
467    v: &Option<Vec<Url>>,
468    fmt: &mut std::fmt::Formatter,
469) -> Result<(), std::fmt::Error> {
470    match v {
471        Some(urls) => {
472            write!(fmt, "Some(")?;
473            fmt_urls(urls, fmt)?;
474            write!(fmt, ")")?;
475        },
476        None => {
477            write!(fmt, "None")?;
478        },
479    }
480    Ok(())

483#[derive(Clone, Copy, Debug, PartialEq, Eq)]
484pub struct Ratio {
485    pub numerator: u64,
486    pub denominator: u64,
487}
488
489impl From<Ratio> for (u64, u64) {
490    fn from(r: Ratio) -> Self {
491        (r.numerator, r.denominator)
492    }
493}
494
495impl Display for Ratio {
496    fn fmt(&self, f: &mut Formatter) -> fmt::Result {
497        write!(f, "{}:{}", self.numerator, self.denominator)
498    }
499}
500
501impl PartialOrd for Ratio {
502    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
503        Some(self.cmp(other))
504    }
505}
506
507impl Ord for Ratio {
508    fn cmp(&self, other: &Self) -> Ordering {
509        (self.numerator * other.denominator).cmp(&(other.numerator * self.denominator))
510    }
511}

513#[derive(Clone, Debug)]
514struct ModuleArgs(Vec<String>);
515
516impl ModuleArgs {
517    fn parse(&self) -> Modules {
518        match self.try_parse() {
519            Ok(modules) => modules,
520            Err(err) => err.exit(),
521        }
522    }
523
524    fn try_parse(&self) -> Result<Modules, clap::Error> {
525        let mut modules = Modules::default();
526        let mut curr = self.0.clone();
527        let mut provided = Default::default();
528
529        while !curr.is_empty() {
530            // The first argument (the program name) is used only for help generation. We include a
531            // `--` so that the generated usage will look like `sequencer -- <command>` which is the
532            // way these commands must be invoked due to the use of `raw` arguments.
533            let module = SequencerModule::try_parse_from(
534                once("sequencer --").chain(curr.iter().map(|s| s.as_str())),
535            )?;
536            match module {
537                SequencerModule::Storage(m) => {
538                    curr = m.add(&mut modules.storage_fs, &mut provided)?
539                },
540                SequencerModule::StorageFs(m) => {
541                    curr = m.add(&mut modules.storage_fs, &mut provided)?
542                },
543                SequencerModule::StorageSql(m) => {
544                    curr = m.add(&mut modules.storage_sql, &mut provided)?
545                },
546                SequencerModule::Http(m) => curr = m.add(&mut modules.http, &mut provided)?,
547                SequencerModule::Query(m) => curr = m.add(&mut modules.query, &mut provided)?,
548                SequencerModule::Submit(m) => curr = m.add(&mut modules.submit, &mut provided)?,
549                SequencerModule::Status(m) => curr = m.add(&mut modules.status, &mut provided)?,
550                SequencerModule::Catchup(m) => curr = m.add(&mut modules.catchup, &mut provided)?,
551                SequencerModule::Config(m) => curr = m.add(&mut modules.config, &mut provided)?,
552                SequencerModule::HotshotEvents(m) => {
553                    curr = m.add(&mut modules.hotshot_events, &mut provided)?
554                },
555                SequencerModule::Explorer(m) => {
556                    curr = m.add(&mut modules.explorer, &mut provided)?
557                },
558            }
559        }
560
561        Ok(modules)
562    }
563}

565trait ModuleInfo: Args + FromArgMatches {
566    const NAME: &'static str;
567    fn requires() -> Vec<&'static str>;
568}
569
570macro_rules! module {
571    ($name:expr, $opt:ty $(,requires: $($req:expr),*)?) => {
572        impl ModuleInfo for $opt {
573            const NAME: &'static str = $name;
574
575            fn requires() -> Vec<&'static str> {
576                vec![$($($req),*)?]
577            }
578        }
579    };
580}

603impl<Options: ModuleInfo> Module<Options> {
604    /// Add this as an optional module. Return the next optional module args.
605    fn add(
606        self,
607        options: &mut Option<Options>,
608        provided: &mut HashSet<&'static str>,
609    ) -> Result<Vec<String>, clap::Error> {
610        if options.is_some() {
611            return Err(clap::Error::raw(
612                ErrorKind::TooManyValues,
613                format!("optional module {} can only be started once", Options::NAME),
614            ));
615        }
616        for req in Options::requires() {
617            if !provided.contains(&req) {
618                return Err(clap::Error::raw(
619                    ErrorKind::MissingRequiredArgument,
620                    format!("module {} is missing required module {req}", Options::NAME),
621                ));
622            }
623        }
624        *options = Some(*self.options);
625        provided.insert(Options::NAME);
626        Ok(self.modules)
627    }
628}
