27/// State that checks the light client state update and the signature collection
28#[derive(Default)]
29struct StateRelayServerState {
30    /// Minimum weight to form an available state signature bundle
31    threshold: U256,
32    /// Stake table
33    known_nodes: HashMap<StateVerKey, U256>,
34    /// Signatures bundles for each block height
35    bundles: HashMap<u64, HashMap<LightClientState, StateSignaturesBundle>>,
36
37    /// The latest state signatures bundle whose total weight exceeds the threshold
38    latest_available_bundle: Option<StateSignaturesBundle>,
39    /// The block height of the latest available state signature bundle
40    latest_block_height: Option<u64>,
41
42    /// A ordered queue of block heights, used for garbage collection.
43    queue: BTreeSet<u64>,
44
45    /// shutdown signal
46    shutdown: Option<oneshot::Receiver<()>>,
47}
48
49impl StateRelayServerState {
50    pub fn new(threshold: U256) -> Self {
51        Self {
52            threshold,
53            ..Default::default()
54        }
55    }
56
57    pub fn with_shutdown_signal(
58        mut self,
59        shutdown_listener: Option<oneshot::Receiver<()>>,
60    ) -> Self {
61        if self.shutdown.is_some() {
62            panic!("A shutdown signal is already registered and can not be registered twice");
63        }
64        self.shutdown = shutdown_listener;
65        self
66    }
67}

74pub trait StateRelayServerDataSource {
75    /// Get the latest available signatures bundle.
76    /// # Errors
77    /// Errors if there's no available signatures bundle.
78    fn get_latest_signature_bundle(&self) -> Result<StateSignaturesBundle, Error>;
79
80    /// Post a signature to the relay server
81    /// # Errors
82    /// Errors if the signature is invalid, already posted, or no longer needed.
83    fn post_signature(
84        &mut self,
85        key: StateVerKey,
86        state: LightClientState,
87        signature: StateSignature,
88    ) -> Result<(), Error>;
89}

91impl StateRelayServerDataSource for StateRelayServerState {
92    fn get_latest_signature_bundle(&self) -> Result<StateSignaturesBundle, Error> {
93        match &self.latest_available_bundle {
94            Some(bundle) => Ok(bundle.clone()),
95            None => Err(tide_disco::error::ServerError::catch_all(
96                StatusCode::NOT_FOUND,
97                "The light client state signatures are not ready.".to_owned(),
98            )),
99        }
100    }
101
102    fn post_signature(
103        &mut self,
104        key: StateVerKey,
105        state: LightClientState,
106        signature: StateSignature,
107    ) -> Result<(), Error> {
108        if state.block_height <= self.latest_block_height.unwrap_or(0) {
109            // This signature is no longer needed
110            return Ok(());
111        }
112        let one = U256::one();
113        let weight = self.known_nodes.get(&key).unwrap_or(&one);
114        // TODO(Chengyu): We don't know where to fetch the stake table yet.
115        // Related issue: [https://github.com/EspressoSystems/espresso-sequencer/issues/1022]
116        // .ok_or(tide_disco::error::ServerError::catch_all(
117        //     StatusCode::Unauthorized,
118        //     "The posted key is not found in the stake table.".to_owned(),
119        // ))?;
120        let state_msg: [FieldType; 3] = (&state).into();
121        if StateSignatureScheme::verify(&(), &key, state_msg, &signature).is_err() {
122            return Err(tide_disco::error::ServerError::catch_all(
123                StatusCode::BAD_REQUEST,
124                "The posted signature is not valid.".to_owned(),
125            ));
126        }
127        let block_height = state.block_height;
128        // TODO(Chengyu): this serialization should be removed once `LightClientState` implements `Eq`.
129        let bundles_at_height = self.bundles.entry(block_height).or_insert_with(|| {
130            self.queue.insert(block_height);
131            Default::default()
132        });
133        let bundle = bundles_at_height
134            .entry(state.clone())
135            .or_insert(StateSignaturesBundle {
136                state,
137                signatures: Default::default(),
138                accumulated_weight: U256::from(0),
139            });
140        tracing::debug!(
141            "Accepting new signature for block height {} from {}.",
142            block_height,
143            key
144        );
145        match bundle.signatures.entry(key) {
146            std::collections::hash_map::Entry::Occupied(_) => {
147                // A signature is already posted for this key with this state
148                return Err(tide_disco::error::ServerError::catch_all(
149                    StatusCode::BAD_REQUEST,
150                    "A signature of this light client state is already posted at this block height for this key.".to_owned(),
151                ));
152            },
153            std::collections::hash_map::Entry::Vacant(entry) => {
154                entry.insert(signature);
155                bundle.accumulated_weight += *weight;
156            },
157        }
158
159        if bundle.accumulated_weight >= self.threshold {
160            tracing::info!(
161                "State signature bundle at block height {} is ready to serve.",
162                block_height
163            );
164            self.latest_block_height = Some(block_height);
165            self.latest_available_bundle = Some(bundle.clone());
166            while let Some(height) = self.queue.pop_first() {
167                self.bundles.remove(&height);
168                if height == block_height {
169                    break;
170                }
171            }
172        }
173        Ok(())
174    }
175}

177/// configurability options for the web server
178#[derive(Args, Default)]
179pub struct Options {
180    #[arg(
181        long = "state-relay-server-api-path",
182        env = "STATE_RELAY_SERVER_API_PATH"
183    )]
184    /// path to API
185    pub api_path: Option<PathBuf>,
186}

188/// Set up APIs for relay server
189fn define_api<State, ApiVer: StaticVersionType + 'static>(
190    options: &Options,
191    _: ApiVer,
192) -> Result<Api<State, Error, ApiVer>, ApiError>
193where
194    State: 'static + Send + Sync + ReadState + WriteState,
195    <State as ReadState>::State: Send + Sync + StateRelayServerDataSource,
196{
197    let mut api = match &options.api_path {
198        Some(path) => Api::<State, Error, ApiVer>::from_file(path)?,
199        None => {
200            let toml: toml::Value = toml::from_str(include_str!(
201                "../../api/state_relay_server.toml"
202            ))
203            .map_err(|err| ApiError::CannotReadToml {
204                reason: err.to_string(),
205            })?;
206            Api::<State, Error, ApiVer>::new(toml)?
207        },
208    };
209
210    api.get("getlateststate", |_req, state| {
211        async move { state.get_latest_signature_bundle() }.boxed()
212    })?
213    .post("poststatesignature", |req, state| {
214        async move {
215            let StateSignatureRequestBody {
216                key,
217                state: lcstate,
218                signature,
219            } = req
220                .body_auto::<StateSignatureRequestBody, ApiVer>(ApiVer::instance())
221                .map_err(Error::from_request_error)?;
222            state.post_signature(key, lcstate, signature)
223        }
224        .boxed()
225    })?;
226
227    Ok(api)
228}

230pub async fn run_relay_server<ApiVer: StaticVersionType + 'static>(
231    shutdown_listener: Option<oneshot::Receiver<()>>,
232    threshold: U256,
233    url: Url,
234    bind_version: ApiVer,
235) -> std::io::Result<()> {
236    let options = Options::default();
237
238    let api = define_api(&options, bind_version).unwrap();
239
240    // We don't have a stake table yet, putting some temporary value here.
241    // Related issue: [https://github.com/EspressoSystems/espresso-sequencer/issues/1022]
242    let state =
243        State::new(StateRelayServerState::new(threshold).with_shutdown_signal(shutdown_listener));
244    let mut app = App::<State, Error>::with_state(state);
245
246    app.register_module("api", api).unwrap();
247
248    let app_future = app.serve(url, bind_version);
249
250    app_future.await
251}
