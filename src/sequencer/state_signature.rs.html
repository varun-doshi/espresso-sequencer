29/// A relay server that's collecting and serving the light client state signatures

32/// Capacity for the in memory signature storage.
33const SIGNATURE_STORAGE_CAPACITY: usize = 100;
34
35#[derive(Debug)]
36pub struct StateSigner<ApiVer: StaticVersionType> {
37    /// Key for signing a new light client state
38    sign_key: StateSignKey,
39
40    /// Key for verifying a light client state
41    ver_key: StateVerKey,
42
43    /// The most recent light client state signatures
44    signatures: RwLock<StateSignatureMemStorage>,
45
46    /// Commitment for current fixed stake table
47    #[allow(dead_code)] // although not used today, might need it for dynamic stake table later
48    stake_table_comm: StakeTableCommitmentType,
49
50    /// The state relay server url
51    relay_server_client: Option<Client<ServerError, ApiVer>>,
52}
53
54impl<ApiVer: StaticVersionType> StateSigner<ApiVer> {
55    pub fn new(
56        sign_key: StateSignKey,
57        ver_key: StateVerKey,
58        stake_table_comm: StakeTableCommitmentType,
59    ) -> Self {
60        Self {
61            sign_key,
62            ver_key,
63            stake_table_comm,
64            signatures: Default::default(),
65            relay_server_client: Default::default(),
66        }
67    }
68
69    /// Connect to the given state relay server to send signed HotShot states to.
70    pub fn with_relay_server(mut self, url: Url) -> Self {
71        self.relay_server_client = Some(Client::new(url));
72        self
73    }
74
75    pub(super) async fn handle_event(&self, event: &Event<SeqTypes>) {
76        let EventType::Decide { leaf_chain, .. } = &event.event else {
77            return;
78        };
79        let Some(LeafInfo { leaf, .. }) = leaf_chain.first() else {
80            return;
81        };
82        match leaf
83            .block_header()
84            .get_light_client_state(leaf.view_number())
85        {
86            Ok(state) => {
87                let signature = self.sign_new_state(&state).await;
88                tracing::debug!("New leaves decided. Latest block height: {}", leaf.height(),);
89
90                if let Some(client) = &self.relay_server_client {
91                    let request_body = StateSignatureRequestBody {
92                        key: self.ver_key.clone(),
93                        state,
94                        signature,
95                    };
96                    if let Err(error) = client
97                        .post::<()>("api/state")
98                        .body_binary(&request_body)
99                        .unwrap()
100                        .send()
101                        .await
102                    {
103                        tracing::warn!("Error posting signature to the relay server: {:?}", error);
104                    }
105                }
106            },
107            Err(err) => {
108                tracing::error!("Error generating light client state: {:?}", err)
109            },
110        }
111    }
112
113    /// Return a signature of a light client state at given height.
114    pub async fn get_state_signature(&self, height: u64) -> Option<StateSignatureRequestBody> {
115        let pool_guard = self.signatures.read().await;
116        pool_guard.get_signature(height)
117    }
118
119    /// Sign the light client state at given height and store it.
120    async fn sign_new_state(&self, state: &LightClientState) -> StateSignature {
121        let msg: [CircuitField; 3] = state.into();
122        let signature =
123            StateSignatureScheme::sign(&(), &self.sign_key, msg, &mut rand::thread_rng()).unwrap();
124        let mut pool_guard = self.signatures.write().await;
125        pool_guard.push(
126            state.block_height,
127            StateSignatureRequestBody {
128                key: self.ver_key.clone(),
129                state: state.clone(),
130                signature: signature.clone(),
131            },
132        );
133        tracing::debug!(
134            "New signature added for block height {}",
135            state.block_height
136        );
137        signature
138    }
139}

141/// A rolling in-memory storage for the most recent light client state signatures.
142#[derive(Debug, Default)]
143pub struct StateSignatureMemStorage {
144    pool: HashMap<u64, StateSignatureRequestBody>,
145    deque: VecDeque<u64>,
146}
147
148impl StateSignatureMemStorage {
149    pub fn push(&mut self, height: u64, signature: StateSignatureRequestBody) {
150        self.pool.insert(height, signature);
151        self.deque.push_back(height);
152        if self.pool.len() > SIGNATURE_STORAGE_CAPACITY {
153            self.pool.remove(&self.deque.pop_front().unwrap());
154        }
155    }
156
157    pub fn get_signature(&self, height: u64) -> Option<StateSignatureRequestBody> {
158        self.pool.get(&height).cloned()
159    }
160}

162/// Type for stake table commitment
163pub type StakeTableCommitmentType = (CircuitField, CircuitField, CircuitField);
164
165/// Helper function for stake table commitment
166pub fn static_stake_table_commitment(
167    known_nodes_with_stakes: &[PeerConfig<SeqTypes>],
168    capacity: usize,
169) -> (CircuitField, CircuitField, CircuitField) {
170    let mut st = StakeTable::<BLSPubKey, StateVerKey, CircuitField>::new(capacity);
171    known_nodes_with_stakes.iter().for_each(|peer| {
172        // This `unwrap()` won't fail unless number of entries exceeds `capacity`
173        st.register(
174            *peer.stake_table_entry.key(),
175            peer.stake_table_entry.stake(),
176            peer.state_ver_key.clone(),
177        )
178        .unwrap();
179    });
180    st.advance();
181    st.advance();
182    // This `unwrap()` won't fail
183    st.commitment(SnapshotVersion::LastEpochStart).unwrap()
184}
