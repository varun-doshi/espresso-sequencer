21/// An external message that can be sent to or received from a node
22#[derive(Debug, Serialize, Deserialize, Clone)]
23pub enum ExternalMessage {
24    RequestResponse(Vec<u8>),
25}
26
27/// The external event handler
28#[derive(Clone)]
29pub struct ExternalEventHandler<V: Versions> {
30    /// The sender to the request-response protocol
31    request_response_sender: Sender<Bytes>,
32
33    /// The type phantom
34    phantom: PhantomData<V>,
35}
36
37// The different types of outbound messages (broadcast or direct)
38#[derive(Debug)]
39#[allow(dead_code)]
40pub enum OutboundMessage {
41    Direct(MessageKind<SeqTypes>, PubKey),
42    Broadcast(MessageKind<SeqTypes>),
43}
44
45impl<V: Versions> ExternalEventHandler<V> {
46    /// Creates a new `ExternalEventHandler` with the given network
47    pub async fn new<N: ConnectedNetwork<PubKey>>(
48        tasks: &mut TaskList,
49        request_response_sender: Sender<Bytes>,
50        outbound_message_receiver: Receiver<OutboundMessage>,
51        network: Arc<N>,
52        public_key: PubKey,
53        hotshot_upgrade_lock: UpgradeLock<SeqTypes, V>,
54    ) -> Result<Self> {
55        // Spawn the outbound message handling loop
56        tasks.spawn(
57            "ExternalEventHandler",
58            Self::outbound_message_loop(
59                outbound_message_receiver,
60                network,
61                public_key,
62                hotshot_upgrade_lock,
63            ),
64        );
65
66        Ok(Self {
67            request_response_sender,
68            phantom: PhantomData,
69        })
70    }
71
72    /// Handles an event
73    ///
74    /// # Errors
75    /// If the message type is unknown or if there is an error serializing or deserializing the message
76    pub async fn handle_event(&self, external_message_bytes: &[u8]) -> Result<()> {
77        // Deserialize the external message
78        let external_message = bincode::deserialize(external_message_bytes)
79            .with_context(|| "Failed to deserialize external message")?;
80
81        // Match the type
82        match external_message {
83            ExternalMessage::RequestResponse(request_response) => {
84                // Send the inner message to the request-response protocol
85                self.request_response_sender
86                    .send(request_response.into())
87                    .await?;
88            },
89        }
90        Ok(())
91    }
92
93    /// The main loop for sending outbound messages.
94    async fn outbound_message_loop<N: ConnectedNetwork<PubKey>>(
95        mut receiver: Receiver<OutboundMessage>,
96        network: Arc<N>,
97        public_key: PubKey,
98        hotshot_upgrade_lock: UpgradeLock<SeqTypes, V>,
99    ) {
100        while let Some(message) = receiver.recv().await {
101            // Match the message type
102            match message {
103                OutboundMessage::Direct(message, recipient) => {
104                    // Wrap it in the real message type
105                    let message_inner = Message {
106                        sender: public_key,
107                        kind: message,
108                    };
109
110                    // Serialize it
111                    let message_bytes = match hotshot_upgrade_lock.serialize(&message_inner).await {
112                        Ok(message_bytes) => message_bytes,
113                        Err(err) => {
114                            tracing::warn!("Failed to serialize direct message: {}", err);
115                            continue;
116                        },
117                    };
118
119                    // Send the message to the recipient
120                    if let Err(err) = network.direct_message(message_bytes, recipient).await {
121                        tracing::error!("Failed to send message: {:?}", err);
122                    };
123                },
124
125                OutboundMessage::Broadcast(message) => {
126                    // Wrap it in the real message type
127                    let message_inner = Message {
128                        sender: public_key,
129                        kind: message,
130                    };
131
132                    // Serialize it
133                    let message_bytes = match hotshot_upgrade_lock.serialize(&message_inner).await {
134                        Ok(message_bytes) => message_bytes,
135                        Err(err) => {
136                            tracing::warn!("Failed to serialize broadcast message: {}", err);
137                            continue;
138                        },
139                    };
140
141                    // Broadcast the message to the global topic
142                    if let Err(err) = network
143                        .broadcast_message(message_bytes, Topic::Global, BroadcastDelay::None)
144                        .await
145                    {
146                        tracing::error!("Failed to broadcast message: {:?}", err);
147                    };
148                },
149            }
150        }
151    }
152}
