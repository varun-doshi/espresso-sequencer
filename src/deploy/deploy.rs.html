16/// Deploy contracts needed to run the sequencer.
17///
18/// This script deploys contracts needed to run the sequencer to an L1. It outputs a .env file
19/// containing the addresses of the deployed contracts.
20///
21/// This script can also be used to do incremental deployments. The only contract addresses
22/// needed to configure the sequencer network are ESPRESSO_SEQUENCER_FEE_CONTRACT_PROXY_ADDRESS,
23/// ESPRESSO_SEQUENCER_LIGHT_CLIENT_PROXY_ADDRESS and (soon) PERMISSIONED_STAKE_TABLE_ADDRESS.
24/// These contracts, however, have dependencies, and a full deployment involves several
25/// contracts. Some of these contracts, especially libraries may already have been deployed, or
26/// perhaps one of the top-level contracts has been deployed and we only need to deploy the other
27/// one.
28///
29/// It is possible to pass in the addresses of already deployed contracts, in which case those
30/// addresses will be used in place of deploying a new contract wherever that contract is required
31/// in the deployment process. The generated .env file will include all the addresses passed in as
32/// well as those newly deployed.
33#[derive(Clone, Debug, Parser)]
34struct Options {
35    /// A JSON-RPC endpoint for the L1 to deploy to.
36    #[clap(
37        short,
38        long,
39        env = "ESPRESSO_SEQUENCER_L1_PROVIDER",
40        default_value = "http://localhost:8545"
41    )]
42    rpc_url: Url,
43
44    /// Request rate when polling L1.
45    #[clap(
46        long,
47        env = "ESPRESSO_SEQUENCER_L1_POLLING_INTERVAL",
48        default_value = "7s",
49        value_parser = parse_duration,
50    )]
51    pub l1_polling_interval: Duration,
52
53    /// URL of a sequencer node that is currently providing the HotShot config.
54    /// This is used to initialize the stake table.
55    #[clap(
56        long,
57        env = "ESPRESSO_SEQUENCER_URL",
58        default_value = "http://localhost:24000"
59    )]
60    pub sequencer_url: Url,
61
62    /// Mnemonic for an L1 wallet.
63    ///
64    /// This wallet is used to deploy the contracts, so the account indicated by ACCOUNT_INDEX must
65    /// be funded with with ETH.
66    #[clap(
67        long,
68        name = "MNEMONIC",
69        env = "ESPRESSO_SEQUENCER_ETH_MNEMONIC",
70        default_value = "test test test test test test test test test test test junk"
71    )]
72    mnemonic: String,
73
74    /// Address for the multisig wallet that will be the admin
75    ///
76    /// If provided, this the multisig wallet that will be able to upgrade contracts and execute
77    /// admin only functions on contracts. If not provided, admin power for all contracts will be
78    /// held by the account used to deploy the contracts (determined from MNEMONIC, ACCOUNT_INDEX).
79    #[clap(
80        long,
81        name = "MULTISIG_ADDRESS",
82        env = "ESPRESSO_SEQUENCER_ETH_MULTISIG_ADDRESS"
83    )]
84    multisig_address: Option<Address>,
85
86    /// Account index in the L1 wallet generated by MNEMONIC to use when deploying the contracts.
87    #[clap(
88        long,
89        name = "ACCOUNT_INDEX",
90        env = "ESPRESSO_DEPLOYER_ACCOUNT_INDEX",
91        default_value = "0"
92    )]
93    account_index: u32,
94
95    /// Only deploy the given groups of related contracts.
96    #[clap(long, value_delimiter = ',')]
97    only: Option<Vec<ContractGroup>>,
98
99    /// Write deployment results to OUT as a .env file.
100    ///
101    /// If not provided, the results will be written to stdout.
102    #[clap(short, long, name = "OUT", env = "ESPRESSO_DEPLOYER_OUT_PATH")]
103    out: Option<PathBuf>,
104
105    #[clap(flatten)]
106    contracts: DeployedContracts,
107
108    /// If toggled, launch a mock prover contract with a smaller verification key.
109    #[clap(short, long)]
110    pub use_mock_contract: bool,
111
112    /// Stake table capacity for the prover circuit
113    #[clap(short, long, env = "ESPRESSO_SEQUENCER_STAKE_TABLE_CAPACITY", default_value_t = STAKE_TABLE_CAPACITY)]
114    pub stake_table_capacity: usize,
115
116    /// Permissioned prover address for light client contract.
117    ///
118    /// If the light client contract is being deployed and this is set, the prover will be
119    /// permissioned so that only this address can update the light client state. Otherwise, proving
120    /// will be permissionless.
121    ///
122    /// If the light client contract is not being deployed, this option is ignored.
123    #[clap(long, env = "ESPRESSO_SEQUENCER_PERMISSIONED_PROVER")]
124    permissioned_prover: Option<Address>,
125
126    /// A toml file with the initial stake table.
127    ///
128    /// Schema:
129    ///
130    /// public_keys = [
131    ///   {
132    ///     stake_table_key = "BLS_VER_KEY~...",
133    ///     state_ver_key = "SCHNORR_VER_KEY~...",
134    ///     da = true,
135    ///     stake = 1, # this value is ignored, but needs to be set
136    ///   },
137    /// ]
138    #[clap(long, env = "ESPRESSO_SEQUENCER_INITIAL_PERMISSIONED_STAKE_TABLE_PATH")]
139    initial_stake_table_path: Option<PathBuf>,
140
141    /// Exit escrow period for the stake table contract.
142    ///
143    /// This is the period for which stake table contract will retain funds after withdrawals have
144    /// been requested. It should be set to a value that is at least 3 hotshot epochs plus ample
145    /// time to allow for submission of slashing evidence. Initially it will probably be around one
146    /// week.
147    #[clap(long, env = "ESPRESSO_SEQUENCER_STAKE_TABLE_EXIT_ESCROW_PERIOD", value_parser = parse_duration)]
148    exit_escrow_period: Option<Duration>,
149
150    /// The address that the tokens will be minted to.
151    ///
152    /// If unset the tokens will be minted to the deployer account.
153    #[clap(long, env = "ESP_TOKEN_INITIAL_GRANT_RECIPIENT_ADDRESS")]
154    initial_token_grant_recipient: Option<Address>,
155
156    #[clap(flatten)]
157    logging: logging::Config,
158}

160#[tokio::main]
161async fn main() -> anyhow::Result<()> {
162    let opt = Options::parse();
163    opt.logging.init();
164
165    let contracts = Contracts::from(opt.contracts);
166
167    let sequencer_url = opt.sequencer_url.clone();
168
169    let genesis = light_client_genesis(&sequencer_url, opt.stake_table_capacity).boxed();
170
171    let initial_stake_table = if let Some(path) = opt.initial_stake_table_path {
172        tracing::info!("Loading initial stake table from {:?}", path);
173        Some(PermissionedStakeTableConfig::<SeqTypes>::from_toml_file(&path)?.into())
174    } else {
175        None
176    };
177
178    let contracts = deploy(
179        opt.rpc_url,
180        opt.l1_polling_interval,
181        opt.mnemonic,
182        opt.account_index,
183        opt.multisig_address,
184        opt.use_mock_contract,
185        opt.only,
186        genesis,
187        opt.permissioned_prover,
188        contracts,
189        initial_stake_table,
190        opt.exit_escrow_period,
191        opt.initial_token_grant_recipient,
192    )
193    .await?;
194
195    if let Some(out) = &opt.out {
196        let file = File::options()
197            .create(true)
198            .truncate(true)
199            .write(true)
200            .open(out)?;
201        contracts.write(file)?;
202    } else {
203        contracts.write(stdout())?;
204    }
205
206    Ok(())
207}
