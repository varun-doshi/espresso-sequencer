1//! The following is the main `Marshal` binary, which just instantiates and runs
2//! a `Marshal` object.
3
4use anyhow::{Context, Result};
5use cdn_marshal::{Config, Marshal};
6use clap::Parser;
7use espresso_types::{parse_size, SeqTypes};
8use sequencer::network::cdn::ProductionDef;
9use tracing_subscriber::EnvFilter;
10
11#[derive(Parser, Debug)]
12#[command(author, version, about, long_about = None)]
13/// The main component of the push CDN.
14struct Args {
15    /// The discovery client endpoint (including scheme) to connect to
16    #[arg(short, long, env = "ESPRESSO_CDN_MARSHAL_DISCOVERY_ENDPOINT")]
17    discovery_endpoint: String,
18
19    /// The port to bind to for connections (from users)
20    #[arg(
21        short,
22        long,
23        default_value_t = 1737,
24        env = "ESPRESSO_CDN_MARSHAL_BIND_PORT"
25    )]
26    bind_port: u16,
27
28    /// The endpoint to bind to for externalizing metrics (in `IP:port` form). If not provided,
29    /// metrics are not exposed.
30    #[arg(short, long, env = "ESPRESSO_CDN_MARSHAL_METRICS_BIND_ENDPOINT")]
31    metrics_bind_endpoint: Option<String>,
32
33    /// The path to the CA certificate
34    /// If not provided, a local, pinned CA is used
35    #[arg(long, env = "ESPRESSO_CDN_MARSHAL_CA_CERT_PATH")]
36    ca_cert_path: Option<String>,
37
38    /// The path to the CA key
39    /// If not provided, a local, pinned CA is used
40    #[arg(long, env = "ESPRESSO_CDN_MARSHAL_CA_KEY_PATH")]
41    ca_key_path: Option<String>,
42
43    /// The size of the global memory pool. This is the maximum number of bytes that
44    /// can be allocated at once for all connections. A connection will block if it
45    /// tries to allocate more than this amount until some memory is freed.
46    #[arg(
47        long,
48        default_value = "1GB",
49        value_parser = parse_size,
50        env = "ESPRESSO_CDN_MARSHAL_GLOBAL_MEMORY_POOL_SIZE"
51    )]
52    global_memory_pool_size: u64,
53}
54
55#[tokio::main]
56async fn main() -> Result<()> {
57    // Parse command-line arguments
58    let args = Args::parse();
59
60    // Initialize tracing
61    if std::env::var("RUST_LOG_FORMAT") == Ok("json".to_string()) {
62        tracing_subscriber::fmt()
63            .with_env_filter(EnvFilter::from_default_env())
64            .json()
65            .init();
66    } else {
67        tracing_subscriber::fmt()
68            .with_env_filter(EnvFilter::from_default_env())
69            .init();
70    }
71
72    // Cast the memory pool size to a `usize`
73    let global_memory_pool_size =
74        usize::try_from(args.global_memory_pool_size).with_context(|| {
75            format!(
76                "Failed to convert global memory pool size to usize: {}",
77                args.global_memory_pool_size
78            )
79        })?;
80
81    // Create a new `Config`
82    let config = Config {
83        discovery_endpoint: args.discovery_endpoint,
84        bind_endpoint: format!("0.0.0.0:{}", args.bind_port),
85        metrics_bind_endpoint: args.metrics_bind_endpoint,
86        ca_cert_path: args.ca_cert_path,
87        ca_key_path: args.ca_key_path,
88        global_memory_pool_size: Some(global_memory_pool_size),
89    };
90
91    // Create new `Marshal` from the config
92    let marshal = Marshal::<ProductionDef<SeqTypes>>::new(config).await?;
93
94    // Start the main loop, consuming it
95    marshal.start().await?;
96
97    Ok(())
98}
