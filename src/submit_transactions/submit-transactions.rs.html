33/// Submit random transactions to an Espresso Sequencer.
34#[derive(Clone, Debug, Parser)]
35struct Options {
36    /// Minimum size of transaction to submit.
37    ///
38    /// The size of each transaction will be chosen uniformly between MIN_SIZE and MAX_SIZE.
39    #[clap(long, name = "MIN_SIZE", default_value = "1", value_parser = parse_size, env = "ESPRESSO_SUBMIT_TRANSACTIONS_MIN_SIZE")]
40    min_size: u64,
41
42    /// Maximum size of transaction to submit.
43    ///
44    /// The size of each transaction will be chosen uniformly between MIN_SIZE and MAX_SIZE.
45    #[clap(long, name = "MAX_SIZE", default_value = "1kb", value_parser = parse_size, env = "ESPRESSO_SUBMIT_TRANSACTIONS_MAX_SIZE")]
46    max_size: u64,
47
48    /// Minimum size of batch of transactions to submit.
49    ///
50    /// Batches will be a random count between MIN_BATCH_SIZE and MAX_BATCH_SIZE, with a falling distribution favoring smaller batches.
51    /// This is by selecting a random size S on each iteration I since last batch, and collecting a batch whenever that S <= I.
52    #[clap(long, name = "MIN_BATCH_SIZE", default_value = "1", value_parser = parse_size, env = "ESPRESSO_SUBMIT_TRANSACTIONS_MIN_BATCH_SIZE")]
53    min_batch_size: u64,
54
55    /// Maximum size of batch of transactions to submit.
56    ///
57    /// Batches will be a random count between MIN_BATCH_SIZE and MAX_BATCH_SIZE, with a falling distribution favoring smaller batches.
58    #[clap(long, name = "MAX_BATCH_SIZE", default_value = "20", value_parser = parse_size, env = "ESPRESSO_SUBMIT_TRANSACTIONS_MAX_BATCH_SIZE")]
59    max_batch_size: u64,
60
61    /// Minimum namespace ID to submit to.
62    #[clap(
63        long,
64        default_value = "10000",
65        env = "ESPRESSO_SUBMIT_TRANSACTIONS_MIN_NAMESPACE"
66    )]
67    min_namespace: u32,
68
69    /// Maximum namespace ID to submit to.
70    #[clap(
71        long,
72        default_value = "10010",
73        env = "ESPRESSO_SUBMIT_TRANSACTIONS_MAX_NAMESPACE"
74    )]
75    max_namespace: u32,
76
77    /// Mean delay between submitting transactions.
78    ///
79    /// The delay after each transaction will be sampled from an exponential distribution with mean
80    /// DELAY.
81    #[clap(long, name = "DELAY", value_parser = parse_duration, default_value = "30s", env = "ESPRESSO_SUBMIT_TRANSACTIONS_DELAY")]
82    delay: Duration,
83
84    /// Maximum number of unprocessed transaction submissions.
85    ///
86    /// This can be used to apply backpressure so that the tasks submitting transactions do not get
87    /// too far ahead of the task processing results.
88    #[clap(
89        long,
90        default_value = "1000",
91        env = "ESPRESSO_SUBMIT_TRANSACTIONS_CHANNEL_BOUND"
92    )]
93    channel_bound: usize,
94
95    /// Seed for reproducible randomness.
96    #[clap(long, env = "ESPRESSO_SUBMIT_TRANSACTIONS_SEED")]
97    seed: Option<u64>,
98
99    /// Number of parallel tasks to run.
100    #[clap(
101        short,
102        long,
103        default_value = "1",
104        env = "ESPRESSO_SUBMIT_TRANSACTIONS_JOBS"
105    )]
106    jobs: usize,
107
108    /// Number of accumulated pending transactions which should trigger a warning.
109    #[clap(
110        long,
111        default_value = "10",
112        env = "ESPRESSO_SUBMIT_TRANSACTIONS_PENDING_TRANSACTIONS_WARNING_THRESHOLD"
113    )]
114    pending_transactions_warning_threshold: usize,
115
116    /// Duration after which we should warn about a pending transaction.
117    #[clap(long, value_parser = parse_duration, default_value = "30s", env = "ESPRESSO_SUBMIT_TRANSACTIONS_SLOW_TRANSACTION_WARNING_THRESHOLD")]
118    slow_transaction_warning_threshold: Duration,
119
120    /// Enable an HTTP server with a healthcheck endpoint on this port.
121    #[clap(short, long, env = "ESPRESSO_SUBMIT_TRANSACTIONS_PORT")]
122    port: Option<u16>,
123
124    /// Alternative URL to submit transactions to, if not the query service URL.
125    #[clap(long, env = "ESPRESSO_SUBMIT_TRANSACTIONS_SUBMIT_URL")]
126    submit_url: Option<Url>,
127
128    /// URL of the query service.
129    #[clap(env = "ESPRESSO_SEQUENCER_URL")]
130    url: Url,
131
132    /// Relay num_nodes for benchmark results output
133    #[cfg(feature = "benchmarking")]
134    #[clap(short, long, env = "ESPRESSO_ORCHESTRATOR_NUM_NODES")]
135    num_nodes: NonZeroUsize,
136
137    /// The first block that benchmark starts counting in
138    #[cfg(feature = "benchmarking")]
139    #[clap(short, long, env = "ESPRESSO_BENCH_START_BLOCK")]
140    benchmark_start_block: NonZeroUsize,
141
142    /// The final block that benchmark counts in
143    #[cfg(feature = "benchmarking")]
144    #[clap(short, long, env = "ESPRESSO_BENCH_END_BLOCK")]
145    benchmark_end_block: NonZeroUsize,
146
147    #[clap(flatten)]
148    logging: logging::Config,
149}

151impl Options {
152    fn submit_url(&self) -> Url {
153        self.submit_url
154            .clone()
155            .unwrap_or_else(|| self.url.join("submit").unwrap())
156    }
157    fn use_public_mempool(&self) -> bool {
158        self.submit_url.is_none()
159    }
160}

162#[tokio::main]
163async fn main() {
164    let opt = Options::parse();
165    opt.logging.init();
166
167    tracing::warn!("starting load generator for sequencer {}", opt.url);
168
169    let (sender, mut receiver) = mpsc::channel(opt.channel_bound);
170
171    let seed = opt.seed.unwrap_or_else(random_seed);
172    tracing::info!("PRNG seed: {seed}");
173    let mut rng = ChaChaRng::seed_from_u64(seed);
174
175    // Subscribe to block stream so we can check that our transactions are getting sequenced.
176    let client = Client::<Error, SequencerApiVersion>::new(opt.url.clone());
177    let block_height: usize = client.get("status/block-height").send().await.unwrap();
178    let mut blocks = client
179        .socket(&format!("availability/stream/blocks/{}", block_height - 1))
180        .subscribe()
181        .await
182        .unwrap();
183    tracing::info!("listening for blocks starting at {block_height}");
184
185    // Spawn tasks to submit transactions.
186    for _ in 0..opt.jobs {
187        spawn(submit_transactions(
188            opt.clone(),
189            sender.clone(),
190            ChaChaRng::from_rng(&mut rng).unwrap(),
191            SequencerApiVersion::instance(),
192        ));
193    }
194
195    // Start healthcheck endpoint once tasks are running.
196    if let Some(port) = opt.port {
197        spawn(server(port, SequencerApiVersion::instance()));
198    }
199
200    // Keep track of the results.
201    let mut pending = HashMap::new();
202    let mut total_latency = Duration::default();
203    let mut total_transactions = 0;
204
205    // Keep track of the latency after warm up for benchmarking
206    #[cfg(feature = "benchmarking")]
207    let mut num_block = 0;
208    #[cfg(feature = "benchmarking")]
209    let mut benchmark_total_latency = Duration::default();
210    #[cfg(feature = "benchmarking")]
211    let mut benchmark_minimum_latency = Duration::default();
212    #[cfg(feature = "benchmarking")]
213    let mut benchmark_maximum_latency = Duration::default();
214    #[cfg(feature = "benchmarking")]
215    let mut benchmark_total_transactions = 0;
216    #[cfg(feature = "benchmarking")]
217    let mut benchmark_finish = false;
218    #[cfg(feature = "benchmarking")]
219    let mut total_throughput = 0;
220    #[cfg(feature = "benchmarking")]
221    let mut start: Instant = Instant::now(); // will be re-assign once has_started turned to true
222    #[cfg(feature = "benchmarking")]
223    let mut has_started: bool = false;
224
225    while let Some(block) = blocks.next().await {
226        let block: BlockQueryData<SeqTypes> = match block {
227            Ok(block) => block,
228            Err(err) => {
229                tracing::warn!("error getting block: {err}");
230                continue;
231            },
232        };
233        let received_at = Instant::now();
234        tracing::debug!("got block {}", block.height());
235        #[cfg(feature = "benchmarking")]
236        {
237            num_block += 1;
238            if !has_started && (num_block as usize) >= usize::from(opt.benchmark_start_block) {
239                has_started = true;
240                start = Instant::now();
241            }
242        }
243
244        // Get all transactions which were submitted before this block.
245        while let Ok(Some(tx)) = receiver.try_next() {
246            pending.insert(tx.hash, tx.submitted_at);
247        }
248
249        // Clear pending transactions from the block.
250        for (_, tx) in block.enumerate() {
251            if let Some(submitted_at) = pending.remove(&tx.commit()) {
252                let latency = received_at - submitted_at;
253                tracing::info!(
254                    "got transaction {} in block {}, latency {latency:?}",
255                    tx.commit(),
256                    block.height()
257                );
258                total_latency += latency;
259                total_transactions += 1;
260                tracing::info!("average latency: {:?}", total_latency / total_transactions);
261                #[cfg(feature = "benchmarking")]
262                {
263                    if has_started && !benchmark_finish {
264                        benchmark_minimum_latency = if total_transactions == 0 {
265                            latency
266                        } else {
267                            std::cmp::min(benchmark_minimum_latency, latency)
268                        };
269                        benchmark_maximum_latency = if total_transactions == 0 {
270                            latency
271                        } else {
272                            std::cmp::max(benchmark_maximum_latency, latency)
273                        };
274
275                        benchmark_total_latency += latency;
276                        benchmark_total_transactions += 1;
277                        // Transaction = NamespaceId(u64) + payload(Vec<u8>)
278                        let payload_length = tx.into_payload().len();
279                        let tx_sz = payload_length * std::mem::size_of::<u8>() // size of payload
280                        + std::mem::size_of::<u64>() // size of the namespace
281                        + std::mem::size_of::<Transaction>(); // size of the struct wrapper
282                        total_throughput += tx_sz;
283                    }
284                }
285            }
286        }
287
288        #[cfg(feature = "benchmarking")]
289        if !benchmark_finish && (num_block as usize) >= usize::from(opt.benchmark_end_block) {
290            let block_range = format!("{}~{}", opt.benchmark_start_block, opt.benchmark_end_block,);
291            let transaction_size_range_in_bytes = format!("{}~{}", opt.min_size, opt.max_size,);
292            let transactions_per_batch_range = format!(
293                "{}~{}",
294                (opt.jobs as u64 * opt.min_batch_size),
295                (opt.jobs as u64 * opt.max_batch_size),
296            );
297            let benchmark_average_latency = benchmark_total_latency / benchmark_total_transactions;
298            let avg_transaction_size = total_throughput as u32 / benchmark_total_transactions;
299            let total_time_elapsed_in_sec = start.elapsed(); // in seconds
300            let avg_throughput_bytes_per_sec = (total_throughput as u64)
301                / std::cmp::max(total_time_elapsed_in_sec.as_secs(), 1u64);
302            // Open the CSV file in append mode
303            let results_csv_file = OpenOptions::new()
304                .create(true)
305                .append(true) // Open in append mode
306                .open("scripts/benchmarks_results/results.csv")
307                .unwrap();
308            // Open a file for writing
309            let mut wtr = Writer::from_writer(results_csv_file);
310            let mut pub_or_priv_pool = "private";
311            if opt.use_public_mempool() {
312                pub_or_priv_pool = "public";
313            }
314            let _ = wtr.write_record([
315                "total_nodes",
316                "da_committee_size",
317                "block_range",
318                "transaction_size_range_in_bytes",
319                "transaction_per_batch_range",
320                "pub_or_priv_pool",
321                "avg_latency_in_sec",
322                "minimum_latency_in_sec",
323                "maximum_latency_in_sec",
324                "avg_throughput_bytes_per_sec",
325                "total_transactions",
326                "avg_transaction_size_in_bytes",
327                "total_time_elapsed_in_sec",
328            ]);
329            let _ = wtr.write_record(&[
330                opt.num_nodes.to_string(),
331                opt.num_nodes.to_string(),
332                block_range,
333                transaction_size_range_in_bytes,
334                transactions_per_batch_range,
335                pub_or_priv_pool.to_string(),
336                benchmark_average_latency.as_secs().to_string(),
337                benchmark_minimum_latency.as_secs().to_string(),
338                benchmark_maximum_latency.as_secs().to_string(),
339                avg_throughput_bytes_per_sec.to_string(),
340                benchmark_total_transactions.to_string(),
341                avg_transaction_size.to_string(),
342                total_time_elapsed_in_sec.as_secs().to_string(),
343            ]);
344            let _ = wtr.flush();
345            println!(
346                "Latency results successfully saved in scripts/benchmarks_results/results.csv"
347            );
348            benchmark_finish = true;
349        }
350
351        // If a lot of transactions are pending, it might indicate the sequencer is struggling to
352        // finalize them. We should warn about this.
353        if pending.len() >= opt.pending_transactions_warning_threshold {
354            tracing::warn!(
355                "transactions are not being finalized or being finalized too slowly, {} pending",
356                pending.len()
357            );
358        } else {
359            tracing::debug!("{} transactions still pending", pending.len());
360
361            // Even if we are not accumulating transactions, it is still possible that some
362            // individual transactions are not being finalized. Warn about any transaction which has
363            // been pending for too long.
364            for (tx, submitted_at) in &pending {
365                let duration = received_at - *submitted_at;
366                if duration >= opt.slow_transaction_warning_threshold {
367                    tracing::warn!("transaction {tx} has been pending for {duration:?}");
368                }
369            }
370        }
371    }
372    tracing::info!(
373        "block stream ended with {} transactions still pending",
374        pending.len()
375    );
376}

380    submitted_at: Instant,

383async fn submit_transactions<ApiVer: StaticVersionType>(
384    opt: Options,
385    mut sender: Sender<SubmittedTransaction>,
386    mut rng: ChaChaRng,
387    _: ApiVer,
388) {
389    let url = opt.submit_url();
390    tracing::info!(%url, "starting load generator task");
391    let client = Client::<Error, ApiVer>::new(url);
392
393    // Create an exponential distribution for sampling delay times. The distribution should have
394    // mean `opt.delay`, or parameter `\lambda = 1 / opt.delay`.
395    let delay_distr = rand_distr::Exp::<f64>::new(1f64 / opt.delay.as_millis() as f64).unwrap();
396
397    let mut txns = Vec::new();
398    let mut hashes = Vec::new();
399    loop {
400        let tx = random_transaction(&opt, &mut rng);
401        let hash = tx.commit();
402        tracing::info!(
403            "Adding transaction {hash} for namespace {} of size {}",
404            tx.namespace(),
405            tx.payload().len()
406        );
407        txns.push(tx);
408        hashes.push(hash);
409
410        let randomized_batch_size = if opt.use_public_mempool() {
411            1
412        } else {
413            rng.gen_range(opt.min_batch_size..=opt.max_batch_size)
414        };
415        let txns_batch_count = txns.len() as u64;
416        if randomized_batch_size <= txns_batch_count {
417            if let Err(err) = if txns_batch_count == 1 {
418                // occasionally test the 'submit' endpoint, just for coverage
419                client
420                    .post::<()>("submit")
421                    .body_binary(&txns[0])
422                    .unwrap()
423                    .send()
424                    .await
425            } else {
426                client
427                    .post::<()>("batch")
428                    .body_binary(&txns)
429                    .unwrap()
430                    .send()
431                    .await
432            } {
433                tracing::error!(
434                    ?err,
435                    "failed to submit batch of {txns_batch_count} transactions"
436                );
437            } else {
438                tracing::info!("submitted batch of {txns_batch_count} transactions");
439                let submitted_at = Instant::now();
440                for hash in hashes.iter() {
441                    sender
442                        .send(SubmittedTransaction {
443                            hash: *hash,
444                            submitted_at,
445                        })
446                        .await
447                        .ok();
448                }
449            }
450            txns.clear();
451            hashes.clear();
452
453            let delay = Duration::from_millis(delay_distr.sample(&mut rng) as u64);
454            tracing::info!("sleeping for {delay:?}");
455            sleep(delay).await;
456        }
457    }
458}

460async fn server<ApiVer: StaticVersionType + 'static>(port: u16, bind_version: ApiVer) {
461    if let Err(err) = App::<(), ServerError>::with_state(())
462        .serve(format!("0.0.0.0:{port}"), bind_version)
463        .await
464    {
465        tracing::error!("web server exited: {err}");
466    }

469fn random_transaction(opt: &Options, rng: &mut ChaChaRng) -> Transaction {
470    // TODO instead use NamespaceId::random, but that does not allow us to
471    // enforce `gen_range(opt.min_namespace..=opt.max_namespace)`
472    let namespace = rng.gen_range(opt.min_namespace..=opt.max_namespace);
473
474    let len = rng.gen_range(opt.min_size..=opt.max_size);
475    let mut payload = vec![0; len as usize];
476    rng.fill_bytes(&mut payload);
477
478    Transaction::new(namespace.into(), payload)
479}

481fn random_seed() -> u64 {
482    ChaChaRng::from_entropy().next_u64()
