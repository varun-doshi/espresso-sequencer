56    pub async fn deploy() -> Result<Self> {
57        let exit_escrow_period = Duration::from_secs(1);
58        let port = portpicker::pick_unused_port().unwrap();
59        // Spawn anvil
60        let provider = ProviderBuilder::new()
61            .with_recommended_fillers()
62            .on_anvil_with_wallet_and_config(|anvil| anvil.port(port).arg("--accounts").arg("20"));
63        let rpc_url = format!("http://localhost:{}", port).parse()?;
64        let deployer_address = provider.default_signer_address();
65
66        // `EspToken.sol`
67        let token_impl = EspToken::deploy(provider.clone()).await?;
68        let data = token_impl
69            .initialize(deployer_address, deployer_address)
70            .calldata()
71            .clone();
72
73        let proxy = ERC1967Proxy::deploy(provider.clone(), *token_impl.address(), data).await?;
74        let token = EspToken::new(*proxy.address(), provider.clone());
75
76        // `StakeTable.sol`
77        let stake_table_impl = StakeTable::deploy(provider.clone()).await?;
78        let data = stake_table_impl
79            .initialize(
80                *token.address(),
81                "0xFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFF".parse()?, // fake LC address
82                U256::from(exit_escrow_period.as_secs()),
83                deployer_address,
84            )
85            .calldata()
86            .clone();
87
88        let proxy =
89            ERC1967Proxy::deploy(provider.clone(), *stake_table_impl.address(), data).await?;
90        let stake_table = StakeTable::new(*proxy.address(), provider.clone());
91
92        // Approve the stake table contract so it can transfer tokens to itself
93        let receipt = token
94            .approve(*stake_table.address(), parse_ether("1000000")?)
95            .send()
96            .await?
97            .get_receipt()
98            .await?;
99        assert!(receipt.status());
100
101        let bls_key_pair = BLSKeyPair::generate(&mut rand::thread_rng());
102        let schnorr_key_pair = SchnorrKeyPair::generate(&mut rand::thread_rng());
103        Ok(Self {
104            provider,
105            deployer_address,
106            token,
107            stake_table,
108            exit_escrow_period,
109            rpc_url,
110            bls_key_pair,
111            schnorr_key_pair,
112            commission: Commission::try_from("12.34")?,
113        })
114    }

200            .arg("--stake-table-address")
201            .arg(self.stake_table.address().to_string());

206#[cfg(test)]

210    #[tokio::test]
211    async fn test_deploy() -> Result<()> {
212        let system = TestSystem::deploy().await?;
213        // sanity check that we can fetch the exit escrow period
214        assert_eq!(
215            system.stake_table.exitEscrowPeriod().call().await?._0,
216            U256::from(system.exit_escrow_period.as_secs())
217        );
218
219        let to = "0x1111111111111111111111111111111111111111".parse()?;
220
221        // sanity check that we can transfer tokens
222        system.transfer(to, U256::from(123)).await?;
223
224        // sanity check that we can fetch the balance
225        assert_eq!(system.balance(to).await?, U256::from(123));
226
227        Ok(())
228    }
