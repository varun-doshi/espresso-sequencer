44pub async fn decide_leaf_chain(decided_view: u64) -> Arc<Vec<LeafInfo<TestTypes>>> {
45    decide_leaf_chain_with_transactions(decided_view, vec![transaction()]).await
46}

48pub async fn decide_leaf_chain_with_transactions(
49    decided_view: u64,
50    transactions: Vec<TestTransaction>,
51) -> Arc<Vec<LeafInfo<TestTypes>>> {
52    let (da_proposal, quorum_proposal) =
53        proposals_with_transactions(decided_view, transactions).await;
54    let mut leaf = Leaf2::from_quorum_proposal(&quorum_proposal);
55    let payload = <TestBlockPayload as BlockPayload<TestTypes>>::from_bytes(
56        &da_proposal.encoded_transactions,
57        &da_proposal.metadata,
58    );
59    leaf.fill_block_payload_unchecked(payload);
60    Arc::new(vec![LeafInfo {
61        leaf,
62        state: Default::default(),
63        delta: None,
64        vid_share: None,
65    }])
66}

74/// Create mock pair of DA and Quorum proposals with given transactions
75pub async fn proposals_with_transactions(
76    view: u64,
77    transactions: Vec<TestTransaction>,
78) -> (DaProposal2<TestTypes>, QuorumProposalWrapper<TestTypes>) {
79    let epoch = None;
80    let view_number = <TestTypes as NodeType>::View::new(view);
81    let upgrade_lock = UpgradeLock::<TestTypes, TestVersions>::new();
82    let validated_state = TestValidatedState::default();
83    let instance_state = TestInstanceState::default();
84
85    let (payload, metadata) = <TestBlockPayload as BlockPayload<TestTypes>>::from_transactions(
86        transactions.clone(),
87        &validated_state,
88        &instance_state,
89    )
90    .await
91    .unwrap();
92    let encoded_transactions = TestTransaction::encode(&transactions);
93
94    let header = TestBlockHeader::new(
95        &Leaf::<TestTypes>::genesis::<TestVersions>(&Default::default(), &Default::default())
96            .await
97            .into(),
98        vid_commitment::<TestVersions>(
99            &encoded_transactions,
100            &metadata.encode(),
101            GENESIS_VID_NUM_STORAGE_NODES,
102            <TestVersions as Versions>::Base::VERSION,
103        ),
104        <TestBlockPayload as BlockPayload<TestTypes>>::builder_commitment(&payload, &metadata),
105        metadata,
106    );
107
108    let genesis_qc = QuorumCertificate::<TestTypes>::genesis::<TestVersions>(
109        &TestValidatedState::default(),
110        &TestInstanceState::default(),
111    )
112    .await
113    .to_qc2();
114    let parent_proposal = QuorumProposalWrapper {
115        proposal: QuorumProposal2 {
116            block_header: header,
117            view_number: ViewNumber::new(view_number.saturating_sub(1)),
118            justify_qc: genesis_qc,
119            upgrade_certificate: None,
120            view_change_evidence: None,
121            next_drb_result: None,
122            next_epoch_justify_qc: None,
123            epoch,
124        },
125    };
126    let leaf = Leaf2::from_quorum_proposal(&parent_proposal);
127
128    let quorum_data = QuorumData2 {
129        leaf_commit: leaf.commit(),
130        epoch,
131        block_number: Some(leaf.height()),
132    };
133
134    let versioned_data = VersionedVoteData::<_, _, _>::new_infallible(
135        quorum_data.clone(),
136        view_number,
137        &upgrade_lock,
138    )
139    .await;
140
141    let commitment = Commitment::from_raw(versioned_data.commit().into());
142
143    let justify_qc =
144        QuorumCertificate2::new(quorum_data, commitment, view_number, None, PhantomData);
145
146    (
147        DaProposal2 {
148            encoded_transactions: encoded_transactions.into(),
149            metadata,
150            view_number,
151            epoch,
152            epoch_transition_indicator: EpochTransitionIndicator::NotInTransition,
153        },
154        QuorumProposalWrapper {
155            proposal: QuorumProposal2 {
156                block_header: leaf.block_header().clone(),
157                view_number,
158                justify_qc,
159                upgrade_certificate: None,
160                view_change_evidence: None,
161                next_drb_result: None,
162                next_epoch_justify_qc: None,
163                epoch,
164            },
165        },
166    )
167}

169pub fn builder_state(view: u64) -> Arc<BuilderState<TestTypes>> {
170    let references = parent_references(view);
171    let (_, receiver) = broadcast(TEST_CHANNEL_BUFFER_SIZE);
172    BuilderState::new(
173        references,
174        Duration::from_secs(1),
175        receiver,
176        TestValidatedState::default(),
177    )
178}

180/// Generate references for given view number with random
181/// commitments for use in testing code
182pub fn parent_references(view: u64) -> ParentBlockReferences<TestTypes> {
183    let rng = &mut thread_rng();
184    ParentBlockReferences {
185        view_number: <TestTypes as NodeType>::View::new(view),
186        leaf_commit: random_commitment(rng),
187        vid_commitment: hotshot_types::data::VidCommitment::V0(
188            advz_scheme(TEST_NUM_NODES_IN_VID_COMPUTATION)
189                .commit_only(rng.sample_iter(Standard).take(100).collect::<Vec<_>>())
190                .unwrap(),
191        ),
192        builder_commitment: BuilderCommitment::from_bytes(
193            rng.sample_iter(Standard).take(32).collect::<Vec<_>>(),
194        ),
195        tx_count: rng.gen(),
196        last_nonempty_view: None,
197    }
