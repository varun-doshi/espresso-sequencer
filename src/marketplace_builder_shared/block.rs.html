15/// Enum to hold the different sources of the transaction
16#[derive(Clone, Debug, PartialEq)]
17pub enum TransactionSource {
18    /// Transaction from private mempool
19    Private,
20    /// Transaction from public mempool
21    Public,
22}
23
24/// [`ReceivedTransaction`] represents receipt information concerning a received
25/// [`NodeType::Transaction`].
26#[derive(Debug, Clone)]
27pub struct ReceivedTransaction<Types: NodeType> {
28    /// the transaction
29    pub transaction: Types::Transaction,
30    /// transaction's hash
31    pub commit: Commitment<Types::Transaction>,
32    /// transaction's estimated length
33    pub min_block_size: u64,
34    /// transaction's source
35    pub source: TransactionSource,
36    /// received time
37    pub time_in: Instant,
38}
39
40impl<Types: NodeType> ReceivedTransaction<Types> {
41    pub fn new(transaction: Types::Transaction, source: TransactionSource) -> Self {
42        Self {
43            commit: transaction.commit(),
44            min_block_size: transaction.minimum_block_size(),
45            source,
46            time_in: Instant::now(),
47            transaction,
48        }
49    }
50}

52/// Unique identifier for a block
53#[derive(Clone, Debug, Hash, PartialEq, Eq)]
54pub struct BlockId<Types: NodeType> {
55    /// Block hash
56    pub hash: BuilderCommitment,
57    /// Block view
58    pub view: Types::View,
59}
60
61impl<Types: NodeType> std::fmt::Display for BlockId<Types> {
62    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
63        write!(
64            f,
65            "Block({}@{})",
66            hex::encode(self.hash.as_ref()),
67            *self.view
68        )
69    }
70}

72/// Unique identifier for a builder state
73///
74/// Builder state is identified by the VID commitment
75/// and view of the block it targets to extend, i.e.
76/// builder with given state ID assumes blocks/bundles it's building
77/// are going to be included immediately after the parent block.
78#[derive(Clone, Debug, Hash, PartialEq, Eq)]
79pub struct BuilderStateId<Types: NodeType> {
80    /// View number of the parent block
81    pub parent_view: Types::View,
82    /// VID commitment of the parent block
83    pub parent_commitment: VidCommitment,
84}
85
86impl<Types: NodeType> std::fmt::Display for BuilderStateId<Types> {
87    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
88        write!(
89            f,
90            "BuilderState({}@{})",
91            self.parent_commitment, *self.parent_view
92        )
93    }
94}

96/// References to the parent block that is extended to spawn the new builder state.
97#[derive(derive_more::Debug, Clone, PartialEq, Eq)]
98pub struct ParentBlockReferences<Types: NodeType> {
99    /// View on which the parent block was proposed
100    pub view_number: Types::View,
101    /// VID commitment of the parent block payload
102    pub vid_commitment: VidCommitment,
103    /// Leaf commitment of the parent leaf
104    pub leaf_commit: Commitment<Leaf2<Types>>,
105    /// Builder commitment of the parent block payload
106    pub builder_commitment: BuilderCommitment,
107    /// Number of transactions included in the parent block
108    pub tx_count: usize,
109    /// Last known view that had a block with transactions
110    pub last_nonempty_view: Option<Types::View>,
111}
112
113impl<Types> ParentBlockReferences<Types>
114where
115    Types: NodeType,
116{
117    /// Create mock references for bootstrap (don't correspond to a real block)
118    pub fn bootstrap() -> Self {
119        Self {
120            view_number: Types::View::genesis(),
121            vid_commitment: VidCommitment::default(),
122            leaf_commit: fake_commitment(),
123            builder_commitment: BuilderCommitment::from_bytes([]),
124            tx_count: 0,
125            last_nonempty_view: None,
126        }
127    }
128}
129
130// implement display for the derived info
131impl<Types: NodeType> std::fmt::Display for ParentBlockReferences<Types> {
132    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
133        write!(f, "View Number: {:?}", self.view_number)
134    }
