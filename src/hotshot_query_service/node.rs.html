13//! A node's view of a HotShot chain
14//!
15//! The node API provides a subjective view of the HotShot blockchain, from the perspective of
16//! one particular node. It provides access to information that the
17//! [availability](crate::availability) API does not, because this information depends on the
18//! perspective of the node observing it, and may be subject to eventual consistency. For example,
19//! `/node/block-height` may return smaller counts than expected, if the node being queried is not
20//! fully synced with the entire history of the chain. However, the node will _eventually_ sync and
21//! return the expected counts.
22

40#[derive(Debug)]
41pub struct Options {
42    pub api_path: Option<PathBuf>,
43
44    /// Additional API specification files to merge with `node-api-path`.
45    ///
46    /// These optional files may contain route definitions for application-specific routes that have
47    /// been added as extensions to the basic node API.
48    pub extensions: Vec<toml::Value>,
49
50    /// The maximum number of headers which can be loaded in a single `header/window` query.
51    pub window_limit: usize,
52}
53
54impl Default for Options {
55    fn default() -> Self {
56        Self {
57            api_path: None,
58            extensions: vec![],
59            window_limit: 500,
60        }
61    }
62}

64#[derive(Clone, Debug, From, Snafu, Deserialize, Serialize)]
65#[snafu(visibility(pub))]
66pub enum Error {
67    Request {
68        source: RequestError,
69    },
70    #[snafu(display("{source}"))]
71    Query {
72        source: QueryError,
73    },
74    #[snafu(display("error fetching VID share for block {block}: {source}"))]
75    #[from(ignore)]
76    QueryVid {
77        source: QueryError,
78        block: String,
79    },
80    #[snafu(display(
81        "error fetching window starting from {start} and ending at time {end}: {source}"
82    ))]
83    #[from(ignore)]
84    QueryWindow {
85        source: QueryError,
86        start: String,
87        end: u64,
88    },
89    Custom {
90        message: String,
91        status: StatusCode,
92    },
93}
94
95impl Error {
96    pub fn internal<M: Display>(message: M) -> Self {
97        Self::Custom {
98            message: message.to_string(),
99            status: StatusCode::INTERNAL_SERVER_ERROR,
100        }
101    }
102
103    pub fn status(&self) -> StatusCode {
104        match self {
105            Self::Request { .. } => StatusCode::BAD_REQUEST,
106            Self::Query { source, .. }
107            | Self::QueryVid { source, .. }
108            | Self::QueryWindow { source, .. } => source.status(),
109            Self::Custom { status, .. } => *status,
110        }
111    }
112}

114pub fn define_api<State, Types: NodeType, Ver: StaticVersionType + 'static>(
115    options: &Options,
116    _: Ver,
117) -> Result<Api<State, Error, Ver>, ApiError>
118where
119    State: 'static + Send + Sync + ReadState,
120    <State as ReadState>::State: NodeDataSource<Types> + Send + Sync,
121{
122    let mut api = load_api::<State, Error, Ver>(
123        options.api_path.as_ref(),
124        include_str!("../api/node.toml"),
125        options.extensions.clone(),
126    )?;
127    let window_limit = options.window_limit;
128    api.with_version("0.0.1".parse().unwrap())
129        .get("block_height", |_req, state| {
130            async move { state.block_height().await.context(QuerySnafu) }.boxed()
131        })?
132        .get("count_transactions", |req, state| {
133            async move {
134                let from: Bound<usize> = match req.opt_integer_param("from")? {
135                    Some(from) => Bound::Included(from),
136                    None => Bound::Unbounded,
137                };
138                let to = match req.opt_integer_param("to")? {
139                    Some(to) => Bound::Included(to),
140                    None => Bound::Unbounded,
141                };
142                Ok(state.count_transactions_in_range((from, to)).await?)
143            }
144            .boxed()
145        })?
146        .get("payload_size", |req, state| {
147            async move {
148                let from: Bound<usize> = match req.opt_integer_param("from")? {
149                    Some(from) => Bound::Included(from),
150                    None => Bound::Unbounded,
151                };
152                let to = match req.opt_integer_param("to")? {
153                    Some(to) => Bound::Included(to),
154                    None => Bound::Unbounded,
155                };
156                Ok(state.payload_size_in_range((from, to)).await?)
157            }
158            .boxed()
159        })?
160        .get("get_vid_share", |req, state| {
161            async move {
162                let id = if let Some(height) = req.opt_integer_param("height")? {
163                    BlockId::Number(height)
164                } else if let Some(hash) = req.opt_blob_param("hash")? {
165                    BlockId::Hash(hash)
166                } else {
167                    BlockId::PayloadHash(req.blob_param("payload-hash")?)
168                };
169                state.vid_share(id).await.context(QueryVidSnafu {
170                    block: id.to_string(),
171                })
172            }
173            .boxed()
174        })?
175        .get("sync_status", |_req, state| {
176            async move { state.sync_status().await.context(QuerySnafu) }.boxed()
177        })?
178        .get("get_header_window", move |req, state| {
179            async move {
180                let start = if let Some(height) = req.opt_integer_param("height")? {
181                    WindowStart::Height(height)
182                } else if let Some(hash) = req.opt_blob_param("hash")? {
183                    WindowStart::Hash(hash)
184                } else {
185                    WindowStart::Time(req.integer_param("start")?)
186                };
187                let end = req.integer_param("end")?;
188                state
189                    .get_header_window(start, end, window_limit)
190                    .await
191                    .context(QueryWindowSnafu {
192                        start: format!("{start:?}"),
193                        end,
194                    })
195            }
196            .boxed()
197        })?
198        .get("get_limits", move |_req, _state| {
199            async move { Ok(Limits { window_limit }) }.boxed()
200        })?;
201    Ok(api)
202}

238    #[tokio::test(flavor = "multi_thread")]
239    async fn test_api() {
240        setup_test();
241
242        let window_limit = 78;
243
244        // Create the consensus network.
245        let mut network = MockNetwork::<MockDataSource, MockVersions>::init().await;
246        let mut events = network.handle().event_stream();
247        network.start().await;
248
249        // Start the web server.
250        let port = pick_unused_port().unwrap();
251        let mut app = App::<_, Error>::with_state(ApiState::from(network.data_source()));
252        app.register_module(
253            "node",
254            define_api(
255                &Options {
256                    window_limit,
257                    ..Default::default()
258                },
259                MockBase::instance(),
260            )
261            .unwrap(),
262        )
263        .unwrap();
264        network.spawn(
265            "server",
266            app.serve(format!("0.0.0.0:{}", port), MockBase::instance()),
267        );
268
269        // Start a client.
270        let client = Client::<Error, MockBase>::new(
271            format!("http://localhost:{}/node", port).parse().unwrap(),
272        );
273        assert!(client.connect(Some(Duration::from_secs(60))).await);
274
275        // Check limits endpoint.
276        assert_eq!(
277            client.get::<Limits>("limits").send().await.unwrap(),
278            Limits { window_limit }
279        );
280
281        // Wait until a few blocks have been sequenced.
282        let block_height = loop {
283            let block_height = client.get::<usize>("block-height").send().await.unwrap();
284            if block_height > network.num_nodes() {
285                break block_height;
286            }
287            sleep(Duration::from_secs(1)).await;
288        };
289
290        // We test these counters with non-trivial values in `data_source.rs`, here we just want to
291        // make sure the API handlers are working, so a response of 0 is fine.
292        assert_eq!(
293            client
294                .get::<u64>("transactions/count")
295                .send()
296                .await
297                .unwrap(),
298            0
299        );
300        assert_eq!(
301            client
302                .get::<u64>("payloads/total-size")
303                .send()
304                .await
305                .unwrap(),
306            0
307        );
308
309        let mut headers = vec![];
310
311        // Get VID share for each block.
312        tracing::info!(block_height, "checking VID shares");
313        'outer: while let Some(event) = events.next().await {
314            let EventType::Decide { leaf_chain, .. } = event.event else {
315                continue;
316            };
317            for LeafInfo {
318                leaf, vid_share, ..
319            } in leaf_chain.iter().rev()
320            {
321                headers.push(leaf.block_header().clone());
322                if leaf.block_header().block_number >= block_height as u64 {
323                    break 'outer;
324                }
325                tracing::info!(height = leaf.block_header().block_number, "checking share");
326
327                let share = client
328                    .get::<VidShare>(&format!("vid/share/{}", leaf.block_header().block_number))
329                    .send()
330                    .await
331                    .unwrap();
332                if let Some(vid_share) = vid_share.as_ref() {
333                    let VidDisperseShare::V0(new_share) = vid_share else {
334                        panic!("VID share is not V0");
335                    };
336                    assert_eq!(share, VidShare::V0(new_share.share.clone()));
337                }
338
339                // Query various other ways.
340                assert_eq!(
341                    share,
342                    client
343                        .get(&format!("vid/share/hash/{}", leaf.block_header().commit()))
344                        .send()
345                        .await
346                        .unwrap()
347                );
348                assert_eq!(
349                    share,
350                    client
351                        .get(&format!(
352                            "vid/share/payload-hash/{}",
353                            leaf.block_header().payload_commitment
354                        ))
355                        .send()
356                        .await
357                        .unwrap()
358                );
359            }
360        }
361
362        // Check time window queries. The various edge cases are thoroughly tested for each
363        // individual data source. In this test, we just smoketest API parameter handling. Sleep 2
364        // seconds to ensure a new header is produced with a timestamp after the latest one in
365        // `headers`
366        sleep(Duration::from_secs(2)).await;
367        let first_header = &headers[0];
368        let last_header = &headers.last().unwrap();
369        let window: TimeWindowQueryData<Header<MockTypes>> = client
370            .get(&format!(
371                "header/window/{}/{}",
372                first_header.timestamp,
373                last_header.timestamp + 1
374            ))
375            .send()
376            .await
377            .unwrap();
378        assert!(window.window.contains(first_header));
379        assert!(window.window.contains(last_header));
380        assert!(window.next.is_some());
381
382        // Query for the same window other ways.
383        assert_eq!(
384            window,
385            client
386                .get(&format!(
387                    "header/window/from/0/{}",
388                    last_header.timestamp + 1
389                ))
390                .send()
391                .await
392                .unwrap()
393        );
394        assert_eq!(
395            window,
396            client
397                .get(&format!(
398                    "header/window/from/hash/{}/{}",
399                    first_header.commit(),
400                    last_header.timestamp + 1
401                ))
402                .send()
403                .await
404                .unwrap()
405        );
406
407        // In this simple test, the node should be fully synchronized.
408        let sync_status = client
409            .get::<SyncStatus>("sync-status")
410            .send()
411            .await
412            .unwrap();
413        assert_eq!(sync_status.missing_blocks, 0);
414        assert_eq!(sync_status.missing_leaves, 0);
415
416        network.shut_down().await;
417    }

419    #[tokio::test(flavor = "multi_thread")]
420    async fn test_aggregate_ranges() {
421        setup_test();
422
423        // Create the consensus network.
424        let mut network = MockNetwork::<MockSqlDataSource, MockVersions>::init().await;
425        let mut events = network.handle().event_stream();
426        network.start().await;
427
428        // Start the web server.
429        let port = pick_unused_port().unwrap();
430        let mut app = App::<_, Error>::with_state(ApiState::from(network.data_source()));
431        app.register_module(
432            "node",
433            define_api(&Default::default(), MockBase::instance()).unwrap(),
434        )
435        .unwrap();
436        network.spawn(
437            "server",
438            app.serve(format!("0.0.0.0:{}", port), MockBase::instance()),
439        );
440
441        // Start a client.
442        let client =
443            Client::<Error, MockBase>::new(format!("http://localhost:{}", port).parse().unwrap());
444        assert!(client.connect(Some(Duration::from_secs(60))).await);
445
446        // Wait until a few transactions have been sequenced.
447        let mut tx_heights = vec![];
448        let mut tx_sizes = vec![];
449        for i in [1, 2] {
450            let txn = mock_transaction(vec![0; i]);
451            let hash = txn.commit();
452
453            network.submit_transaction(txn).await;
454
455            let leaf = 'outer: loop {
456                let EventType::Decide { leaf_chain, .. } = events.next().await.unwrap().event
457                else {
458                    continue;
459                };
460                for info in leaf_chain.iter().rev() {
461                    let leaf = &info.leaf;
462                    if BlockPayload::<MockTypes>::transaction_commitments(
463                        &leaf.block_payload().unwrap(),
464                        BlockHeader::<MockTypes>::metadata(leaf.block_header()),
465                    )
466                    .contains(&hash)
467                    {
468                        break 'outer leaf.clone();
469                    }
470                }
471
472                tracing::info!("waiting for tx {i}");
473                sleep(Duration::from_secs(1)).await;
474            };
475            tx_heights.push(leaf.height());
476            tx_sizes.push(leaf.block_payload().unwrap().encode().len());
477        }
478        tracing::info!(?tx_heights, ?tx_sizes, "transactions sequenced");
479
480        // Wait for the aggregator to process the inserted blocks.
481        while let Err(err) = client
482            .get::<usize>(&format!("node/transactions/count/{}", tx_heights[1]))
483            .send()
484            .await
485        {
486            if err.status() == StatusCode::NOT_FOUND {
487                tracing::info!(?tx_heights, "waiting for aggregator");
488                sleep(Duration::from_secs(1)).await;
489                continue;
490            } else {
491                panic!("unexpected error: {err:#}");
492            }
493        }
494
495        // Range including empty blocks (genesis block) only
496        assert_eq!(
497            0,
498            client
499                .get::<usize>("node/transactions/count/0")
500                .send()
501                .await
502                .unwrap()
503        );
504        assert_eq!(
505            0,
506            client
507                .get::<usize>("node/payloads/size/0")
508                .send()
509                .await
510                .unwrap()
511        );
512
513        // First transaction only
514        assert_eq!(
515            1,
516            client
517                .get::<usize>(&format!("node/transactions/count/{}", tx_heights[0]))
518                .send()
519                .await
520                .unwrap()
521        );
522        assert_eq!(
523            tx_sizes[0],
524            client
525                .get::<usize>(&format!("node/payloads/size/{}", tx_heights[0]))
526                .send()
527                .await
528                .unwrap()
529        );
530
531        // Last transaction only
532        assert_eq!(
533            1,
534            client
535                .get::<usize>(&format!(
536                    "node/transactions/count/{}/{}",
537                    tx_heights[0] + 1,
538                    tx_heights[1]
539                ))
540                .send()
541                .await
542                .unwrap()
543        );
544        assert_eq!(
545            tx_sizes[1],
546            client
547                .get::<usize>(&format!(
548                    "node/payloads/size/{}/{}",
549                    tx_heights[0] + 1,
550                    tx_heights[1]
551                ))
552                .send()
553                .await
554                .unwrap()
555        );
556
557        // All transactions
558        assert_eq!(
559            2,
560            client
561                .get::<usize>("node/transactions/count",)
562                .send()
563                .await
564                .unwrap()
565        );
566        assert_eq!(
567            tx_sizes[0] + tx_sizes[1],
568            client
569                .get::<usize>("node/payloads/size",)
570                .send()
571                .await
572                .unwrap()
573        );
574
575        network.shut_down().await;
576    }

578    #[tokio::test(flavor = "multi_thread")]
579    async fn test_extensions() {
580        setup_test();
581
582        let dir = TempDir::with_prefix("test_node_extensions").unwrap();
583        let data_source = ExtensibleDataSource::new(
584            MockDataSource::create(dir.path(), Default::default())
585                .await
586                .unwrap(),
587            0,
588        );
589
590        // Create the API extensions specification.
591        let extensions = toml! {
592            [route.post_ext]
593            PATH = ["/ext/:val"]
594            METHOD = "POST"
595            ":val" = "Integer"
596
597            [route.get_ext]
598            PATH = ["/ext"]
599            METHOD = "GET"
600        };
601
602        let mut api =
603            define_api::<RwLock<ExtensibleDataSource<MockDataSource, u64>>, MockTypes, MockBase>(
604                &Options {
605                    extensions: vec![extensions.into()],
606                    ..Default::default()
607                },
608                MockBase::instance(),
609            )
610            .unwrap();
611        api.get("get_ext", |_, state| {
612            async move { Ok(*state.as_ref()) }.boxed()
613        })
614        .unwrap()
615        .post("post_ext", |req, state| {
616            async move {
617                *state.as_mut() = req.integer_param("val")?;
618                Ok(())
619            }
620            .boxed()
621        })
622        .unwrap();
623
624        let mut app = App::<_, Error>::with_state(RwLock::new(data_source));
625        app.register_module("node", api).unwrap();
626
627        let port = pick_unused_port().unwrap();
628        let _server = BackgroundTask::spawn(
629            "server",
630            app.serve(format!("0.0.0.0:{}", port), MockBase::instance()),
631        );
632
633        let client = Client::<Error, MockBase>::new(
634            format!("http://localhost:{}/node", port).parse().unwrap(),
635        );
636        assert!(client.connect(Some(Duration::from_secs(60))).await);
637
638        assert_eq!(client.get::<u64>("ext").send().await.unwrap(), 0);
639        client.post::<()>("ext/42").send().await.unwrap();
640        assert_eq!(client.get::<u64>("ext").send().await.unwrap(), 42);
641
642        // Ensure we can still access the built-in functionality.
643        let sync_status: SyncStatus = client.get("sync-status").send().await.unwrap();
644        assert!(sync_status.is_fully_synced(), "{sync_status:?}");
645    }
