44/// A specific action that can be targeted to inject an error.
45#[derive(Clone, Copy, Debug, PartialEq, Eq)]
46pub enum FailableAction {
47    // TODO currently we implement failable actions for the availability methods, but if needed we
48    // can always add more variants for other actions.
49    GetHeader,
50    GetLeaf,
51    GetLeaves,
52    GetBlock,
53    GetPayload,
54    GetPayloadMetadata,
55    GetVidCommon,
56    GetVidCommonMetadata,
57    GetHeaderRange,
58    GetLeafRange,
59    GetBlockRange,
60    GetPayloadRange,
61    GetPayloadMetadataRange,
62    GetVidCommonRange,
63    GetVidCommonMetadataRange,
64    GetTransaction,
65    FirstAvailableLeaf,
66
67    /// Target any action for failure.
68    Any,
69}
70
71impl FailableAction {
72    /// Should `self` being targeted for failure cause `action` to fail?
73    fn matches(self, action: Self) -> bool {
74        // Fail if this is the action specifically targeted for failure or if we are failing any
75        // action right now.
76        self == action || self == Self::Any
77    }
78}
79
80#[derive(Clone, Copy, Debug, Default)]
81enum FailureMode {
82    #[default]
83    Never,
84    Once(FailableAction),
85    Always(FailableAction),
86}
87
88impl FailureMode {
89    fn maybe_fail(&mut self, action: FailableAction) -> QueryResult<()> {
90        match self {
91            Self::Once(fail_action) if fail_action.matches(action) => {
92                *self = Self::Never;
93            },
94            Self::Always(fail_action) if fail_action.matches(action) => {},
95            _ => return Ok(()),
96        }
97
98        Err(QueryError::Error {
99            message: "injected error".into(),
100        })
101    }
102}
103
104#[derive(Debug, Default)]
105struct Failure {
106    on_read: FailureMode,
107    on_write: FailureMode,
108    on_commit: FailureMode,
109    on_begin_writable: FailureMode,
110    on_begin_read_only: FailureMode,
111}
112
113/// Storage wrapper for error injection.
114#[derive(Clone, Debug)]
115pub struct FailStorage<S> {
116    inner: S,
117    failure: Arc<Mutex<Failure>>,
118}
119
120impl<S> From<S> for FailStorage<S> {
121    fn from(inner: S) -> Self {
122        Self {
123            inner,
124            failure: Default::default(),
125        }
126    }
127}
128
129impl<S> FailStorage<S> {
130    pub async fn fail_reads(&self, action: FailableAction) {
131        self.failure.lock().await.on_read = FailureMode::Always(action);
132    }
133
134    pub async fn fail_writes(&self, action: FailableAction) {
135        self.failure.lock().await.on_write = FailureMode::Always(action);
136    }
137
138    pub async fn fail_commits(&self, action: FailableAction) {
139        self.failure.lock().await.on_commit = FailureMode::Always(action);
140    }
141
142    pub async fn fail_begins_writable(&self, action: FailableAction) {
143        self.failure.lock().await.on_begin_writable = FailureMode::Always(action);
144    }
145
146    pub async fn fail_begins_read_only(&self, action: FailableAction) {
147        self.failure.lock().await.on_begin_read_only = FailureMode::Always(action);
148    }
149
150    pub async fn fail(&self, action: FailableAction) {
151        let mut failure = self.failure.lock().await;
152        failure.on_read = FailureMode::Always(action);
153        failure.on_write = FailureMode::Always(action);
154        failure.on_commit = FailureMode::Always(action);
155        failure.on_begin_writable = FailureMode::Always(action);
156        failure.on_begin_read_only = FailureMode::Always(action);
157    }
158
159    pub async fn pass_reads(&self) {
160        self.failure.lock().await.on_read = FailureMode::Never;
161    }
162
163    pub async fn pass_writes(&self) {
164        self.failure.lock().await.on_write = FailureMode::Never;
165    }
166
167    pub async fn pass_commits(&self) {
168        self.failure.lock().await.on_commit = FailureMode::Never;
169    }
170
171    pub async fn pass_begins_writable(&self) {
172        self.failure.lock().await.on_begin_writable = FailureMode::Never;
173    }
174
175    pub async fn pass_begins_read_only(&self) {
176        self.failure.lock().await.on_begin_read_only = FailureMode::Never;
177    }
178
179    pub async fn pass(&self) {
180        let mut failure = self.failure.lock().await;
181        failure.on_read = FailureMode::Never;
182        failure.on_write = FailureMode::Never;
183        failure.on_commit = FailureMode::Never;
184        failure.on_begin_writable = FailureMode::Never;
185        failure.on_begin_read_only = FailureMode::Never;
186    }
187
188    pub async fn fail_one_read(&self, action: FailableAction) {
189        self.failure.lock().await.on_read = FailureMode::Once(action);
190    }
191
192    pub async fn fail_one_write(&self, action: FailableAction) {
193        self.failure.lock().await.on_write = FailureMode::Once(action);
194    }
195
196    pub async fn fail_one_commit(&self, action: FailableAction) {
197        self.failure.lock().await.on_commit = FailureMode::Once(action);
198    }
199
200    pub async fn fail_one_begin_writable(&self, action: FailableAction) {
201        self.failure.lock().await.on_begin_writable = FailureMode::Once(action);
202    }
203
204    pub async fn fail_one_begin_read_only(&self, action: FailableAction) {
205        self.failure.lock().await.on_begin_read_only = FailureMode::Once(action);
206    }
207}

209impl<S> VersionedDataSource for FailStorage<S>
210where
211    S: VersionedDataSource,
212{
213    type Transaction<'a>
214        = Transaction<S::Transaction<'a>>
215    where
216        Self: 'a;
217    type ReadOnly<'a>
218        = Transaction<S::ReadOnly<'a>>
219    where
220        Self: 'a;
221
222    async fn write(&self) -> anyhow::Result<<Self as VersionedDataSource>::Transaction<'_>> {
223        self.failure
224            .lock()
225            .await
226            .on_begin_writable
227            .maybe_fail(FailableAction::Any)?;
228        Ok(Transaction {
229            inner: self.inner.write().await?,
230            failure: self.failure.clone(),
231        })
232    }
233
234    async fn read(&self) -> anyhow::Result<<Self as VersionedDataSource>::ReadOnly<'_>> {
235        self.failure
236            .lock()
237            .await
238            .on_begin_read_only
239            .maybe_fail(FailableAction::Any)?;
240        Ok(Transaction {
241            inner: self.inner.read().await?,
242            failure: self.failure.clone(),
243        })
244    }
245}
