13use std::{fmt::Debug, time::Duration};
14
15use anyhow::bail;
16use async_trait::async_trait;
17
18#[derive(Clone, Debug)]
19pub struct PrunerCfg {
20    pruning_threshold: Option<u64>,
21    minimum_retention: Duration,
22    target_retention: Duration,
23    batch_size: u64,
24    max_usage: u16,
25    interval: Duration,
26    state_tables: Vec<String>,
27}
28
29#[async_trait]
30pub trait PruneStorage: PrunerConfig {
31    type Pruner: Default + Send;
32
33    async fn get_disk_usage(&self) -> anyhow::Result<u64> {
34        Ok(0)
35    }
36
37    async fn prune(&self, _pruner: &mut Self::Pruner) -> anyhow::Result<Option<u64>> {
38        Ok(None)
39    }
40}
41
42#[async_trait]
43pub trait PrunedHeightStorage: Sized {
44    async fn load_pruned_height(&mut self) -> anyhow::Result<Option<u64>> {
45        Ok(None)
46    }
47}
48
49#[async_trait]
50pub trait PrunedHeightDataSource: Sized {
51    async fn load_pruned_height(&self) -> anyhow::Result<Option<u64>> {
52        Ok(None)
53    }
54}
55
56pub trait PrunerConfig {
57    fn set_pruning_config(&mut self, _cfg: PrunerCfg) {}
58    fn get_pruning_config(&self) -> Option<PrunerCfg> {
59        None
60    }
61}
62
63impl PrunerCfg {
64    pub fn new() -> Self {
65        Default::default()
66    }
67
68    pub fn validate(&self) -> anyhow::Result<()> {
69        if let Some(pruning_threshold) = self.pruning_threshold {
70            if pruning_threshold == 0 {
71                bail!("pruning_threshold must be greater than 0 or set to None")
72            }
73        }
74
75        if self.max_usage > 10000 {
76            bail!("max_usage must be less than or equal to 10000")
77        }
78
79        Ok(())
80    }
81
82    pub fn with_state_tables(mut self, state_tables: Vec<String>) -> Self {
83        self.state_tables = state_tables;
84        self
85    }
86
87    pub fn with_pruning_threshold(mut self, pruning_threshold: u64) -> Self {
88        self.pruning_threshold = Some(pruning_threshold);
89        self
90    }
91
92    pub fn with_minimum_retention(mut self, minimum_retention: Duration) -> Self {
93        self.minimum_retention = minimum_retention;
94        self
95    }
96
97    pub fn with_target_retention(mut self, target_retention: Duration) -> Self {
98        self.target_retention = target_retention;
99        self
100    }
101
102    pub fn with_batch_size(mut self, batch_size: u64) -> Self {
103        self.batch_size = batch_size;
104        self
105    }
106
107    pub fn with_max_usage(mut self, max_usage: u16) -> Self {
108        self.max_usage = max_usage;
109        self
110    }
111
112    pub fn with_interval(mut self, interval: Duration) -> Self {
113        self.interval = interval;
114        self
115    }
116
117    /// Disk space threshold (in bytes).
118    ///
119    /// If the disk usage exceeds this threshold, pruning of data starts from
120    /// the oldest data and continues until the disk usage falls below `MAX_USAGE
121    /// or until the oldest data is younger than `MINIMUM_RETENTION`
122    pub fn pruning_threshold(&self) -> Option<u64> {
123        self.pruning_threshold
124    }
125
126    /// Minimum data retention period
127    ///
128    /// Data younger than this is never pruned, regardless of disk usage.
129    pub fn minimum_retention(&self) -> Duration {
130        self.minimum_retention
131    }
132
133    /// Target data retention period
134    ///
135    /// This is the ideal period for which data should be retained
136    /// data younger than this and older than `MINIMUM_RETENTION` may be pruned if disk usage exceeds the `pruning_threshold`.
137    pub fn target_retention(&self) -> Duration {
138        self.target_retention
139    }
140
141    /// Number of blocks to remove in a single pruning operation.
142    pub fn batch_size(&self) -> u64 {
143        self.batch_size
144    }
145
146    /// Maximum disk usage (in basis points).
147    ///
148    /// Pruning stops once the disk usage falls below this value, even if
149    /// some data older than the `MINIMUM_RETENTION` remains. Values range
150    /// from 0 (0%) to 10000 (100%).
151    pub fn max_usage(&self) -> u16 {
152        self.max_usage
153    }
154
155    /// Pruning interval
156    pub fn interval(&self) -> Duration {
157        self.interval
158    }
159
160    /// State tables to prune
161    pub fn state_tables(&self) -> Vec<String> {
162        self.state_tables.clone()
163    }
164}
165
166impl Default for PrunerCfg {
167    fn default() -> Self {
168        Self {
169            // 3 TB
170            pruning_threshold: Some(3 * 10_u64.pow(12)),
171            // 1 day
172            minimum_retention: Duration::from_secs(24 * 3600),
173            // 7 days
174            target_retention: Duration::from_secs(7 * 24 * 3600),
175            batch_size: 30000,
176            // 80%
177            max_usage: 8000,
178            // 1.5 hour
179            interval: Duration::from_secs(5400),
180            state_tables: Vec::new(),
181        }
182    }
183}
