17//! atomically. A transaction also provides all the immutable query functionality of a regular
18//! database connection, so that the updated state of the database can be queried midway through a
19//! transaction.

93pub trait TransactionMode: Send + Sync {

96    ) -> impl Future<Output = anyhow::Result<()>> + Send;

101impl TransactionMode for Write {
102    #[allow(unused_variables)]
103    async fn begin(conn: &mut <Db as Database>::Connection) -> anyhow::Result<()> {
104        // SQLite automatically sets the read/write mode of a transactions based on the statements
105        // in it. However, there is still a good reason to explicitly enable write mode right from
106        // the start: if a transaction first executes a read statement and then a write statement,
107        // it will be upgraded from a read transaction to a write transaction. Because this involves
108        // obtaining a different kind of lock while already holding one, it can cause a deadlock,
109        // e.g.:
110        // * Transaction A executes a read statement, obtaining a read lock
111        // * Transaction B executes a write statement and begins waiting for a write lock
112        // * Transaction A executes a write statement and begins waiting for a write lock
113        //
114        // Transaction A can never obtain its write lock because it must first wait for transaction
115        // B to get a write lock, which cannot happen because B is in turn waiting for A to release
116        // its read lock.
117        //
118        // This type of deadlock cannot happen if transaction A immediately starts as a write, since
119        // it will then only ever try to acquire one type of lock (a write lock). By working with
120        // this restriction (transactions are either readers or writers, but never upgradable), we
121        // avoid deadlock, we more closely imitate the concurrency semantics of postgres, and we
122        // take advantage of the SQLite busy timeout, which may allow a transaction to acquire a
123        // lock and succeed (after a small delay), even when there was a conflicting transaction in
124        // progress. Whereas a deadlock is always an automatic rollback.
125        //
126        // The proper way to begin a write transaction in SQLite is with `BEGIN IMMEDIATE`. However,
127        // sqlx does not expose any way to customize the `BEGIN` statement that starts a
128        // transaction. A servicable workaround is to perform some write statement before performing
129        // any read statement, ensuring that the first lock we acquire is exclusive. A write
130        // statement that has no actual effect on the database is suitable for this purpose, hence
131        // the `WHERE false`.
132        #[cfg(feature = "embedded-db")]
133        conn.execute("UPDATE pruned_height SET id = id WHERE false")
134            .await?;
135
136        // With Postgres things are much more straightforward: just tell Postgres we want a write
137        // transaction immediately after opening it.
138        #[cfg(not(feature = "embedded-db"))]
139        conn.execute("SET TRANSACTION ISOLATION LEVEL SERIALIZABLE")
140            .await?;
141
142        Ok(())
143    }
144
145    fn display() -> &'static str {
146        "write"
147    }
148}

150impl TransactionMode for Read {
151    #[allow(unused_variables)]
152    async fn begin(conn: &mut <Db as Database>::Connection) -> anyhow::Result<()> {
153        // With Postgres, we explicitly set the transaction mode to specify that we want the
154        // strongest possible consistency semantics in case of competing transactions
155        // (SERIALIZABLE), and we want to wait until this is possible rather than failing
156        // (DEFERRABLE).
157        //
158        // With SQLite, there is nothing to be done here, as SQLite automatically starts
159        // transactions in read-only mode, and always has serializable concurrency unless we
160        // explicitly opt in to dirty reads with a pragma.
161        #[cfg(not(feature = "embedded-db"))]
162        conn.execute("SET TRANSACTION ISOLATION LEVEL SERIALIZABLE, READ ONLY, DEFERRABLE")
163            .await?;
164
165        Ok(())
166    }
167
168    fn display() -> &'static str {
169        "read-only"
170    }
171}

173#[derive(Clone, Copy, Debug)]
174enum CloseType {
175    Commit,
176    Revert,
177    Drop,
178}
179
180#[derive(Debug)]
181struct TransactionMetricsGuard<Mode> {
182    started_at: Instant,
183    metrics: PoolMetrics,
184    close_type: CloseType,
185    _mode: PhantomData<Mode>,
186}
187
188impl<Mode: TransactionMode> TransactionMetricsGuard<Mode> {
189    fn begin(metrics: PoolMetrics) -> Self {
190        let started_at = Instant::now();
191        tracing::trace!(mode = Mode::display(), ?started_at, "begin");
192        metrics.open_transactions.update(1);
193
194        Self {
195            started_at,
196            metrics,
197            close_type: CloseType::Drop,
198            _mode: Default::default(),
199        }
200    }
201
202    fn set_closed(&mut self, t: CloseType) {
203        self.close_type = t;
204    }
205}
206
207impl<Mode> Drop for TransactionMetricsGuard<Mode> {
208    fn drop(&mut self) {
209        self.metrics
210            .transaction_durations
211            .add_point((self.started_at.elapsed().as_millis() as f64) / 1000.);
212        self.metrics.open_transactions.update(-1);
213        match self.close_type {
214            CloseType::Commit => self.metrics.commits.add(1),
215            CloseType::Revert => self.metrics.reverts.add(1),
216            CloseType::Drop => self.metrics.drops.add(1),
217        }
218        tracing::trace!(started_at = ?self.started_at, reason = ?self.close_type, "close");
219    }
220}

231impl<Mode: TransactionMode> Transaction<Mode> {
232    pub(super) async fn new(pool: &Pool<Db>, metrics: PoolMetrics) -> anyhow::Result<Self> {
233        let mut inner = pool.begin().await?;
234        let metrics = TransactionMetricsGuard::begin(metrics);
235        Mode::begin(inner.as_mut()).await?;
236        Ok(Self { inner, metrics })
237    }
238}

240impl<Mode: TransactionMode> update::Transaction for Transaction<Mode> {
241    async fn commit(mut self) -> anyhow::Result<()> {
242        self.inner.commit().await?;
243        self.metrics.set_closed(CloseType::Commit);
244        Ok(())
245    }
246    fn revert(mut self) -> impl Future + Send {
247        async move {
248            self.inner.rollback().await.unwrap();
249            self.metrics.set_closed(CloseType::Revert);
250        }
251    }

254/// A collection of parameters which can be bound to a SQL query.
255///
256/// This trait allows us to carry around hetergenous lists of parameters (e.g. tuples) and bind them
257/// to a query at the last moment before executing. This means we can manipulate the parameters
258/// independently of the query before executing it. For example, by requiring a trait bound of
259/// `Params<'p> + Clone`, we get a list (or tuple) of parameters which can be cloned and then bound
260/// to a query, which allows us to keep a copy of the parameters around in order to retry the query
261/// if it fails.
262///
263/// # Lifetimes
264///
265/// A SQL [`Query`] with lifetime `'q` borrows from both it's SQL statement (`&'q str`) and its
266/// parameters (bound via `bind<'q>`). Sometimes, though, it is necessary for the statement and its
267/// parameters to have different (but overlapping) lifetimes. For example, the parameters might be
268/// passed in and owned by the caller, while the query string is constructed in the callee and its
269/// lifetime is limited to the callee scope. (See for example the [`upsert`](Transaction::upsert)
270/// function which does exactly this.)
271///
272/// We could rectify this situation with a trait bound like `P: for<'q> Params<'q>`, meaning `P`
273/// must be bindable to a query with a lifetime chosen by the callee. However, when `P` is an
274/// associated type, such as an element of an iterator, as in
275/// `<I as IntoIter>::Item: for<'q> Params<'q>`, [a current limitation](https://blog.rust-lang.org/2022/10/28/gats-stabilization.html#implied-static-requirement-from-higher-ranked-trait-bounds.)
276/// in the Rust compiler then requires `P: 'static`, which we don't necessarily want: the caller
277/// should be able to pass in a reference to avoid expensive cloning.
278///
279/// So, instead, we work around this by making it explicit in the [`Params`] trait that the lifetime
280/// of the query we're binding to (`'q`) may be different than the lifetime of the parameters (`'p`)
281/// as long as the parameters outlive the duration of the query (the `'p: 'q`) bound on the
282/// [`bind`](Self::bind) function.
283pub trait Params<'p> {
284    fn bind<'q, 'r>(
285        self,
286        q: &'q mut Separated<'r, 'p, Db, &'static str>,
287    ) -> &'q mut Separated<'r, 'p, Db, &'static str>
288    where
289        'p: 'r;
290}
291
292/// A collection of parameters with a statically known length.
293///
294/// This is a simple trick for enforcing at compile time that a list of parameters has a certain
295/// length, such as matching the length of a list of column names. This can prevent easy mistakes
296/// like leaving out a parameter. It is implemented for tuples up to length 8.
297pub trait FixedLengthParams<'p, const N: usize>: Params<'p> {}
298
299macro_rules! impl_tuple_params {
300    ($n:literal, ($($t:ident,)+)) => {
301        impl<'p,  $($t),+> Params<'p> for ($($t,)+)
302        where $(
303            $t: 'p +  Encode<'p, Db> + Type<Db>
304        ),+{
305            fn bind<'q, 'r>(self, q: &'q mut Separated<'r, 'p, Db, &'static str>) ->   &'q mut Separated<'r, 'p, Db, &'static str>
306            where 'p: 'r,
307            {
308                #[allow(non_snake_case)]
309                let ($($t,)+) = self;
310                q $(
311                    .push_bind($t)
312                )+
313
314            }
315        }
316
317        impl<'p, $($t),+> FixedLengthParams<'p, $n> for ($($t,)+)
318        where $(
319            $t: 'p + for<'q> Encode<'q, Db> + Type<Db>
320        ),+ {
321        }
322    };
323}
324
325impl_tuple_params!(1, (T,));
326impl_tuple_params!(2, (T1, T2,));
327impl_tuple_params!(3, (T1, T2, T3,));
328impl_tuple_params!(4, (T1, T2, T3, T4,));
329impl_tuple_params!(5, (T1, T2, T3, T4, T5,));
330impl_tuple_params!(6, (T1, T2, T3, T4, T5, T6,));
331impl_tuple_params!(7, (T1, T2, T3, T4, T5, T6, T7,));
332impl_tuple_params!(8, (T1, T2, T3, T4, T5, T6, T7, T8,));

334pub fn build_where_in<'a, I>(
335    query: &'a str,
336    column: &'a str,
337    values: I,
338) -> QueryResult<(queries::QueryBuilder<'a>, String)>
339where
340    I: IntoIterator,
341    I::Item: 'a + Encode<'a, Db> + Type<Db>,
342{
343    let mut builder = queries::QueryBuilder::default();
344    let params = values
345        .into_iter()
346        .map(|v| Ok(format!("{} ", builder.bind(v)?)))
347        .collect::<QueryResult<Vec<String>>>()?;
348
349    if params.is_empty() {
350        return Err(QueryError::Error {
351            message: "failed to build WHERE IN query. No parameter found ".to_string(),
352        });
353    }
354
355    let sql = format!(
356        "{query} where {column} IN ({}) ",
357        params.into_iter().join(",")
358    );
359
360    Ok((builder, sql))
361}

363/// Low-level, general database queries and mutation.
364impl Transaction<Write> {
365    pub async fn upsert<'p, const N: usize, R>(
366        &mut self,
367        table: &str,
368        columns: [&str; N],
369        pk: impl IntoIterator<Item = &str>,
370        rows: R,
371    ) -> anyhow::Result<()>
372    where
373        R: IntoIterator,
374        R::Item: 'p + FixedLengthParams<'p, N>,
375    {
376        let set_columns = columns
377            .iter()
378            .map(|col| format!("{col} = excluded.{col}"))
379            .join(",");
380        let columns_str = columns
381            .into_iter()
382            .map(|col| format!("\"{col}\""))
383            .join(",");
384        let pk = pk.into_iter().join(",");
385
386        let mut query_builder =
387            QueryBuilder::new(format!("INSERT INTO \"{table}\" ({columns_str}) "));
388        let mut num_rows = 0;
389
390        query_builder.push_values(rows, |mut b, row| {
391            num_rows += 1;
392            row.bind(&mut b);
393        });
394
395        if num_rows == 0 {
396            tracing::warn!("trying to upsert 0 rows, this has no effect");
397            return Ok(());
398        }
399        query_builder.push(format!(" ON CONFLICT ({pk}) DO UPDATE SET {set_columns}"));
400
401        let res = self.execute(query_builder.build()).await?;
402        let stmt = query_builder.sql();
403        let rows_modified = res.rows_affected() as usize;
404
405        if rows_modified != num_rows {
406            tracing::error!(
407                stmt,
408                "unexpected number of rows modified: expected {num_rows} but got {rows_modified}"
409            );
410        }
411        Ok(())
412    }
413}

415/// Query service specific mutations.
416impl Transaction<Write> {
417    /// Delete a batch of data for pruning.
418    pub(super) async fn delete_batch(
419        &mut self,
420        state_tables: Vec<String>,
421        height: u64,
422    ) -> anyhow::Result<()> {
423        self.execute(query("DELETE FROM header WHERE height <= $1").bind(height as i64))
424            .await?;
425
426        // prune merklized state tables
427        // only delete nodes having created < h AND
428        // is not the newest node with its position
429        for state_table in state_tables {
430            self.execute(
431                query(&format!(
432                    "
433                DELETE FROM {state_table} WHERE (path, created) IN
434                (SELECT path, created FROM 
435                (SELECT path, created, 
436                ROW_NUMBER() OVER (PARTITION BY path ORDER BY created DESC) as rank 
437                FROM {state_table} WHERE created <= $1) ranked_nodes WHERE rank != 1)"
438                ))
439                .bind(height as i64),
440            )
441            .await?;
442        }
443
444        self.save_pruned_height(height).await?;
445        Ok(())
446    }
447
448    /// Record the height of the latest pruned header.
449    pub(super) async fn save_pruned_height(&mut self, height: u64) -> anyhow::Result<()> {
450        // id is set to 1 so that there is only one row in the table.
451        // height is updated if the row already exists.
452        self.upsert(
453            "pruned_height",
454            ["id", "last_height"],
455            ["id"],
456            [(1i32, height as i64)],
457        )
458        .await
459    }

462impl<Types> UpdateAvailabilityStorage<Types> for Transaction<Write>
463where
464    Types: NodeType,
465    Payload<Types>: QueryablePayload<Types>,
466    Header<Types>: QueryableHeader<Types>,
467{
468    async fn insert_leaf(&mut self, leaf: LeafQueryData<Types>) -> anyhow::Result<()> {
469        let height = leaf.height();
470        let view = leaf.leaf().view_number().u64();
471
472        // Ignore the leaf if it is below the pruned height. This can happen if, for instance, the
473        // fetcher is racing with the pruner.
474        if let Some(pruned_height) = self.load_pruned_height().await? {
475            if height <= pruned_height {
476                tracing::info!(
477                    height,
478                    pruned_height,
479                    "ignoring leaf which is already pruned"
480                );
481                return Ok(());
482            }
483        }
484
485        // While we don't necessarily have the full block for this leaf yet, we can initialize the
486        // header table with block metadata taken from the leaf.
487        let header_json = serde_json::to_value(leaf.leaf().block_header())
488            .context("failed to serialize header")?;
489        self.upsert(
490            "header",
491            ["height", "hash", "payload_hash", "data", "timestamp"],
492            ["height"],
493            [(
494                height as i64,
495                leaf.block_hash().to_string(),
496                leaf.leaf().block_header().payload_commitment().to_string(),
497                header_json,
498                leaf.leaf().block_header().timestamp() as i64,
499            )],
500        )
501        .await?;
502
503        // Similarly, we can initialize the payload table with a null payload, which can help us
504        // distinguish between blocks that haven't been produced yet and blocks we haven't received
505        // yet when answering queries.
506        // We don't overwrite the payload if it already exists.
507        // During epoch transition in PoS, the same height block is sent multiple times.
508        // The first block may have the payload, but subsequent blocks might be missing it.
509        // Overwriting would cause the payload to be lost since the block height is the same
510        let query = query("INSERT INTO payload (height) VALUES ($1) ON CONFLICT DO NOTHING")
511            .bind(height as i64);
512        query.execute(self.as_mut()).await?;
513
514        // Finally, we insert the leaf itself, which references the header row we created.
515        // Serialize the full leaf and QC to JSON for easy storage.
516        let leaf_json = serde_json::to_value(leaf.leaf()).context("failed to serialize leaf")?;
517        let qc_json = serde_json::to_value(leaf.qc()).context("failed to serialize QC")?;
518        self.upsert(
519            "leaf2",
520            ["height", "view", "hash", "block_hash", "leaf", "qc"],
521            ["height", "view"],
522            [(
523                height as i64,
524                view as i64,
525                leaf.hash().to_string(),
526                leaf.block_hash().to_string(),
527                leaf_json,
528                qc_json,
529            )],
530        )
531        .await?;
532
533        Ok(())
534    }
535
536    async fn insert_block(&mut self, block: BlockQueryData<Types>) -> anyhow::Result<()> {
537        let height = block.height();
538
539        // Ignore the block if it is below the pruned height. This can happen if, for instance, the
540        // fetcher is racing with the pruner.
541        if let Some(pruned_height) = self.load_pruned_height().await? {
542            if height <= pruned_height {
543                tracing::info!(
544                    height,
545                    pruned_height,
546                    "ignoring block which is already pruned"
547                );
548                return Ok(());
549            }
550        }
551
552        // The header and payload tables should already have been initialized when we inserted the
553        // corresponding leaf. All we have to do is add the payload itself and its size.
554        let payload = block.payload.encode();
555
556        self.upsert(
557            "payload",
558            ["height", "data", "size", "num_transactions"],
559            ["height"],
560            [(
561                height as i64,
562                payload.as_ref().to_vec(),
563                block.size() as i32,
564                block.num_transactions() as i32,
565            )],
566        )
567        .await?;
568
569        // Index the transactions in the block.
570        let mut rows = vec![];
571        for (txn_ix, txn) in block.enumerate() {
572            let txn_ix =
573                serde_json::to_value(&txn_ix).context("failed to serialize transaction index")?;
574            rows.push((txn.commit().to_string(), height as i64, txn_ix));
575        }
576        if !rows.is_empty() {
577            self.upsert(
578                "transactions",
579                ["hash", "block_height", "idx"],
580                ["block_height", "idx"],
581                rows,
582            )
583            .await?;
584        }
585
586        Ok(())
587    }

589    async fn insert_vid(
590        &mut self,
591        common: VidCommonQueryData<Types>,
592        share: Option<VidShare>,
593    ) -> anyhow::Result<()> {
594        let height = common.height();
595
596        // Ignore the object if it is below the pruned height. This can happen if, for instance, the
597        // fetcher is racing with the pruner.
598        if let Some(pruned_height) = self.load_pruned_height().await? {
599            if height <= pruned_height {
600                tracing::info!(
601                    height,
602                    pruned_height,
603                    "ignoring VID common which is already pruned"
604                );
605                return Ok(());
606            }
607        }
608
609        let common_data =
610            bincode::serialize(common.common()).context("failed to serialize VID common data")?;
611        if let Some(share) = share {
612            let share_data = bincode::serialize(&share).context("failed to serialize VID share")?;
613            self.upsert(
614                "vid2",
615                ["height", "common", "share"],
616                ["height"],
617                [(height as i64, common_data, share_data)],
618            )
619            .await
620        } else {
621            // Don't touch the `share` column at all if we don't have a share to insert. It's
622            // possible that this column already exists, and we are just upserting the common data,
623            // in which case we don't want to overwrite the share with NULL.
624            self.upsert(
625                "vid2",
626                ["height", "common"],
627                ["height"],
628                [(height as i64, common_data)],
629            )
630            .await
631        }
632    }

650
651    async fn insert_merkle_nodes(
652        &mut self,
653        proof: MerkleProof<State::Entry, State::Key, State::T, ARITY>,
654        traversal_path: Vec<usize>,
655        block_number: u64,
656    ) -> anyhow::Result<()> {
657        let pos = proof.pos;
658        let path = proof.proof;
659
660        let name = State::state_type();
661        let block_number = block_number as i64;
662
663        let mut traversal_path = traversal_path.iter().map(|n| *n as i32);
664
665        // All the nodes are collected here, They depend on the hash ids which are returned after
666        // hashes are upserted in the db
667        let mut nodes = Vec::new();
668        let mut hashset = HashSet::new();
669
670        for node in path.iter() {
671            match node {
672                MerkleNode::Empty => {
673                    let index = serde_json::to_value(pos.clone())
674                        .decode_error("malformed merkle position")?;
675                    // The node path represents the sequence of nodes from the root down to a specific node.
676                    // Therefore, the traversal path needs to be reversed
677                    // The root node path is an empty array.
678                    let node_path = traversal_path.clone().rev().collect();
679                    nodes.push((
680                        Node {
681                            path: node_path,
682                            idx: Some(index),
683                            ..Default::default()
684                        },
685                        None,
686                        [0_u8; 32].to_vec(),
687                    ));
688                    hashset.insert([0_u8; 32].to_vec());
689                },
690                MerkleNode::ForgettenSubtree { .. } => {
691                    bail!("Node in the Merkle path contains a forgetten subtree");
692                },
693                MerkleNode::Leaf { value, pos, elem } => {
694                    let mut leaf_commit = Vec::new();
695                    // Serialize the leaf node hash value into a vector
696                    value
697                        .serialize_compressed(&mut leaf_commit)
698                        .decode_error("malformed merkle leaf commitment")?;
699
700                    let path = traversal_path.clone().rev().collect();
701
702                    let index = serde_json::to_value(pos.clone())
703                        .decode_error("malformed merkle position")?;
704                    let entry =
705                        serde_json::to_value(elem).decode_error("malformed merkle element")?;
706
707                    nodes.push((
708                        Node {
709                            path,
710                            idx: Some(index),
711                            entry: Some(entry),
712                            ..Default::default()
713                        },
714                        None,
715                        leaf_commit.clone(),
716                    ));
717
718                    hashset.insert(leaf_commit);
719                },
720                MerkleNode::Branch { value, children } => {
721                    // Get hash
722                    let mut branch_hash = Vec::new();
723                    value
724                        .serialize_compressed(&mut branch_hash)
725                        .decode_error("malformed merkle branch hash")?;
726
727                    // We only insert the non-empty children in the children field of the table
728                    // BitVec is used to separate out Empty children positions
729                    let mut children_bitvec = BitVec::new();
730                    let mut children_values = Vec::new();
731                    for child in children {
732                        let child = child.as_ref();
733                        match child {
734                            MerkleNode::Empty => {
735                                children_bitvec.push(false);
736                            },
737                            MerkleNode::Branch { value, .. }
738                            | MerkleNode::Leaf { value, .. }
739                            | MerkleNode::ForgettenSubtree { value } => {
740                                let mut hash = Vec::new();
741                                value
742                                    .serialize_compressed(&mut hash)
743                                    .decode_error("malformed merkle node hash")?;
744
745                                children_values.push(hash);
746                                // Mark the entry as 1 in bitvec to indicate a non-empty child
747                                children_bitvec.push(true);
748                            },
749                        }
750                    }
751
752                    // insert internal node
753                    let path = traversal_path.clone().rev().collect();
754                    nodes.push((
755                        Node {
756                            path,
757                            children: None,
758                            children_bitvec: Some(children_bitvec),
759                            ..Default::default()
760                        },
761                        Some(children_values.clone()),
762                        branch_hash.clone(),
763                    ));
764                    hashset.insert(branch_hash);
765                    hashset.extend(children_values);
766                },
767            }
768
769            // advance the traversal path for the internal nodes at each iteration
770            // The final node would be the Root node where this iterator is exhausted
771            traversal_path.next();
772        }
773        // We build a hashset to avoid duplicate entries
774        let hashes = hashset.into_iter().collect::<Vec<Vec<u8>>>();
775
776        // insert all the hashes into database
777        // It returns all the ids inserted in the order they were inserted
778        // We use the hash ids to insert all the nodes
779        let (query, sql) = build_hash_batch_insert(&hashes)?;
780        let nodes_hash_ids: HashMap<Vec<u8>, i32> = query
781            .query_as(&sql)
782            .fetch(self.as_mut())
783            .try_collect()
784            .await?;
785
786        // Updates the node fields
787        for (node, children, hash) in &mut nodes {
788            node.created = block_number;
789            node.hash_id = *nodes_hash_ids.get(&*hash).ok_or(QueryError::Error {
790                message: "Missing node hash".to_string(),
791            })?;
792
793            if let Some(children) = children {
794                let children_hashes = children
795                    .iter()
796                    .map(|c| nodes_hash_ids.get(c).copied())
797                    .collect::<Option<Vec<i32>>>()
798                    .ok_or(QueryError::Error {
799                        message: "Missing child hash".to_string(),
800                    })?;
801
802                node.children = Some(children_hashes.into());
803            }
804        }
805
806        Node::upsert(name, nodes.into_iter().map(|(n, ..)| n), self).await?;
807
808        Ok(())
809    }
810}

813impl<Mode: TransactionMode> PrunedHeightStorage for Transaction<Mode> {
814    async fn load_pruned_height(&mut self) -> anyhow::Result<Option<u64>> {
815        let Some((height,)) =
816            query_as::<(i64,)>("SELECT last_height FROM pruned_height ORDER BY id DESC LIMIT 1")
817                .fetch_optional(self.as_mut())
818                .await?
819        else {
820            return Ok(None);
821        };
822        Ok(Some(height as u64))
823    }
