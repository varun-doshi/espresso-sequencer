73/// Embed migrations from the given directory into the current binary for PostgreSQL or SQLite.
74///
75/// The macro invocation `include_migrations!(path)` evaluates to an expression of type `impl
76/// Iterator<Item = Migration>`. Each migration must be a text file which is an immediate child of
77/// `path`, and there must be no non-migration files in `path`. The migration files must have names
78/// of the form `V${version}__${name}.sql`, where `version` is a positive integer indicating how the
79/// migration is to be ordered relative to other migrations, and `name` is a descriptive name for
80/// the migration.
81///
82/// `path` should be an absolute path. It is possible to give a path relative to the root of the
83/// invoking crate by using environment variable expansions and the `CARGO_MANIFEST_DIR` environment
84/// variable.
85///
86/// As an example, this is the invocation used to load the default migrations from the
87/// `hotshot-query-service` crate. The migrations are located in a directory called `migrations` at
88/// - PostgreSQL migrations are in `/migrations/postgres`.
89/// - SQLite migrations are in `/migrations/sqlite`.
90///
91/// ```
92/// # use hotshot_query_service::data_source::sql::{include_migrations, Migration};
93/// // For PostgreSQL
94/// #[cfg(not(feature = "embedded-db"))]
95///  let mut migrations: Vec<Migration> =
96///     include_migrations!("$CARGO_MANIFEST_DIR/migrations/postgres").collect();
97/// // For SQLite
98/// #[cfg(feature = "embedded-db")]
99/// let mut migrations: Vec<Migration> =
100///     include_migrations!("$CARGO_MANIFEST_DIR/migrations/sqlite").collect();
101///    
102///     migrations.sort();
103///     assert_eq!(migrations[0].version(), 10);
104///     assert_eq!(migrations[0].name(), "init_schema");
105/// ```
106///
107/// Note that a similar macro is available from Refinery:
108/// [embed_migrations](https://docs.rs/refinery/0.8.11/refinery/macro.embed_migrations.html). This
109/// macro differs in that it evaluates to an iterator of [migrations](Migration), making it an
110/// expression macro, while `embed_migrations` is a statement macro that defines a module which
111/// provides access to the embedded migrations only indirectly via a
112/// [`Runner`](https://docs.rs/refinery/0.8.11/refinery/struct.Runner.html). The direct access to
113/// migrations provided by [`include_migrations`] makes this macro easier to use with
114/// [`Config::migrations`], for combining custom migrations with [`default_migrations`].
115#[macro_export]
116macro_rules! include_migrations {
117    ($dir:tt) => {
118        $crate::data_source::storage::sql::include_dir!($dir)
119            .files()
120            .map(|file| {
121                let path = file.path();
122                let name = path
123                    .file_name()
124                    .and_then(std::ffi::OsStr::to_str)
125                    .unwrap_or_else(|| {
126                        panic!(
127                            "migration file {} must have a non-empty UTF-8 name",
128                            path.display()
129                        )
130                    });
131                let sql = file
132                    .contents_utf8()
133                    .unwrap_or_else(|| panic!("migration file {name} must use UTF-8 encoding"));
134                $crate::data_source::storage::sql::Migration::unapplied(name, sql)
135                    .expect("invalid migration")
136            })
137    };
138}

140/// The migrations required to build the default schema for this version of [`SqlStorage`].
141pub fn default_migrations() -> Vec<Migration> {
142    #[cfg(not(feature = "embedded-db"))]
143    let mut migrations =
144        include_migrations!("$CARGO_MANIFEST_DIR/migrations/postgres").collect::<Vec<_>>();
145
146    #[cfg(feature = "embedded-db")]
147    let mut migrations =
148        include_migrations!("$CARGO_MANIFEST_DIR/migrations/sqlite").collect::<Vec<_>>();
149
150    // Check version uniqueness and sort by version.
151    validate_migrations(&mut migrations).expect("default migrations are invalid");
152
153    // Check that all migration versions are multiples of 100, so that custom migrations can be
154    // inserted in between.
155    for m in &migrations {
156        if m.version() <= 30 {
157            // An older version of this software used intervals of 10 instead of 100. This was
158            // changed to allow more custom migrations between each default migration, but we must
159            // still accept older migrations that followed the older rule.
160            assert!(
161                m.version() > 0 && m.version() % 10 == 0,
162                "legacy default migration version {} is not a positive multiple of 10",
163                m.version()
164            );
165        } else {
166            assert!(
167                m.version() % 100 == 0,
168                "default migration version {} is not a multiple of 100",
169                m.version()
170            );
171        }
172    }
173
174    migrations
175}

177/// Validate and preprocess a sequence of migrations.
178///
179/// * Ensure all migrations have distinct versions
180/// * Ensure migrations are sorted by increasing version
181fn validate_migrations(migrations: &mut [Migration]) -> Result<(), Error> {
182    migrations.sort_by_key(|m| m.version());
183
184    // Check version uniqueness.
185    for (prev, next) in migrations.iter().zip(migrations.iter().skip(1)) {
186        if next <= prev {
187            return Err(Error::msg(format!(
188                "migration versions are not strictly increasing ({prev}->{next})"
189            )));
190        }
191    }
192
193    Ok(())
194}

196/// Add custom migrations to a default migration sequence.
197///
198/// Migrations in `custom` replace migrations in `default` with the same version. Otherwise, the two
199/// sequences `default` and `custom` are merged so that the resulting sequence is sorted by
200/// ascending version number. Each of `default` and `custom` is assumed to be the output of
201/// [`validate_migrations`]; that is, each is sorted by version and contains no duplicate versions.
202fn add_custom_migrations(
203    default: impl IntoIterator<Item = Migration>,
204    custom: impl IntoIterator<Item = Migration>,
205) -> impl Iterator<Item = Migration> {
206    default
207        .into_iter()
208        // Merge sorted lists, joining pairs of equal version into `EitherOrBoth::Both`.
209        .merge_join_by(custom, |l, r| l.version().cmp(&r.version()))
210        // Prefer the custom migration for a given version when both default and custom versions
211        // are present.
212        .map(|pair| pair.reduce(|_, custom| custom))
213}

500impl SqlStorage {
501    pub fn pool(&self) -> Pool<Db> {
502        self.pool.clone()
503    }
504    /// Connect to a remote database.
505    pub async fn connect(mut config: Config) -> Result<Self, Error> {
506        let metrics = PrometheusMetrics::default();
507        let pool_metrics = PoolMetrics::new(&*metrics.subgroup("sql".into()));
508        let pool = config.pool_opt.clone();
509        let pruner_cfg = config.pruner_cfg;
510
511        // re-use the same pool if present and return early
512        if let Some(pool) = config.pool {
513            return Ok(Self {
514                metrics,
515                pool_metrics,
516                pool,
517                pruner_cfg,
518            });
519        }
520
521        #[cfg(not(feature = "embedded-db"))]
522        let schema = config.schema.clone();
523        #[cfg(not(feature = "embedded-db"))]
524        let pool = pool.after_connect(move |conn, _| {
525            let schema = config.schema.clone();
526            async move {
527                query(&format!("SET search_path TO {schema}"))
528                    .execute(conn)
529                    .await?;
530                Ok(())
531            }
532            .boxed()
533        });
534
535        #[cfg(feature = "embedded-db")]
536        if config.reset {
537            std::fs::remove_file(config.db_opt.get_filename())?;
538        }
539
540        let pool = pool.connect_with(config.db_opt).await?;
541
542        // Create or connect to the schema for this query service.
543        let mut conn = pool.acquire().await?;
544
545        #[cfg(not(feature = "embedded-db"))]
546        if config.reset {
547            query(&format!("DROP SCHEMA IF EXISTS {} CASCADE", schema))
548                .execute(conn.as_mut())
549                .await?;
550        }
551
552        #[cfg(not(feature = "embedded-db"))]
553        query(&format!("CREATE SCHEMA IF NOT EXISTS {}", schema))
554            .execute(conn.as_mut())
555            .await?;
556
557        // Get migrations and interleave with custom migrations, sorting by version number.
558        validate_migrations(&mut config.migrations)?;
559        let migrations =
560            add_custom_migrations(default_migrations(), config.migrations).collect::<Vec<_>>();
561
562        // Get a migration runner. Depending on the config, we can either use this to actually run
563        // the migrations or just check if the database is up to date.
564        let runner = refinery::Runner::new(&migrations).set_grouped(true);
565
566        if config.no_migrations {
567            // We've been asked not to run any migrations. Abort if the DB is not already up to
568            // date.
569            let last_applied = runner
570                .get_last_applied_migration_async(&mut Migrator::from(&mut conn))
571                .await?;
572            let last_expected = migrations.last();
573            if last_applied.as_ref() != last_expected {
574                return Err(Error::msg(format!(
575                    "DB is out of date: last applied migration is {last_applied:?}, but expected {last_expected:?}"
576                )));
577            }
578        } else {
579            // Run migrations using `refinery`.
580            match runner.run_async(&mut Migrator::from(&mut conn)).await {
581                Ok(report) => {
582                    tracing::info!("ran DB migrations: {report:?}");
583                },
584                Err(err) => {
585                    tracing::error!("DB migrations failed: {:?}", err.report());
586                    Err(err)?;
587                },
588            }
589        }
590
591        if config.archive {
592            // If running in archive mode, ensure the pruned height is set to 0, so the fetcher will
593            // reconstruct previously pruned data.
594            query("DELETE FROM pruned_height WHERE id = 1")
595                .execute(conn.as_mut())
596                .await?;
597        }
598
599        conn.close().await?;
600
601        Ok(Self {
602            pool,
603            pool_metrics,
604            metrics,
605            pruner_cfg,
606        })
607    }
608}

626impl SqlStorage {
627    async fn get_minimum_height(&self) -> QueryResult<Option<u64>> {
628        let mut tx = self.read().await.map_err(|err| QueryError::Error {
629            message: err.to_string(),
630        })?;
631        let (Some(height),) =
632            query_as::<(Option<i64>,)>("SELECT MIN(height) as height FROM header")
633                .fetch_one(tx.as_mut())
634                .await?
635        else {
636            return Ok(None);
637        };
638        Ok(Some(height as u64))
639    }
640
641    async fn get_height_by_timestamp(&self, timestamp: i64) -> QueryResult<Option<u64>> {
642        let mut tx = self.read().await.map_err(|err| QueryError::Error {
643            message: err.to_string(),
644        })?;
645
646        // We order by timestamp and then height, even though logically this is no different than
647        // just ordering by height, since timestamps are monotonic. The reason is that this order
648        // allows the query planner to efficiently solve the where clause and presort the results
649        // based on the timestamp index. The remaining sort on height, which guarantees a unique
650        // block if multiple blocks have the same timestamp, is very efficient, because there are
651        // never more than a handful of blocks with the same timestamp.
652        let Some((height,)) = query_as::<(i64,)>(
653            "SELECT height FROM header
654              WHERE timestamp <= $1
655              ORDER BY timestamp DESC, height DESC
656              LIMIT 1",
657        )
658        .bind(timestamp)
659        .fetch_optional(tx.as_mut())
660        .await?
661        else {
662            return Ok(None);
663        };
664        Ok(Some(height as u64))
665    }
666}

673        let mut tx = self.read().await?;
674
675        #[cfg(not(feature = "embedded-db"))]
676        let query = "SELECT pg_database_size(current_database())";
677
678        #[cfg(feature = "embedded-db")]
679        let query = "
680            SELECT( (SELECT page_count FROM pragma_page_count) * (SELECT * FROM pragma_page_size)) AS total_bytes";
681
682        let row = tx.fetch_one(query).await?;
683        let size: i64 = row.get(0);
684
685        Ok(size as u64)
686    }
687
688    /// Note: The prune operation may not immediately free up space even after rows are deleted.
689    /// This is because a vacuum operation may be necessary to reclaim more space.
690    /// PostgreSQL already performs auto vacuuming, so we are not including it here
691    /// as running a vacuum operation can be resource-intensive.
692    async fn prune(&self, pruner: &mut Pruner) -> anyhow::Result<Option<u64>> {
693        let cfg = self.get_pruning_config().ok_or(QueryError::Error {
694            message: "Pruning config not found".to_string(),
695        })?;
696        let batch_size = cfg.batch_size();
697        let max_usage = cfg.max_usage();
698        let state_tables = cfg.state_tables();
699
700        // If a pruner run was already in progress, some variables may already be set,
701        // depending on whether a batch was deleted and which batch it was (target or minimum retention).
702        // This enables us to resume the pruner run from the exact heights.
703        // If any of these values are not set, they can be loaded from the database if necessary.
704        let mut minimum_retention_height = pruner.minimum_retention_height;
705        let mut target_height = pruner.target_height;
706        let mut height = match pruner.pruned_height {
707            Some(h) => h,
708            None => {
709                let Some(height) = self.get_minimum_height().await? else {
710                    tracing::info!("database is empty, nothing to prune");
711                    return Ok(None);
712                };
713
714                height
715            },
716        };
717
718        // Prune data exceeding target retention in batches
719        if pruner.target_height.is_none() {
720            let th = self
721                .get_height_by_timestamp(
722                    Utc::now().timestamp() - (cfg.target_retention().as_secs()) as i64,
723                )
724                .await?;
725            target_height = th;
726            pruner.target_height = target_height;
727        };
728
729        if let Some(target_height) = target_height {
730            if height < target_height {
731                height = min(height + batch_size, target_height);
732                let mut tx = self.write().await?;
733                tx.delete_batch(state_tables, height).await?;
734                tx.commit().await.map_err(|e| QueryError::Error {
735                    message: format!("failed to commit {e}"),
736                })?;
737
738                pruner.pruned_height = Some(height);
739                return Ok(Some(height));
740            }
741        }
742
743        #[cfg(feature = "embedded-db")]
744        {
745            let mut conn = self.pool().acquire().await?;
746            query("VACUUM").execute(conn.as_mut()).await?;
747            conn.close().await?;
748        }
749
750        // If threshold is set, prune data exceeding minimum retention in batches
751        // This parameter is needed for SQL storage as there is no direct way to get free space.
752        if let Some(threshold) = cfg.pruning_threshold() {
753            let usage = self.get_disk_usage().await?;
754
755            // Prune data exceeding minimum retention in batches starting from minimum height
756            // until usage is below threshold
757            if usage > threshold {
758                tracing::warn!(
759                    "Disk usage {usage} exceeds pruning threshold {:?}",
760                    cfg.pruning_threshold()
761                );
762
763                if minimum_retention_height.is_none() {
764                    minimum_retention_height = self
765                        .get_height_by_timestamp(
766                            Utc::now().timestamp() - (cfg.minimum_retention().as_secs()) as i64,
767                        )
768                        .await?;
769
770                    pruner.minimum_retention_height = minimum_retention_height;
771                }
772
773                if let Some(min_retention_height) = minimum_retention_height {
774                    if (usage as f64 / threshold as f64) > (f64::from(max_usage) / 10000.0)
775                        && height < min_retention_height
776                    {
777                        height = min(height + batch_size, min_retention_height);
778                        let mut tx = self.write().await?;
779                        tx.delete_batch(state_tables, height).await?;
780                        tx.commit().await.map_err(|e| QueryError::Error {
781                            message: format!("failed to commit {e}"),
782                        })?;
783
784                        #[cfg(feature = "embedded-db")]
785                        {
786                            let mut conn = self.pool().acquire().await?;
787                            query("VACUUM").execute(conn.as_mut()).await?;
788                            conn.close().await?;
789                        }
790
791                        pruner.pruned_height = Some(height);
792
793                        return Ok(Some(height));
794                    }
795                }
796            }
797        }
798
799        Ok(None)
800    }

855                    "SELECT leaf, qc, common as vid_common, share as vid_share FROM leaf INNER JOIN vid on leaf.height = vid.height ORDER BY leaf.height LIMIT {} OFFSET {}",
856                    limit, offset
857                ))

862

865            }
866

868            let mut vid_rows = Vec::new();

872                let qc = row.try_get("qc")?;
873                let leaf1: Leaf<Types> = serde_json::from_value(leaf1)?;

877                let qc2: QuorumCertificate2<Types> = qc.to_qc2();

882                    serde_json::to_value(leaf2.clone()).context("failed to serialize leaf2")?;
883                let qc2_json = serde_json::to_value(qc2).context("failed to serialize QC2")?;
884

886                let vid_common_bytes: Vec<u8> = row.try_get("vid_common")?;
887                let vid_share_bytes: Vec<u8> = row.try_get("vid_share")?;

890                    .context("failed to deserialize vid_share")?;
891
892                let new_vid_share_bytes = bincode::serialize(&VidShare::V0(vid_share))
893                    .context("failed to serialize vid_share")?;
894
895                let vid_common: ADVZCommon = bincode::deserialize(&vid_common_bytes)
896                    .context("failed to deserialize vid_common")?;
897
898                let new_vid_common_bytes = bincode::serialize(&VidCommon::V0(vid_common))
899                    .context("failed to serialize vid_common")?;

910                    leaf2.block_header().commit().to_string(),

912                    qc2_json,
913                ));
914            }
915

923                    .push_bind(row.1)
924                    .push_bind(row.2)
925                    .push_bind(row.3)
926                    .push_bind(row.4)
927                    .push_bind(row.5);
928            });

931
932            let mut tx = self.write().await.map_err(|err| QueryError::Error {
933                message: err.to_string(),
934            })?;
935
936            query.execute(tx.as_mut()).await?;
937
938            tx.commit().await?;
939            tracing::warn!("inserted {} rows into leaf2 table", offset);
940            // migrate vid
941            let mut query_builder: sqlx::QueryBuilder<Db> =
942                sqlx::QueryBuilder::new("INSERT INTO vid2 (height, common, share) ");
943
944            query_builder.push_values(vid_rows.into_iter(), |mut b, row| {
945                b.push_bind(row.0).push_bind(row.1).push_bind(row.2);
946            });
947
948            let query = query_builder.build();
949
950            let mut tx = self.write().await.map_err(|err| QueryError::Error {
951                message: err.to_string(),
952            })?;
953
954            query.execute(tx.as_mut()).await?;
955
956            tx.commit().await?;
957
958            tracing::warn!("inserted {} rows into vid2 table", offset);
959
960            if rows.len() < limit {
961                break;
962            }
963
964            offset += limit;
965        }
966
967        let mut tx = self.write().await.map_err(|err| QueryError::Error {
968            message: err.to_string(),
969        })?;
970
971        tracing::warn!("query service types migration is completed!");
972
973        tx.upsert(
974            "types_migration",
975            ["id", "completed"],
976            ["id"],
977            [(0_i64, true)],
978        )
979        .await?;
980
981        tracing::info!("updated types_migration table");
982
983        tx.commit().await?;
984        Ok(())
985    }

1184                    .stdout(Stdio::null())
1185                    .stderr(Stdio::null())
1186                    .status()
1187                    // We should ensure the exit status. A simple `unwrap`
1188                    // would panic on unrelated errors (such as network
1189                    // connection failures)
1190                    .and_then(|status| {
1191                        status
1192                            .success()
1193                            .then_some(true)

1195                            .ok_or(std::io::Error::from_raw_os_error(666))

1204                // However, on some systems, there is a slight delay before the port is exposed via
1205                // host networking. We don't need to check again that the database is ready on the
1206                // host (and maybe can't, because the host might not have pg_isready installed), but
1207                // we can ensure the port is open by just establishing a TCP connection.
1208                while let Err(err) =
1209                    TcpStream::connect(format!("{}:{}", self.host, self.port)).await
1210                {
1211                    tracing::warn!("database is ready, but port is not available to host: {err:#}");
1212                    sleep(Duration::from_millis(100)).await;
1213                }
1214            })
1215            .await
1216            {
1217                panic!(
1218                    "failed to connect to TmpDb within configured timeout {timeout_duration:?}: {err:#}\n{}",
1219                    "Consider increasing the timeout by setting SQL_TMP_DB_CONNECT_TIMEOUT"
1220                );
1221            }
