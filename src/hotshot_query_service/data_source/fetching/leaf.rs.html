48impl<Types> FetchRequest for LeafId<Types>
49where
50    Types: NodeType,
51{
52    fn might_exist(self, heights: Heights) -> bool {
53        if let LeafId::Number(n) = self {
54            heights.might_exist(n as u64)
55        } else {
56            true
57        }
58    }
59}

61#[async_trait]
62impl<Types> Fetchable<Types> for LeafQueryData<Types>
63where
64    Types: NodeType,
65    Payload<Types>: QueryablePayload<Types>,
66{
67    type Request = LeafId<Types>;
68
69    fn satisfies(&self, req: Self::Request) -> bool {
70        match req {
71            LeafId::Number(n) => self.height() == n as u64,
72            LeafId::Hash(h) => self.hash() == h,
73        }
74    }
75
76    async fn passive_fetch(
77        notifiers: &Notifiers<Types>,
78        req: Self::Request,
79    ) -> BoxFuture<'static, Option<Self>> {
80        notifiers
81            .leaf
82            .wait_for(move |leaf| leaf.satisfies(req))
83            .await
84            .into_future()
85            .boxed()
86    }
87
88    async fn active_fetch<S, P>(
89        tx: &mut impl AvailabilityStorage<Types>,
90        fetcher: Arc<Fetcher<Types, S, P>>,
91        req: Self::Request,
92    ) -> anyhow::Result<()>
93    where
94        S: VersionedDataSource + 'static,
95        for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
96        for<'a> S::ReadOnly<'a>:
97            AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
98        P: AvailabilityProvider<Types>,
99    {
100        fetch_leaf_with_callbacks(tx, fetcher, req, None).await
101    }
102
103    async fn load<S>(storage: &mut S, req: Self::Request) -> QueryResult<Self>
104    where
105        S: AvailabilityStorage<Types>,
106    {
107        storage.get_leaf(req).await
108    }
109}

111pub(super) async fn fetch_leaf_with_callbacks<Types, S, P, I>(
112    tx: &mut impl AvailabilityStorage<Types>,
113    fetcher: Arc<Fetcher<Types, S, P>>,
114    req: LeafId<Types>,
115    callbacks: I,
116) -> anyhow::Result<()>
117where
118    Types: NodeType,
119    Payload<Types>: QueryablePayload<Types>,
120    S: VersionedDataSource + 'static,
121    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
122    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
123    P: AvailabilityProvider<Types>,
124    I: IntoIterator<Item = LeafCallback<Types, S, P>> + Send + 'static,
125    I::IntoIter: Send,
126{
127    match req {
128        LeafId::Number(n) => {
129            // We need the next leaf in the chain so we can figure out what hash we expect for this
130            // leaf, so we can fetch it securely from an untrusted provider.
131            let next = (n + 1) as u64;
132            let next = match tx.first_available_leaf(next).await {
133                Ok(leaf) if leaf.height() == next => leaf,
134                Ok(leaf) => {
135                    // If we don't have the immediate successor leaf, but we have some later leaf,
136                    // then we can't trigger this exact fetch, but we can fetch the (apparently)
137                    // missing parent of the leaf we do have, which will trigger a chain of fetches
138                    // that eventually reaches all the way back to the desired leaf.
139                    tracing::debug!(
140                        n,
141                        fetching = leaf.height() - 1,
142                        "do not have necessary leaf; trigger fetch of a later leaf"
143                    );
144
145                    let mut callbacks = vec![LeafCallback::Leaf {
146                        fetcher: fetcher.clone(),
147                    }];
148
149                    if !fetcher.leaf_only {
150                        callbacks.push(
151                            HeaderCallback::Payload {
152                                fetcher: fetcher.clone(),
153                            }
154                            .into(),
155                        );
156                        callbacks.push(
157                            HeaderCallback::VidCommon {
158                                fetcher: fetcher.clone(),
159                            }
160                            .into(),
161                        );
162                    }
163
164                    fetcher.leaf_fetcher.clone().spawn_fetch(
165                        request::LeafRequest::new(
166                            leaf.height() - 1,
167                            leaf.leaf().parent_commitment(),
168                            leaf.leaf().justify_qc().commit(),
169                        ),
170                        fetcher.provider.clone(),
171                        // After getting the leaf, grab the other data as well; that will be missing
172                        // whenever the leaf was.
173                        callbacks,
174                    );
175                    return Ok(());
176                },
177                Err(QueryError::Missing | QueryError::NotFound) => {
178                    // We successfully queried the database, but the next leaf wasn't there. We
179                    // know for sure that based on the current state of the DB, we cannot fetch this
180                    // leaf.
181                    tracing::debug!(n, "not fetching leaf with unknown successor");
182                    return Ok(());
183                },
184                Err(QueryError::Error { message }) => {
185                    // An error occurred while querying the database. We don't know if we need to
186                    // fetch the leaf or not. Return an error so we can try again.
187                    bail!("failed to fetch successor for leaf {n}: {message}");
188                },
189            };
190
191            let fetcher = fetcher.clone();
192            fetcher.leaf_fetcher.clone().spawn_fetch(
193                request::LeafRequest::new(
194                    n as u64,
195                    next.leaf().parent_commitment(),
196                    next.leaf().justify_qc().commit(),
197                ),
198                fetcher.provider.clone(),
199                once(LeafCallback::Leaf { fetcher }).chain(callbacks),
200            );
201        },
202        LeafId::Hash(h) => {
203            // We don't actively fetch leaves when requested by hash, because we have no way of
204            // knowing whether a leaf with such a hash actually exists, and we don't want to bother
205            // peers with requests for non-existent leaves.
206            tracing::debug!("not fetching unknown leaf {h}");
207        },
208    }
209
210    Ok(())
211}

213/// Trigger a fetch of the parent of the given `leaf`, if it is missing.
214///
215/// Leaves have a unique constraint among fetchable objects: we cannot fetch a given leaf at height
216/// `h` unless we have its child at height `h + 1`. This is because the child, through its
217/// `parent_commitment`, tells us what the hash of the parent should be, which lets us authenticate
218/// it when fetching from an untrusted provider. Thus, requests for leaf `h` might block if `h + 1`
219/// is not available. To ensure all these requests are eventually unblocked, and all leaves are
220/// eventually fetched, we call this function whenever we receive leaf `h + 1` to check if we need
221/// to then fetch leaf `h`.
222pub(super) fn trigger_fetch_for_parent<Types, S, P>(
223    fetcher: &Arc<Fetcher<Types, S, P>>,
224    leaf: &LeafQueryData<Types>,
225) where
226    Types: NodeType,
227    Payload<Types>: QueryablePayload<Types>,
228    S: VersionedDataSource + 'static,
229    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
230    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
231    P: AvailabilityProvider<Types>,
232{
233    let height = leaf.height();
234    let parent = leaf.leaf().parent_commitment();
235    let parent_qc = leaf.leaf().justify_qc().commit();
236
237    // Check that there is a parent to fetch.
238    if height == 0 {
239        return;
240    }
241
242    // Spawn an async task; we're triggering a fire-and-forget fetch of a leaf that might now be
243    // available; we don't need to block the caller on this.
244    let fetcher = fetcher.clone();
245    let span = tracing::info_span!("fetch parent leaf", height, %parent, %parent_qc);
246    spawn(
247        async move {
248            // Check if we already have the parent.
249            match fetcher.storage.read().await {
250                Ok(mut tx) => {
251                    // Don't bother fetching a pruned leaf.
252                    if let Ok(pruned_height) = tx.load_pruned_height().await {
253                        if pruned_height.is_some_and(|ph| height <= ph) {
254                            tracing::info!(
255                                height,
256                                ?pruned_height,
257                                "not fetching pruned parent leaf"
258                            );
259                            return;
260                        }
261                    }
262
263                    if tx.get_leaf(((height - 1) as usize).into()).await.is_ok() {
264                        return;
265                    }
266                },
267                Err(err) => {
268                    // If we can't open a transaction, we can't be sure that we already have the
269                    // parent, so we fall through to fetching it just to be safe.
270                    tracing::warn!(
271                        height,
272                        %parent,
273                        "error opening transaction to check for parent leaf: {err:#}",
274                    );
275                },
276            }
277
278            tracing::info!(height, %parent, "received new leaf; fetching missing parent");
279            fetcher.leaf_fetcher.clone().spawn_fetch(
280                request::LeafRequest::new(height - 1, parent, parent_qc),
281                fetcher.provider.clone(),
282                // After getting the leaf, grab the other data as well; that will be missing
283                // whenever the leaf was.
284                [
285                    LeafCallback::Leaf {
286                        fetcher: fetcher.clone(),
287                    },
288                    HeaderCallback::Payload {
289                        fetcher: fetcher.clone(),
290                    }
291                    .into(),
292                    HeaderCallback::VidCommon {
293                        fetcher: fetcher.clone(),
294                    }
295                    .into(),
296                ],
297            );
298        }
299        .instrument(span),
300    );
301}

341#[derive(Derivative, From)]
342#[derivative(Debug(bound = ""))]
343pub(super) enum LeafCallback<Types: NodeType, S, P> {
344    /// Callback when fetching the leaf for its own sake.
345    #[from(ignore)]
346    Leaf {
347        #[derivative(Debug = "ignore")]
348        fetcher: Arc<Fetcher<Types, S, P>>,
349    },
350    /// Callback when fetching the leaf in order to then look up something else.
351    Continuation {
352        callback: HeaderCallback<Types, S, P>,
353    },
354}
355
356impl<Types: NodeType, S, P> PartialEq for LeafCallback<Types, S, P> {
357    fn eq(&self, other: &Self) -> bool {
358        self.cmp(other).is_eq()
359    }
360}
361
362impl<Types: NodeType, S, P> Eq for LeafCallback<Types, S, P> {}
363
364impl<Types: NodeType, S, P> Ord for LeafCallback<Types, S, P> {
365    fn cmp(&self, other: &Self) -> Ordering {
366        match (self, other) {
367            // Store leaves in the database before storing derived objects.
368            (Self::Leaf { .. }, Self::Continuation { .. }) => Ordering::Less,
369            (Self::Continuation { .. }, Self::Leaf { .. }) => Ordering::Greater,
370
371            (Self::Continuation { callback: cb1 }, Self::Continuation { callback: cb2 }) => {
372                cb1.cmp(cb2)
373            },
374            _ => Ordering::Equal,
375        }
376    }
377}
378
379impl<Types: NodeType, S, P> PartialOrd for LeafCallback<Types, S, P> {
380    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
381        Some(self.cmp(other))
382    }
383}

385impl<Types: NodeType, S, P> Callback<LeafQueryData<Types>> for LeafCallback<Types, S, P>
386where
387    Payload<Types>: QueryablePayload<Types>,
388    S: VersionedDataSource + 'static,
389    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
390    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
391    P: AvailabilityProvider<Types>,
392{
393    async fn run(self, leaf: LeafQueryData<Types>) {
394        match self {
395            Self::Leaf { fetcher } => {
396                tracing::info!("fetched leaf {}", leaf.height());
397                // Trigger a fetch of the parent leaf, if we don't already have it.
398                trigger_fetch_for_parent(&fetcher, &leaf);
399                fetcher.store_and_notify(leaf).await;
400            },
401            Self::Continuation { callback } => callback.run(leaf.leaf.block_header().clone()),
402        }
403    }
