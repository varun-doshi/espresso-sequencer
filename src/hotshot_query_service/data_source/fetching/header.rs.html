41impl<Types: NodeType> From<LeafQueryData<Types>> for HeaderQueryData<Types> {
42    fn from(leaf: LeafQueryData<Types>) -> Self {
43        let header = leaf.header().clone();
44
45        Self { header }
46    }
47}
48
49fn satisfies_header_req_from_leaf<Types>(leaf: &LeafQueryData<Types>, req: BlockId<Types>) -> bool
50where
51    Types: NodeType,
52    Payload<Types>: QueryablePayload<Types>,
53{
54    HeaderQueryData::satisfies(&HeaderQueryData::new(leaf.header().clone()), req)
55}
56
57#[async_trait]
58impl<Types> Fetchable<Types> for HeaderQueryData<Types>
59where
60    Types: NodeType,
61    Payload<Types>: QueryablePayload<Types>,
62{
63    type Request = BlockId<Types>;
64
65    fn satisfies(&self, req: Self::Request) -> bool {
66        let header = self.header();
67        match req {
68            BlockId::Number(n) => header.block_number() as usize == n,
69            BlockId::Hash(h) => header.commit() == h,
70            BlockId::PayloadHash(h) => header.payload_commitment() == h,
71        }
72    }
73
74    async fn passive_fetch(
75        notifiers: &Notifiers<Types>,
76        req: Self::Request,
77    ) -> BoxFuture<'static, Option<Self>> {
78        notifiers
79            .leaf
80            .wait_for(move |leaf| satisfies_header_req_from_leaf(leaf, req))
81            .await
82            .into_future()
83            .map(|leaf| leaf.map(HeaderQueryData::from))
84            .boxed()
85    }
86
87    async fn active_fetch<S, P>(
88        tx: &mut impl AvailabilityStorage<Types>,
89        fetcher: Arc<Fetcher<Types, S, P>>,
90        req: Self::Request,
91    ) -> anyhow::Result<()>
92    where
93        S: VersionedDataSource + 'static,
94        for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
95        for<'a> S::ReadOnly<'a>:
96            AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
97        P: AvailabilityProvider<Types>,
98    {
99        // Note: if leaf only mode is enabled
100        // the payload callback will not do any active fetching and just return
101        // This is because we don't have payload fetcher for leaf only mode
102        fetch_header_and_then(
103            tx,
104            req,
105            HeaderCallback::Payload {
106                fetcher: fetcher.clone(),
107            },
108        )
109        .await
110    }
111
112    async fn load<S>(storage: &mut S, req: Self::Request) -> QueryResult<Self>
113    where
114        S: AvailabilityStorage<Types>,
115    {
116        storage.get_header(req).await.map(|header| Self { header })
117    }
118}

120#[derive(Derivative)]
121#[derivative(Debug(bound = ""))]
122pub(super) enum HeaderCallback<Types, S, P>
123where
124    Types: NodeType,
125{
126    /// Callback when fetching the leaf in order to then look up the corresponding block.
127    Payload {
128        #[derivative(Debug = "ignore")]
129        fetcher: Arc<Fetcher<Types, S, P>>,
130    },
131    /// Callback when fetching the leaf in order to then look up the corresponding VID common data.
132    VidCommon {
133        #[derivative(Debug = "ignore")]
134        fetcher: Arc<Fetcher<Types, S, P>>,
135    },
136}
137
138impl<Types: NodeType, S, P> PartialEq for HeaderCallback<Types, S, P> {
139    fn eq(&self, other: &Self) -> bool {
140        self.cmp(other).is_eq()
141    }
142}
143
144impl<Types: NodeType, S, P> Eq for HeaderCallback<Types, S, P> {}
145
146impl<Types: NodeType, S, P> PartialOrd for HeaderCallback<Types, S, P> {
147    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
148        Some(self.cmp(other))
149    }
150}
151
152impl<Types: NodeType, S, P> Ord for HeaderCallback<Types, S, P> {
153    fn cmp(&self, other: &Self) -> Ordering {
154        // Arbitrarily, we choose to run payload callbacks before VID callbacks.
155        match (self, other) {
156            (Self::Payload { .. }, Self::VidCommon { .. }) => Ordering::Less,
157            (Self::VidCommon { .. }, Self::Payload { .. }) => Ordering::Greater,
158            _ => Ordering::Equal,
159        }
160    }
161}
162
163impl<Types, S, P> HeaderCallback<Types, S, P>
164where
165    Types: NodeType,
166    Payload<Types>: QueryablePayload<Types>,
167    S: VersionedDataSource + 'static,
168    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
169    P: AvailabilityProvider<Types>,
170{
171    fn fetcher(&self) -> Arc<Fetcher<Types, S, P>> {
172        match self {
173            Self::Payload { fetcher } => fetcher.clone(),
174            Self::VidCommon { fetcher } => fetcher.clone(),
175        }
176    }
177
178    pub(super) fn run(self, header: Header<Types>) {
179        match self {
180            Self::Payload { fetcher } => {
181                tracing::info!(
182                    "fetched leaf {}, will now fetch payload",
183                    header.block_number()
184                );
185                fetch_block_with_header(fetcher, header);
186            },
187            Self::VidCommon { fetcher } => {
188                tracing::info!(
189                    "fetched leaf {}, will now fetch VID common",
190                    header.block_number()
191                );
192                fetch_vid_common_with_header(fetcher, header);
193            },
194        }
195    }
196}

198pub(super) async fn fetch_header_and_then<Types, S, P>(
199    tx: &mut impl AvailabilityStorage<Types>,
200    req: BlockId<Types>,
201    callback: HeaderCallback<Types, S, P>,
202) -> anyhow::Result<()>
203where
204    Types: NodeType,
205    Payload<Types>: QueryablePayload<Types>,
206    S: VersionedDataSource + 'static,
207    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
208    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
209    P: AvailabilityProvider<Types>,
210{
211    // Check if at least the header is available in local storage. If it is, we benefit two ways:
212    // 1. We know for sure the corresponding block exists, so we can unconditionally trigger an
213    //    active fetch without unnecessarily bothering our peers.
214    // 2. We only need to fetch the missing data (e.g. payload or VID common), not the full block.
215    //    Not only is this marginally less data to download, there are some providers that may only
216    //    be able to provide certain data. For example, the HotShot DA committee members may be able
217    //    to provide paylaods, but not full blocks. Or, in the case where VID recovery is needed,
218    //    the VID common data may be available but the full block may not exist anywhere.
219    match tx.get_header(req).await {
220        Ok(header) => {
221            callback.run(header);
222            return Ok(());
223        },
224        Err(QueryError::Missing | QueryError::NotFound) => {
225            // We successfully queried the database, but the header wasn't there. Fall through to
226            // fetching it.
227            tracing::debug!(?req, "header not available locally; trying fetch");
228        },
229        Err(QueryError::Error { message }) => {
230            // An error occurred while querying the database. We don't know if we need to fetch the
231            // header or not. Return an error so we can try again.
232            bail!("failed to fetch header for block {req:?}: {message}");
233        },
234    }
235
236    // If the header is _not_ present, we may still be able to fetch the request, but we need to
237    // fetch the header (in fact, the entire leaf) first. This is because we have an invariant that
238    // we should not store derived objects in the database unless we already have the corresponding
239    // header and leaf.
240    match req {
241        BlockId::Number(n) => {
242            fetch_leaf_with_callbacks(tx, callback.fetcher(), n.into(), [callback.into()]).await?;
243        },
244        BlockId::Hash(h) => {
245            // Given only the hash, we cannot tell if the corresponding leaf actually exists, since
246            // we don't have a corresponding header. Therefore, we will not spawn an active fetch.
247            tracing::debug!("not fetching unknown block {h}");
248        },
249        BlockId::PayloadHash(h) => {
250            // Same as above, we don't fetch a block with a payload that is not known to exist.
251            tracing::debug!("not fetching block with unknown payload {h}");
252        },
253    }
254
255    Ok(())
