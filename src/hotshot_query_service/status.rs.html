13//! Queries for node-specific state and uncommitted data.
14//!
15//! Unlike the [availability](crate::availability) and [node](crate::node) APIs, which deal only
16//! with committed data (albeit with different consistency properties), the status API offers a
17//! glimpse into internal consensus state and uncommitted data. Here you can find low-level
18//! information about a particular node, such as consensus and networking metrics.
19//!
20//! The status API is intended to be a lightweight way to inspect the activities and health of a
21//! consensus node. It is the only API that can be run without any persistent storage, and its
22//! memory overhead is also very low. As a consequence, it only serves two types of data:
23//! * snapshots of the state right now, with no way to query historical snapshots
24//! * summary statistics
25

41#[derive(Default)]
42pub struct Options {
43    pub api_path: Option<PathBuf>,
44
45    /// Additional API specification files to merge with `status-api-path`.
46    ///
47    /// These optional files may contain route definitions for application-specific routes that have
48    /// been added as extensions to the basic status API.
49    pub extensions: Vec<toml::Value>,
50}
51
52#[derive(Clone, Debug, From, Snafu, Deserialize, Serialize)]
53pub enum Error {
54    Request { source: RequestError },
55    Internal { reason: String },
56}
57
58impl Error {
59    pub fn status(&self) -> StatusCode {
60        match self {
61            Self::Request { .. } => StatusCode::BAD_REQUEST,
62            Self::Internal { .. } => StatusCode::INTERNAL_SERVER_ERROR,
63        }
64    }
65}
66
67fn internal<M: Display>(msg: M) -> Error {
68    Error::Internal {
69        reason: msg.to_string(),
70    }
71}
72
73pub fn define_api<State, Ver: StaticVersionType + 'static>(
74    options: &Options,
75    _: Ver,
76) -> Result<Api<State, Error, Ver>, ApiError>
77where
78    State: 'static + Send + Sync + ReadState,
79    <State as ReadState>::State: Send + Sync + StatusDataSource,
80{
81    let mut api = load_api::<State, Error, Ver>(
82        options.api_path.as_ref(),
83        include_str!("../api/status.toml"),
84        options.extensions.clone(),
85    )?;
86    api.with_version("0.0.1".parse().unwrap())
87        .get("block_height", |_, state| {
88            async { state.block_height().await.map_err(internal) }.boxed()
89        })?
90        .get("success_rate", |_, state| {
91            async { state.success_rate().await.map_err(internal) }.boxed()
92        })?
93        .get("time_since_last_decide", |_, state| {
94            async {
95                state
96                    .elapsed_time_since_last_decide()
97                    .await
98                    .map_err(internal)
99            }
100            .boxed()
101        })?
102        .metrics("metrics", |_, state| {
103            async { Ok(Cow::Borrowed(state.metrics())) }.boxed()
104        })?;
105    Ok(api)
106}

133    #[tokio::test(flavor = "multi_thread")]
134    async fn test_api() {
135        setup_test();
136
137        // Create the consensus network.
138        let mut network = MockNetwork::<MockDataSource, MockVersions>::init().await;
139
140        // Start the web server.
141        let port = pick_unused_port().unwrap();
142        let mut app = App::<_, Error>::with_state(ApiState::from(network.data_source()));
143        app.register_module(
144            "status",
145            define_api(&Default::default(), MockBase::instance()).unwrap(),
146        )
147        .unwrap();
148        network.spawn(
149            "server",
150            app.serve(format!("0.0.0.0:{}", port), MockBase::instance()),
151        );
152
153        // Start a client.
154        let url = Url::from_str(&format!("http://localhost:{}/status", port)).unwrap();
155        let client = Client::<Error, MockBase>::new(url.clone());
156        assert!(client.connect(Some(Duration::from_secs(60))).await);
157
158        // The block height is initially zero.
159        assert_eq!(client.get::<u64>("block-height").send().await.unwrap(), 0);
160
161        // Test Prometheus export.
162        // Create `reqwest` client that allows redirects
163        let reqwest_client = reqwest::Client::builder()
164            .redirect(Policy::limited(5))
165            .build()
166            .unwrap();
167
168        // Ask for the Prometheus data
169        let res = reqwest_client
170            .get(format!("{url}/metrics"))
171            .send()
172            .await
173            .unwrap();
174
175        // Make sure it has the correct response code
176        assert_eq!(res.status(), StatusCode::OK);
177        let prometheus = res.text().await.unwrap();
178        let lines = prometheus.lines().collect::<Vec<_>>();
179        assert!(
180            lines.contains(&"consensus_current_view 0"),
181            "Missing consensus_current_view in metrics:\n{}",
182            prometheus
183        );
184
185        // Start the validators and wait for the block to be finalized.
186        network.start().await;
187
188        // Check updated block height.
189        // being updated and the decide event being published. Retry this a few times until it
190        // succeeds.
191        while client.get::<u64>("block-height").send().await.unwrap() <= 1 {
192            tracing::info!("waiting for block height to update");
193            sleep(Duration::from_secs(1)).await;
194        }
195        let success_rate = client.get::<f64>("success-rate").send().await.unwrap();
196        // If metrics are populating correctly, we should get a finite number. If not, we might get
197        // NaN or infinity due to division by 0.
198        assert!(success_rate.is_finite(), "{success_rate}");
199        // We know at least some views have been successful, since we finalized a block.
200        assert!(success_rate > 0.0, "{success_rate}");
201
202        network.shut_down().await;
203    }

205    #[tokio::test(flavor = "multi_thread")]
206    async fn test_extensions() {
207        setup_test();
208
209        let dir = TempDir::with_prefix("test_status_extensions").unwrap();
210        let data_source = ExtensibleDataSource::new(
211            MockDataSource::create(dir.path(), Default::default())
212                .await
213                .unwrap(),
214            0,
215        );
216
217        let extensions = toml! {
218            [route.post_ext]
219            PATH = ["/ext/:val"]
220            METHOD = "POST"
221            ":val" = "Integer"
222
223            [route.get_ext]
224            PATH = ["/ext"]
225            METHOD = "GET"
226        };
227
228        let mut api = define_api::<RwLock<ExtensibleDataSource<MockDataSource, u64>>, MockBase>(
229            &Options {
230                extensions: vec![extensions.into()],
231                ..Default::default()
232            },
233            MockBase::instance(),
234        )
235        .unwrap();
236        api.get("get_ext", |_, state| {
237            async move { Ok(*state.as_ref()) }.boxed()
238        })
239        .unwrap()
240        .post("post_ext", |req, state| {
241            async move {
242                *state.as_mut() = req.integer_param("val")?;
243                Ok(())
244            }
245            .boxed()
246        })
247        .unwrap();
248
249        let mut app = App::<_, Error>::with_state(RwLock::new(data_source));
250        app.register_module("status", api).unwrap();
251
252        let port = pick_unused_port().unwrap();
253        let _server = BackgroundTask::spawn(
254            "server",
255            app.serve(format!("0.0.0.0:{}", port), MockBase::instance()),
256        );
257
258        let client = Client::<Error, MockBase>::new(
259            format!("http://localhost:{}/status", port).parse().unwrap(),
260        );
261        assert!(client.connect(Some(Duration::from_secs(60))).await);
262
263        assert_eq!(client.get::<u64>("ext").send().await.unwrap(), 0);
264        client.post::<()>("ext/42").send().await.unwrap();
265        assert_eq!(client.get::<u64>("ext").send().await.unwrap(), 42);
266
267        // Ensure we can still access the built-in functionality.
268        assert_eq!(client.get::<u64>("block-height").send().await.unwrap(), 0);
269    }
