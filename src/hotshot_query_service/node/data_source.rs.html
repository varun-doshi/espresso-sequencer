13//! Data for the [`node`](super) API.
14//!
15//! This module is just an alternative view of the same data provided by the
16//! [`availability`](crate::availability) API. It provides more insight into what data the node
17//! actually has at present, as opposed to trying to present a perfect view of an abstract chain,
18//! fetching data from other sources as needed. It is also more liberal with provided aggregate
19//! counts and statistics which may be inaccurate if data is missing.
20//!
21//! Due to this relationship with the availability module, this module has its own [data source
22//! trait](`NodeDataSource`) but not its own update trait. The node data source is expected to read
23//! its data from the same underlying database as the availability API, and as such the data is
24//! updated implicitly via the [availability API update
25//! trait](crate::availability::UpdateAvailabilityData).
26
27use std::ops::RangeBounds;
28
29use async_trait::async_trait;
30use derivative::Derivative;
31use derive_more::From;
32use hotshot_types::{data::VidShare, traits::node_implementation::NodeType};
33
34use super::query_data::{BlockHash, BlockId, SyncStatus, TimeWindowQueryData};
35use crate::{Header, QueryResult};
36
37#[derive(Derivative, From)]
38#[derivative(Copy(bound = ""), Debug(bound = ""))]
39pub enum WindowStart<Types: NodeType> {
40    #[from(ignore)]
41    Time(u64),
42    #[from(ignore)]
43    Height(u64),
44    Hash(BlockHash<Types>),
45}
46
47impl<Types: NodeType> Clone for WindowStart<Types> {
48    fn clone(&self) -> Self {
49        *self
50    }
51}
52
53#[async_trait]
54pub trait NodeDataSource<Types: NodeType> {
55    async fn block_height(&self) -> QueryResult<usize>;
56    async fn count_transactions_in_range(
57        &self,
58        range: impl RangeBounds<usize> + Send,
59    ) -> QueryResult<usize>;
60    async fn payload_size_in_range(
61        &self,
62        range: impl RangeBounds<usize> + Send,
63    ) -> QueryResult<usize>;
64    async fn vid_share<ID>(&self, id: ID) -> QueryResult<VidShare>
65    where
66        ID: Into<BlockId<Types>> + Send + Sync;
67    async fn get_header_window(
68        &self,
69        start: impl Into<WindowStart<Types>> + Send + Sync,
70        end: u64,
71        limit: usize,
72    ) -> QueryResult<TimeWindowQueryData<Header<Types>>>;
73
74    /// Search the database for missing objects and generate a report.
75    async fn sync_status(&self) -> QueryResult<SyncStatus>;
76
77    async fn count_transactions(&self) -> QueryResult<usize> {
78        self.count_transactions_in_range(0..).await
79    }
80
81    async fn payload_size(&self) -> QueryResult<usize> {
82        self.payload_size_in_range(0..).await
83    }
84}
