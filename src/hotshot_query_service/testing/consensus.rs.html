84impl<D: DataSourceLifeCycle + UpdateStatusData, V: Versions> MockNetwork<D, V> {
85    pub async fn init() -> Self {
86        Self::init_with_config(|_| {}, false).await
87    }

90        Self::init_with_config(|_| {}, true).await

93    pub async fn init_with_config(
94        update_config: impl FnOnce(&mut HotShotConfig<MockTypes>),
95        leaf_only: bool,
96    ) -> Self {
97        let (pub_keys, priv_keys): (Vec<_>, Vec<_>) = (0..NUM_NODES)
98            .map(|i| BLSPubKey::generated_from_seed_indexed([0; 32], i as u64))
99            .unzip();
100        let num_staked_nodes = NonZeroUsize::new(pub_keys.len()).unwrap();
101        let state_key_pairs = (0..num_staked_nodes.into())
102            .map(|i| StateKeyPair::generate_from_seed_indexed([0; 32], i as u64))
103            .collect::<Vec<_>>();
104        let master_map = MasterMap::new();
105        let stake = 1u64;
106        let known_nodes_with_stake = (0..num_staked_nodes.into())
107            .map(|id| PeerConfig {
108                stake_table_entry: pub_keys[id].stake_table_entry(U256::from(stake)),
109                state_ver_key: state_key_pairs[id].ver_key(),
110            })
111            .collect::<Vec<_>>();
112
113        let membership = Arc::new(RwLock::new(MockMembership::new(
114            known_nodes_with_stake.clone(),
115            known_nodes_with_stake.clone(),
116        )));
117
118        // Pick a random, unused port for the builder server
119        let builder_port = portpicker::pick_unused_port().expect("No ports available");
120
121        // Create the bind URL from the random port
122        let builder_url =
123            Url::parse(&format!("http://0.0.0.0:{builder_port}")).expect("Failed to parse URL");
124
125        // Start the builder server
126        let builder_task =
127            <SimpleBuilderImplementation as TestBuilderImplementation<MockTypes>>::start(
128                NUM_NODES,
129                builder_url.clone(),
130                (),
131                Default::default(),
132            )
133            .await;
134
135        let mut config = HotShotConfig {
136            builder_urls: vec1::vec1![builder_url.clone()],
137            fixed_leader_for_gpuvid: 0,
138            num_nodes_with_stake: num_staked_nodes,
139            known_nodes_with_stake: known_nodes_with_stake.clone(),
140            next_view_timeout: 10000,
141            num_bootstrap: 0,
142            da_staked_committee_size: pub_keys.len(),
143            known_da_nodes: known_nodes_with_stake.clone(),
144            data_request_delay: Duration::from_millis(200),
145            view_sync_timeout: Duration::from_millis(250),
146            start_threshold: (
147                known_nodes_with_stake.len() as u64,
148                known_nodes_with_stake.len() as u64,
149            ),
150            builder_timeout: Duration::from_secs(1),
151            start_proposing_view: 0,
152            stop_proposing_view: 0,
153            start_voting_view: 0,
154            stop_voting_view: 0,
155            start_proposing_time: 0,
156            stop_proposing_time: 0,
157            start_voting_time: 0,
158            stop_voting_time: 0,
159            epoch_height: 10,
160            epoch_start_block: 0,
161        };
162        update_config(&mut config);
163
164        let nodes = join_all(
165            priv_keys
166                .into_iter()
167                .enumerate()
168                .map(|(node_id, priv_key)| {
169                    let membership = membership.clone();
170                    let config = config.clone();
171
172                    let pub_keys = pub_keys.clone();
173                    let master_map = master_map.clone();
174                    let state_priv_keys = state_key_pairs
175                        .iter()
176                        .map(|kp| kp.sign_key())
177                        .collect::<Vec<_>>();
178
179                    let span = info_span!("initialize node", node_id);
180                    async move {
181                        let storage = D::create(node_id).await;
182                        let data_source = if leaf_only {
183                            D::leaf_only_ds(&storage).await
184                        } else {
185                            D::connect(&storage).await
186                        };
187
188                        let network = Arc::new(MemoryNetwork::new(
189                            &pub_keys[node_id],
190                            &master_map.clone(),
191                            &[Topic::Global, Topic::Da],
192                            None,
193                        ));
194
195                        let hs_storage: TestStorage<MockTypes> = TestStorage::default();
196                        membership
197                            .write()
198                            .await
199                            .set_first_epoch(ViewNumber::new(0), INITIAL_DRB_RESULT);
200                        let memberships =
201                            EpochMembershipCoordinator::new(membership, config.epoch_height);
202
203                        let hotshot = SystemContext::init(
204                            pub_keys[node_id],
205                            priv_key,
206                            state_priv_keys[node_id].clone(),
207                            node_id as u64,
208                            config,
209                            memberships,
210                            network,
211                            HotShotInitializer::from_genesis::<V>(
212                                TestInstanceState::default(),
213                                0,
214                                0,
215                                vec![],
216                            )
217                            .await
218                            .unwrap(),
219                            ConsensusMetricsValue::new(&*data_source.populate_metrics()),
220                            hs_storage,
221                            MarketplaceConfig {
222                                auction_results_provider: Arc::new(
223                                    TestAuctionResultsProvider::default(),
224                                ),
225                                fallback_builder_url: Url::from_str("https://some.url").unwrap(),
226                            },
227                        )
228                        .await
229                        .unwrap()
230                        .0;
231
232                        MockNode {
233                            hotshot,
234                            data_source,
235                            storage,
236                        }
237                    }
238                    .instrument(span)
239                }),
240        )
241        .await;
242
243        // Hook the builder up to the event stream from the first node
244
245        builder_task.start(Box::new(nodes[0].hotshot.event_stream()));
246
247        let mut network = Self {
248            nodes,
249            pub_keys,
250            tasks: Default::default(),
251        };
252        D::setup(&mut network).await;
253        network
254    }
255}

257impl<D: DataSourceLifeCycle, V: Versions> MockNetwork<D, V> {
258    pub fn handle(&self) -> &SystemContextHandle<MockTypes, MockNodeImpl, V> {
259        &self.nodes[0].hotshot
260    }
261
262    pub async fn submit_transaction(&self, tx: MockTransaction) {
263        self.handle().submit_transaction(tx).await.unwrap();
264    }
265
266    pub fn num_nodes(&self) -> usize {
267        self.pub_keys.len()
268    }
269
270    pub fn proposer(&self, i: usize) -> SignatureKey<MockTypes> {
271        self.pub_keys[i]
272    }
273
274    pub fn data_source_index(&self, i: usize) -> D {
275        self.nodes[i].data_source.clone()
276    }
277
278    pub fn data_source(&self) -> D {
279        self.data_source_index(0)
280    }
281
282    pub fn storage(&self) -> &D::Storage {
283        &self.nodes[0].storage
284    }
285
286    pub fn spawn(&mut self, name: impl Display, task: impl Future + Send + 'static) {
287        self.tasks.push(BackgroundTask::spawn(name, task));
288    }
289
290    pub async fn shut_down(mut self) {
291        self.shut_down_impl().await
292    }
293
294    async fn shut_down_impl(&mut self) {
295        for node in &mut self.nodes {
296            node.hotshot.shut_down().await;
297        }
298    }
299}

301impl<D: DataSourceLifeCycle, V: Versions> MockNetwork<D, V> {
302    pub async fn start(&mut self) {
303        // Spawn the update tasks.
304        for (i, node) in self.nodes.iter_mut().enumerate() {
305            let ds = node.data_source.clone();
306            let mut events = node.hotshot.event_stream();
307            self.tasks.push(BackgroundTask::spawn(
308                format!("update node {i}"),
309                async move {
310                    while let Some(event) = events.next().await {
311                        tracing::info!(node = i, event = ?event.event, "EVENT");
312                        {
313                            ds.handle_event(&event).await;
314                        }
315                        yield_now().await;
316                    }
317                },
318            ));
319        }
320
321        join_all(
322            self.nodes
323                .iter()
324                .map(|node| node.hotshot.hotshot.start_consensus()),
325        )
326        .await;
327    }
328}

330impl<D: DataSourceLifeCycle, V: Versions> Drop for MockNetwork<D, V> {
331    fn drop(&mut self) {
332        if let Ok(handle) = Handle::try_current() {
333            block_in_place(move || handle.block_on(self.shut_down_impl()));
334        }
335    }
336}

338#[async_trait]
339pub trait DataSourceLifeCycle: Clone + Send + Sync + Sized + 'static {
340    /// Backing storage for the data source.
341    ///
342    /// This can be used to connect to data sources to the same underlying data. It must be kept
343    /// alive as long as the related data sources are open.
344    type Storage: Send + Sync;
345
346    async fn create(node_id: usize) -> Self::Storage;
347    async fn connect(storage: &Self::Storage) -> Self;
348    async fn reset(storage: &Self::Storage) -> Self;
349    async fn handle_event(&self, event: &Event<MockTypes>);
350    async fn leaf_only_ds(_storage: &Self::Storage) -> Self {
351        panic!("not supported")
352    }
353
354    /// Setup runs after setting up the network but before starting a test.
355    async fn setup<V: Versions>(_network: &mut MockNetwork<Self, V>) {}
356}
