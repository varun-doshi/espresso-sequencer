41/// Event streaming handle for a [`SystemContext`] instance running in the background
42///
43/// This type provides the means to message and interact with a background [`SystemContext`] instance,
44/// allowing the ability to receive [`Event`]s from it, send transactions to it, and interact with
45/// the underlying storage.
46pub struct SystemContextHandle<TYPES: NodeType, I: NodeImplementation<TYPES>, V: Versions> {
47    /// The [sender](Sender) and [receiver](Receiver),
48    /// to allow the application to communicate with HotShot.
49    pub(crate) output_event_stream: (Sender<Event<TYPES>>, InactiveReceiver<Event<TYPES>>),
50
51    /// access to the internal event stream, in case we need to, say, shut something down
52    #[allow(clippy::type_complexity)]
53    pub(crate) internal_event_stream: (
54        Sender<Arc<HotShotEvent<TYPES>>>,
55        InactiveReceiver<Arc<HotShotEvent<TYPES>>>,
56    ),
57    /// registry for controlling consensus tasks
58    pub(crate) consensus_registry: ConsensusTaskRegistry<HotShotEvent<TYPES>>,
59
60    /// registry for controlling network tasks
61    pub(crate) network_registry: NetworkTaskRegistry,
62
63    /// Internal reference to the underlying [`SystemContext`]
64    pub hotshot: Arc<SystemContext<TYPES, I, V>>,
65
66    /// Reference to the internal storage for consensus datum.
67    pub(crate) storage: Arc<RwLock<I::Storage>>,
68
69    /// Networks used by the instance of hotshot
70    pub network: Arc<I::Network>,
71
72    /// Memberships used by consensus
73    pub membership_coordinator: EpochMembershipCoordinator<TYPES>,
74
75    /// Number of blocks in an epoch, zero means there are no epochs
76    pub epoch_height: u64,
77}

82    /// Adds a hotshot consensus-related task to the `SystemContextHandle`.
83    pub fn add_task<S: TaskState<Event = HotShotEvent<TYPES>> + 'static>(&mut self, task_state: S) {
84        let task = Task::new(
85            task_state,
86            self.internal_event_stream.0.clone(),
87            self.internal_event_stream.1.activate_cloned(),
88        );
89
90        self.consensus_registry.run_task(task);
91    }

94    pub fn event_stream(&self) -> impl Stream<Item = Event<TYPES>> {
95        self.output_event_stream.1.activate_cloned()
96    }

98    /// Message other participants with a serialized message from the application
99    /// Receivers of this message will get an `Event::ExternalMessageReceived` via
100    /// the event stream.
101    ///
102    /// # Errors
103    /// Errors if serializing the request fails, or the request fails to be sent
104    pub async fn send_external_message(
105        &self,
106        msg: Vec<u8>,
107        recipients: RecipientList<TYPES::SignatureKey>,
108    ) -> Result<()> {
109        let message = Message {
110            sender: self.public_key().clone(),
111            kind: MessageKind::External(msg),
112        };
113        let serialized_message = self.hotshot.upgrade_lock.serialize(&message).await?;
114
115        match recipients {
116            RecipientList::Broadcast => {
117                self.network
118                    .broadcast_message(serialized_message, Topic::Global, BroadcastDelay::None)
119                    .await?;
120            },
121            RecipientList::Direct(recipient) => {
122                self.network
123                    .direct_message(serialized_message, recipient)
124                    .await?;
125            },
126            RecipientList::Many(recipients) => {
127                self.network
128                    .da_broadcast_message(serialized_message, recipients, BroadcastDelay::None)
129                    .await?;
130            },
131        }
132        Ok(())
133    }

135    /// Request a proposal from the all other nodes.  Will block until some node
136    /// returns a valid proposal with the requested commitment.  If nobody has the
137    /// proposal this will block forever
138    ///
139    /// # Errors
140    /// Errors if signing the request for proposal fails
141    pub fn request_proposal(
142        &self,
143        view: TYPES::View,
144        leaf_commitment: Commitment<Leaf2<TYPES>>,
145    ) -> Result<impl futures::Future<Output = Result<Proposal<TYPES, QuorumProposalWrapper<TYPES>>>>>
146    {
147        // We need to be able to sign this request before submitting it to the network. Compute the
148        // payload first.
149        let signed_proposal_request = ProposalRequestPayload {
150            view_number: view,
151            key: self.public_key().clone(),
152        };
153
154        // Finally, compute the signature for the payload.
155        let signature = TYPES::SignatureKey::sign(
156            self.private_key(),
157            signed_proposal_request.commit().as_ref(),
158        )?;
159
160        let membership_coordinator = self.membership_coordinator.clone();
161        let receiver = self.internal_event_stream.1.activate_cloned();
162        let sender = self.internal_event_stream.0.clone();
163        let epoch_height = self.epoch_height;
164        Ok(async move {
165            // First, broadcast that we need a proposal
166            broadcast_event(
167                HotShotEvent::QuorumProposalRequestSend(signed_proposal_request, signature).into(),
168                &sender,
169            )
170            .await;
171            loop {
172                let hs_event = EventDependency::new(
173                    receiver.clone(),
174                    Box::new(move |event| {
175                        let event = event.as_ref();
176                        if let HotShotEvent::QuorumProposalResponseRecv(quorum_proposal) = event {
177                            quorum_proposal.data.view_number() == view
178                        } else {
179                            false
180                        }
181                    }),
182                )
183                .completed()
184                .await
185                .ok_or(anyhow!("Event dependency failed to get event"))?;
186
187                // Then, if it's `Some`, make sure that the data is correct
188                if let HotShotEvent::QuorumProposalResponseRecv(quorum_proposal) = hs_event.as_ref()
189                {
190                    let maybe_epoch = option_epoch_from_block_number::<TYPES>(
191                        quorum_proposal.data.proposal.epoch.is_some(),
192                        quorum_proposal.data.block_header().block_number(),
193                        epoch_height,
194                    );
195                    let membership = match membership_coordinator
196                        .membership_for_epoch(maybe_epoch)
197                        .await
198                    {
199                        Result::Ok(m) => m,
200                        Err(e) => {
201                            tracing::warn!(e.message);
202                            continue;
203                        },
204                    };
205                    // Make sure that the quorum_proposal is valid
206                    if let Err(err) = quorum_proposal.validate_signature(&membership).await {
207                        tracing::warn!("Invalid Proposal Received after Request.  Err {:?}", err);
208                        continue;
209                    }
210                    let proposed_leaf = Leaf2::from_quorum_proposal(&quorum_proposal.data);
211                    let commit = proposed_leaf.commit();
212                    if commit == leaf_commitment {
213                        return Ok(quorum_proposal.clone());
214                    }
215                    tracing::warn!("Proposal received from request has different commitment than expected.\nExpected = {:?}\nReceived{:?}", leaf_commitment, commit);
216                }
217            }
218        })
219    }

305    /// Shut down the inner hotshot and wait until all background threads are closed.
306    pub async fn shut_down(&mut self) {
307        // this is required because `SystemContextHandle` holds an inactive receiver and
308        // `broadcast_direct` below can wait indefinitely
309        self.internal_event_stream.0.set_await_active(false);
310        let _ = self
311            .internal_event_stream
312            .0
313            .broadcast_direct(Arc::new(HotShotEvent::Shutdown))
314            .await
315            .inspect_err(|err| tracing::error!("Failed to send shutdown event: {err}"));
316
317        tracing::error!("Shutting down the network!");
318        self.hotshot.network.shut_down().await;
319
320        tracing::error!("Shutting down network tasks!");
321        self.network_registry.shutdown().await;
322
323        tracing::error!("Shutting down consensus!");
324        self.consensus_registry.shutdown().await;
325    }

333    /// Wrapper for `HotShotConsensusApi`'s `leader` function
334    ///
335    /// # Errors
336    /// Returns an error if the leader cannot be calculated
337    #[allow(clippy::unused_async)] // async for API compatibility reasons
338    pub async fn leader(
339        &self,
340        view_number: TYPES::View,
341        epoch_number: Option<TYPES::Epoch>,
342    ) -> Result<TYPES::SignatureKey> {
343        self.hotshot
344            .membership_coordinator
345            .membership_for_epoch(epoch_number)
346            .await?
347            .leader(view_number)
348            .await
349            .context("Failed to lookup leader")
350    }
