59pub struct CdnMetricsValue {
60    /// The number of failed messages

67        // Create a subgroup for the CDN

69
70        // Create the CDN-specific metrics

72            num_failed_messages: subgroup.create_counter("num_failed_messages".into(), None),

84/// A wrapped `SignatureKey`. We need to implement the Push CDN's `SignatureScheme`
85/// trait in order to sign and verify messages to/from the CDN.
86#[derive(Clone, Eq, PartialEq)]
87pub struct WrappedSignatureKey<T: SignatureKey + 'static>(pub T);
88impl<T: SignatureKey> SignatureScheme for WrappedSignatureKey<T> {
89    type PrivateKey = T::PrivateKey;
90    type PublicKey = Self;
91
92    /// Sign a message of arbitrary data and return the serialized signature
93    fn sign(
94        private_key: &Self::PrivateKey,
95        namespace: &str,
96        message: &[u8],
97    ) -> anyhow::Result<Vec<u8>> {
98        // Combine the namespace and message into a single byte array
99        let message = [namespace.as_bytes(), message].concat();
100
101        let signature = T::sign(private_key, &message)?;
102        Ok(bincode_opts().serialize(&signature)?)
103    }
104
105    /// Verify a message of arbitrary data and return the result
106    fn verify(
107        public_key: &Self::PublicKey,
108        namespace: &str,
109        message: &[u8],
110        signature: &[u8],
111    ) -> bool {
112        // Deserialize the signature
113        let signature: T::PureAssembledSignatureType = match bincode_opts().deserialize(signature) {
114            Ok(key) => key,
115            Err(_) => return false,
116        };
117
118        // Combine the namespace and message into a single byte array
119        let message = [namespace.as_bytes(), message].concat();
120
121        public_key.0.validate(&signature, &message)
122    }
123}
124
125/// We need to implement the `Serializable` so the Push CDN can serialize the signatures
126/// and public keys and send them over the wire.
127impl<T: SignatureKey> Serializable for WrappedSignatureKey<T> {
128    fn serialize(&self) -> anyhow::Result<Vec<u8>> {
129        Ok(self.0.to_bytes())
130    }
131
132    fn deserialize(serialized: &[u8]) -> anyhow::Result<Self> {
133        Ok(WrappedSignatureKey(T::from_bytes(serialized)?))
134    }
135}

220impl<K: SignatureKey + 'static> PushCdnNetwork<K> {
221    /// Create a new `PushCdnNetwork` (really a client) from a marshal endpoint, a list of initial
222    /// topics we are interested in, and our wrapped keypair that we use to authenticate with the
223    /// marshal.
224    ///
225    /// # Errors
226    /// If we fail to build the config
227    pub fn new(
228        marshal_endpoint: String,
229        topics: Vec<Topic>,
230        keypair: KeyPair<WrappedSignatureKey<K>>,
231        metrics: CdnMetricsValue,
232    ) -> anyhow::Result<Self> {
233        // Build config
234        let config = ClientConfig {
235            endpoint: marshal_endpoint,
236            subscribed_topics: topics.into_iter().map(|t| t as u8).collect(),
237            keypair: keypair.clone(),
238            use_local_authority: true,
239        };
240
241        // Create the client from the config
242        let client = Client::new(config);
243
244        Ok(Self {
245            client,
246            metrics: Arc::from(metrics),
247            internal_queue: Arc::new(Mutex::new(VecDeque::new())),
248            public_key: keypair.public_key.0,
249            // Start unpaused
250            #[cfg(feature = "hotshot-testing")]
251            is_paused: Arc::from(AtomicBool::new(false)),
252        })
253    }
254
255    /// Broadcast a message to members of the particular topic. Does not retry.
256    ///
257    /// # Errors
258    /// - If we fail to serialize the message
259    /// - If we fail to send the broadcast message.
260    async fn broadcast_message(&self, message: Vec<u8>, topic: Topic) -> Result<(), NetworkError> {
261        // If we're paused, don't send the message
262        #[cfg(feature = "hotshot-testing")]
263        if self.is_paused.load(Ordering::Relaxed) {
264            return Ok(());
265        }
266
267        // Send the message
268        if let Err(err) = self
269            .client
270            .send_broadcast_message(vec![topic as u8], message)
271            .await
272        {
273            return Err(NetworkError::MessageReceiveError(format!(
274                "failed to send broadcast message: {err}"
275            )));
276        };
277
278        Ok(())
279    }
280}

283impl<TYPES: NodeType> TestableNetworkingImplementation<TYPES>

286    /// Generate n Push CDN clients, a marshal, and two brokers (that run locally).
287    /// Uses a `SQLite` database instead of Redis.
288    #[allow(clippy::too_many_lines)]
289    fn generator(
290        _expected_node_count: usize,
291        _num_bootstrap: usize,
292        _network_id: usize,
293        da_committee_size: usize,
294        _reliability_config: Option<Box<dyn NetworkReliability>>,
295        _secondary_network_delay: Duration,
296    ) -> AsyncGenerator<Arc<Self>> {
297        // The configuration we are using for testing is 2 brokers & 1 marshal
298
299        // A keypair shared between brokers
300        let (broker_public_key, broker_private_key) =
301            TYPES::SignatureKey::generated_from_seed_indexed([0u8; 32], 1337);
302
303        // Get the OS temporary directory
304        let temp_dir = std::env::temp_dir();
305
306        // Create an SQLite file inside of the temporary directory
307        let discovery_endpoint = temp_dir
308            .join(Path::new(&format!(
309                "test-{}.sqlite",
310                StdRng::from_entropy().next_u64()
311            )))
312            .to_string_lossy()
313            .into_owned();
314
315        // Pick some unused public ports
316        let public_address_1 = format!(
317            "127.0.0.1:{}",
318            portpicker::pick_unused_port().expect("could not find an open port")
319        );
320        let public_address_2 = format!(
321            "127.0.0.1:{}",
322            portpicker::pick_unused_port().expect("could not find an open port")
323        );
324
325        // 2 brokers
326        for i in 0..2 {
327            // Get the ports to bind to
328            let private_port = portpicker::pick_unused_port().expect("could not find an open port");
329
330            // Extrapolate addresses
331            let private_address = format!("127.0.0.1:{private_port}");
332            let (public_address, other_public_address) = if i == 0 {
333                (public_address_1.clone(), public_address_2.clone())
334            } else {
335                (public_address_2.clone(), public_address_1.clone())
336            };
337
338            // Calculate the broker identifiers
339            let broker_identifier = format!("{public_address}/{public_address}");
340            let other_broker_identifier = format!("{other_public_address}/{other_public_address}");
341
342            // Configure the broker
343            let config: BrokerConfig<TestingDef<TYPES::SignatureKey>> = BrokerConfig {
344                public_advertise_endpoint: public_address.clone(),
345                public_bind_endpoint: public_address,
346                private_advertise_endpoint: private_address.clone(),
347                private_bind_endpoint: private_address,
348                metrics_bind_endpoint: None,
349                keypair: KeyPair {
350                    public_key: WrappedSignatureKey(broker_public_key.clone()),
351                    private_key: broker_private_key.clone(),
352                },
353                discovery_endpoint: discovery_endpoint.clone(),
354
355                user_message_hook: NoMessageHook,
356                broker_message_hook: NoMessageHook,
357
358                ca_cert_path: None,
359                ca_key_path: None,
360                // 1GB
361                global_memory_pool_size: Some(1024 * 1024 * 1024),
362            };
363
364            // Create and spawn the broker
365            spawn(async move {
366                let broker: Broker<TestingDef<TYPES::SignatureKey>> =
367                    Broker::new(config).await.expect("broker failed to start");
368
369                // If we are the first broker by identifier, we need to sleep a bit
370                // for discovery to happen first
371                if other_broker_identifier > broker_identifier {
372                    sleep(Duration::from_secs(2)).await;
373                }
374
375                // Error if we stopped unexpectedly
376                if let Err(err) = broker.start().await {
377                    error!("broker stopped: {err}");
378                }
379            });
380        }
381
382        // Get the port to use for the marshal
383        let marshal_port = portpicker::pick_unused_port().expect("could not find an open port");
384
385        // Configure the marshal
386        let marshal_endpoint = format!("127.0.0.1:{marshal_port}");
387        let marshal_config = MarshalConfig {
388            bind_endpoint: marshal_endpoint.clone(),
389            discovery_endpoint,
390            metrics_bind_endpoint: None,
391            ca_cert_path: None,
392            ca_key_path: None,
393            // 1GB
394            global_memory_pool_size: Some(1024 * 1024 * 1024),
395        };
396
397        // Spawn the marshal
398        spawn(async move {
399            let marshal: Marshal<TestingDef<TYPES::SignatureKey>> = Marshal::new(marshal_config)
400                .await
401                .expect("failed to spawn marshal");
402
403            // Error if we stopped unexpectedly
404            if let Err(err) = marshal.start().await {
405                error!("marshal stopped: {err}");
406            }
407        });
408
409        // This function is called for each client we spawn
410        Box::pin({
411            move |node_id| {
412                // Clone this so we can pin the future
413                let marshal_endpoint = marshal_endpoint.clone();
414
415                Box::pin(async move {
416                    // Derive our public and priate keys from our index
417                    let private_key =
418                        TYPES::SignatureKey::generated_from_seed_indexed([0u8; 32], node_id).1;
419                    let public_key = TYPES::SignatureKey::from_private(&private_key);
420
421                    // Calculate if we're DA or not
422                    let topics = if node_id < da_committee_size as u64 {
423                        vec![Topic::Da as u8, Topic::Global as u8]
424                    } else {
425                        vec![Topic::Global as u8]
426                    };
427
428                    // Configure our client
429                    let client_config: ClientConfig<ClientDef<TYPES::SignatureKey>> =
430                        ClientConfig {
431                            keypair: KeyPair {
432                                public_key: WrappedSignatureKey(public_key.clone()),
433                                private_key,
434                            },
435                            subscribed_topics: topics,
436                            endpoint: marshal_endpoint,
437                            use_local_authority: true,
438                        };
439
440                    // Create our client
441                    Arc::new(PushCdnNetwork {
442                        client: Client::new(client_config),
443                        metrics: Arc::new(CdnMetricsValue::default()),
444                        internal_queue: Arc::new(Mutex::new(VecDeque::new())),
445                        public_key,
446                        #[cfg(feature = "hotshot-testing")]
447                        is_paused: Arc::from(AtomicBool::new(false)),
448                    })
449                })
450            }
451        })
452    }

456        None
457    }

460#[async_trait]
461impl<K: SignatureKey + 'static> ConnectedNetwork<K> for PushCdnNetwork<K> {
462    /// Pause sending and receiving on the PushCDN network.
463    fn pause(&self) {
464        #[cfg(feature = "hotshot-testing")]
465        self.is_paused.store(true, Ordering::Relaxed);
466    }
467
468    /// Resume sending and receiving on the PushCDN network.
469    fn resume(&self) {
470        #[cfg(feature = "hotshot-testing")]
471        self.is_paused.store(false, Ordering::Relaxed);
472    }
473
474    /// Wait for the client to initialize the connection
475    async fn wait_for_ready(&self) {
476        let _ = self.client.ensure_initialized().await;
477    }
478
479    /// TODO: shut down the networks. Unneeded for testing.
480    fn shut_down<'a, 'b>(&'a self) -> BoxSyncFuture<'b, ()>
481    where
482        'a: 'b,
483        Self: 'b,
484    {
485        boxed_sync(async move { self.client.close().await })
486    }
487
488    /// Broadcast a message to all members of the quorum.
489    ///
490    /// # Errors
491    /// - If we fail to serialize the message
492    /// - If we fail to send the broadcast message.
493    async fn broadcast_message(
494        &self,
495        message: Vec<u8>,
496        topic: HotShotTopic,
497        _broadcast_delay: BroadcastDelay,
498    ) -> Result<(), NetworkError> {
499        // If we're paused, don't send the message
500        #[cfg(feature = "hotshot-testing")]
501        if self.is_paused.load(Ordering::Relaxed) {
502            return Ok(());
503        }
504        self.broadcast_message(message, topic.into())
505            .await
506            .inspect_err(|_e| {
507                self.metrics.num_failed_messages.add(1);
508            })
509    }
510
511    /// Broadcast a message to all members of the DA committee.
512    ///
513    /// # Errors
514    /// - If we fail to serialize the message
515    /// - If we fail to send the broadcast message.
516    async fn da_broadcast_message(
517        &self,
518        message: Vec<u8>,
519        _recipients: Vec<K>,
520        _broadcast_delay: BroadcastDelay,
521    ) -> Result<(), NetworkError> {
522        // If we're paused, don't send the message
523        #[cfg(feature = "hotshot-testing")]
524        if self.is_paused.load(Ordering::Relaxed) {
525            return Ok(());
526        }
527        self.broadcast_message(message, Topic::Da)
528            .await
529            .inspect_err(|_e| {
530                self.metrics.num_failed_messages.add(1);
531            })
532    }
533
534    /// Send a direct message to a node with a particular key. Does not retry.
535    ///
536    /// - If we fail to serialize the message
537    /// - If we fail to send the direct message
538    async fn direct_message(&self, message: Vec<u8>, recipient: K) -> Result<(), NetworkError> {
539        // If the message is to ourselves, just add it to the internal queue
540        if recipient == self.public_key {
541            self.internal_queue.lock().push_back(message);
542            return Ok(());
543        }
544
545        // If we're paused, don't send the message
546        #[cfg(feature = "hotshot-testing")]
547        if self.is_paused.load(Ordering::Relaxed) {
548            return Ok(());
549        }
550
551        // Send the message
552        if let Err(e) = self
553            .client
554            .send_direct_message(&WrappedSignatureKey(recipient), message)
555            .await
556        {
557            self.metrics.num_failed_messages.add(1);
558            return Err(NetworkError::MessageSendError(format!(
559                "failed to send direct message: {e}"
560            )));
561        };
562
563        Ok(())
564    }
565
566    /// Receive a message. Is agnostic over `transmit_type`, which has an issue
567    /// to be removed anyway.
568    ///
569    /// # Errors
570    /// - If we fail to receive messages. Will trigger a retry automatically.
571    async fn recv_message(&self) -> Result<Vec<u8>, NetworkError> {
572        // If we have a message in the internal queue, return it
573        if let Some(message) = self.internal_queue.lock().pop_front() {
574            return Ok(message);
575        }
576
577        // Receive a message from the network
578        let message = self.client.receive_message().await;
579
580        // If we're paused, receive but don't process messages
581        #[cfg(feature = "hotshot-testing")]
582        if self.is_paused.load(Ordering::Relaxed) {
583            sleep(Duration::from_millis(100)).await;
584            return Ok(vec![]);
585        }
586
587        // If it was an error, wait a bit and retry
588        let message = match message {
589            Ok(message) => message,
590            Err(error) => {
591                return Err(NetworkError::MessageReceiveError(format!(
592                    "failed to receive message: {error}"
593                )));
594            },
595        };
596
597        // Extract the underlying message
598        let (PushCdnMessage::Broadcast(Broadcast { message, topics: _ })
599        | PushCdnMessage::Direct(Direct {
600            message,
601            recipient: _,
602        })) = message
603        else {
604            return Ok(vec![]);
605        };
606
607        Ok(message)
608    }
609
610    /// Do nothing here, as we don't need to look up nodes.
611    fn queue_node_lookup(
612        &self,
613        _view_number: ViewNumber,
614        _pk: K,
615    ) -> Result<(), TrySendError<Option<(ViewNumber, K)>>> {
616        Ok(())
617    }
618}
