18use hotshot_utils::anytrace::Result;

21#[derive(Clone, Debug, Eq, PartialEq, Hash)]
22
23/// The static committee election
24pub struct StaticCommitteeLeaderForTwoViews<T: NodeType> {
25    /// The nodes eligible for leadership.
26    /// NOTE: This is currently a hack because the DA leader needs to be the quorum
27    /// leader but without voting rights.
28    eligible_leaders: Vec<PeerConfig<T>>,
29
30    /// The nodes on the committee and their stake
31    stake_table: Vec<PeerConfig<T>>,
32
33    /// The nodes on the committee and their stake
34    da_stake_table: Vec<PeerConfig<T>>,
35
36    /// The nodes on the committee and their stake, indexed by public key
37    indexed_stake_table: BTreeMap<T::SignatureKey, PeerConfig<T>>,
38
39    /// The nodes on the committee and their stake, indexed by public key
40    indexed_da_stake_table: BTreeMap<T::SignatureKey, PeerConfig<T>>,
41}
42
43impl<TYPES: NodeType> Membership<TYPES> for StaticCommitteeLeaderForTwoViews<TYPES> {
44    type Error = hotshot_utils::anytrace::Error;
45    /// Create a new election
46    fn new(committee_members: Vec<PeerConfig<TYPES>>, da_members: Vec<PeerConfig<TYPES>>) -> Self {
47        // For each eligible leader, get the stake table entry
48        let eligible_leaders: Vec<PeerConfig<TYPES>> = committee_members
49            .iter()
50            .filter(|&member| member.stake_table_entry.stake() > U256::zero())
51            .cloned()
52            .collect();
53
54        // For each member, get the stake table entry
55        let members: Vec<PeerConfig<TYPES>> = committee_members
56            .iter()
57            .filter(|&member| member.stake_table_entry.stake() > U256::zero())
58            .cloned()
59            .collect();
60
61        // For each member, get the stake table entry
62        let da_members: Vec<PeerConfig<TYPES>> = da_members
63            .iter()
64            .filter(|&member| member.stake_table_entry.stake() > U256::zero())
65            .cloned()
66            .collect();
67
68        // Index the stake table by public key
69        let indexed_stake_table: BTreeMap<TYPES::SignatureKey, PeerConfig<TYPES>> = members
70            .iter()
71            .map(|member| {
72                (
73                    TYPES::SignatureKey::public_key(&member.stake_table_entry),
74                    member.clone(),
75                )
76            })
77            .collect();
78
79        // Index the stake table by public key
80        let indexed_da_stake_table: BTreeMap<TYPES::SignatureKey, PeerConfig<TYPES>> = da_members
81            .iter()
82            .map(|member| {
83                (
84                    TYPES::SignatureKey::public_key(&member.stake_table_entry),
85                    member.clone(),
86                )
87            })
88            .collect();
89
90        Self {
91            eligible_leaders,
92            stake_table: members,
93            da_stake_table: da_members,
94            indexed_stake_table,
95            indexed_da_stake_table,
96        }
97    }
98
99    /// Get the stake table for the current view
100    fn stake_table(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> Vec<PeerConfig<TYPES>> {
101        self.stake_table.clone()
102    }
103
104    /// Get the stake table for the current view
105    fn da_stake_table(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> Vec<PeerConfig<TYPES>> {
106        self.da_stake_table.clone()
107    }
108
109    /// Get all members of the committee for the current view
110    fn committee_members(
111        &self,
112        _view_number: <TYPES as NodeType>::View,
113        _epoch: Option<<TYPES as NodeType>::Epoch>,
114    ) -> BTreeSet<<TYPES as NodeType>::SignatureKey> {
115        self.stake_table
116            .iter()
117            .map(|sc| TYPES::SignatureKey::public_key(&sc.stake_table_entry))
118            .collect()
119    }
120
121    /// Get all members of the committee for the current view
122    fn da_committee_members(
123        &self,
124        _view_number: <TYPES as NodeType>::View,
125        _epoch: Option<<TYPES as NodeType>::Epoch>,
126    ) -> BTreeSet<<TYPES as NodeType>::SignatureKey> {
127        self.da_stake_table
128            .iter()
129            .map(|da| TYPES::SignatureKey::public_key(&da.stake_table_entry))
130            .collect()
131    }
132
133    /// Get the stake table entry for a public key
134    fn stake(
135        &self,
136        pub_key: &<TYPES as NodeType>::SignatureKey,
137        _epoch: Option<<TYPES as NodeType>::Epoch>,
138    ) -> Option<PeerConfig<TYPES>> {
139        // Only return the stake if it is above zero
140        self.indexed_stake_table.get(pub_key).cloned()
141    }
142
143    /// Get DA the stake table entry for a public key
144    fn da_stake(
145        &self,
146        pub_key: &<TYPES as NodeType>::SignatureKey,
147        _epoch: Option<<TYPES as NodeType>::Epoch>,
148    ) -> Option<PeerConfig<TYPES>> {
149        // Only return the stake if it is above zero
150        self.indexed_da_stake_table.get(pub_key).cloned()
151    }
152
153    /// Check if a node has stake in the committee
154    fn has_stake(
155        &self,
156        pub_key: &<TYPES as NodeType>::SignatureKey,
157        _epoch: Option<<TYPES as NodeType>::Epoch>,
158    ) -> bool {
159        self.indexed_stake_table
160            .get(pub_key)
161            .is_some_and(|x| x.stake_table_entry.stake() > U256::zero())
162    }
163
164    /// Check if a node has stake in the committee
165    fn has_da_stake(
166        &self,
167        pub_key: &<TYPES as NodeType>::SignatureKey,
168        _epoch: Option<<TYPES as NodeType>::Epoch>,
169    ) -> bool {
170        self.indexed_da_stake_table
171            .get(pub_key)
172            .is_some_and(|x| x.stake_table_entry.stake() > U256::zero())
173    }
174
175    /// Index the vector of public keys with the current view number
176    fn lookup_leader(
177        &self,
178        view_number: <TYPES as NodeType>::View,
179        _epoch: Option<<TYPES as NodeType>::Epoch>,
180    ) -> Result<TYPES::SignatureKey> {
181        let index =
182            usize::try_from((*view_number / 2) % self.eligible_leaders.len() as u64).unwrap();
183        let res = self.eligible_leaders[index].clone();
184
185        Ok(TYPES::SignatureKey::public_key(&res.stake_table_entry))
186    }
187
188    /// Get the total number of nodes in the committee
189    fn total_nodes(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> usize {
190        self.stake_table.len()
191    }
192
193    /// Get the total number of DA nodes in the committee
194    fn da_total_nodes(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> usize {
195        self.da_stake_table.len()
196    }
197
198    /// Get the voting success threshold for the committee
199    fn success_threshold(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> U256 {
200        U256::from(((self.stake_table.len() as u64 * 2) / 3) + 1)
201    }
202
203    /// Get the voting success threshold for the committee
204    fn da_success_threshold(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> U256 {
205        U256::from(((self.da_stake_table.len() as u64 * 2) / 3) + 1)
206    }
207
208    /// Get the voting failure threshold for the committee
209    fn failure_threshold(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> U256 {
210        U256::from(((self.stake_table.len() as u64) / 3) + 1)
211    }
212
213    /// Get the voting upgrade threshold for the committee
214    fn upgrade_threshold(&self, _epoch: Option<<TYPES as NodeType>::Epoch>) -> U256 {
215        U256::from(((self.stake_table.len() as u64 * 9) / 10) + 1)
216    }
217    fn has_stake_table(&self, _epoch: TYPES::Epoch) -> bool {
218        true
219    }
220    fn has_randomized_stake_table(&self, _epoch: TYPES::Epoch) -> bool {
221        true
222    }
223
224    fn add_drb_result(&mut self, _epoch: <TYPES as NodeType>::Epoch, _drb_result: DrbResult) {}
225
226    fn set_first_epoch(&mut self, _epoch: TYPES::Epoch, _initial_drb_result: DrbResult) {}
227}
