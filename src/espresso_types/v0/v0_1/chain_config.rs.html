18/// Global variables for an Espresso blockchain.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
20pub struct ChainConfig {
21    /// Espresso chain ID
22    pub chain_id: ChainId,
23
24    /// Maximum size in bytes of a block
25    pub max_block_size: BlockSize,
26
27    /// Minimum fee in WEI per byte of payload
28    pub base_fee: FeeAmount,
29
30    /// Fee contract address on L1.
31    ///
32    /// This is optional so that fees can easily be toggled on/off, with no need to deploy a
33    /// contract when they are off. In a future release, after fees are switched on and thoroughly
34    /// tested, this may be made mandatory.
35    pub fee_contract: Option<Address>,
36
37    /// Account that receives sequencing fees.
38    ///
39    /// This account in the Espresso fee ledger will always receive every fee paid in Espresso,
40    /// regardless of whether or not their is a `fee_contract` deployed. Once deployed, the fee
41    /// contract can decide what to do with tokens locked in this account in Espresso.
42    pub fee_recipient: FeeAccount,
43}

45#[derive(Clone, Debug, Copy, PartialEq, Deserialize, Serialize, Eq, Hash)]
46pub struct ResolvableChainConfig {
47    pub(crate) chain_config: Either<ChainConfig, Commitment<ChainConfig>>,
48}
49
50impl Committable for ChainConfig {
51    fn tag() -> String {
52        "CHAIN_CONFIG".to_string()
53    }
54
55    fn commit(&self) -> Commitment<Self> {
56        let comm = committable::RawCommitmentBuilder::new(&Self::tag())
57            .fixed_size_field("chain_id", &self.chain_id.to_fixed_bytes())
58            .u64_field("max_block_size", *self.max_block_size)
59            .fixed_size_field("base_fee", &self.base_fee.to_fixed_bytes())
60            .fixed_size_field("fee_recipient", &self.fee_recipient.to_fixed_bytes());
61        let comm = if let Some(addr) = self.fee_contract {
62            comm.u64_field("fee_contract", 1).fixed_size_bytes(&addr.0)
63        } else {
64            comm.u64_field("fee_contract", 0)
65        };
66        comm.finalize()
67    }
68}
69
70impl ResolvableChainConfig {
71    pub fn commit(&self) -> Commitment<ChainConfig> {
72        match self.chain_config {
73            Either::Left(config) => config.commit(),
74            Either::Right(commitment) => commitment,
75        }
76    }
77    pub fn resolve(self) -> Option<ChainConfig> {
78        match self.chain_config {
79            Either::Left(config) => Some(config),
80            Either::Right(_) => None,
81        }
82    }
83}
84
85impl From<Commitment<ChainConfig>> for ResolvableChainConfig {
86    fn from(value: Commitment<ChainConfig>) -> Self {
87        Self {
88            chain_config: Either::Right(value),
89        }
90    }
91}
92
93impl From<ChainConfig> for ResolvableChainConfig {
94    fn from(value: ChainConfig) -> Self {
95        Self {
96            chain_config: Either::Left(value),
97        }
98    }
99}
