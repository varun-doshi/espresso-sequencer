11/// Each variant represents a specific version of a namespace proof.
12#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
13pub enum NsProof {
14    V0(ADVZNsProof),
15    V1(AvidMNsProof),
16}
17
18impl NsProof {
19    pub fn new(payload: &Payload, index: &NsIndex, common: &VidCommon) -> Option<NsProof> {
20        match common {
21            VidCommon::V0(common) => Some(NsProof::V0(ADVZNsProof::new(payload, index, common)?)),
22            VidCommon::V1(common) => Some(NsProof::V1(AvidMNsProof::new(payload, index, common)?)),
23        }
24    }
25
26    pub fn verify(
27        &self,
28        ns_table: &NsTable,
29        commit: &VidCommitment,
30        common: &VidCommon,
31    ) -> Option<(Vec<Transaction>, NamespaceId)> {
32        match (self, common) {
33            (Self::V0(proof), VidCommon::V0(common)) => proof.verify(ns_table, commit, common),
34            (Self::V1(proof), VidCommon::V1(common)) => proof.verify(ns_table, commit, common),
35            _ => {
36                tracing::error!("Incompatible version of VidCommon and NsProof.");
37                None
38            },
39        }
40    }
41
42    pub fn export_all_txs(&self, ns_id: &NamespaceId) -> Vec<Transaction> {
43        match self {
44            Self::V0(proof) => proof.export_all_txs(ns_id),
45            Self::V1(proof) => {
46                NsPayload::from_bytes_slice(&proof.0.ns_payload).export_all_txs(ns_id)
47            },
48        }
49    }
50}
