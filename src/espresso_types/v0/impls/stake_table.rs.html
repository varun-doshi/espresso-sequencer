44/// Create the consensus and DA stake tables from L1 events
45///
46/// This is a pure function, to make it easily testable.
47///
48/// We expect have at most a few hundred EVM events in the
49/// PermissionedStakeTable contract over the liftetime of the contract so it
50/// should not significantly affect performance to fetch all events and
51/// perform the computation in this functions once per epoch.
52pub fn from_l1_events<I: Iterator<Item = StakeTableEvent>>(
53    events: I,
54) -> anyhow::Result<IndexMap<Address, Validator<BLSPubKey>>> {
55    let mut validators = IndexMap::new();
56    let mut bls_keys = HashSet::new();
57    let mut schnorr_keys = HashSet::new();
58    for event in events {
59        tracing::debug!("Processing stake table event: {:?}", event);
60        match event {
61            StakeTableEvent::Register(ValidatorRegistered {
62                account,
63                blsVk,
64                schnorrVk,
65                commission,
66            }) => {
67                // TODO(abdul): BLS and Schnorr signature keys verification
68                let stake_table_key = bls_alloy_to_jf2(blsVk.clone());
69                let state_ver_key = edward_bn254point_to_state_ver(schnorrVk.clone());
70                // TODO(MA): The stake table contract currently enforces that each bls key is only used once. We will
71                // move this check to the confirmation layer and remove it from the contract. Once we have the signature
72                // check in this functions we can skip if a BLS key, or Schnorr key was previously used.
73                if bls_keys.contains(&stake_table_key) {
74                    bail!("bls key {} already used", stake_table_key.to_string());
75                };
76
77                // The contract does *not* enforce that each schnorr key is only used once.
78                if schnorr_keys.contains(&state_ver_key) {
79                    tracing::warn!("schnorr key {} already used", state_ver_key.to_string());
80                };
81
82                bls_keys.insert(stake_table_key);
83                schnorr_keys.insert(state_ver_key.clone());
84
85                match validators.entry(account) {
86                    indexmap::map::Entry::Occupied(_occupied_entry) => {
87                        bail!("validator {:#x} already registered", *account)
88                    },
89                    indexmap::map::Entry::Vacant(vacant_entry) => vacant_entry.insert(Validator {
90                        account,
91                        stake_table_key,
92                        state_ver_key,
93                        stake: U256::from(0_u64),
94                        commission,
95                        delegators: HashMap::default(),
96                    }),
97                };
98            },
99            StakeTableEvent::Deregister(exit) => {
100                validators
101                    .shift_remove(&exit.validator)
102                    .with_context(|| format!("validator {:#x} not found", exit.validator))?;
103            },
104            StakeTableEvent::Delegate(delegated) => {
105                let Delegated {
106                    delegator,
107                    validator,
108                    amount,
109                } = delegated;
110                let validator_entry = validators
111                    .get_mut(&validator)
112                    .with_context(|| format!("validator {validator:#x} not found"))?;
113
114                if amount.is_zero() {
115                    tracing::warn!("delegator {delegator:?} has 0 stake");
116                    continue;
117                }
118                // Increase stake
119                validator_entry.stake += amount;
120                // Add delegator to the set
121                validator_entry.delegators.insert(delegator, amount);
122            },
123            StakeTableEvent::Undelegate(undelegated) => {
124                let Undelegated {
125                    delegator,
126                    validator,
127                    amount,
128                } = undelegated;
129                let validator_entry = validators
130                    .get_mut(&validator)
131                    .with_context(|| format!("validator {validator:#x} not found"))?;
132
133                validator_entry.stake = validator_entry
134                    .stake
135                    .checked_sub(amount)
136                    .with_context(|| "stake is less than undelegated amount")?;
137
138                let delegator_stake = validator_entry
139                    .delegators
140                    .get_mut(&delegator)
141                    .with_context(|| format!("delegator {delegator:#x} not found"))?;
142                *delegator_stake = delegator_stake
143                    .checked_sub(amount)
144                    .with_context(|| "delegator_stake is less than undelegated amount")?;
145
146                if delegator_stake.is_zero() {
147                    // if delegator stake is 0, remove from set
148                    validator_entry.delegators.remove(&delegator);
149                }
150            },
151            StakeTableEvent::KeyUpdate(update) => {
152                let ConsensusKeysUpdated {
153                    account,
154                    blsVK,
155                    schnorrVK,
156                } = update;
157                let validator = validators
158                    .get_mut(&account)
159                    .with_context(|| "validator {account:#x} not found")?;
160                let bls = bls_alloy_to_jf2(blsVK);
161                let state_ver_key = edward_bn254point_to_state_ver(schnorrVK);
162
163                validator.stake_table_key = bls;
164                validator.state_ver_key = state_ver_key;
165            },
166        }
167    }
168
169    select_validators(&mut validators)?;
170
171    Ok(validators)
172}

174fn select_validators(
175    validators: &mut IndexMap<Address, Validator<BLSPubKey>>,
176) -> anyhow::Result<()> {
177    // Remove invalid validators first
178    validators.retain(|address, validator| {
179        if validator.delegators.is_empty() {
180            tracing::info!("Validator {address:?} does not have any delegator");
181            return false;
182        }
183
184        if validator.stake.is_zero() {
185            tracing::info!("Validator {address:?} does not have any stake");
186            return false;
187        }
188
189        true
190    });
191
192    if validators.is_empty() {
193        bail!("No valid validators found");
194    }
195
196    // Find the maximum stake
197    let maximum_stake = validators
198        .values()
199        .map(|v| v.stake)
200        .max()
201        .context("Failed to determine max stake")?;
202
203    let minimum_stake = maximum_stake
204        .checked_div(U256::from(VID_TARGET_TOTAL_STAKE))
205        .context("div err")?;
206
207    // Collect validators that meet the minimum stake criteria
208    let mut valid_stakers: Vec<_> = validators
209        .iter()
210        .filter(|(_, v)| v.stake >= minimum_stake)
211        .map(|(addr, v)| (*addr, v.stake))
212        .collect();
213
214    // Sort by stake (descending order)
215    valid_stakers.sort_by_key(|(_, stake)| std::cmp::Reverse(*stake));
216
217    // Keep only the top 100 stakers
218    if valid_stakers.len() > 100 {
219        valid_stakers.truncate(100);
220    }
221
222    // Retain only the selected validators
223    let selected_addresses: HashSet<_> = valid_stakers.iter().map(|(addr, _)| *addr).collect();
224    validators.retain(|address, _| selected_addresses.contains(address));
225
226    Ok(())
227}

230pub enum StakeTableEvent {
231    Register(ValidatorRegistered),
232    Deregister(ValidatorExit),
233    Delegate(Delegated),
234    Undelegate(Undelegated),

238impl std::fmt::Debug for StakeTableEvent {
239    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
240        match self {
241            StakeTableEvent::Register(event) => write!(f, "Register({:?})", event.account),
242            StakeTableEvent::Deregister(event) => write!(f, "Deregister({:?})", event.validator),
243            StakeTableEvent::Delegate(event) => write!(f, "Delegate({:?})", event.delegator),
244            StakeTableEvent::Undelegate(event) => write!(f, "Undelegate({:?})", event.delegator),
245            StakeTableEvent::KeyUpdate(event) => write!(f, "KeyUpdate({:?})", event.account),
246        }
247    }

250impl StakeTableEvent {
251    pub fn sort_events(
252        registrations: Vec<(ValidatorRegistered, Log)>,
253        deregistrations: Vec<(ValidatorExit, Log)>,
254        delegations: Vec<(Delegated, Log)>,
255        undelegated_events: Vec<(Undelegated, Log)>,
256        keys_update: Vec<(ConsensusKeysUpdated, Log)>,
257    ) -> anyhow::Result<BTreeMap<(u64, u64), StakeTableEvent>> {
258        let mut map = BTreeMap::new();
259        for (registration, log) in registrations {
260            map.insert(
261                (
262                    log.block_number.context("block number")?,
263                    log.log_index.context("log index")?,
264                ),
265                registration.into(),
266            );
267        }
268        for (dereg, log) in deregistrations {
269            map.insert(
270                (
271                    log.block_number.context("block number")?,
272                    log.log_index.context("log index")?,
273                ),
274                dereg.into(),
275            );
276        }
277        for (delegation, log) in delegations {
278            map.insert(
279                (
280                    log.block_number.context("block number")?,
281                    log.log_index.context("log index")?,
282                ),
283                delegation.into(),
284            );
285        }
286        for (undelegated, log) in undelegated_events {
287            map.insert(
288                (
289                    log.block_number.context("block number")?,
290                    log.log_index.context("log index")?,
291                ),
292                undelegated.into(),
293            );
294        }
295
296        for (update, log) in keys_update {
297            map.insert(
298                (
299                    log.block_number.context("block number")?,
300                    log.log_index.context("log index")?,
301                ),
302                update.into(),
303            );
304        }
305        Ok(map)
306    }

377    /// Updates `Self.stake_table` with stake_table for
378    /// `Self.contract_address` at `l1_block_height`. This is intended
379    /// to be called before calling `self.stake()` so that
380    /// `Self.stake_table` only needs to be updated once in a given
381    /// life-cycle but may be read from many times.
382    fn update_stake_table(
383        &mut self,
384        epoch: EpochNumber,
385        validators: IndexMap<Address, Validator<BLSPubKey>>,
386    ) {
387        let mut address_mapping = HashMap::new();
388        let stake_table = validators
389            .values()
390            .map(|v| {
391                address_mapping.insert(v.stake_table_key, v.account);
392                (
393                    v.stake_table_key,
394                    PeerConfig {
395                        stake_table_entry: BLSPubKey::stake_table_entry(
396                            &v.stake_table_key,
397                            v.stake.to_ethers(),
398                        ),
399                        state_ver_key: v.state_ver_key.clone(),
400                    },
401                )
402            })
403            .collect();
404
405        self.state.insert(
406            epoch,
407            EpochCommittee {
408                eligible_leaders: self.non_epoch_committee.eligible_leaders.clone(),
409                stake_table,
410                validators,
411                address_mapping,
412            },
413        );
414    }

451    // We need a constructor to match our concrete type.
452    pub fn new_stake(
453        // TODO remove `new` from trait and rename this to `new`.
454        // https://github.com/EspressoSystems/HotShot/commit/fcb7d54a4443e29d643b3bbc53761856aef4de8b
455        committee_members: Vec<PeerConfig<SeqTypes>>,
456        da_members: Vec<PeerConfig<SeqTypes>>,
457        l1_client: L1Client,
458        contract_address: Option<Address>,
459        peers: Arc<dyn StateCatchup>,
460        persistence: impl MembershipPersistence,
461    ) -> Self {
462        // For each eligible leader, get the stake table entry
463        let eligible_leaders: Vec<_> = committee_members
464            .iter()
465            .filter(|&peer_config| {
466                peer_config.stake_table_entry.stake() > ethers::types::U256::zero()
467            })
468            .cloned()
469            .collect();
470
471        // For each member, get the stake table entry
472        let stake_table: Vec<_> = committee_members
473            .iter()
474            .filter(|&peer_config| {
475                peer_config.stake_table_entry.stake() > ethers::types::U256::zero()
476            })
477            .cloned()
478            .collect();
479
480        // For each member, get the stake table entry
481        let da_members: Vec<_> = da_members
482            .iter()
483            .filter(|&peer_config| {
484                peer_config.stake_table_entry.stake() > ethers::types::U256::zero()
485            })
486            .cloned()
487            .collect();
488
489        // Index the stake table by public key
490        let indexed_stake_table: HashMap<PubKey, _> = stake_table
491            .iter()
492            .map(|peer_config| {
493                (
494                    PubKey::public_key(&peer_config.stake_table_entry),
495                    peer_config.clone(),
496                )
497            })
498            .collect();
499
500        // Index the stake table by public key
501        let indexed_da_members: HashMap<PubKey, _> = da_members
502            .iter()
503            .map(|peer_config| {
504                (
505                    PubKey::public_key(&peer_config.stake_table_entry),
506                    peer_config.clone(),
507                )
508            })
509            .collect();
510
511        let members = NonEpochCommittee {
512            eligible_leaders,
513            stake_table,
514            da_members,
515            indexed_stake_table,
516            indexed_da_members,
517        };
518
519        let mut map = HashMap::new();
520        let epoch_committee = EpochCommittee {
521            eligible_leaders: members.eligible_leaders.clone(),
522            stake_table: members
523                .stake_table
524                .iter()
525                .map(|x| (PubKey::public_key(&x.stake_table_entry), x.clone()))
526                .collect(),
527            validators: Default::default(),
528            address_mapping: HashMap::new(),
529        };
530        map.insert(Epoch::genesis(), epoch_committee.clone());
531        // TODO: remove this, workaround for hotshot asking for stake tables from epoch 1
532        map.insert(Epoch::genesis() + 1u64, epoch_committee.clone());
533
534        Self {
535            non_epoch_committee: members,
536            state: map,
537            l1_client,
538            contract_address,
539            randomized_committees: BTreeMap::new(),
540            peers,
541            persistence: Arc::new(persistence),
542            first_epoch: None,
543        }
544    }
545    fn get_stake_table(&self, epoch: &Option<Epoch>) -> Option<Vec<PeerConfig<SeqTypes>>> {
546        if let Some(epoch) = epoch {
547            self.state
548                .get(epoch)
549                .map(|committee| committee.stake_table.clone().into_values().collect())
550        } else {
551            Some(self.non_epoch_committee.stake_table.clone())
552        }
553    }
554
555    /// Get the stake table by epoch. Try to load from DB and fall back to fetching from l1.
556    async fn get_stake_table_by_epoch(
557        &self,
558        epoch: Epoch,
559        contract_address: Address,
560        l1_block: u64,
561    ) -> Result<IndexMap<alloy::primitives::Address, Validator<BLSPubKey>>, GetStakeTablesError>
562    {
563        if let Some(stake_tables) = self
564            .persistence
565            .load_stake(epoch)
566            .await
567            .map_err(GetStakeTablesError::PersistenceLoadError)?
568        {
569            Ok(stake_tables)
570        } else {
571            self.l1_client
572                .get_stake_table(contract_address, l1_block)
573                .await
574                .map_err(GetStakeTablesError::L1ClientFetchError)
575        }
576    }

674        self.da_stake(pub_key, epoch)
675            .map(|x| x.stake_table_entry.stake() > ethers::types::U256::zero())
676            .unwrap_or_default()
677    }

679    /// Index the vector of public keys with the current view number
680    fn lookup_leader(
681        &self,
682        view_number: <SeqTypes as NodeType>::View,
683        epoch: Option<Epoch>,
684    ) -> Result<PubKey, Self::Error> {
685        if let Some(epoch) = epoch {
686            let Some(randomized_committee) = self.randomized_committees.get(&epoch) else {
687                tracing::error!(
688                    "We are missing the randomized committee for epoch {}",
689                    epoch
690                );
691                return Err(LeaderLookupError);
692            };
693
694            Ok(PubKey::public_key(&select_randomized_leader(
695                randomized_committee,
696                *view_number,
697            )))
698        } else {
699            let leaders = &self.non_epoch_committee.eligible_leaders;
700
701            let index = *view_number as usize % leaders.len();
702            let res = leaders[index].clone();
703            Ok(PubKey::public_key(&res.stake_table_entry))
704        }
705    }

717    /// Get the voting success threshold for the committee
718    fn success_threshold(&self, epoch: Option<Epoch>) -> primitive_types::U256 {
719        let total_stake = self.total_stake(epoch);
720        if total_stake < primitive_types::U256::max_value() / 2 {
721            ((total_stake * 2) / 3) + 1
722        } else {
723            ((total_stake / 3) * 2) + 2
724        }
725    }
726
727    /// Get the voting success threshold for the committee
728    fn da_success_threshold(&self, epoch: Option<Epoch>) -> primitive_types::U256 {
729        let total_stake = self.total_da_stake(epoch);
730        if total_stake < primitive_types::U256::max_value() / 2 {
731            ((total_stake * 2) / 3) + 1
732        } else {
733            ((total_stake / 3) * 2) + 2
734        }
735    }
736
737    /// Get the voting failure threshold for the committee
738    fn failure_threshold(&self, epoch: Option<Epoch>) -> primitive_types::U256 {
739        let total_stake = self.total_stake(epoch);
740
741        (total_stake / 3) + 1
742    }
743
744    /// Get the voting upgrade threshold for the committee
745    fn upgrade_threshold(&self, epoch: Option<Epoch>) -> primitive_types::U256 {
746        let total_stake = self.total_stake(epoch);
747
748        let normal_threshold = self.success_threshold(epoch);
749        let higher_threshold = if total_stake < primitive_types::U256::max_value() / 9 {
750            (total_stake * 9) / 10
751        } else {
752            (total_stake / 10) * 9
753        };
754
755        max(higher_threshold, normal_threshold)
756    }

758    #[allow(refining_impl_trait)]
759    async fn add_epoch_root(
760        &self,
761        epoch: Epoch,
762        block_header: Header,
763    ) -> Option<Box<dyn FnOnce(&mut Self) + Send>> {
764        let Some(address) = self.contract_address else {
765            tracing::debug!("`add_epoch_root` called with `self.contract_address` value of `None`");
766            return None;
767        };
768
769        let stake_tables = self
770            .get_stake_table_by_epoch(epoch, address, block_header.height())
771            .await
772            .inspect_err(|e| {
773                tracing::error!(?e, "`add_epoch_root`, error retrieving stake table");
774            })
775            .ok()?;
776
777        if let Err(e) = self
778            .persistence
779            .store_stake(epoch, stake_tables.clone())
780            .await
781        {
782            tracing::error!(?e, "`add_epoch_root`, error storing stake table");
783        }
784
785        Some(Box::new(move |committee: &mut Self| {
786            committee.update_stake_table(epoch, stake_tables);
787        }))
788    }

794    fn has_randomized_stake_table(&self, epoch: Epoch) -> bool {
795        match self.first_epoch {
796            None => true,
797            Some(first_epoch) => {
798                if epoch < first_epoch {
799                    self.state.contains_key(&epoch)
800                } else {
801                    self.randomized_committees.contains_key(&epoch)
802                }
803            },
804        }

807    async fn get_epoch_root(
808        membership: Arc<RwLock<Self>>,
809        block_height: u64,
810        epoch: Epoch,
811    ) -> anyhow::Result<Header> {
812        let peers = membership.read().await.peers.clone();
813        let stake_table = membership.read().await.stake_table(Some(epoch)).clone();
814        let success_threshold = membership.read().await.success_threshold(Some(epoch));
815        // Fetch leaves from peers
816        let leaf: Leaf2 = peers
817            .fetch_leaf(block_height, stake_table.clone(), success_threshold)
818            .await?;
819
820        Ok(leaf.block_header().clone())
821    }
822
823    async fn get_epoch_drb(
824        membership: Arc<RwLock<Self>>,
825        block_height: u64,
826        epoch: Epoch,
827    ) -> anyhow::Result<DrbResult> {
828        let peers = membership.read().await.peers.clone();
829        let stake_table = membership.read().await.stake_table(Some(epoch)).clone();
830        let success_threshold = membership.read().await.success_threshold(Some(epoch));
831
832        tracing::debug!(
833            "Getting DRB for epoch {:?}, block height {:?}",
834            epoch,
835            block_height
836        );
837        let drb_leaf = peers
838            .fetch_leaf(block_height, stake_table, success_threshold)
839            .await?;
840
841        let Some(drb) = drb_leaf.next_drb_result else {
842            tracing::error!(
843          "We received a leaf that should contain a DRB result, but the DRB result is missing: {:?}",
844          drb_leaf
845        );
846
847            bail!("DRB leaf is missing the DRB result.");
848        };
849
850        Ok(drb)
851    }
852
853    fn add_drb_result(&mut self, epoch: Epoch, drb: DrbResult) {
854        let Some(raw_stake_table) = self.state.get(&epoch) else {
855            tracing::error!("add_drb_result({}, {:?}) was called, but we do not yet have the stake table for epoch {}", epoch, drb, epoch);
856            return;
857        };
858
859        let leaders = raw_stake_table
860            .eligible_leaders
861            .clone()
862            .into_iter()
863            .map(|peer_config| peer_config.stake_table_entry)
864            .collect::<Vec<_>>();
865        let randomized_committee = generate_stake_cdf(leaders, drb);
866
867        self.randomized_committees
868            .insert(epoch, randomized_committee);
869    }
870
871    fn set_first_epoch(&mut self, epoch: Epoch, initial_drb_result: DrbResult) {
872        self.first_epoch = Some(epoch);
873
874        let epoch_committee = self.state.get(&Epoch::genesis()).unwrap().clone();
875        self.state.insert(epoch, epoch_committee.clone());
876        self.state.insert(epoch + 1, epoch_committee);
877        self.add_drb_result(epoch, initial_drb_result);
878        self.add_drb_result(epoch + 1, initial_drb_result);
879    }
