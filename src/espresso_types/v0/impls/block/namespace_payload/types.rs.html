1//! Types related to a namespace payload and its transaction table.
2//!
3//! All code that needs to know the binary format of a namespace payload and its
4//! transaction table is restricted to this file.
5//!
6//! There are many newtypes in this file to facilitate transaction proofs.
7//!
8//! # Binary format of a namespace payload
9//!
10//! Any sequence of bytes is a valid [`NsPayload`].
11//!
12//! A namespace payload consists of two concatenated byte sequences:
13//! - `tx_table`: transaction table
14//! - `tx_payloads`: transaction payloads
15//!
16//! # Transaction table
17//!
18//! Byte lengths for the different items that could appear in a `tx_table` are
19//! specified in local private constants [`NUM_TXS_BYTE_LEN`],
20//! [`TX_OFFSET_BYTE_LEN`].
21//!
22//! ## Number of entries in the transaction table
23//!
24//! The first [`NUM_TXS_BYTE_LEN`] bytes of the `tx_table` indicate the number
25//! `n` of entries in the table as a little-endian unsigned integer. If the
26//! entire namespace payload byte length is smaller than [`NUM_TXS_BYTE_LEN`]
27//! then the missing bytes are zero-padded.
28//!
29//! The bytes in the namespace payload beyond the first [`NUM_TXS_BYTE_LEN`]
30//! bytes encode entries in the `tx_table`. Each entry consumes exactly
31//! [`TX_OFFSET_BYTE_LEN`] bytes.
32//!
33//! The number `n` could be anything, including a number much larger than the
34//! number of entries that could fit in the namespace payload. As such, the
35//! actual number of entries in the `tx_table` is defined as the minimum of `n`
36//! and the maximum number of whole `tx_table` entries that could fit in the
37//! namespace payload.
38//!
39//! The `tx_payloads` consist of any bytes in the namespace payload beyond the
40//! `tx_table`.
41//!
42//! ## Transaction table entry
43//!
44//! Each entry in the `tx_table` is exactly [`TX_OFFSET_BYTE_LEN`] bytes. These
45//! bytes indicate the end-index of a transaction in the namespace payload
46//! bytes. This end-index is a little-endian unsigned integer.
47//!
48//! This offset is relative to the end of the `tx_table` within the current
49//! namespace.
50//!
51//! ### Example
52//!
53//! Suppose a block payload has 3000 bytes and 3 namespaces of 1000 bytes each.
54//! Suppose the `tx_table` for final namespace in the block has byte length 100,
55//! and suppose an entry in that `tx_table` indicates an end-index of `10`. The
56//! actual end-index of that transaction relative to the current namespace is
57//! `110`: `10` bytes for the offset plus `100` bytes for the `tx_table`.
58//! Relative to the entire block payload, the end-index of that transaction is
59//! `2110`: `10` bytes for the offset plus `100` bytes for the `tx_table` plus
60//! `2000` bytes for this namespace.
61//!
62//! # How to deduce a transaction's byte range
63//!
64//! In order to extract the payload bytes of a single transaction `T` from the
65//! namespace payload one needs both the start- and end-indices for `T`.
66//!
67//! See [`TxPayloadRange::new`] for clarification. What follows is a description
68//! of what's implemented in [`TxPayloadRange::new`].
69//!
70//! If `T` occupies the `i`th entry in the `tx_table` for `i>0` then the
71//! start-index for `T` is defined as the end-index of the `(i-1)`th entry in
72//! the table.
73//!
74//! Thus, both start- and end-indices for any transaction `T` can be read from a
75//! contiguous, constant-size byte range in the `tx_table`. This property
76//! facilitates transaction proofs.
77//!
78//! The start-index of the 0th entry in the table is implicitly defined to be
79//! `0`.
80//!
81//! The start- and end-indices `(declared_start, declared_end)` declared in the
82//! `tx_table` could be anything. As such, the actual start- and end-indices
83//! `(start, end)` are defined so as to ensure that the byte range is
84//! well-defined and in-bounds for the namespace payload:
85//! ```ignore
86//! end = min(declared_end, namespace_payload_byte_length)
87//! start = min(declared_start, end)
88//! ```
89//!
90//! To get the byte range for `T` relative to the current namespace, the above
91//! range is translated by the byte length of the `tx_table` *as declared in the
92//! `tx_table` itself*, suitably truncated to fit within the current namespace.
93//!
94//! In particular, if the `tx_table` declares a huge number `n` of entries that
95//! cannot fit into the namespace payload then all transactions in this
96//! namespace have a zero-length byte range whose start- and end-indices are
97//! both `namespace_payload_byte_length`.
98//!
99//! In a "honestly-prepared" `tx_table` the end-index of the final transaction
100//! equals the byte length of the namespace payload minus the byte length of the
101//! `tx_table`. (Otherwise the namespace payload might have bytes that are not
102//! included in any transaction.)
103//!
104//! It is possible that a `tx_table` table could indicate two distinct
105//! transactions whose byte ranges overlap, though no "honestly-prepared"
106//! `tx_table` would do this.

121impl NumTxs {
122    /// Returns the minimum of:
123    /// - `num_txs`
124    /// - The maximum number of tx table entries that could fit in a namespace
125    ///   whose byte length is `byte_len`.
126    pub fn new(num_txs: &NumTxsUnchecked, byte_len: &NsPayloadByteLen) -> Self {
127        Self(std::cmp::min(
128            // Number of txs declared in the tx table
129            num_txs.0,
130            // Max number of tx table entries that could fit in the namespace payload
131            byte_len.0.saturating_sub(NUM_TXS_BYTE_LEN) / TX_OFFSET_BYTE_LEN,
132        ))
133    }
134
135    pub fn in_bounds(&self, index: &TxIndex) -> bool {
136        index.0 < self.0
137    }
138}

189impl TxTableEntries {
190    const TWO_ENTRIES_BYTE_LEN: usize = 2 * TX_OFFSET_BYTE_LEN;
191
192    pub fn to_payload_bytes(&self) -> Vec<u8> {
193        let mut bytes = Vec::with_capacity(Self::TWO_ENTRIES_BYTE_LEN);
194        if let Some(prev) = self.prev {
195            bytes.extend(usize_to_bytes::<TX_OFFSET_BYTE_LEN>(prev));
196        }
197        bytes.extend(usize_to_bytes::<TX_OFFSET_BYTE_LEN>(self.cur));
198        bytes
199    }
200}
201
202impl FromNsPayloadBytes<'_> for TxTableEntries {
203    fn from_payload_bytes(bytes: &[u8]) -> Self {
204        match bytes.len() {
205            TX_OFFSET_BYTE_LEN => Self {
206                cur: usize_from_bytes::<TX_OFFSET_BYTE_LEN>(bytes),
207                prev: None,
208            },
209            Self::TWO_ENTRIES_BYTE_LEN => Self {
210                cur: usize_from_bytes::<TX_OFFSET_BYTE_LEN>(&bytes[TX_OFFSET_BYTE_LEN..]),
211                prev: Some(usize_from_bytes::<TX_OFFSET_BYTE_LEN>(
212                    &bytes[..TX_OFFSET_BYTE_LEN],
213                )),
214            },
215            len => panic!(
216                "unexpected bytes len {} should be either {} or {}",
217                len,
218                TX_OFFSET_BYTE_LEN,
219                Self::TWO_ENTRIES_BYTE_LEN
220            ),
221        }
222    }
223}

225impl TxTableEntriesRange {
226    pub fn new(index: &TxIndex) -> Self {
227        let start = if index.0 == 0 {
228            // Special case: the desired range includes only one entry from
229            // the tx table: the first entry. This entry starts immediately
230            // following the bytes that encode the tx table length.
231            NUM_TXS_BYTE_LEN
232        } else {
233            // The desired range starts at the beginning of the previous tx
234            // table entry.
235            (index.0 - 1)
236                .saturating_mul(TX_OFFSET_BYTE_LEN)
237                .saturating_add(NUM_TXS_BYTE_LEN)
238        };
239        // The desired range ends at the end of this transaction's tx table entry
240        let end = index
241            .0
242            .saturating_add(1)
243            .saturating_mul(TX_OFFSET_BYTE_LEN)
244            .saturating_add(NUM_TXS_BYTE_LEN);
245        Self(start..end)
246    }
247}

269impl TxPayloadRange {
270    pub fn new(
271        num_txs: &NumTxsUnchecked,
272        tx_table_entries: &TxTableEntries,
273        byte_len: &NsPayloadByteLen,
274    ) -> Self {
275        let tx_table_byte_len = num_txs
276            .0
277            .saturating_mul(TX_OFFSET_BYTE_LEN)
278            .saturating_add(NUM_TXS_BYTE_LEN);
279        let end = tx_table_entries
280            .cur
281            .saturating_add(tx_table_byte_len)
282            .min(byte_len.0);
283        let start = tx_table_entries
284            .prev
285            .unwrap_or(0)
286            .saturating_add(tx_table_byte_len)
287            .min(end);
288        Self(start..end)
289    }
290}

326impl NsPayloadBuilder {
327    /// Add a transaction's payload to this namespace
328    pub fn append_tx(&mut self, tx: Transaction) {
329        self.tx_bodies.extend(tx.into_payload());
330        self.tx_table_entries
331            .extend(usize_to_bytes::<TX_OFFSET_BYTE_LEN>(self.tx_bodies.len()));
332    }
333
334    /// Serialize to bytes and consume self.
335    pub fn into_bytes(self) -> Vec<u8> {
336        let mut result = Vec::with_capacity(
337            NUM_TXS_BYTE_LEN + self.tx_table_entries.len() + self.tx_bodies.len(),
338        );
339        let num_txs = NumTxsUnchecked(self.tx_table_entries.len() / TX_OFFSET_BYTE_LEN);
340        result.extend(num_txs.to_payload_bytes());
341        result.extend(self.tx_table_entries);
342        result.extend(self.tx_bodies);
343        result
344    }
345
346    /// Byte length of a tx table header.
347    pub const fn tx_table_header_byte_len() -> usize {
348        NUM_TXS_BYTE_LEN
349    }
350
351    /// Byte length of a single tx table entry.
352    pub const fn tx_table_entry_byte_len() -> usize {
353        TX_OFFSET_BYTE_LEN
354    }
