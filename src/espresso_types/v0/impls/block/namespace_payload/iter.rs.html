29impl<'a> Iter<'a> {
30    pub fn new(block: &'a Payload) -> Self {
31        Self {
32            ns_iter: NsIter::new(&block.ns_table().len()).peekable(),
33            tx_iter: None,
34            block,
35        }
36    }
37}
38
39impl Iterator for Iter<'_> {
40    type Item = Index;
41
42    fn next(&mut self) -> Option<Self::Item> {
43        loop {
44            let Some(ns_index) = self.ns_iter.peek() else {
45                break None; // ns_iter consumed
46            };
47
48            if let Some(tx_index) = self
49                .tx_iter
50                .get_or_insert_with(|| self.block.ns_payload(ns_index).iter())
51                .next()
52            {
53                break Some(Index {
54                    ns_index: ns_index.clone(),
55                    tx_index,
56                });
57            }
58
59            self.tx_iter = None; // unset `tx_iter`; it's consumed for this namespace
60            self.ns_iter.next();
61        }
62    }
63}
