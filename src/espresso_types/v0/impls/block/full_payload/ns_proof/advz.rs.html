17impl ADVZNsProof {
18    /// Returns the payload bytes for the `index`th namespace, along with a
19    /// proof of correctness for those bytes. Returns `None` on error.
20    ///
21    /// The namespace payload [`NsPayloadOwned`] is included as a hidden field
22    /// in the returned [`NsProof`]. A conventional API would instead return
23    /// `(NsPayload, NsProof)` and [`NsProof`] would not contain the namespace
24    /// payload.
25    /// ([`TxProof::new`](crate::block::namespace_payload::TxProof::new)
26    /// conforms to this convention.) In the future we should change this API to
27    /// conform to convention. But that would require a change to our RPC
28    /// endpoint API at [`endpoints`](crate::api::endpoints), which is a hassle.
29    pub fn new(payload: &Payload, index: &NsIndex, common: &ADVZCommon) -> Option<ADVZNsProof> {
30        let payload_byte_len = payload.byte_len();
31        if !payload_byte_len.is_consistent(common) {
32            tracing::warn!(
33                "payload byte len {} inconsistent with common {}",
34                payload_byte_len,
35                ADVZScheme::get_payload_byte_len(common)
36            );
37            return None; // error: payload byte len inconsistent with common
38        }
39        if !payload.ns_table().in_bounds(index) {
40            tracing::warn!("ns_index {:?} out of bounds", index);
41            return None; // error: index out of bounds
42        }
43        let ns_payload_range = payload.ns_table().ns_range(index, &payload_byte_len);
44
45        // TODO vid_scheme() arg should be u32 to match get_num_storage_nodes
46        // https://github.com/EspressoSystems/HotShot/issues/3298
47        let vid = advz_scheme(
48            ADVZScheme::get_num_storage_nodes(common).try_into().ok()?, // error: failure to convert u32 to usize
49        );
50
51        let ns_proof = if ns_payload_range.as_block_range().is_empty() {
52            None
53        } else {
54            Some(
55                vid.payload_proof(payload.encode(), ns_payload_range.as_block_range())
56                    .ok()?, // error: internal to payload_proof()
57            )
58        };
59
60        Some(ADVZNsProof {
61            ns_index: index.clone(),
62            ns_payload: payload.read_ns_payload(&ns_payload_range).to_owned(),
63            ns_proof,
64        })
65    }
66
67    /// Verify a [`NsProof`] against a payload commitment. Returns `None` on
68    /// error or if verification fails.
69    ///
70    /// There is no [`NsPayload`](crate::block::namespace_payload::NsPayload)
71    /// arg because this data is already included in the [`NsProof`]. See
72    /// [`NsProof::new`] for discussion.
73    ///
74    /// If verification is successful then return `(Vec<Transaction>,
75    /// NamespaceId)` obtained by post-processing the underlying
76    /// [`NsPayload`](crate::block::namespace_payload::NsPayload). Why? This
77    /// method might be run by a client in a WASM environment who might be
78    /// running non-Rust code, in which case the client is unable to perform
79    /// this post-processing himself.
80    pub fn verify(
81        &self,
82        ns_table: &NsTable,
83        commit: &VidCommitment,
84        common: &ADVZCommon,
85    ) -> Option<(Vec<Transaction>, NamespaceId)> {
86        match commit {
87            VidCommitment::V0(commit) => {
88                ADVZScheme::is_consistent(commit, common).ok()?;
89                if !ns_table.in_bounds(&self.ns_index) {
90                    return None; // error: index out of bounds
91                }
92
93                let range = ns_table
94                    .ns_range(&self.ns_index, &PayloadByteLen::from_vid_common(common))
95                    .as_block_range();
96
97                match (&self.ns_proof, range.is_empty()) {
98                    (Some(proof), false) => {
99                        // TODO advz_scheme() arg should be u32 to match get_num_storage_nodes
100                        // https://github.com/EspressoSystems/HotShot/issues/3298
101                        let vid = advz_scheme(
102                            ADVZScheme::get_num_storage_nodes(common).try_into().ok()?, // error: failure to convert u32 to usize
103                        );
104
105                        vid.payload_verify(
106                            Statement {
107                                payload_subslice: self.ns_payload.as_bytes_slice(),
108                                range,
109                                commit,
110                                common,
111                            },
112                            proof,
113                        )
114                        .ok()? // error: internal to payload_verify()
115                        .ok()?; // verification failure
116                    },
117                    (None, true) => {}, // 0-length namespace, nothing to verify
118                    (None, false) => {
119                        tracing::error!(
120                            "ns verify: missing proof for nonempty ns payload range {:?}",
121                            range
122                        );
123                        return None;
124                    },
125                    (Some(_), true) => {
126                        tracing::error!("ns verify: unexpected proof for empty ns payload range");
127                        return None;
128                    },
129                }
130
131                // verification succeeded, return some data
132                let ns_id = ns_table.read_ns_id_unchecked(&self.ns_index);
133                Some((self.ns_payload.export_all_txs(&ns_id), ns_id))
134            },
135            VidCommitment::V1(_) => None,
136        }
137    }
138
139    /// Return all transactions in the namespace whose payload is proven by
140    /// `self`. The namespace ID for each returned [`Transaction`] is set to
141    /// `ns_id`.
142    ///
143    /// # Design warning
144    ///
145    /// This method relies on a promise that a [`NsProof`] stores the entire
146    /// namespace payload. If in the future we wish to remove the payload from a
147    /// [`NsProof`] then this method can no longer be supported.
148    ///
149    /// In that case, use the following a workaround:
150    /// - Given a [`NamespaceId`], get a [`NsIndex`] `i` via
151    ///   [`NsTable::find_ns_id`].
152    /// - Use `i` to get a
153    ///   [`NsPayload`](crate::block::namespace_payload::NsPayload) `p` via
154    ///   [`Payload::ns_payload`].
155    /// - Use `p` to get the desired [`Vec<Transaction>`] via
156    ///   [`NsPayload::export_all_txs`](crate::block::namespace_payload::NsPayload::export_all_txs).
157    ///
158    /// This workaround duplicates the work done in [`NsProof::new`]. If you
159    /// don't like that then you could instead hack [`NsProof::new`] to return a
160    /// pair `(NsProof, Vec<Transaction>)`.
161    pub fn export_all_txs(&self, ns_id: &NamespaceId) -> Vec<Transaction> {
162        self.ns_payload.export_all_txs(ns_id)
163    }
164}

166#[cfg(test)]
167mod tests {
168    use futures::future;
169    use hotshot::{helpers::initialize_logging, traits::BlockPayload};
170    use hotshot_types::{
171        data::VidCommitment,
172        traits::EncodeBytes,
173        vid::advz::{advz_scheme, ADVZScheme},
174    };
175    use jf_vid::{VidDisperse, VidScheme};
176
177    use crate::{v0::impls::block::test::ValidTest, v0_1::ADVZNsProof, Payload};
178
179    #[tokio::test(flavor = "multi_thread")]
180    async fn ns_proof() {
181        let test_cases = vec![
182            vec![
183                vec![5, 8, 8],
184                vec![7, 9, 11],
185                vec![10, 5, 8],
186                vec![7, 8, 9],
187                vec![],
188            ],
189            vec![vec![1, 2, 3], vec![4, 5, 6]],
190            vec![],
191        ];
192
193        initialize_logging();
194
195        let mut rng = jf_utils::test_rng();
196        let mut tests = ValidTest::many_from_tx_lengths(test_cases, &mut rng);
197
198        struct BlockInfo {
199            block: Payload,
200            vid: VidDisperse<ADVZScheme>,
201            ns_proofs: Vec<ADVZNsProof>,
202        }
203
204        let blocks: Vec<BlockInfo> = {
205            // compute blocks separately to avoid async error `captured variable
206            // cannot escape `FnMut` closure body` caused by mutable variable `vid`
207            // below.
208            let blocks_only = future::join_all(tests.iter().map(|t| async {
209                Payload::from_transactions(t.all_txs(), &Default::default(), &Default::default())
210                    .await
211                    .unwrap()
212                    .0
213            }))
214            .await;
215
216            let mut vid = advz_scheme(10);
217            blocks_only
218                .into_iter()
219                .map(|block| {
220                    let vid = vid.disperse(block.encode()).unwrap();
221                    let ns_proofs: Vec<ADVZNsProof> = block
222                        .ns_table()
223                        .iter()
224                        .map(|ns_index| ADVZNsProof::new(&block, &ns_index, &vid.common).unwrap())
225                        .collect();
226                    BlockInfo {
227                        block,
228                        vid,
229                        ns_proofs,
230                    }
231                })
232                .collect()
233        };
234
235        // sanity: verify all valid namespace proofs
236        for (
237            BlockInfo {
238                block,
239                vid,
240                ns_proofs,
241            },
242            test,
243        ) in blocks.iter().zip(tests.iter_mut())
244        {
245            for ns_proof in ns_proofs.iter() {
246                let ns_id = block.ns_table().read_ns_id(&ns_proof.ns_index).unwrap();
247                let txs = test
248                    .nss
249                    .remove(&ns_id)
250                    .unwrap_or_else(|| panic!("namespace {} missing from test", ns_id));
251
252                // verify ns_proof
253                let (ns_proof_txs, ns_proof_ns_id) = ns_proof
254                    .verify(
255                        block.ns_table(),
256                        &VidCommitment::V0(vid.commit),
257                        &vid.common,
258                    )
259                    .unwrap_or_else(|| panic!("namespace {} proof verification failure", ns_id));
260
261                assert_eq!(ns_proof_ns_id, ns_id);
262                assert_eq!(ns_proof_txs, txs);
263            }
264        }
265
266        assert!(blocks.len() >= 2, "need at least 2 test_cases");
267
268        let ns_proof_0_0 = &blocks[0].ns_proofs[0];
269        let ns_table_0 = blocks[0].block.ns_table();
270        let ns_table_1 = blocks[1].block.ns_table();
271        let vid_commit_0 = &VidCommitment::V0(blocks[0].vid.commit);
272        let vid_commit_1 = &VidCommitment::V0(blocks[1].vid.commit);
273        let vid_common_0 = &blocks[0].vid.common;
274        let vid_common_1 = &blocks[1].vid.common;
275
276        // mix and match ns_table, vid_commit, vid_common
277        {
278            // wrong ns_table
279            assert!(ns_proof_0_0
280                .verify(ns_table_1, vid_commit_0, vid_common_0)
281                .is_none());
282
283            // wrong vid commitment
284            assert!(ns_proof_0_0
285                .verify(ns_table_0, vid_commit_1, vid_common_0)
286                .is_none());
287
288            // wrong vid common
289            assert!(ns_proof_0_0
290                .verify(ns_table_0, vid_commit_0, vid_common_1)
291                .is_none());
292
293            // wrong ns_proof
294            assert!(ns_proof_0_0
295                .verify(ns_table_1, vid_commit_1, vid_common_1)
296                .is_none());
297        }
298
299        // hack the proof
300        {
301            ns_proof_0_0
302                .verify(ns_table_0, vid_commit_0, vid_common_0)
303                .expect("sanity: correct proof should succeed");
304
305            let wrong_ns_index_ns_proof_0_0 = ADVZNsProof {
306                ns_index: blocks[0].ns_proofs[1].ns_index.clone(),
307                ..ns_proof_0_0.clone()
308            };
309            assert!(wrong_ns_index_ns_proof_0_0
310                .verify(ns_table_0, vid_commit_0, vid_common_0)
311                .is_none());
312
313            let wrong_ns_payload_ns_proof_0_0 = ADVZNsProof {
314                ns_payload: blocks[0].ns_proofs[1].ns_payload.clone(),
315                ..ns_proof_0_0.clone()
316            };
317            assert!(wrong_ns_payload_ns_proof_0_0
318                .verify(ns_table_0, vid_commit_0, vid_common_0)
319                .is_none());
320
321            let wrong_proof_ns_proof_0_0 = ADVZNsProof {
322                ns_proof: blocks[0].ns_proofs[1].ns_proof.clone(),
323                ..ns_proof_0_0.clone()
324            };
325            assert!(wrong_proof_ns_proof_0_0
326                .verify(ns_table_0, vid_commit_0, vid_common_0)
327                .is_none());
328        }
329    }
