1//! This module contains the namespace proof implementation for the new AvidM scheme.
2
3use hotshot_types::{data::VidCommitment, vid::avidm::AvidMCommon};
4use vid::avid_m::namespaced::NsAvidMScheme;
5
6use crate::{v0_3::AvidMNsProof, NamespaceId, NsIndex, NsPayload, NsTable, Payload, Transaction};
7
8impl AvidMNsProof {
9    pub fn new(payload: &Payload, index: &NsIndex, common: &AvidMCommon) -> Option<AvidMNsProof> {
10        let payload_byte_len = payload.byte_len();
11        let index = index.0;
12        let ns_table = payload.ns_table();
13        let ns_table = ns_table
14            .iter()
15            .map(|index| ns_table.ns_range(&index, &payload_byte_len).0)
16            .collect::<Vec<_>>();
17
18        if index >= ns_table.len() {
19            tracing::warn!("ns_index {:?} out of bounds", index);
20            return None; // error: index out of bounds
21        }
22
23        if ns_table[index].is_empty() {
24            None
25        } else {
26            match NsAvidMScheme::namespace_proof(common, &payload.raw_payload, index, ns_table) {
27                Ok(proof) => Some(AvidMNsProof(proof)),
28                Err(e) => {
29                    tracing::error!("error generating namespace proof: {:?}", e);
30                    None
31                },
32            }
33        }
34    }
35
36    /// Unlike the ADVZ scheme, this function won't fail with a wrong `ns_table`.
37    /// It only uses `ns_table` to get the namespace id.
38    pub fn verify(
39        &self,
40        ns_table: &NsTable,
41        commit: &VidCommitment,
42        common: &AvidMCommon,
43    ) -> Option<(Vec<Transaction>, NamespaceId)> {
44        match commit {
45            VidCommitment::V1(commit) => {
46                match NsAvidMScheme::verify_namespace_proof(common, commit, &self.0) {
47                    Ok(Ok(_)) => {
48                        let ns_id = ns_table.read_ns_id(&NsIndex(self.0.ns_index))?;
49                        let ns_payload = NsPayload::from_bytes_slice(&self.0.ns_payload);
50                        Some((ns_payload.export_all_txs(&ns_id), ns_id))
51                    },
52                    Ok(Err(_)) => None,
53                    Err(e) => {
54                        tracing::warn!("error verifying namespace proof: {:?}", e);
55                        None
56                    },
57                }
58            },
59            _ => None,
60        }
61    }
62}

64/// Copied from ADVZNsProof tests.
65#[cfg(test)]
66mod tests {
67    use futures::future;
68    use hotshot::{helpers::initialize_logging, traits::BlockPayload};
69    use hotshot_types::{
70        data::VidCommitment,
71        traits::EncodeBytes,

75    use crate::{v0::impls::block::test::ValidTest, v0_3::AvidMNsProof, NsIndex, Payload};
76
77    #[tokio::test(flavor = "multi_thread")]
78    async fn ns_proof() {
79        let test_cases = vec![
80            vec![
81                vec![5, 8, 8],
82                vec![7, 9, 11],
83                vec![10, 5, 8],
84                vec![7, 8, 9],
85                vec![],
86            ],
87            vec![vec![1, 2, 3], vec![4, 5, 6]],
88            vec![],
89        ];
90
91        initialize_logging();
92
93        let mut rng = jf_utils::test_rng();
94        let mut tests = ValidTest::many_from_tx_lengths(test_cases, &mut rng);
95
96        let param = AvidMParam::new(5usize, 10usize).unwrap();
97
98        struct BlockInfo {
99            block: Payload,
100            vid_commit: VidCommitment,
101            ns_proofs: Vec<AvidMNsProof>,
102        }
103
104        let blocks: Vec<BlockInfo> = future::join_all(tests.iter().map(|t| async {
105            let block =
106                Payload::from_transactions(t.all_txs(), &Default::default(), &Default::default())
107                    .await
108                    .unwrap()
109                    .0;
110            let payload_byte_len = block.byte_len();
111            let ns_table = block.ns_table();
112            let ns_table = ns_table
113                .iter()
114                .map(|index| ns_table.ns_range(&index, &payload_byte_len).0)
115                .collect::<Vec<_>>();
116            let vid_commit = AvidMScheme::commit(&param, &block.encode(), ns_table).unwrap();
117            let ns_proofs: Vec<AvidMNsProof> = block
118                .ns_table()
119                .iter()
120                .map(|ns_index| AvidMNsProof::new(&block, &ns_index, &param).unwrap())
121                .collect();
122            BlockInfo {
123                block,
124                vid_commit: VidCommitment::V1(vid_commit),
125                ns_proofs,
126            }
127        }))
128        .await;
129
130        // sanity: verify all valid namespace proofs
131        for (
132            BlockInfo {
133                block,
134                vid_commit,
135                ns_proofs,
136            },
137            test,
138        ) in blocks.iter().zip(tests.iter_mut())
139        {
140            for ns_proof in ns_proofs.iter() {
141                let ns_id = block
142                    .ns_table()
143                    .read_ns_id(&NsIndex(ns_proof.0.ns_index))
144                    .unwrap();
145                let txs = test
146                    .nss
147                    .remove(&ns_id)
148                    .unwrap_or_else(|| panic!("namespace {} missing from test", ns_id));
149
150                // verify ns_proof
151                let (ns_proof_txs, ns_proof_ns_id) = ns_proof
152                    .verify(block.ns_table(), vid_commit, &param)
153                    .unwrap_or_else(|| panic!("namespace {} proof verification failure", ns_id));
154
155                assert_eq!(ns_proof_ns_id, ns_id);
156                assert_eq!(ns_proof_txs, txs);
157            }
158        }
159
160        assert!(blocks.len() >= 2, "need at least 2 test_cases");
161
162        let ns_proof_0_0 = &blocks[0].ns_proofs[0];
163        let ns_table_0 = blocks[0].block.ns_table();
164        let ns_table_1 = blocks[1].block.ns_table();
165        let vid_commit_1 = &blocks[1].vid_commit;
166
167        // mix and match ns_table, vid_commit, vid_common
168        {
169            // wrong vid commitment
170            assert!(ns_proof_0_0
171                .verify(ns_table_0, vid_commit_1, &param)
172                .is_none());
173
174            // wrong ns_proof
175            assert!(ns_proof_0_0
176                .verify(ns_table_1, vid_commit_1, &param)
177                .is_none());
178        }
179    }
