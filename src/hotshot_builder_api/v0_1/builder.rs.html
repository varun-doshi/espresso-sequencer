44#[derive(Clone, Debug, Error, Deserialize, Serialize)]
45pub enum BuildError {
46    #[error("The requested resource does not exist or is not known to this builder service")]
47    NotFound,
48    #[error("The requested resource exists but is not currently available")]
49    Missing,
50    #[error("Error trying to fetch the requested resource: {0}")]
51    Error(String),
52}
53
54/// Enum to keep track on status of a transaction
55#[derive(Clone, Debug, Deserialize, Serialize, PartialEq)]
56pub enum TransactionStatus {
57    Pending,
58    Sequenced { leaf: u64 },
59    Rejected { reason: String }, // Rejection reason is in the String format
60    Unknown,
61}
62
63#[derive(Clone, Debug, Error, Deserialize, Serialize)]
64pub enum Error {
65    #[error("Error processing request: {0}")]
66    Request(#[from] RequestError),
67    #[error("Error building block from {resource}: {source}")]
68    BlockAvailable {
69        source: BuildError,
70        resource: String,
71    },
72    #[error("Error claiming block {resource}: {source}")]
73    BlockClaim {
74        source: BuildError,
75        resource: String,
76    },
77    #[error("Error unpacking transactions: {0}")]
78    TxnUnpack(RequestError),
79    #[error("Error submitting transaction: {0}")]
80    TxnSubmit(BuildError),
81    #[error("Error getting builder address: {0}")]
82    BuilderAddress(#[from] BuildError),
83    #[error("Error getting transaction status: {0}")]
84    TxnStat(BuildError),
85    #[error("Custom error {status}: {message}")]
86    Custom { message: String, status: StatusCode },
87}
88
89impl tide_disco::error::Error for Error {
90    fn catch_all(status: StatusCode, msg: String) -> Self {
91        Error::Custom {
92            message: msg,
93            status,
94        }
95    }
96
97    fn status(&self) -> StatusCode {
98        match self {
99            Error::Request { .. } => StatusCode::BAD_REQUEST,
100            Error::BlockAvailable { source, .. } | Error::BlockClaim { source, .. } => match source
101            {
102                BuildError::NotFound => StatusCode::NOT_FOUND,
103                BuildError::Missing => StatusCode::NOT_FOUND,
104                BuildError::Error { .. } => StatusCode::INTERNAL_SERVER_ERROR,
105            },
106            Error::TxnUnpack { .. } => StatusCode::BAD_REQUEST,
107            Error::TxnSubmit { .. } => StatusCode::INTERNAL_SERVER_ERROR,
108            Error::Custom { .. } => StatusCode::INTERNAL_SERVER_ERROR,
109            Error::BuilderAddress { .. } => StatusCode::INTERNAL_SERVER_ERROR,
110            Error::TxnStat { .. } => StatusCode::INTERNAL_SERVER_ERROR,
111        }
112    }
113}

115pub(crate) fn try_extract_param<T: for<'a> TryFrom<&'a TaggedBase64>>(
116    params: &RequestParams,
117    param_name: &str,
118) -> Result<T, Error> {
119    params
120        .param(param_name)?
121        .as_tagged_base64()?
122        .try_into()
123        .map_err(|_| Error::Custom {
124            message: format!("Invalid {param_name}"),
125            status: StatusCode::UNPROCESSABLE_ENTITY,
126        })
127}

129pub fn define_api<State, Types: NodeType>(
130    options: &Options,
131) -> Result<Api<State, Error, Version>, ApiError>
132where
133    State: 'static + Send + Sync + ReadState,
134    <State as ReadState>::State: Send + Sync + BuilderDataSource<Types>,
135{
136    let mut api = load_api::<State, Error, Version>(
137        options.api_path.as_ref(),
138        include_str!("../../api/v0_1/builder.toml"),
139        options.extensions.clone(),
140    )?;
141    api.with_version("0.1.0".parse().unwrap())
142        .get("available_blocks", |req, state| {
143            async move {
144                let hash = req.blob_param("parent_hash")?;
145                let view_number = req.integer_param("view_number")?;
146                let signature = try_extract_param(&req, "signature")?;
147                let sender = try_extract_param(&req, "sender")?;
148                state
149                    .available_blocks(&hash, view_number, sender, &signature)
150                    .await
151                    .map_err(|source| Error::BlockAvailable {
152                        source,
153                        resource: hash.to_string(),
154                    })
155            }
156            .boxed()
157        })?
158        .get("claim_block", |req, state| {
159            async move {
160                let block_hash: BuilderCommitment = req.blob_param("block_hash")?;
161                let view_number = req.integer_param("view_number")?;
162                let signature = try_extract_param(&req, "signature")?;
163                let sender = try_extract_param(&req, "sender")?;
164                state
165                    .claim_block(&block_hash, view_number, sender, &signature)
166                    .await
167                    .map_err(|source| Error::BlockClaim {
168                        source,
169                        resource: block_hash.to_string(),
170                    })
171            }
172            .boxed()
173        })?
174        .get("claim_block_with_num_nodes", |req, state| {
175            async move {
176                let block_hash: BuilderCommitment = req.blob_param("block_hash")?;
177                let view_number = req.integer_param("view_number")?;
178                let signature = try_extract_param(&req, "signature")?;
179                let sender = try_extract_param(&req, "sender")?;
180                let num_nodes = req.integer_param("num_nodes")?;
181                state
182                    .claim_block_with_num_nodes(
183                        &block_hash,
184                        view_number,
185                        sender,
186                        &signature,
187                        num_nodes,
188                    )
189                    .await
190                    .map_err(|source| Error::BlockClaim {
191                        source,
192                        resource: block_hash.to_string(),
193                    })
194            }
195            .boxed()
196        })?
197        .get("claim_header_input", |req, state| {
198            async move {
199                let block_hash: BuilderCommitment = req.blob_param("block_hash")?;
200                let view_number = req.integer_param("view_number")?;
201                let signature = try_extract_param(&req, "signature")?;
202                let sender = try_extract_param(&req, "sender")?;
203                state
204                    .claim_block_header_input(&block_hash, view_number, sender, &signature)
205                    .await
206                    .map_err(|source| Error::BlockClaim {
207                        source,
208                        resource: block_hash.to_string(),
209                    })
210            }
211            .boxed()
212        })?
213        .get("claim_header_input_v2", |req, state| {
214            async move {
215                let block_hash: BuilderCommitment = req.blob_param("block_hash")?;
216                let view_number = req.integer_param("view_number")?;
217                let signature = try_extract_param(&req, "signature")?;
218                let sender = try_extract_param(&req, "sender")?;
219                let out = state
220                    .claim_block_header_input(&block_hash, view_number, sender, &signature)
221                    .await
222                    .map_err(|source| Error::BlockClaim {
223                        source,
224                        resource: block_hash.to_string(),
225                    });
226
227                out.map(|input| AvailableBlockHeaderInputV2::<Types> {
228                    fee_signature: input.fee_signature,
229                    sender: input.sender,
230                })
231            }
232            .boxed()
233        })?
234        .get("builder_address", |_req, state| {
235            async move { state.builder_address().await.map_err(|e| e.into()) }.boxed()
236        })?;
237    Ok(api)
238}

240pub fn submit_api<State, Types: NodeType, Ver: StaticVersionType + 'static>(
241    options: &Options,
242) -> Result<Api<State, Error, Ver>, ApiError>
243where
244    State: 'static + Send + Sync + ReadState,
245    <State as ReadState>::State: Send + Sync + AcceptsTxnSubmits<Types>,
246{
247    let mut api = load_api::<State, Error, Ver>(
248        options.api_path.as_ref(),
249        include_str!("../../api/v0_1/submit.toml"),
250        options.extensions.clone(),
251    )?;
252    api.with_version("0.0.1".parse().unwrap())
253        .at("submit_txn", |req: RequestParams, state| {
254            async move {
255                let tx = req
256                    .body_auto::<<Types as NodeType>::Transaction, Ver>(Ver::instance())
257                    .map_err(Error::TxnUnpack)?;
258                let hash = tx.commit();
259                state
260                    .read(|state| state.submit_txns(vec![tx]))
261                    .await
262                    .map_err(Error::TxnSubmit)?;
263                Ok(hash)
264            }
265            .boxed()
266        })?
267        .at("submit_batch", |req: RequestParams, state| {
268            async move {
269                let txns = req
270                    .body_auto::<Vec<<Types as NodeType>::Transaction>, Ver>(Ver::instance())
271                    .map_err(Error::TxnUnpack)?;
272                let hashes = txns.iter().map(|tx| tx.commit()).collect::<Vec<_>>();
273                state
274                    .read(|state| state.submit_txns(txns))
275                    .await
276                    .map_err(Error::TxnSubmit)?;
277                Ok(hashes)
278            }
279            .boxed()
280        })?
281        .get("get_status", |req: RequestParams, state| {
282            async move {
283                let tx = req
284                    .body_auto::<<Types as NodeType>::Transaction, Ver>(Ver::instance())
285                    .map_err(Error::TxnUnpack)?;
286                let hash = tx.commit();
287                state.txn_status(hash).await.map_err(Error::TxnStat)
288            }
289            .boxed()
290        })?;
291    Ok(api)
