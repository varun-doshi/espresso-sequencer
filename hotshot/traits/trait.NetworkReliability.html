fn chaos_send_msg( &self, msg: Vec<u8>, send_fn: Arc<dyn Fn(Vec<u8>) -> Pin<Box<dyn Future<Output = ()> + Sync + Send>> + Sync + Send>, ) -> Pin<Box<dyn Future<Output = ()> + Sync + Send>>
